class Solution:
    def combinationSum4(self, nums: List[int], target: int) -> int:
        dp = [0] * (target+1)
        dp[-1] = 1 # base case

        for amt in range(target, -1, -1):
            for n in nums:
                if amt + n <= target:
                    dp[amt] += dp[amt+n]
        
        return dp[0]
