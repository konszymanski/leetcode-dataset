class Solution(object):
    def combinationSum4(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        dp = [0] * (target + 1)
        length = len(nums)
        
        for i in range (1, target + 1):
            for j in range(length):
                if i == nums[j]:
                    dp[i] += 1
                if i > nums[j]:
                    dp[i] += dp[i - nums[j]]
        
        return dp[target]
