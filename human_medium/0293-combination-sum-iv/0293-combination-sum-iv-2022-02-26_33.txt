class Solution:
    def combinationSum4(self, nums: List[int], target: int) -> int:
        
        #Index = current target amount we\'re working with
        #Value = number of ways we can get to this value from the numbers we have
        dp = [0] * (target + 1)
        
        #We seed dp with just one of each of the numbers we have
        for num in nums:
            if target - num >= 0:
                dp[num] = 1
                
        #Now we iterate through dp to check every single amount
        for targetAmount in range(target + 1):
            #And we also want to iterate through every single number so that we have every possible combination
            for num in nums:
                
                #If the targetAmount - num is greater than zero, we\'ve found a potential combiantion
                if targetAmount - num >= 0:
                    #if so, we add dp[targetAmount - num]. This value may end up being zero
                    dp[targetAmount] += dp[targetAmount - num]
        
        #We just return all the unique ways we can make up target
        return dp[target]
