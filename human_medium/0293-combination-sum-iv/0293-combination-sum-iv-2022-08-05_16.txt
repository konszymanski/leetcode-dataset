class Solution:
    def combinationSum4(self, nums: List[int], target: int) -> int:
        
        dp = defaultdict(int)
        
        return self.rec(nums, target, dp)
    
    def rec(self, nums, target, dp):
        
        if target == 0:
            return 1
        
        if target < 0:
            return 0
        
        if target in dp:
            return dp[target]
        
        for num in nums:
            dp[target] += self.rec(nums, target-num, dp)
        
        return dp[target]
