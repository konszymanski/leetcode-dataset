class Solution:
    def combinationSum4(self, nums: List[int], target: int) -> int:
        dp = [0 for i in range(target+1)]
        nums.sort()
        for i in range(1, target+1):
            if i in nums:
                dp[i] = 1
            dp[i] += sum([dp[i-nums[j]] for j in range(len(nums)) if i-nums[j]>0 ])
        return dp[-1]
