class Solution:
    def combinationSum4(self, nums: List[int], target: int) -> int:
        dp = [0] * (target + 1)
        for i in range(1, target + 1):
            for each in nums:
                if i == each:
                    dp[i] += 1
                elif i - each > 0:
                    dp[i] += dp[i - each]
        return dp[target]
