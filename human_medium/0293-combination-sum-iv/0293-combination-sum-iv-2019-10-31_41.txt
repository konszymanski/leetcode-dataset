count = 0
dp = {}

def dp_recur(nums, target, dp):
	if target in dp:
		return dp[target]
	else:
		count = 0
		for x in nums:
			if x <= target:
				new_target = target - x
				if new_target == 0:
					count += 1
					continue 
				count += dp_recur(nums, new_target, dp)

		dp[target] = count
	return count

count = dp_recur(nums, target, dp)
# print (dp)

return count
