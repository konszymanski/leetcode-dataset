# time complexity O(nm), where n=target, m=len(nums)
class Solution:
    def combinationSum4(self, nums: List[int], target: int) -> int:
        
        # dp array storing number of combinations for amount \'i\'
        d = [0 for i in range(target+1)]

        # update from amount 1 to target
        for i in range(1, target+1):
            # for each number given
            for num in nums:
                # if amount is larger than number
                if i-num > 0:
                    # add number of combimations at amount \'i-num\'
                    d[i] += d[i-num]
                # add one since there exist num equal to amount \'i\'
                elif i-num == 0:
                    d[i] += 1

        # return number of combinations at amount \'i\'
        return d[target]
