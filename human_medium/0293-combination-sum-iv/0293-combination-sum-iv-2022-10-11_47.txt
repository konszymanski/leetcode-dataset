int combinationSum4(vector<int>& nums, int target) {
	vector<uint> combinationCache(target + 1, 0);
	combinationCache[0] = 1;

	for(int currSum = 1; currSum <= target; currSum++){
		for(int num : nums){
			int prevSum = currSum - num;
			if(prevSum < 0 || !combinationCache[prevSum]){
				continue;
			}

			combinationCache[currSum] += combinationCache[prevSum];
		}
	}

	return combinationCache[target];
}
