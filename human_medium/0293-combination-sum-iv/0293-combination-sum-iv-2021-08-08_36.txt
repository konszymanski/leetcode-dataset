class Solution(object):
    def combinationSum4(self, nums, target):
        """
        :type nums: List[int]
        :type target: int
        :rtype: int
        """
        
        counter = Counter(nums)
        dp = []
        for i in range(target+1):
            dp.append(0)
        
        all_nums = set(nums)
        all_nums = list(all_nums)
		#remove duplicates, and then sort
		#sets are unordered in nature
        all_nums.sort()
        #no duplicates
        for step in range(1, target+1):
            step_sum = 0
            
            for num in all_nums:
                if num > step:
                    break
                else:
                    prev = step - num
                    count = counter[num]
                    if prev == 0:
                        step_sum += count
                    else:
						#can use a number with the same value for however many times it appears in nums
                        step_sum += dp[prev] * count
                
            dp[step] = step_sum
        
       
        return dp[target]
