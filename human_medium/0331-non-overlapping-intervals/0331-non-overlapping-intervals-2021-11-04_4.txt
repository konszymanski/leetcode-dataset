def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort()
        cnt = 0
        prevEnd = -math.inf
        for currStart,currEnd in intervals:
            if prevEnd > currStart: # overlapping condition
                cnt +=1 # removed one interval
                prevEnd = min(prevEnd,currEnd) # interval with longer interval is removed => consider interval with min end time
            else:
                prevEnd = currEnd
        return cnt
