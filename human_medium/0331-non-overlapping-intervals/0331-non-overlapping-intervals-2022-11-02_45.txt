# Just the variation of merge-innterval problem 
        # only key take away is we\'d like to minimize the interval so we have to remove minimum no. of intervals
        T(c)=O(nlog(n)*n)
        
        intervals.sort()
        # print(intervals)
        res=[]
        count=0
        for i in range(len(intervals)):
            if len(res)!=0 and intervals[i][0]<res[-1][1]:
                count+=1
                
                # minimize the 2nd value by taking min of interval and res
                res[-1][1]=min(res[-1][1],intervals[i][1])
            else:
                res.append(intervals[i])
        return count
