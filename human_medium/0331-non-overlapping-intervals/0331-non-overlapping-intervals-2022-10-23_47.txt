class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort()
        answer = 0
        for i in range(1, len(intervals)):
            if intervals[i][0] < intervals[i-1][1]:
                if intervals[i][1] > intervals[i-1][1]:
                    intervals[i] = intervals[i-1]
                answer += 1
        return answer
