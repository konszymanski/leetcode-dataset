class Solution:
    
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
       
        # Init
        n = len(intervals)
        intervals.sort()
        c = 0
        i = 0
        
        # Loop for each elements
        while i < n:
            
            # For indexes not 0
            if i:
                
                # Current interva
                curr = intervals[i]
                
                # In case of overlapping intervals
                if prev[1] > curr[0]:
                    
                    c += 1 # remove one of them, increment counter
                    
                    prev = prev if prev[1] < curr[1] else curr # remove the one which is longer
                    
                else: # In case of non-overlapping intervals update prev
                    
                    prev = curr # prev = curr
            
            else: # Base case
                
                prev = intervals[i] # prev
            
            # Increment index
            i += 1
       
        return c
