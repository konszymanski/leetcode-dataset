class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        heap=intervals
        heapq.heapify(heap)
        count=0    
        if heap:
            prevEnd=heapq.heappop(heap)[1]
            
        while heap:
            curr=heapq.heappop(heap)
            if curr[0]>=prevEnd:
                prevEnd=curr[1]
            else:
                count+=1
                prevEnd=min(curr[1],prevEnd)
        return count
