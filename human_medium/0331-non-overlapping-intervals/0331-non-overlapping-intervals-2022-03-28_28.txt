class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort()
        temp=intervals[0]
        count=0
        for i in range(1, len(intervals)):
            if intervals[i][0]<temp[1]:
                if temp[1]>intervals[i][1]:
                    temp=intervals[i]
                count+=1
                continue
            temp=intervals[i]
        return count
