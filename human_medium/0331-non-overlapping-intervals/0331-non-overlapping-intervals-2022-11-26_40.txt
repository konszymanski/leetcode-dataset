class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        # sort in ascending order by end time
        intervals.sort(key = lambda x: x[1])
        end, res = -math.inf, 0
        for interval in intervals:
            if end > interval[0]:
                res += 1
            else:
                end = interval[1]
        return res
