class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        sorted_intervals = sorted(intervals)
        non_conflicting = []

        for i in sorted_intervals:
            if not non_conflicting or i[0] >= non_conflicting[-1][1]:
                non_conflicting.append(i)
            elif i[1] < non_conflicting[-1][1]:
                non_conflicting.pop()
                non_conflicting.append(i)

        return len(sorted_intervals) - len(non_conflicting)
