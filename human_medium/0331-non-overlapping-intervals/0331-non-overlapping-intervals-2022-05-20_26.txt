class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort(key=lambda x: [x[1], x[0]])

        curr_end, count = -float(\'inf\'), 0
        for s, e in intervals:
            if s < curr_end:
                count += 1
            else:
                curr_end = e

        return count
