class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        n = len(intervals)
        if n<=0:
            return 0
        intervals.sort()
        ans = 0
        end = intervals[0][1]
        print(intervals)
        for i in range(1, n):
            if intervals[i][0] < end:
                ans += 1
                end = min(end, intervals[i][1])
            else:
                end = intervals[i][1]
        return ans
