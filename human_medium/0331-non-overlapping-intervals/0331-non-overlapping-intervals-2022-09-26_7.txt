class Solution:
    def __init__(self):
        self.hasBeenSorted = False
    
    def eraseOverlapIntervals(self, intervals):
        
        if not self.hasBeenSorted:
            intervals.sort()
            self.hasBeenSorted = True
        
        #Base case
		if len(intervals) == 1:
            return 0

        #First element doesn\'t overlap: it doesn\'t impact the final result
        if intervals[0][1] <= intervals[1][0]:
            return self.eraseOverlapIntervals(intervals[1:])
        
        #First element overlaps: either we remove it, or we remove the next one
		if intervals[1][1] > intervals[0][1]:
			intervals.pop(1)
		else:
			intervals.pop(0)
		return 1 + self.eraseOverlapIntervals(intervals)
