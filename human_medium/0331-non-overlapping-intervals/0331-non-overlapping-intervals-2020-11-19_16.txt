class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        ans, cur = 0, 0
        intervals.sort(key = lambda x: x[0])
        for i in range(1,len(intervals)):
            if intervals[cur][1] > intervals[i][0]:
                cur = cur if intervals[i][1] > intervals[cur][1] else i
                ans += 1
            else:
                cur = i
        return ans
