class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        
        countOverlap = 0
        intervals.sort()

        prevStart, prevEnd = intervals[0]
        
        for start, end in intervals[1:]:
            if start < prevEnd:
                countOverlap += 1
                prevEnd = min(prevEnd, end)
            
            else:
                prevEnd = end
            
        return countOverlap
