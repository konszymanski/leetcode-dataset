class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort()
        count = 0
        prev = intervals[0]
        
        for i in range(1, len(intervals)):
            cur = intervals[i]
            
            # if they overlap increment the count
            if prev[1] > cur[0]:
                count += 1
                
                # if prev interval has larger end value, shift prev to cur
                # this will make sure we removed the largest interval greedly
                if prev[1] > cur[1]:
                    prev = cur              
            else:
                prev = cur
                
        return  count
