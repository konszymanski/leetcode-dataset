class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        count = 0
        intervals.sort()
        _, hi = intervals[0]
        for i in range(1, len(intervals)):
            start, end = intervals[i]
            if start >= hi:
                hi = end
            else:
                hi = min(hi, end)
                count += 1
        return count
