class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        max_count = len(intervals)
        if max_count  <= 1:
            return 0
        arr = sorted(intervals, key=lambda x: x[1])
        counter = 0
        last_end = float("-inf")
        for elem in arr:
            if elem[0] >= last_end:
                last_end = elem[1]
                counter += 1
        return max_count - counter
