class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        sorted_intervals = sorted(intervals, key=lambda val: val[0])
        index, count = 1, 0
        # perform greedy search
        prev_interval = sorted_intervals[0]
        while index < len(intervals):
            if prev_interval[1] > sorted_intervals[index][0]:
                if prev_interval[1] >= sorted_intervals[index][1]:
                    prev_interval = sorted_intervals[index]
                count += 1
            else:
                prev_interval = sorted_intervals[index]
            index += 1
        return count
