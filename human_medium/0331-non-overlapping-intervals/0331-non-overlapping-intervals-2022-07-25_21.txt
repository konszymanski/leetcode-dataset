class Solution:
    def eraseOverlapIntervals(self, intervals):
        if not intervals:
            return 0
        intervals.sort()
        r = intervals[0][1]
        ans = 0
        for l in intervals[1:]:
            if l[0] < r:
                ans +=1
                r = min(r,l[1])
            else:
                r= l[1]
        return ans
