class Solution:
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        # sort intervals by start -> O(nlogn)
        intervals.sort(key=lambda intr: intr[0])
        
        overlap_count = 0
        last_end = intervals[0][1]
                
        # iterate intervals -> O(n)
        for start, end in intervals[1:]:
            # if curr interval overlap prev interval
            if last_end > start:
                # increase counter
                overlap_count += 1
                
                # remove the bigger interval (all we actually is updating the last end value)
                last_end = min(last_end, end)
            else:
                last_end = end
        
        return overlap_count
