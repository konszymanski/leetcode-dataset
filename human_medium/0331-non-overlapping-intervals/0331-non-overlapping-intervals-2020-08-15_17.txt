class Solution(object):
    def eraseOverlapIntervals(self, intervals):
        """
        :type intervals: List[List[int]]
        :rtype: int
        """
        # Sort the list in the intervals with the second item in the list item in ascending order
        if len(intervals) == 0:
            return 0
        sort_by_finish_time = sorted(intervals, key = lambda lst: lst[1])
        # Initialize max_intervals which will contail all compatible items
        max_intervals = [sort_by_finish_time.pop(0)]
        # Loop through the rest of the sorted items and add compatible item to max_intervals
        for item in sort_by_finish_time:
            current = max_intervals[-1]
            if item[0] >= current[1]:
                max_intervals.append(item)
        return len(intervals) - len(max_intervals)
