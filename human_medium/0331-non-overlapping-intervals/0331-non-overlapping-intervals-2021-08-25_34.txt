class Solution:            
    def eraseOverlapIntervals(self, intervals: List[List[int]]) -> int:
        intervals.sort()    
		# Like other intervals question, let\'s get the intervals sorted in asending order. 
        
        keep = [intervals[0]]
        for i in range(1, len(intervals)):
			# We start from comparing the previous interval with current interval 
			# from the second interval in the intervals array. 
            last = keep.pop()
            curr = intervals[i]
            if last[1] > curr[0]:
				# These two interval are overlap. 
                if last[1] > curr[1]:
					# If the previous interval end time is later then curr one, the best way to do is delete the previous interval. 
					# Thus, we do not put the previous one into stack 
                    keep.append(curr)
                else:
					# If the previous interval end time is earlier then curr one, the best way to do is delete the curr interval. 
					# Thus, we do not put the curr one into stack 
                    keep.append(last)
            else:
				# There is no overlap. We put both intervals into stack 
                keep.append(last)
                keep.append(intervals[i])
        return len(intervals) - len(keep)
