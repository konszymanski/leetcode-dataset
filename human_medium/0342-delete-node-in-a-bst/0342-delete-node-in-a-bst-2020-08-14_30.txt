class Solution:
    def deleteNode(self, root: TreeNode, key: int) -> TreeNode:
        
        # edge case
        if not root:return None
        
        # this method will return the Correct Node which will be replaced by the deleted Node
        def helper(root, add_point):            
            
            if not root:
                return add_point            
            
            tmp = root            
            while tmp.left:
                tmp = tmp.left            
            tmp.left = add_point        
            
            return root
        
        
        # if root is the key then root.right will be the root to be returned
        if root.val == key: return helper(root.right, root.left)
        

        dq = deque()
        dq.append(root)        
        
        while dq:            
            cur_root = dq.popleft()
                        
            # if key is in left side
            if cur_root.left and key < cur_root.val:
                if cur_root.left.val == key:                
                    cur_root.left = helper(cur_root.left.right, cur_root.left.left)        
                    break                    
                else:
                    dq.append(cur_root.left)
                        
            # if key is in right side
            if cur_root.right and key > cur_root.val:
                if cur_root.right.val == key:                    
                    cur_root.right = helper(cur_root.right.right, cur_root.right.left)    
                    break                    
                else:
                    dq.append(cur_root.right)
                                
        return root
