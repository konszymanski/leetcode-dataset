class Solution:
    def deleteNode(self, root: Optional[TreeNode], val: int) -> Optional[TreeNode]:
        if (root == None):
            return None
        if (root.val == val):
            return self.helper(root)
        dummy = root
        while root:
            if (root.val > val):
                if (root.left != None and root.left.val == val):
                    root.left = self.helper(root.left)
                    break
                else:
                    root = root.left
            else:
                if (root.right != None and root.right.val == val):
                    root.right = self.helper(root.right)
                    break
                else:
                    root = root.right

        return dummy
        
    def helper(self,root):
        if (root.left == None):
            return root.right
        elif (root.right == None):
            return root.left

        right_child = root.right
        last_right = self.check_last_right(root.left)
        last_right.right = right_child
        return root.left


    def check_last_right(self,root):
        if root.right == None:
            return root
        return self.check_last_right(root.right)
