class Solution(object):
    def getMinNode(self, root):
        minNode = root
        while minNode.left != None:
            minNode = minNode.left
        return minNode
    
    def deleteNode(self, root, key):
        """
        :type root: TreeNode
        :type key: int
        :rtype: TreeNode
        """
        if root == None:
            return None
        
        if root.val == key:
            # when the target node only has one child or no child
            if root.left == None:
                return root.right
            if root.right == None:
                return root.left
            # when the target node has two children
            minNode = self.getMinNode(root.right) # get the min node on the right child so we can swap it with the target root
            root.right = self.deleteNode(root.right, minNode.val) # remove the min node from the tree
            minNode.left = root.left
            minNode.right = root.right # these two steps re-link the min node to the tree
            root = minNode # redirect the root to the min node because we\'ll return it at the end of the method
        elif key < root.val:
            root.left = self.deleteNode(root.left, key)
        elif key > root.val:
            root.right = self.deleteNode(root.right, key)
        return root
