class Solution:
    def deleteNode(self, root: TreeNode, key: int) -> TreeNode:
        return self.delnode(root, key)        
        
    def delnode(self, root, key):
        if not root:
            return
        if root.val > key:
            root.left = self.delnode(root.left, key)
        elif root.val < key:
            root.right = self.delnode(root.right, key)
        else:
            if not root.left:
                return root.right
            elif not root.right:
                return root.left
            l = root.left
            r = root.right
            root = self.right_smallest(root)
            root.left = l
            root.right = r if root is not r else r.right
        return root
                
            
    def right_smallest(self, root): #return and delete the right smallest node
        parent = root
        ptr = root.right
        if not ptr.left:
            return ptr
        while ptr.left:
            parent = ptr
            ptr = ptr.left
        parent.left = None if not ptr.right else ptr.right
        return ptr
