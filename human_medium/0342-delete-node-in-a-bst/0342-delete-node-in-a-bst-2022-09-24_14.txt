# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def deleteNode(self, root: Optional[TreeNode], key: int) -> Optional[TreeNode]:
        if not root: return 
        parent = TreeNode(0)
        parent.left = root
        nodes = [(root, parent, True)]
        while nodes:
            n, p, left = nodes.pop()
            if n.val > key and n.left: nodes.append((n.left, n, True))
            elif n.val == key:
                if not (n.left or n.right):
                    if left: p.left = None
                    else: p.right = None
                elif not n.left:
                    if left: p.left = n.right
                    else: p.right = n.right
                elif not n.right:
                    if left: p.left = n.left
                    else: p.right = n.left
                else:
                    L = n.left
                    while L.right: 
                        L = L.right
                    if left: p.left, L.right = n.left, n.right
                    else: p.right, L.right = n.left, n.right
            elif n.right: nodes.append((n.right, n, False))
        return parent.left
