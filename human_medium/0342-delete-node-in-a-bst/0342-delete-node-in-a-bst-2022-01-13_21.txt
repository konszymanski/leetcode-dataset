def deleteNode(self, root: Optional[TreeNode], key: int) -> Optional[TreeNode]:
        if not root:
            return 
        dummy = TreeNode(0)           # this is in case we want to delete the root element
        dummy.right = root
        ptr = root
        prev = dummy               # works as the parent pointer while deleting the node
        while ptr:
            if key == ptr.val:
                #deal with it
                if not ptr.left:
                    if ptr.val < prev.val:
                        prev.left = ptr.right
                    else:
                        prev.right = ptr.right
                elif not ptr.right:
                    if ptr.val < prev.val:
                        prev.left = ptr.left
                    else:
                        prev.right = ptr.left
                else:
                    temp = ptr.left
                    ptr2 = ptr.right
                    while ptr2.left:
                        ptr2 = ptr2.left
                    ptr2.left = temp
                    if ptr.val < prev.val:
                        prev.left = ptr.right
                    else:
                        prev.right = ptr.right        
                break
            elif key > ptr.val:
                prev = ptr
                ptr = ptr.right
            else:
                prev = ptr
                ptr = ptr.left
                
        return dummy.right
