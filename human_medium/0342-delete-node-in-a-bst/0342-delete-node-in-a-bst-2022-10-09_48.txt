# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def deleteNode(self, root: Optional[TreeNode], key: int) -> Optional[TreeNode]:
        if not root:
            return None
        
        if root.val < key:
            root.right = self.deleteNode(root.right, key)
        elif root.val > key:
            root.left = self.deleteNode(root.left, key)
        else:
            # case 1: to_delete_node is a leaf node
            if not root.left and not root.right:
                return None
            # case 2: to_delete_node has only one child node
            elif not root.left:
                return root.right
            elif not root.right:
                return root.left
            # case 3:
            else:
                # get min_node in right subtree
                min_node = root.right
                while min_node.left:
                    min_node = min_node.left
                # replace the to_delete_node with min_node
                root.val = min_node.val
                # delete the min_node 
                root.right = self.deleteNode(root.right, min_node.val)
        return root
