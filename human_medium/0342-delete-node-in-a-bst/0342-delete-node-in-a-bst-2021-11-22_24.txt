# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def deleteNode(self, root: Optional[TreeNode], key: int) -> Optional[TreeNode]:
        def is_leaf(node):
            return not node.left and not node.right
        
        def get_successor(node):
            curr = node.right
            while curr.left:
                curr = curr.left
            return curr
        
        def get_predecessor(node):
            curr = node.left
            while curr.right:
                curr = curr.right
            return curr
            
        def rec(node, key):
            if not node:
                return None
            
            if key > node.val:
                node.right = rec(node.right, key)
                return node
            
            if key < node.val:
                node.left = rec(node.left, key)
                return node
            
            if is_leaf(node):
                return None
            
            if node.right:
                succ = get_successor(node)
                node.val = succ.val
                node.right = rec(node.right, succ.val)
                return node
            
            if node.left:
                pred = get_predecessor(node)
                node.val = pred.val
                node.left = rec(node.left, pred.val)
                return node
            
        return rec(root, key)
