"""
Solution: DFS ---> Recursion....Do this or that
Strategy: 
1. Using the value of the node to delete, find the node in the tree, based on left < value < right
2. when you find the Node (root.value == key): you delete it by returning the node in front of it, either Left or Right
3. if the node has no right node return it\'s left node
4. if right node, Then using a while loop to find the smallest value by the right, which is the leftmost value of the right Node
5. then assign the smallest value to root left
6. and return root right

"""

class Solution(object):
    def deleteNode(self, root, key):
        # Base Case 
        if not root:    return None
        
        # go to the left if value is greater than key
        if key < root.val:
            root.left = self.deleteNode(root.left, key)
            return root
        # go to the right if value is less than than key
        elif key > root.val:
            root.right = self.deleteNode(root.right, key)
            return root
        # we have found the Node (root.value == key)
        else:
            # get the smallest node by the right to maintain the BST after deletion
            smallestNode_in_right = self.findSmallest(root.right)
            
            # if right Node don\'t exist return left
            if not smallestNode_in_right:   return root.left
            
            # 5. then assign the smallest value to root left
            smallestNode_in_right.left = root.left
            
            # Return Right
            return root.right
    
    # Using a while loop to find the leftest Node
    def findSmallest(self, root):
        if not root:    return None
        while root.left:
            root = root.left
        return root
