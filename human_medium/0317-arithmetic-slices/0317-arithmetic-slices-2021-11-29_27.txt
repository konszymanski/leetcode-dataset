class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        if len(nums) < 3:
            return 0
        
        dp = [0] * len(nums)
        res = 0
        
        for idx in range(2, len(nums)):
            if nums[idx - 1] - nums[idx - 2] == nums[idx] - nums[idx - 1]:
                dp[idx] = dp[idx - 1] + 1
                res += dp[idx]
                
        return res
