class Solution:
    def numberOfArithmeticSlices(self, A: List[int]) -> int:
        dp = defaultdict(int)
        res = 0
        
        for i in range(2, len(A)):
            if A[i] - A[i - 1] == A[i - 1] - A[i - 2]:
                dp[i] = dp[i - 1] + 1
                res += dp[i]
        
        return res
