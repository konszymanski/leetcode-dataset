class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        if len(nums)<3:
            return 0
        ans = 0
        for i in range(len(nums)):
            for j in range(i,len(nums)-2):
                if nums[j+1]-nums[j]==nums[j+2]-nums[j+1]:
                    ans+=1
                else:
                    break
                    
            
            
        return ans
