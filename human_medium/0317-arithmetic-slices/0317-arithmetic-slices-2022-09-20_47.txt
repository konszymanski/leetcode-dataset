class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        n =len(nums)
        if n<=2:
            return 0
        ans =0
        c=1
        i=0
        j=0
        while(i<n-1 and j <n ):
            dif =nums[i+1]-nums[i]
            j =i+1 
            while(j<n-1 and nums[j+1]-nums[j]==dif):
                ans+=c
                c+=1
                j+=1 
            i =j
            c=1 
        return ans
