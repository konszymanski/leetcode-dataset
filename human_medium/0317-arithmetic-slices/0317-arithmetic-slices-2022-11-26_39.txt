class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        diff = []
        for i in range(0,len(nums)-1):
            diff.append(nums[i]-nums[i+1])
        temp = 0
        count = 0
        diff.append(1001)
        for k in range(0,len(diff)-1):
            temp += 1
            if diff[k] != diff[k+1]:
                if temp >= 2:
                    count += ((temp - 1) * temp)//2
                temp = 0
        return count
