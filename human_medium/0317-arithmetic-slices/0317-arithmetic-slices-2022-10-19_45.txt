class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        n = len(nums)
        ans = 0
        if n < 3:
            return 0
        
        prev = 0
        curr = 0
        for i in range(2, n):
            curr = 0
            if nums[i]-nums[i-1] == nums[i-1]-nums[i-2]:
                curr = prev + 1
                ans += curr
            
            prev = curr
        return ans
