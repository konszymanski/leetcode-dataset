class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        diff = [nums[i]-nums[i-1] for i in range(1, len(nums))]
        ans = [0] * len(diff)
        
        for i in range(1, len(diff)):
            if diff[i]==diff[i-1]:
                ans[i] = ans[i-1]+1
        return sum(ans)
