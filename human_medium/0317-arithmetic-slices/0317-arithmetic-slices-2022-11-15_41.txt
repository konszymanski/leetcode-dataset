def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        j = 2
        res = 0
        while j<len(nums):
            if nums[j-2] - nums[j-1] == nums[j-1] - nums[j]:
                k = j
                target = nums[j-2] - nums[j-1]
                while k < len(nums) and target == nums[k-1] - nums[k]:
                    k += 1
                res += k-j
                j += 1
            else:
                j += 1
        return res
