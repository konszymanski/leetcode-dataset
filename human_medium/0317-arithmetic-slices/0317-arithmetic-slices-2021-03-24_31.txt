class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
        n = len(nums)
        if n <= 2: return 0
        dp = [2] * n # dp[i]: the length of longest arithmetic slices that end at i th index
        res = 0
        for i in range(2,n):
            if nums[i] - nums[i-1] == nums[i-1] - nums[i-2]:
                dp[i] = dp[i-1] + 1
            if dp[i] >= 3:
                res += dp[i] - 2 # all the subarraies that ends at i th index belong to arithmetic slice. 
        return res
