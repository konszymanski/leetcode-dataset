class Solution:
    def numberOfArithmeticSlices(self, nums: List[int]) -> int:
	
        currentLength = 2
        ans = 0
        
        for i in range(2, len(nums)):
            if nums[i - 1] - nums[i - 2] == nums[i] - nums[i - 1]:
                currentLength += 1
            else:
                ans += (currentLength - 2) * (currentLength - 1) // 2
                currentLength = 2
        ans += (currentLength - 2) * (currentLength - 1) // 2
                
        return ans
