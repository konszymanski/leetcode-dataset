        heights = []

    def minHeight(self, root, graph, path, minHeight, currHeight, memo):
 
        path[root] = True      
        m = 0
        for childNode in graph[root]:
            if path[childNode]:
                continue
            if memo.get(root, {}).get(childNode, None):
                height = memo[root][childNode]
            else:
                height = self.minHeight(childNode, graph, path, minHeight, currHeight+1, memo)
                memo.setdefault(root, {})
                memo[root][childNode] = height

            if height > m:
                m = height
        path[root] = False
        return m + 1
