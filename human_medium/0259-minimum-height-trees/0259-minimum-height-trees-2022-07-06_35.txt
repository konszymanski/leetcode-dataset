Observations
- Find central nodes which have lowest distance from any leaf.
- MHT roots will be mid point(s) of the diameter.
- 2 mid points for even length diameter and 1 for odd.

Approach:
========
- Find any diameter and the mid points will be the MHT root(s).
- To find diameter:
    - From any node, find farthest node. This is one end of the diameter.
    - From that end, find farthest node. This is the other end.
\'\'\'
class Solution:
    def findMinHeightTrees(self, n: int, edges: List[List[int]]) -> List[int]:
        graph = defaultdict(list)
        for u, v in edges:
            graph[u].append(v)
            graph[v].append(u)

        start = 0
        queue = deque()
        visited = [False] * n
        distance = [inf] * n

        queue.append(start)
        visited[start] = True
        distance[start] = 0

        while queue:
            u = queue.popleft()
            for v in graph[u]:
                if not visited[v]:
                    queue.append(v)
                    visited[v] = True
                    distance[v] = distance[u] + 1

        max_dist = max(distance)
        end1 = distance.index(max_dist)
        
        start = end1
        queue = deque()
        visited = [False] * n
        distance = [inf] * n
        prev = [None] * n

        queue.append(start)
        visited[start] = True
        distance[start] = 0
        
        while queue:
            u = queue.popleft()
            for v in graph[u]:
                if not visited[v]:
                    queue.append(v)
                    visited[v] = True
                    distance[v] = distance[u] + 1
                    prev[v] = u

        max_dist = max(distance)
        end2 = distance.index(max_dist)

        path = []

        itr = end2
        while prev[itr] is not None:
            path.append(itr)
            itr = prev[itr]
        path.append(itr)

        path_len = len(path)
        roots = []
        if path_len  % 2 == 0:
            roots.append(path[path_len // 2])
            roots.append(path[path_len // 2 - 1])
        else:
            roots.append(path[path_len // 2])

        return roots
