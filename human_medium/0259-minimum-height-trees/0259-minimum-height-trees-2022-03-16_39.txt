class Solution:
    def findMinHeightTrees(self, n: int, edges: List[List[int]]) -> List[int]:
        if n < 3:
            return list(range(n))
        
        # initialize adjacency list and num of unvisited node in topological sort
        adj = defaultdict(set)
        unseen = n
        
        for i,j in edges:
            adj[i].add(j)
            adj[j].add(i)
            
        dq = deque([i for i in adj if len(adj[i]) == 1])
        
        # Kahn\'s algorithm for topological sort 
        while dq:
            len_ = len(dq)
            
            # if the len of degree 1 vertices are the same as num of unseen nodes we have the solution
            if len_ == unseen:
                return dq
            
            # pop all the vertices with degree 1 in the current graph
            for _ in range(len_):
                leaf = dq.pop()
                
                # update the graph
                while adj[leaf]:
                    parent = adj[leaf].pop()
                    adj[parent].remove(leaf)
                    unseen -=1
                    if len(adj[parent]) == 1:
                        dq.appendleft(parent)
