from collections import defaultdict
class Solution:
	#Selecting median ensures that height of the tree is minimum
    def findMinHeightTrees(self, n: int, edges: List[List[int]]) -> List[int]:  
        if len(edges)==0:
            return [0]
        
        #Make Adj List
        g = defaultdict(list)
        for e in edges:
            g[e[0]].append(e[1])
            g[e[1]].append(e[0])
            
        #Calculate degree and create leaves if indegree = 0(Disjoint nodes) and indegree=1 (leaves - since it is undirected)
        degree = [0]*n
        leaves = []
        for v in range(n):
            if v in g:
                degree[v]=len(g[v])
            else:
                degree[v]=0
            if degree[v]<=1:
                leaves.append(v)
        count = len(leaves)
        #Now remove leaves, while removing them make sure  that you are decreasing child\'s degree by 1. Check if these child nodes are leafs, if so add them and do the same thing until count < n. 
        while count < n:
            newLeaves = []
            for l in leaves:
                degree[l] = 0
                for r in g[l]:
                    degree[r] -= 1
                    if degree[r] == 1:
                        newLeaves.append(r)
            count += len(newLeaves) 
            leaves = [_ for _ in newLeaves]
        return leaves
