class Solution:
	def findMinHeightTrees(self, n, edges):
		g = [set() for _ in range(n)]
		dist = [0] * n
		for u, v in edges: g[u].add(v); g[v].add(u)
		s = [x for x in range(n) if len(g[x]) == 1]
		while s:
			t = []
			for u in s:
				for v in g[u]:
					g[v].remove(u)
					if len(g[v]) == 1:
						t.append(v)
						dist[v] = dist[u] + 1
			s = t
		m = max(dist)
		return [i for i, x in enumerate(dist) if x == m]
