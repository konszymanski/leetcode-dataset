class Solution:
    def findMinHeightTrees(self, n: int, edges):
        def createGraph():
            graph = {}
            for i in range(len(edges)):
                a,b = edges[i]
                if(a in graph):
                    graph[a].append(b)
                else:
                    graph[a]=[b]
                if(b in graph):
                    graph[b].append(a)
                else:
                    graph[b]=[a]
            return graph

        graph = createGraph()
        # print(graph)

        numberOfEdges = [0 for i in range(n)]
        for i in range(n):
            numberOfEdges[i] = len(graph[i])
        currLeaves = []
        for i in range(n):
            if(numberOfEdges[i]==1):
                currLeaves.append(i)
        remainingLeaves = n
        seen = {}
        while(remainingLeaves>2):
            remainingLeaves = remainingLeaves-len(currLeaves)
            newLeaves = []
            while(len(currLeaves)>0):
                node = currLeaves.pop(0)
                for connectedEdge in graph[node]:
                    numberOfEdges[connectedEdge]-=1
                    if(numberOfEdges[connectedEdge]==1):
                        newLeaves.append(connectedEdge)
            print("-->",newLeaves)
            currLeaves = newLeaves

        print(currLeaves)
        return currLeaves
