class Solution:
    def findMinHeightTrees(self, n: int, edges: List[List[int]]) -> List[int]:
        if n <= 2:
            return list(range(n))
        
        g = defaultdict(set)
        for u, v in edges:
            g[u].add(v)
            g[v].add(u)
        
        leaves = [u for u in g if len(g[u]) == 1]
        
        while leaves:
            new_leaves = []
            for u in leaves:
                if not g[u]:
                    continue
                g[(v := g[u].pop())].remove(u)
                if len(g[v]) == 1:
                    new_leaves.append(v)
            if not new_leaves:
                break        
            leaves = new_leaves    
        return leaves
