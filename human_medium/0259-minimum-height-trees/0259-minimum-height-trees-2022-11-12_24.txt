def findMinHeightTrees(self, n: int, edges: List[List[int]]) -> List[int]:
        gr = defaultdict(set)
        for a, b in edges:
            gr[a].add(b)
            gr[b].add(a)
        leafs = [k for k in gr if len(gr[k]) == 1]
        nextleafs = []
        while len(gr) > 2:
            for leaf in leafs:
                nextl = gr[leaf].pop()
                gr[nextl].discard(leaf)
                if len(gr[nextl]) == 1:
                    nextleafs.append(nextl)
                del gr[leaf]
            leafs.clear()
            nextleafs, leafs = leafs, nextleafs
        return gr.keys() or [0]
