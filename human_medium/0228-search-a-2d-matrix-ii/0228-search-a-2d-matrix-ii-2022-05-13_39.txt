class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        # T: O(m + n), S: O(1)
        m, n = len(matrix), len(matrix[0])
        r, c = 0, n - 1
        while r < m and c >= 0:
            curr = matrix[r][c]
            if curr == target:
                return True
            elif curr > target:
                c -= 1
            else:
                r += 1
        return False
