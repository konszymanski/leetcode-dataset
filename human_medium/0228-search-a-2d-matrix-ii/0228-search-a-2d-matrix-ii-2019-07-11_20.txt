# O(m+n) 
    def searchMatrix(self, matrix, target):
        """
        :type matrix: List[List[int]]
        :type target: int
        :rtype: bool
        """
        if not matrix:
            return False
        row, column = 0, len(matrix[0])-1
        while row<len(matrix) and column>=0:
            cur = matrix[row][column]
            if cur == target:
                return True
            elif cur > target:
                column -= 1
            else:
                row += 1
        return False
