def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
	m = len(matrix)
	n = len(matrix[0])
	row,col = 0,n-1 # row and column we are on
	while row < m and 0 <= col: # while we are in-bounds
		element = matrix[row][col] # get the element here
		if element > target: # if it is too big
			col -= 1 # go to the left
		elif element < target: # if it is too small
			row += 1 # go down
		else: # we found the target!
			return True
	return False
