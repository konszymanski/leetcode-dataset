class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        
        def binSearch(nums, target):

            lo, hi = 0, len(nums)

            while lo < hi:
                mid = (lo + hi) // 2
                if nums[mid] == target:
                    return True
                elif nums[mid] < target:
                    lo = mid + 1
                elif nums[mid] > target:
                    hi = mid

            return False
        
        for row in matrix:
            if binSearch(row, target):
                return True 
            
        return False
