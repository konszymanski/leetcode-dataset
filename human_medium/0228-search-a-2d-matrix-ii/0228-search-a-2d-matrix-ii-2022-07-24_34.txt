class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
	    # Define the total row and column counts
        nrows = len(matrix)
        ncols = len(matrix[0])
		# Initialize the rowcounter (i) and column counter(j)
		# We start from top-right corner element, where i = 0, j = ncols-1
        i = 0
        j = ncols-1
		# We loop till i < nrows and j > 0
        while (i < nrows and j >= 0):
		    # If the value is found return true
            if matrix[i][j] == target:
                return True
			# If the current value is more than target value then decrese the column count by 1 
            if matrix[i][j] > target:
                j -= 1
			# If the current value is less than the target value then increase the row count by 1
            if matrix[i][j] < target:
                i += 1
		# return False when the value is not found		
        return False
