class Solution(object):
    def searchMatrix(self, matrix, target):
        M = len(matrix)
        N = len(matrix[0])
        
        r = M - 1
        c = 0
        
        while c < N and r > -1:
            if matrix[r][c] > target:
                r -= 1
            elif matrix[r][c] < target:
                c += 1
            else:
                return True
        
        return False
