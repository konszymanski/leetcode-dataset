class Solution:
    def searchMatrix(self, matrix, target):
        """
        :type matrix: List[List[int]]
        :type target: int
        :rtype: bool
        """
        if len(matrix) < 1:
            return False
        if len(matrix[0]) < 1:
            return False
        for row in matrix:
            if row[0] > target or row[-1] < target:
                continue
            l = 0
            r = len(row) - 1
            while l <= r:
                mid = (l + r) // 2
                if row[mid] == target:
                    return True
                elif row[mid] < target:
                    l = mid + 1
                else:
                    r = mid - 1
        return False
