class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
               
        def binary(l,r):
            
            if l >r:
                return False
            
            m = (l+r)//2
            
            if target > row[m]:
                return binary(m+1,r)
                
            elif target < row[m]:
                return binary(l, m-1)
            
            elif target == row[m]:
                return True          
        
        for row in matrix:
            
            if row[0] <= target and target <= row[-1]:
                
                l,r = 0, len(row)-1

                if binary(l,r): return True        
            
        return False
