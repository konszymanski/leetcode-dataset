class Solution:
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        for a in range(len(matrix[0])):
            top, bottom = 0, len(matrix)-1
            while top <= bottom:
                mid = (top + bottom)//2
                if matrix[mid][a] == target:
                    return True
                if matrix[mid][a] > target:
                    bottom = mid - 1
                else:
                    top = mid + 1
        return False
