class Solution(object):
    def searchMatrix(self, matrix, target):
        n = len(matrix)
        if n==0:return False
        m = len(matrix[0])
        if m==0:return False
        index = m-1
        for i in range(n):
            while index>=0 and matrix[i][index]>target:
                index-=1
            if index == -1:break
            if matrix[i][index]==target:return True
        return False
