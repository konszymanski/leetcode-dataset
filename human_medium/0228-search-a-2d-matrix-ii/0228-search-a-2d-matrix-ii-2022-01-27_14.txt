class Solution:
    """
    
    Localize to the correct value by using the value of a cell
    starting at the top right.
    
    Either the top right (the maximum column, minmum row) and bottom left (maximum
    row, minimum column) can be used to minimize the search space.
    
    In our case we started from the top right. Because it is the max column, 
    if the value at matrix[row][col] is > target, we shrink the column since 
    the target must be to the left.
    
    Since we started in the min row, if the value at matrix[row][col] is < target,
    we expand the row, since the target must be at least in the next row.
    
    
    
    Time complexity: O(m + n). we need max m shrinks and n expands.
    

    """
    
    
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        rows = len(matrix)
        cols = len(matrix[0])
        
        # start at top right
        row = 0
        col = cols - 1
        
        #print (col, cols)
        while row >= 0 and col >= 0 and row < rows and col < cols:
            #print(row, rows, col, cols)
            if matrix[row][col] > target:
                col = col - 1
            elif matrix[row][col] < target:
                row = row + 1
            else:
                return True
            
        return False
