class Solution:
#     Runtime: 175ms 94.99% || memory: 20.3mb 98.38%
    def searchMatrix(self, matrix: List[List[int]], target: int) -> bool:
        col = len(matrix[0])-1
        for item in range(len(matrix)):
            while matrix[item][col] > target and col >= 0:
                col-=1
            if matrix[item][col] == target:
                return True
        return False
