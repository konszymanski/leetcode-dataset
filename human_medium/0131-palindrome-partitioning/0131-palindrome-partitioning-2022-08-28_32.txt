class Solution:
    def partition(self, s: str) -> List[List[str]]:
        def backtrack(s, res, path):
            if not s: res.append(path)
            else:
                for i in range(1, len(s) + 1):
                    if isPalindrome(s[: i]):
                        backtrack(s[i: ], res, path + [s[: i]])
        def isPalindrome(s):
            return s == s[::-1]
        
        res = []
        backtrack(s, res, [])
        return res
