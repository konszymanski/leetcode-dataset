# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rob(self, root: Optional[TreeNode]) -> int:

        def tree(root):

            if not root:
                return 0
            if root in catch:
                return catch[root]

            val = 0
            if root.left:
                val += tree(root.left.left) + tree(root.left.right)
            if root.right:
                val += tree(root.right.left) + tree(root.right.right)
            # meaning max(get first root, get second root) to compare
            catch[root] = max(root.val + val, tree(root.left) + tree(root.right))
            return catch[root]
        catch = {}
        return tree(root)
