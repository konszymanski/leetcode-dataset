    TOPIC:Tree + find sum max based on previous decision, dfs + memo => similar to top-bottom DP
    STEP:
    1.max(cur + exclude cur\'s left and right but include rest of cur subtree, cur.left+right) 
    \'\'\'
    def rob(self, root: Optional[TreeNode]) -> int:
        # create memo map{node: max-sum}
        memo = {} 
        # tree dfs top-< bottom, track up to the root, so return recurion directly
        return self.dfs(root, memo)
        
    def dfs(self, root, memo):
        if not root: return 0
        # check if memo
        if root in memo: return memo[root]
     
        # create val variable to start calculate max sum from now to leaves
        val = 0
        if root.left:
            val += self.dfs(root.left.left, memo) + self.dfs(root.left.right, memo)
        if root.right:
            val += self.dfs(root.right.left, memo) + self.dfs(root.right.right, memo)
            
        memo[root] = max(val + root.val, self.dfs(root.left, memo) + self.dfs(root.right, memo))
        
        return memo[root]
