class Solution:
    def rob(self, root: Optional[TreeNode]) -> int:
        @cache
        def dp(root, turn):
            if not root:
                return 0
            
            withRoot = (root.val + dp(root.left, False) + dp(root.right, False)) if turn else 0
            withoutRoot = dp(root.left, True) + dp(root.right, True)
            return max(withRoot, withoutRoot)
        return dp(root, True)
