class Solution:
    def __init__(self):
        self.cache = {}
		
    def rob(self, root: TreeNode) -> int:
        return max(self.helper(root, 0), self.helper(root, 1))
    
    def helper(self, root, status): # status == 0 means skip
        if not root:
            return 0
        if (root,status) in self.cache:
            return self.cache[(root, status)]
        if status == 1:
            res = root.val + self.helper(root.left, 0) + self.helper(root.right, 0)
        else:
            res = max(self.helper(root.left,0), self.helper(root.left,1)) + max(self.helper(root.right,0), self.helper(root.right,1))
        self.cache[(root, status)] = res
        return res
