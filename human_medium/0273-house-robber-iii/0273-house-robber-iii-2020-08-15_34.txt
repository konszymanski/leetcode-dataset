class Solution:
    def rob(self, root: TreeNode) -> int:
        return max(self.dfs(root))
        
    def dfs(self, node: TreeNode) -> List[int]:
        if not node: 
            return [0, 0]
        
        l, r = self.dfs(node.left), self.dfs(node.right)
        
        return [ max(l) + max(r), node.val + l[0] + r[0]]
