# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def rob(self, root: Optional[TreeNode]) -> int:
        
        dp = {}
        def dfs(node,robbed):
            if (node,robbed) in dp:
                return dp[(node,robbed)]
            if not node.left and not node.right:
                if robbed:
                    return  node.val
                else:
                    return 0
            left ,right = 0,0
            if robbed:
                if node.left:
                    left = dfs(node.left, not robbed)
                if node.right:
                    right =  dfs(node.right, not robbed)
            if not robbed:
                # we can take it or not take it 
                if node.left:
                    left = max(dfs(node.left, robbed),dfs(node.left,not robbed))
                if node.right:
                    right =  max(dfs(node.right,robbed), dfs(node.right, not robbed))
            
            sum_ = left+right
            if robbed:
                dp[(node,robbed)] = node.val + sum_
                return dp[(node,robbed)]
            else:
                dp[(node,robbed)] = sum_
                return dp[(node, robbed)]
        return max(dfs(root,True), dfs(root,False))
