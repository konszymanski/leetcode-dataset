class Solution:
    def rob(self, root: Optional[TreeNode]) -> int:
        
        def dfs(node= root):
            if not node: 
                return (0, 0)
            L, R = dfs(node.left), dfs(node.right)
            return (node.val+L[1]+R[1], max(L)+max(R))
        
        return max(dfs())
