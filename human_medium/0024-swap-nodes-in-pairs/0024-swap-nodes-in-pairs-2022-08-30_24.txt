# Definition for singly-linked list.
# class ListNode(object):
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution(object):
    def swapPairs(self, head):
        """
        :type head: ListNode
        :rtype: ListNode
        """
        arr=[]
        curr=head
        if not head:
            return None
        while curr :
            tmp=curr
            curr=curr.next
            tmp.next=None
            arr.append(tmp)
        for i in range(1,len(arr),2):
            arr[i-1],arr[i]=arr[i],arr[i-1]
        #print(arr)
        head=curr=arr[0]
        for i in range(1,len(arr)):
            curr.next=arr[i]
            curr=curr.next
        return head
