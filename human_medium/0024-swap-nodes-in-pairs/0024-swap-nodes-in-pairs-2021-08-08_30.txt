def solveUtil(head):
    prev=ListNode()
    curr=head
    realhead=head.next
    while curr and curr.next:
        a=curr
        b=curr.next
        temp=a.next.next if a.next.next else None
        #Step 1
		a.next=temp
        b.next=a
        prev.next=b
		#Step 2
        curr=temp
        prev=a
    return realhead
def length(head):
    if head is None:
        return 0
    curr=head
    l=0
    while curr:
        l+=1
        curr=curr.next
    return l
def solve(head):
    if length(head)==0:
        return None
    elif length(head)==1:
        return head
    else:
        return solveUtil(head)
		
class Solution:
	def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
		return solve(head)
