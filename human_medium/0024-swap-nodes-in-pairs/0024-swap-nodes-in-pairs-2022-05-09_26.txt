# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def swapPairs(self, head: Optional[ListNode]) -> Optional[ListNode]:
        # initialize nodes
        dummy = ListNode(0,head)
        prev = dummy
        cur = head
        while cur and cur.next:
            # save the nodes before swapping
            second = cur.next
            nxtPair = cur.next.next        
            # swap nodes
            second.next = cur
            cur.next = nxtPair
            prev.next = second
            # update nodes
            prev = cur
            cur = nxtPair
        # return the head of swapped linked list
        return dummy.next
