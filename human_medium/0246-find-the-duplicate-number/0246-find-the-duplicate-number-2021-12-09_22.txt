class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        i = 0  # we set our initial pointer
        while i < len(nums): # iterate through the given array
            if nums[i] != i + 1: # simply check if given number under i index is not equal to incremented index value
                j = nums[i] - 1 # as written in a description of problem, our array contains number + 1
                if nums[i] != nums[j]: # if they are not equal, just swap them
                    nums[i], nums[j] = nums[j], nums[i]
                else:
                    return nums[i] # our duplicate number
            else: # we don\'t increase our index until it didn\'t go through first statement
                i += 1
