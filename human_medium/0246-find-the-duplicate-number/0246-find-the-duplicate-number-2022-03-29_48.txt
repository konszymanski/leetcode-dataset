class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        duplicateNumber = 0
        for i in range(len(nums)):
            n = abs(nums[i])
            if nums[n-1] < 0:
                duplicateNumber = abs(nums[i])
				break
            else:
                nums[n-1] = -1 * nums[n-1]
        return duplicateNumber
