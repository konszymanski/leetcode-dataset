class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        #cyclic sort
        i = 0
        
        while i < len(nums):
            j = nums[i] - 1
            
            if nums[i] != nums[j]:
                #swap
                nums[i], nums[j] = nums[j], nums[i]
            else:
                i += 1 
                
        for i in range(len(nums)):
            if nums[i] != i + 1:
                return nums[i]
        
        return -1
