class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        # normal speed and double speed
        n, d = nums[nums[0]], nums[nums[nums[0]]]
        while n != d:
            n, d = nums[n], nums[nums[d]]
        # normal speed for both
        d = nums[0]
        while n != d:
            n, d = nums[n], nums[d]
        return n
