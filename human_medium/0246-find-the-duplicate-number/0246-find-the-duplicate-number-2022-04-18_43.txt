class Solution(object):
    def findDuplicate(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        # define two pointer slow1 and fast
        # slow takes one step a time and fast take two steps at a time
        # when we find an node where they intersect we will stop there
        slow1,fast=0,0
        while True:
            slow1=nums[slow1]
            fast=nums[nums[fast]]
            if slow1==fast:
                break
        # we now have the intersection node
        # define a new slow pointer slow2        
        slow2=0
        # slow2 will start from the start node and slow1 will start moving from intersection point
        # according to the floyd\'s algo the node where they intersect is the loop of the graph
        while True:
            slow2=nums[slow2]
            slow1=nums[slow1]
            if slow1==slow2:
                return slow1
