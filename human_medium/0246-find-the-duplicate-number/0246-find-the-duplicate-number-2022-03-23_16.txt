class Solution(object):
    def findDuplicate(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        \'\'\'
        idea --> since all the values in nums are in the range of 1 - len(nums)
                   --> we can use the value in the nums and track them if they have been visited or not by marking thier index location
                   
        for example in nums = [4,4,2,7,8,2,3,1]
        --> when you get 4 --> you go and mark the index location of 4 in nums --> [4,3,2,->7<-,8,2,3,1]
            --> how do you mark the index location ?
                --> just convert the number at that index location to negetive 
                   -->  nums --> [4,3,2,-7,8,2,3,1]
            --> if you encouter 4 again --> when you go and want to change the index location of 4 to negetive 
                --> and if you found out that it has already been converted to negetive --> [4,3,2,-7,8,2,3,1]
                   --> then that means the number of 4 has a duplicate
                   --> you just need to return that 4 value, becuase there is only one duplicate
        \'\'\'
        for num in nums:
            indx = abs(num) # in case if num has already been converted to -ve
            if num > 0 : # has never been visited --> so mark the index location
                nums[indx] *= -1
            else: # has been visited --> so just return that value
                return nums[indx]
