class Solution {
    fun findDuplicate(A: IntArray): Int {
        for (i in 0 until A.size) {
            while (true) {
                var j = A[i] - 1
                if (i == j)
                    break
                if (A[i] == A[j])
                    return A[i]
                A[i] = A[j].also{ A[j] = A[i] }
            }
        }
        return -1
    }
}
