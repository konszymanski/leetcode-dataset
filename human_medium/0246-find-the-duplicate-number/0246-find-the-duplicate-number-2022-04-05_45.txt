class Solution:
    # __main__
    def findDuplicate(self, nums: List[int]) -> int:
        if not(nums):
            return None
        size = len(nums)
        if (size == 1):
            return None
        
        # return self.getResultV1(nums,size)
        # return self.getResultV2(nums,size)
        return self.getResultV3(nums,size)
        # return self.getResultV4(nums,size)
        # return self.getResultV5(nums)
    
    
    # Method 5: Sum Method :
    def getResultV5(self,nums):
        numset = set(nums)
        return (sum(nums)-sum(numset)) // (len(nums)-len(numset))
    
    
    # BEST || Method 4: Tortoise and Hare || Time:O(n), Space;O(1)
    def getResultV4(self,nums,size):
        slow = fast = nums[0]
        while(True):
            slow = nums[slow]               # goto next element;
            fast = nums[nums[fast]]         # goto next.next element;
            if(slow == fast):
                break
        
        fast = nums[0]
        while(slow != fast):
            slow = nums[slow]
            fast = nums[fast]
            
        return slow
    
    
    # BEST || Method 3: Mark Visited Item || Time:O(n),Space: (1)
    def getResultV3(self,nums,size):
        for idx in nums:
            idx = abs(idx)
            if(nums[idx] < 0):
                return idx
            nums[idx] = -nums[idx]
            print("idx:-nums[idx]:",idx,"//",nums[idx])
            
        return size
    
    
    # Method 2: Sorting || Time:O(nLogn), Space: O(1);
    def getResultV2(self,nums,size):
        i = 0
        nums.sort()
        while(i<size):
            if(nums[i] == nums[i+1]):
                return nums[i]
            i += 1
        return None
            
    
    # Method 1: Store and Find || Time: O(n), Space: O(n);
    def getResultV1(self,nums,size):
        dataset = set()                                     # Dict Works too
        for num in nums:
            if num in dataset:
                return num
            else:
                dataset.add(num)
        return None
