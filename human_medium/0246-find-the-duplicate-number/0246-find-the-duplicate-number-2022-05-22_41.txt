class Solution:
    #T=O(n), S=O(1)
    #treat the array as a linked list
	#pigeonhole principle
    def findDuplicate(self, nums: List[int]) -> int:
        #init slow and fast pointers
        s, f = 0, 0
        #shift the pointers until they collide
        while True:
            s = nums[s]
            #fast pointer shifts by 2 units
            f = nums[nums[f]]
            if s == f: 
                break
        
        #reset the fast pointer
        #shift the pointers until they meet again
        #the value at the meeting index would be the duplicate item
        f = 0
        while True:
            s = nums[s]
            f = nums[f]
            if s == f: 
                return s
