class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        i=0
        while i<len(nums):
            if nums[i]-1==i:
                i+=1
                continue
            t = nums[i]-1
            if nums[t]==nums[i]:
                return nums[t]
            nums[t],nums[i]=nums[i],nums[t]
