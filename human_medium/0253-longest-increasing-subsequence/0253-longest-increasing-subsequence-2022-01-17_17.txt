class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        def f(ind,nums,n):
            ans=1
            for i in range(0,ind):
                if nums[i]<nums[ind]:
                    rec_ans=1+f(i,nums,n)
                    ans=max(ans,rec_ans)
            return ans
        n=len(nums)
        ans=float(\'-inf\')
        for i in range(len(nums)-1,-1,-1):
                  ans=max(ans,f(i,nums,n))
        return ans
