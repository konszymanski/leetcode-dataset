class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        n=len(nums)
        dp=[1]*(n)
        maxi=1
        for ind in range(n):
            for prev in range(ind):
                if nums[prev]<nums[ind]:
                    dp[ind]=max(1+dp[prev],dp[ind])
            maxi=max(maxi,dp[ind])
        return maxi
