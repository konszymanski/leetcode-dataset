class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        @cache
        def dfs(k,prev):
	        if k==len(nums):
	        	return 0
	        longest = 0
	        for i in range(k,len(nums)):
	        	if nums[i] > prev:
	        		curr = 1 + dfs(i+1, nums[i])
	        		longest = max(longest, curr)
	        return longest
        return dfs(0,-float("inf"))
