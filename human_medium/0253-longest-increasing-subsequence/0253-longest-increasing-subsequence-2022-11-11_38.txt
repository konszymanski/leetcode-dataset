class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        memo = [2**31]

        for n in reversed(nums):
            for j in range(len(memo) - 1, -1, -1):
                if n > memo[j]:
                    continue
                elif n == memo[j]:
                    break
                elif j == len(memo) - 1:
                    memo.append(n)
                else:
                    memo[j + 1] = n
                break

        return len(memo) - 1
