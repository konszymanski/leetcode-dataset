class Solution:
    def lengthOfLIS(self, nums):
        n=len(nums)
        def rec(i,l,prev):
            # print(i,l,prev)
            if i==n:
                return l
            # p=False
            # if l>0 and nums[i]>prev:
                # p=True
            if nums[i]>prev:
                return max(rec(i+1,l+1,nums[i]),rec(i+1,l,prev))
            
            else:
                return rec(i+1,l,prev)
