from functools import cache

class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        lenNums = len(nums)
        @cache
        def helper(i):
            ids = filter(lambda j:nums[j]>nums[i], range(i,lenNums))
            return max([*map(helper,ids)]+[0]) + 1
        return max(helper(i) for i in range(lenNums))
