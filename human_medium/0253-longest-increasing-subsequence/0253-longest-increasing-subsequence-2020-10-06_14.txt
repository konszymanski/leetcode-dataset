class Solution(object):
    def lengthOfLIS(self, nums):
        if not nums:
            return 0
            
        maxCount = 0
        #this stores the longest subsequence untill the current index
        seq = [1] * len(nums)
        
        for i in range(len(nums)):
            current = nums[i]
            
            for j in range(i):
                prev = nums[j]
                
                if prev < current and seq[j]  + 1 > seq[i]:
                    seq[i]  = seq[j]  + 1
        return max(seq)
