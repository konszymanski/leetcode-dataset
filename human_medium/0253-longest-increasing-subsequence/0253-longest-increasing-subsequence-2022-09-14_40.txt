class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        seq = [] # Not exactly the correct seq but correct length
        seq.append(nums[0])
        
        for num in nums[1:]:
            if seq[-1] < num: seq.append(num)
            else:
                sorted_idx = bisect.bisect_left(seq, num)  # logn
                seq[sorted_idx] = num
        
        return len(seq)
