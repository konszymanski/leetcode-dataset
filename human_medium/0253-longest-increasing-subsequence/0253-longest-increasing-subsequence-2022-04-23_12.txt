class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        # Initialize subsequence
        subseq = [nums[0]]

        for n in nums:
            if n < subseq[0]:
                # Replace first element in subsequence with n
                subseq[0] = n
            elif n > subseq[-1]:
                # Add n onto end of subsequence
                subseq.append(n)
            else:
                # Binary search to replace element in middle of subseq with n
                left = 0
                right = len(subseq) - 1
                middle = (left + right) // 2
                while left < right:
                    if n > subseq[middle]:
                        left = middle + 1
                    else:
                        right = middle
                    middle = (left + right) // 2

                subseq[left] = n

        return len(subseq)
