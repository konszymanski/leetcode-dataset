class Solution(object):
    def lengthOfLIS(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        dp = [1] * len(nums)
        ans = 0
        for i,num in enumerate(nums):
            ans =max(ans,dp[i])
            for j in range(i+1,len(nums)):
                if nums[j] > nums[i]:
                    dp[j] = max(dp[j],dp[i]+1)
        return ans
