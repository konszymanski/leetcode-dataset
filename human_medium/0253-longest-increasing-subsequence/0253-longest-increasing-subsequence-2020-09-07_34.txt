class Solution:
    def lengthOfLIS(self, nums: List[int]) -> int:
        
        @lru_cache
        def memo_solve(i, prev):
            if i == len(nums):
                return 0
            
            ans1 = 0
            if nums[i] > prev:
                ans1 = 1 + memo_solve(i+1, nums[i])
            ans2 = memo_solve(i+1, prev)
            return max(ans1, ans2) if ans1 else ans2
        
        return memo_solve(0, float(\'-inf\'))
