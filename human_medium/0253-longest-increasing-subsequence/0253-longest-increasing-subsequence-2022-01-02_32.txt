def lengthOfLIS(self, nums: List[int]) -> int:
        @cache
        def dp(i):
            if i == 0:
                return 1
            
            target = nums[i]
            ans = 1
            
            for j in range(i):
                if nums[j] < target:
                    ans = max(ans, dp(j) + 1)
                    
            return ans
        
        
        return max(dp(i) 
                   for i in range(len(nums)))
