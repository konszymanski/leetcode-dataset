class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        profit = [0] * len(prices)
        i = 0
        #maxi: most profit if the stock to be sold at index i
        maxi = -float(\'inf\')
        while i < len(prices):
            profit[i] = max(prices[i] + maxi,
                            profit[i-1])
            maxi = max(maxi, -prices[i] + (profit[i-2] if i >= 2 else 0))
            i += 1
        return profit[-1]
