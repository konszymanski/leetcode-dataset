class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        
        cd = 0
        sell = 0
        buy = float(\'-inf\')
        
        for price in prices:
            old_cd = cd
            old_sell = sell
            old_buy = buy
            
            cd = max(old_cd, old_sell)
            sell = old_buy + price
            buy = max(old_buy, old_cd - price)
        
        return max(sell, cd)
