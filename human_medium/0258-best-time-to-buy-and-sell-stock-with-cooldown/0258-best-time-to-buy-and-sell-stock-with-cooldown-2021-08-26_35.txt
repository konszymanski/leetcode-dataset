class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        sold = 0
        hold = -sys.maxsize
        cool = 0
        for price in prices:
            sold_tmp = sold
            hold_tmp = hold
            cool_tmp = cool
            sold = max(sold_tmp, hold_tmp + price)
            hold = max(hold_tmp, cool - price)
            cool = max(cool_tmp, sold_tmp)
        return max(hold, sold)
