class Solution:
    def maxProfit(self, prices: List[int]) -> int:
        # Purpose: find max profit buying and selling stock with cooldown day
        # Method: dp
        # Intuition: 
            # (1) hold: hold before, or cooldown before and buy now.
            # (2) sell: hold before and sell now.
            # (3) cooldown: cooldown before, or sell before.
            
        # init
        cooldown = 0
        sell = 0
        hold = -float(\'inf\')
        
        # find 
        for price in prices:
            
            # store variable
            hold_before = hold
            sell_before = sell
            cooldown_before = cooldown
            
            # run dp
            hold = max(hold_before, cooldown_before - price)
            sell = hold_before + price
            cooldown = max(cooldown_before, sell_before)
        
        # return
        return max(sell, cooldown)
