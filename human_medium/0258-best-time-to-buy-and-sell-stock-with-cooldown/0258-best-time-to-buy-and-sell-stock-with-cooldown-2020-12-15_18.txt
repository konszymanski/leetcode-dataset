def maxProfit(self, prices: List[int]) -> int:
        
        @lru_cache(maxsize=None)
        def recursion(day: int, canBuy: bool) -> int:
            if day >= len(prices):
                return 0
            
            buyOnDay = sellOnDay = 0
            if canBuy:
                buyOnDay = -prices[day] + recursion(day + 1, False)
            else:
                sellOnDay = prices[day] + recursion(day + 2, True)
                
            doNothing = recursion(day + 1, canBuy)
            maxProfit = max(buyOnDay, sellOnDay, doNothing)
            
            return maxProfit
        
        return recursion(0, True)
