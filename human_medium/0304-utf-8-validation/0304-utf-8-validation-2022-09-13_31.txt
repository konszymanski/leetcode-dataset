class Solution(object):
    def validUtf8(self, data):
        temp_binary="00000000"
        for i in range(0,len(data)):
            temp=format(data[i],\'08b\')
            #print(temp)
            if temp[0]==temp[1]==temp[2]==temp[3]==temp[4]==1:
                return False
            data[i]=temp
        i=0
        j=0
        n=len(data)
        #print(data[0])
        while j<len(data):
            if data[j][0]=="0":
                j=j+1
            elif data[j][0:3]=="110":
                if j+1<n and data[j+1][0:2]=="10":
                    j=j+2
                else:
                    return False
            elif data[j][0:4]=="1110":
                if j+2<n and data[j+1][0:2]=="10" and data[j+2][0:2]=="10":
                    j=j+3
                else:
                    return False
            elif data[j][0:5]=="11110":
                if j+3<n and data[j+1][0:2]=="10" and data[j+2][0:2]=="10" and data[j+3][0:2]=="10":
                    j=j+4
                else:
                    return False
            else:
                return False
        return True
