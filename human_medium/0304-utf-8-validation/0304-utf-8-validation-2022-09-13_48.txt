class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        curr = 0 
        for x in data:
            x = bin(x)[2:].zfill(8) 
            if curr: 
                if x.startswith("10"): #will check if the number starts with 10 or not
                    curr -= 1
                else: 
                    return False 
            else:  
                curr = x.find("0") # will find the index of first zero in the string in case not present wil return -1
                if curr not in {0, 2, 3, 4}:  #For byte 1 First zero is at 0,For byte 2 First zero is at index 2 , For Byte 3 first zero is at index 3 and for Byte 4 first zero is at index 4
                    return False
                if curr: 
                    curr -= 1 #Number of Values to check if the they start with "10" or not
                    
        return curr == 0 # If the curr is not 0 then there are not enough number to satisfy all the conditions
