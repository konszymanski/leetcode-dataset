class Solution(object):
    def validUtf8(self,data):
        x=""
        b=[]
        s=""
        L=["10","0","110","1110","11110"]
        for i in data:
            j=0
            x=bin(i).replace("0b","")
            if len(x)<8:
                x="0"*(8-len(x))+x
            b.append(x)
        c=0
        if len(data)>1:
            for i in range(len(b)):
                if c>0:
                    if b[i][:2]!="10":
                        return False
                    c-=1
                elif c==0 and b[i][:2]=="10":
                    return False
                else:
                    z=b[i].find("0")
                    s=b[i][0:z+1]
                    if s not in L:
                        return False
                    c=len(s)-2
                
            if c>0:
                return False
        else:
            if b[0][0]!="0":
                return False
        return True
