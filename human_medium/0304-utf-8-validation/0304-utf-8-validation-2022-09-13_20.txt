def validUtf8(self, data: List[int]) -> bool:
        expecting = 0  # how many extra octets we currently expect
        for byte in data:
            if expecting:
                if byte & 0b11_000000 == 0b10_000000:
                    expecting -= 1
                else:
                    return False
            else:  # not expecting any extra 10xxxxxx octets. Can be any of 4 cases:
                if not byte & 0b1_0000000:  # One octet case
                    continue
                elif byte & 0b111_00000 == 0b110_00000:  # 2 octets case
                    expecting = 1
                elif byte & 0b1111_0000 == 0b1110_0000:  # 3 octets case
                    expecting = 2
                elif byte & 0b11111_000 == 0b11110_000:  # 4 octets case
                    expecting = 3
                else:
                    return False
        return not expecting  # all is good if we get here and expecting==0
