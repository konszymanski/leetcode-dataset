class Solution:
    def validUtf8(self, data: List[int],cnt=0) -> bool:
        for d in data:
            if d&0xf8==0xf8:   # read 111110
                return False
            if d&0xf0==0xf0:   # read 11110
                if cnt>0: return False
                cnt=3
            elif d&0xe0==0xe0: # read 1110
                if cnt>0: return False
                cnt=2
            elif d&0xc0==0xc0: # read 110
                if cnt>0: return False
                cnt=1
            elif d&0x80==0x80: # read 10
                if cnt<=0: return False
                cnt-=1
            elif cnt>0:        # read 0*
                return False
        return True if not cnt else False
