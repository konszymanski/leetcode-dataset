class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        
        check = 0
        for x in (data):
            if x >= 248:
                return False
            
            elif check:
                if 192 > x >= 128:
                    check -= 1
                else:
                    return False
                
            elif x >= 240:
                check = 3
                    
            elif x >= 224:
                check = 2
            
            elif x >= 192:
                check = 1
                
            elif x<128:
                continue
            
            else:
                return False
        
        return True if not check else False
