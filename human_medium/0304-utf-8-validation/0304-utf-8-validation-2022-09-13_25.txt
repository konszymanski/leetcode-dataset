class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        exp_len = 0 # count for expected length
        for item in data:
            # convert integer to binary
            item = bin(item)
            item = item[2:]
            offset = 8-len(item)
            item = \'0\'*offset + item
            # 1 byte char
            if item[0] == \'0\' and exp_len == 0:
                continue
            # Follow up bytes
            elif item.startswith(\'10\') and exp_len > 0:
                exp_len -= 1
            # n byte char
            elif item.startswith(\'11\') and exp_len == 0:
                exp_len = 0
                # get count for expected length
                for ch in item:
                    if ch == \'0\':
                        break
                    exp_len += 1
                # if invalid
                if exp_len > 4:
                    return False
                exp_len -= 1
            else:
                return False
        # check if exp_len is not exhausted
        if exp_len:
            return False
        return True
