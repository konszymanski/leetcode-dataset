class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        # get the least significant 8 bits
        bit = []
        for d in data: bit.append(d & 0b11111111)
        
        # traverse bit
        i = 0
        while i < len(data):
            # check 4 bytes
            if bit[i] >> 3 == 0b11110 and i + 3 < len(data):
                for j in range(i+1, i+4):
                    if bit[j] >> 6 != 0b10: 
                        return False
                i += 4
            # check 3 bytes
            elif bit[i] >> 4 == 0b1110 and i + 2 < len(data):
                for j in range(i+1, i+3):
                    if bit[j] >> 6 != 0b10: 
                        return False
                i += 3
            # check 2 bytes
            elif bit[i] >> 5 == 0b110 and i + 1 < len(data):
                for j in range(i+1, i+2):
                    if bit[j] >> 6 != 0b10: 
                        return False
                i += 2
            # check 1 byte
            elif bit[i] > 0b10000000: return False
            else: i += 1
                
        return True
