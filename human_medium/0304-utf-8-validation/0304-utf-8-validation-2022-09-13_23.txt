class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        def int2bin(n):
            s = str(bin(n)[2:])
            return \'0\' * (8 - len(s)) + s
        
        char_size = cntr = 0        

        for i in data:
            i = int2bin(i)
            print(i)
            if not cntr:
                if i.startswith(\'0\'):
                    cntr = char_size = 1
                elif i.startswith(\'110\'):
                    cntr = char_size = 2
                elif i.startswith(\'1110\'):
                    cntr = char_size = 3
                elif i.startswith(\'11110\'):
                    cntr = char_size = 4
                else:
                    return False
                
            if(cntr < char_size):
                if not i.startswith(\'10\'):
                    return False
                
            cntr -= 1
            
        if cntr:
            return False
        
        return True
