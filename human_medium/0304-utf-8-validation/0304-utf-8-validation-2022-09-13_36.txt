class Solution:
    def validUtf8(self, data: List[int]) -> bool:
	    # Keep track of how many bytes are expected to follow (currently 0)
        expected = 0
        for next_byte in data:
			# Convert to binary string, discard the 0b
            binary = bin(next_byte)[2:]
			# If it starts with zero or has fewer than 8 characters (which means the leading 0s were omitted)
            if (binary[0] == \'0\' or len(binary) < 8):
				# We didn\'t expect any 10... characters, so this is fine
                if expected == 0:
                    continue
				# We expected one and didn\'t get it
                else:
                    return False
			# No bytes expected, so we can start a new multi-byte character
            elif expected == 0:
                if binary[:3] == \'110\':
                    expected = 1
                elif binary[:4] == \'1110\':
                    expected = 2
                elif binary[:5] == \'11110\':
                    expected = 3
				# Any other beginning isn\'t valid
                else:
                    return False
			# This corresponds to a start with 1 with bytes still expected
            else:
				# Only acceptable format for following bytes of a multi-byte character
                if binary[:2] == \'10\':
                    expected -= 1
                else:
                    return False
		# Last thing to check is that no more bytes are expected after all have been processed
        return expected == 0
