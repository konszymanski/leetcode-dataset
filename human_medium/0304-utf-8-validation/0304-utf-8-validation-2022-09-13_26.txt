class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        k = 0
        for x in data:
            if k:
                if x>191 or x<128: return False
                k-=1
            else:
                if x>247: return False
                if x>239: k=3
                elif x>223: k=2
                elif x>191: k=1
                elif x>127 or x<0: return False
        return not k
