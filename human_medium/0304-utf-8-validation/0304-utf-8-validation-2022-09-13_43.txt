import "fmt"
func validUtf8(data []int) bool {
    i:=0
    for i<len(data){
        bin_rep := fmt.Sprintf("%08b", data[i])
        number_of_ones := 0
        for _,b := range(bin_rep){
            if string(b) == "0"{
                break
            }
            number_of_ones+=1
        }
        if number_of_ones == 1 || number_of_ones > 4{
            return false
        }
        for j:=0;j<number_of_ones-1;j++{
            i++
            if i == len(data){
                return false
            }
            bin_rep := fmt.Sprintf("%08b", data[i])
            if bin_rep[:2] != "10"{
                return false
            }
        } 
        i++
    }
    return true
}
