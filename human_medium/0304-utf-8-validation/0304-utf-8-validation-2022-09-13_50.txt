class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        binarySequence = []
        n = len(data)
        i = 0
        for integer in data:
            integerToBinary = bin(integer)[2:]
            while len(integerToBinary) < 8:
                integerToBinary = "0" + integerToBinary
            binarySequence.append(integerToBinary)
        print(binarySequence)
        while i < n:
            nBytes = 0
            while nBytes < 8 and binarySequence[i][nBytes] == \'1\':
                nBytes += 1
            print(nBytes)
            if nBytes == 0:
                i += 1
            elif nBytes == 1:
                return False
            elif nBytes < 5:
                i += 1
                for j in range(nBytes-1):
                    if i >= n or binarySequence[i][:2] != \'10\':
                        return False
                    i += 1
            else:
                return False
        return True
