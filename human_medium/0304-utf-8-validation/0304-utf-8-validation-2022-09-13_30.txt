class Solution:
    def validUtf8(self, data: List[int]) -> bool:
        # Start with 0 count. Count counts the number of sequence which has to be \'10xxxxxx\'
		count = 0
        
        for number in data:
            # Convert number to 8-bit binary format
            binary = format(number, \'08b\')
            
            # When count = 0, we can start with \'0xxxxxxx\' or \'110xxxxx\' or \'1110xxxx\' or \'11110xxx\'
            if count == 0:
                # If \'0xxxxxxx\' we can continue safely
                if binary[0] ==\'0\':
                    continue
                # Otherwise, we have to increase count to account for the number of \'10xxxxxx\' sequences
                else:
                    for i in range(1,5):
                        if binary[i] == \'1\':
                            count += 1
                        else:
                            # Check for case where we have \'10xxxxxx\' as our first number
                            if count == 0:
                                return False
                            break
            
            # Check that count does not exceed 3
            elif count > 3:
                return False
            
            # For count that is between [1,3]
            else:
                # Numbers have to start with \'10xxxxxx\'
                if binary[:2] != \'10\':
                    return False
                else:
                    count -= 1
        
		# If we are left with count >0, this means the sequence is not finished so return False
        return True if count == 0 else False
