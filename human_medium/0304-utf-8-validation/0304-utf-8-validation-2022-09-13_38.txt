def validUtf8(self, data: List[int]) -> bool:
        counter = 0
        for d in data:
            dByte = d & 0xFF
            if counter == 0:
                if (dByte & 0xFF) >> 7 == 0:
                    continue
                elif (dByte & 0xC0) >> 6 == 2:
                    return False
                elif dByte & 0xE0 == 0xC0:
                    counter = 1
                elif dByte & 0xF0 == 0xE0:
                    counter = 2
                elif dByte & 0xF8 == 0xF0:
                    counter = 3
                else:
                    return False
            else:
                if (dByte & 0xC0) >> 6 == 2:
                    counter -= 1
                else:
                    return False
        return True if counter == 0 else False
