class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        nums_sum = sum(nums)
        if nums_sum % 2: 
            return False
        sumToFind = nums_sum // 2
        n = len(nums)
        
        dp = [[-1]*(sumToFind+1) for _ in range(n+1)]
        
        def knapsack(n, reminding_sum):
            if n == 0 or reminding_sum == 0:
                return reminding_sum == 0
            
            elif dp[n][reminding_sum] != -1: 
                return dp[n][reminding_sum]
            
            elif nums[n-1] <= reminding_sum:
                pick = knapsack(n-1, reminding_sum-nums[n-1])
                not_pick = knapsack(n-1, reminding_sum)
                dp[n][reminding_sum] = pick or not_pick
            
            else:
                dp[n][reminding_sum] = knapsack(n-1, reminding_sum)
            
            return dp[n][reminding_sum]
        
        return knapsack(n, sumToFind)
