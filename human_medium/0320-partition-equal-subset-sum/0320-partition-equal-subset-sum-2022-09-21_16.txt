class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        @cache
        def dfs(ind: int, target: int) -> bool:
		    # run out of the list and cannot satisfy the condition
            if ind >= len(nums):
                return False
			# find the target sum
            if target == 0:
                return True
			# cannot find target
            if target < 0:
                return False
			# we sorted nums, so if target is less than the current element, no need to go further
            if target < nums[ind]:
                return False

			# we either "select" the current element or we do not "select" it
            return dfs(ind + 1, target - nums[ind]) or dfs(ind + 1, target)
        
        total = sum(nums)
        nums = sorted(nums)
        if total & 1 != 0:
            return False
        return dfs(0, total >> 1)
