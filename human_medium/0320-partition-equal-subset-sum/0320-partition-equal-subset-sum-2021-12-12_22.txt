class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        n = len(nums)
        s = sum(nums)
        
        if s%2:
            return False

        @cache
        def rec(i, one, two):
            if i == n:
                return not one and not two
            
            if one < 0 or two < 0:
                return False
            
            return rec(i + 1, one - nums[i], two) or rec(i + 1, one, two - nums[i])
        
        return rec(1, s/2 - nums[0], s/2)
