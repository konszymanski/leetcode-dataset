class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        if (ss := sum(nums)) & 1: return False 
        
        @lru_cache(None)
        def fn(i, v):
            """Return True if possible to find subarray of nums[i:] summing to v."""
            if v <= 0: return v == 0
            if i == len(nums): return False 
            return fn(i+1, v) or fn(i+1, v - nums[i])
        
        return fn(0, ss//2)
