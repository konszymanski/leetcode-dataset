class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        if sum(nums) % 2: # check validity
            return False 
			
        dp = [False] * (sum(nums) // 2 + 1)
        dp[0] = True

        for num in nums:
            for i in range(len(dp)-1, 0, -1): # going backwards 
                if num <= i and dp[i-num] and not dp[i]:
                    dp[i] = True
        return dp[-1]
