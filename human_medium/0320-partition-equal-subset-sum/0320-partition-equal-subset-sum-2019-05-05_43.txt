class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        numSum=sum(nums)
        if numSum%2!=0:
            return False
        dp=[0]*(numSum+1)
        dp[0]=1
        
        for num in nums:
            for i in range(numSum,-1,-1):
                if dp[i]:
                    dp[num+i]=1
            if dp[int(numSum/2)]==1:
                return True
        return False
