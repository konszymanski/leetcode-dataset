class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        ans = False
        N = len(nums)
        ts = sum(nums)
        ns = set([0])

        for i in range(N):
            for j in list(ns):
                p = j+nums[i] 
                if ts-p > p:
                    ns.add(p)
                elif ts-p == p:
                    return True
        return False
