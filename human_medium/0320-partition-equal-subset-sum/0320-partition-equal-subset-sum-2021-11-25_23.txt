class Solution:
    def subset_sum_part2(self,arr,target,index):
        if index==len(arr):
            if target==0:
                return True
            else:
                return False

        target-=arr[index]
        if self.subset_sum_part2(arr,target,index+1)==True:
            return True
        target+=arr[index]

        if self.subset_sum_part2(arr,target,index+1)==True:
            return True
    
        return False

    def canPartition(self, nums: List[int]) -> bool:
        sum=0
        for i in range(0,len(nums)):
            sum+=nums[i]
        if sum%2==1:
            return False

        target=sum//2
        index=0
        return self.subset_sum_part2(nums,target,index)
