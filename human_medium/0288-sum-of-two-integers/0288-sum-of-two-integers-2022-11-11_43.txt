class Solution:
    def getSum(self, a: int, b: int) -> int:
        if a >= 0 and b >= 0:
            for i in range(b):
                a += 1
            return a
        elif a < 0 and b >= 0:
            for i in range(b):
                a += 1
            return a
        elif a >= 0 and b <= 0:
            for i in range(-b):
                a -= 1
            return a
        elif a <= 0 and b <= 0:
            for i in range(-b):
                a -= 1
            return a
