class Solution:
    def getSum(self, a: int, b: int) -> int:
        MAX_INT = 0x7FFFFFFF
        MIN_INT = 0x80000000
        MASK = 0x100000000
        
        while b:
            carry = (a & b)
            a = (a ^ b) % MASK
            b = (carry << 1) % MASK
            
        if(a <= MAX_INT):
            return a
        else:
            return ~((a % MIN_INT) ^ MAX_INT)
