class Solution(object):
    def getSum(self, a, b):
        """
        :type a: int
        :type b: int
        :rtype: int
        """
        l = []
        if a > 0:
            for _ in range(a):
                l.append(1)
        else:
            for _ in range(-1*a):
                l.append(-1)
        if b > 0:
            for _ in range(b):
                l.append(1)
        else:
            for _ in range(-1*b):
                l.append(-1)

        return sum(l)
