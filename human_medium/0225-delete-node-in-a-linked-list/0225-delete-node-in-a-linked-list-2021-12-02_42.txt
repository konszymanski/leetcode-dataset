class Solution:
    def deleteNode(self, node):
        # Since we don\'t have the head node, and the ListNode does not have a prev method,
        # we can\'t find the previous node and just repoint it to next, so that means we 
        # can\'t actually delete it, but we can move all the values from the remaining list
        # back one space, which is effectively overwriting it.
        # As there is one node less, we also want to None on penultimate node.next
        while node:
          node.val = node.next.val
          if node.next.next == None:
            node.next = None
            return
          node = node.next
