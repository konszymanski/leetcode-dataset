# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def deleteNode(self, node):
        """
        :type node: ListNode
        :rtype: void Do not return anything, modify node in-place instead.
        """
        while node.next.next: # looping till 2nd last node
            node.val = node.next.val
            node=node.next
        node.val = node.next.val  # updating 2nd last node\'s value
        node.next = None  #removing the last node thus indirectly we have "deleted" that node from the list!
