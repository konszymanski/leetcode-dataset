class Solution:
    def longestSubstring(self, s: str, k: int) -> int:
        n = len(s)
        
        #Base Case
        if n==0 and n<k:
            return 0
        if k<=1:
            return n
            
        counts = Counter(s)
        left = 0
        while left<n and counts[s[left]] >= k:
            left +=1
        if left > n-1:
            return left
        ls1 = self.longestSubstring(s[0:left], k)
        while left<n and counts[s[left]] < k:
            left +=1
        ls2 = self.longestSubstring(s[left:],k) if left<n else 0
        return max(ls1, ls2)
