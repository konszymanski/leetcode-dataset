from collections import Counter
class Solution:
    def longestSubstring(self, s: str, k: int) -> int:
        frequencyMap = Counter(s)
        
        if len(s) == 0:
            return 0
        
        longestLength = 0
        for i in range(len(s)):
            if frequencyMap[s[i]] < k:
                longestLength = max(self.longestSubstring(s[:i], k), self.longestSubstring(s[i+1:], k))
                break
                
            longestLength += 1
        return longestLength
