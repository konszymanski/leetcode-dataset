from collections import Counter

class Solution:
    def longestSubstring(self, s: str, k: int) -> int:
	
        if len(s) < k: return 0
		
		c = Counter(s)
        st = 0
        for p, v in enumerate(s):
            
            if c[v] < k:
                
                return max(self.longestSubstring(s[st:p],k), self.longestSubstring(s[p+1:],k))
        
        return len(s)
