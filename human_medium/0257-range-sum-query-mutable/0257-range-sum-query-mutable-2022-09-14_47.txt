class Node:
    def __init__(self, start, end):
        self.start = start
        self.end = end
        self.sum = 0
        self.left = None
        self.right = None

def build(arr, start, end):
    root = Node(start, end)
    if start == end:
        root.sum = arr[start]
        return root
    mid = start + (end - start)//2
    root.left = build(arr, start, mid)
    root.right = build(arr, mid+1, end)
    root.sum = root.right.sum + root.left.sum
    return root

def update(root, i, val):
    # print(root, i, val)
    if root.start == root.end and root.start == i:
        root.sum = val
        return
    # range
    mid = root.start + (root.end-root.start)//2
    if i <= mid:
        update(root.left, i, val)
    else:
        update(root.right, i, val)
    root.sum = root.left.sum + root.right.sum
    
def get_sum(root, start, end):
    # print(start, end, root.start, root.end)
    if root.start == root.end:
        return root.sum
    mid = root.start + (root.end - root.start)//2
    # print("mid", mid)
    if end <= mid:
        return get_sum(root.left, start, end)
    elif start >= mid+1:
        # print("last")
        return get_sum(root.right, start, end)
    else:
        return get_sum(root.left, start, mid) + get_sum(root.right, mid+1, end)
    
    

class NumArray:

    def __init__(self, nums: List[int]):
        self.tree = build(nums, 0, len(nums)-1)

    def update(self, index: int, val: int) -> None:
        update(self.tree, index, val)

    def sumRange(self, left: int, right: int) -> int:
        return get_sum(self.tree, left, right)
