class NumArray:

    def __init__(self, nums: List[int]):
        self.num_array=nums
        self.pre_sum=sum(self.num_array)  # lets keep sum at the entry
    def update(self, index: int, val: int) -> None:
        self.pre_sum=self.pre_sum - self.num_array[index] + val # we will not calculate the whole sum again we will just remove old value and add newone 
        self.num_array[index]=val
        
    def sumRange(self, left: int, right: int) -> int:
        return self.pre_sum - sum(self.num_array[:left]) - sum(self.num_array[right+1:]) # we will calculate before left sum and after right and minus from whole sum 
    
    
    # thats it here we go as simple as that
