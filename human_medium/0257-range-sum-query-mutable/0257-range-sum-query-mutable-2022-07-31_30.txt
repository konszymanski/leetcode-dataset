def __init__(self, nums: List[int]):
        
        # to store the nums array 
        self.nums = [0] * len(nums)
        
        # to create a fenwick tree
        self.tree = [0] * (len(nums) + 1)
        
        # to call the update function to fill or update the fenwick tree and update the 
        # nums array
        for i in range(len(nums)):
            self.update(i,nums[i])
        

    def update(self, index: int, val: int) -> None:
        # to update the fenwick tree
        # find the difference between the old and new value
        # update the new value in the nums array
        # increment the current index by 1 
        # run the loop while index is in range of the length of nums array
        # update the value of index as below
        # index += (index & -index)
        
        diff = val - self.nums[index]
        self.nums[index] = val
        
        index += 1
        
        while index <= len(self.nums):
            self.tree[index] += diff
            
            index += (index & -index)
            
    # to search the value in the fenwick tree
    def search(self,index):
        
        # to store the total from index 0 to index
        total = 0
        
        # run the loop while index is greater than 0
        # update the value of index as below
        # index -= (index & -index)
        
        while index > 0:
            total += self.tree[index]
            index -= (index & -index)
            
        # return the total
        return total

    def sumRange(self, left: int, right: int) -> int:
        
        # search the value of the from 0 to right and 0 to left and return the difference
        return self.search(right + 1) - self.search(left)
