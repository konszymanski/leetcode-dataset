class Fenwick:
    
    def __init__(self,nums):
        #! initialization of BIT 
        self.sums =  [0] * (nums+1)
    
    def update(self,index,delta):
        while index < len(self.sums):
            self.sums[index] += delta
            index += (index & -index) #! flip the bits
        
    #! query for index   
    def query(self,index): 
        result = 0
        while index > 0:
            result +=  self.sums[index]
            index -= (index & -index)  #! flip the bits
        return result
            
            

class NumArray(object):

    def __init__(self, nums):
        """
        :type nums: List[int]
        """
        self.btree =  Fenwick(len(nums))
        self.nums = nums
        for i,val in enumerate(nums):
            self.btree.update(i+1,val) #! one indexed fenwick tree
            
        

    def update(self, index, val):
        """
        :type index: int
        :type val: int
        :rtype: None
        """
        delta = val - self.nums[index]
        self.btree.update(index+1,delta)
        self.nums[index] = val    

    def sumRange(self, left, right):
        """
        :type left: int
        :type right: int
        :rtype: int
        """
        return (self.btree.query(right+1) - self.btree.query(left))
