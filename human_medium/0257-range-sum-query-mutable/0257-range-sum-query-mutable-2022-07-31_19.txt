class NumArray:
    nums, s, l = [], 0, 0
    
    def __init__(self, nums: List[int]):
        self.nums, self.s, self.l = nums, sum(nums), len(nums)
        
    def update(self, index: int, val: int) -> None:
        self.s, self.nums[index] = self.s - self.nums[index] + val, val

    def sumRange(self, left: int, right: int) -> int:
        return self.s - sum(self.nums[:left]) - sum(self.nums[right + 1:]) if right - left > (self.l / 2) else sum(self.nums[left: right + 1])
