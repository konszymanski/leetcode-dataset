class FenTree():
    
    def __init__(self, n):
        
        # Discard cell_#0, we only use cell_#1 to cell_#n for convenience
        self._array = [ 0 for _ in range(n+1) ]
        
        self._size = len(self._array);

    def getLSB(self, number):
        
        # Get least significant bit of number
        return number & (-number)
        
        
    def update(self, i, delta):
        
        # Update element value of index i
        while i < self._size:
            self._array[i] += delta
            i += self.getLSB(i)
        
        
    def query(self, i):
        
        # Compute and return accumulative sum from 1-st element to i-th element
        summation = 0
        
        while i > 0:
            
            summation += self._array[i]
            i -= self.getLSB(i)
            
        return summation

    
    def rangeSum(self, left, right):
        
        # Compute range sum of [left ... right] inclusively
        return self.query(right) - self.query(left-1)
    
# ----------------------------------------

class NumArray():
    def __init__(self, nums):

        self._input = nums
        self._fenTree = FenTree( len(self._input) )
        
        for idx, number in enumerate(self._input, 1):
            self._fenTree.update(idx, number)

        
        
    def update(self, i, val):
        
        delta = val-self._input[i]
        self._fenTree.update(i+1, delta)
        self._input[i] = val

        
    def sumRange(self, i, j):
        
        return  self._fenTree.rangeSum(i+1, j+1)
