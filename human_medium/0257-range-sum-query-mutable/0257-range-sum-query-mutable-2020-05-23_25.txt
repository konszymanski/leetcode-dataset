class NumArray:
    def __init__(self, nums: List[int]):
        if nums:
            self.nums = nums
            self.sums = self.generatePrefixSums(nums)

    def update(self, i: int, val: int) -> None:        
        self.nums[i] = val
        self.sums = self.generatePrefixSums(self.nums)

    def sumRange(self, i: int, j: int) -> int:
        return self.sums[j + 1] - self.sums[i]
        
    def generatePrefixSums(self, nums) -> List[int]:
        res = []
        res.append(0)
        
        for i in range(1, len(nums) + 1):
            res.append(nums[i - 1] + res[i - 1])
        
        return res
