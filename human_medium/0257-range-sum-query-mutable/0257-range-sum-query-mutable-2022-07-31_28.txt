import numpy


class NumArray:
    
    def __init__(self, nums: List[int]):
        self.nums = numpy.array(nums)
             
    def update(self, index: int, val: int) -> None:
        self.nums[index] = val
        
    def sumRange(self, left: int, right: int) -> int:
        return numpy.sum(self.nums[left:right+1])
