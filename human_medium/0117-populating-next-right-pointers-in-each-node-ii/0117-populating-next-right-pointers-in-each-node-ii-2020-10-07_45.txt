"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: \'Node\' = None, right: \'Node\' = None, next: \'Node\' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

from collections import deque
class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        Q = deque([root]) if root else [] 
        while Q:
            m = len(Q)
            while m:
                m -= 1
                node = Q.popleft()
                if m != 0:
                    node.next = Q[0]
                if node.left:
                    Q.append(node.left)
                if node.right:
                    Q.append(node.right)
        return root
