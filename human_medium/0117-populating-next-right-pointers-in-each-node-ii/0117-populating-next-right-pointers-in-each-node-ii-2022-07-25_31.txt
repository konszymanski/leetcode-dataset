"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: \'Node\' = None, right: \'Node\' = None, next: \'Node\' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:
    def connect(self, root: \'Node\') -> \'Node\':
        if not root: return root
        q = [root]
        
        while q:
            l = len(q)
            
            # update the field of next
            for i in range(l - 1):
                q[i].next = q[i+1]
            
            # pop up all the nodes and append their non-none children
            for i in range(l):
                head = q.pop(0)
                if head.left: q.append(head.left)
                if head.right: q.append(head.right)
        return root
