"""
# Definition for a Node.
class Node:
    def __init__(self, val: int = 0, left: \'Node\' = None, right: \'Node\' = None, next: \'Node\' = None):
        self.val = val
        self.left = left
        self.right = right
        self.next = next
"""

class Solution:   
    def connect(self, root: \'Node\') -> \'Node\':
        def node_list(node):
            while node:
                if node.left:
                    yield node.left
                if node.right:
                    yield node.right
                node = node.next
                
        head = root
        has_head = True
        
        while has_head:
            has_head = False
            temp = head
            for node in node_list(head):
                if not has_head:
                    has_head = True
                    head = node
                else:
                    temp.next = node
                temp = node
        return root
