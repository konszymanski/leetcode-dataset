class Solution(object):
    def integerBreak(self, n):
        """
        :type n: int
        :rtype: int
        """
        dp = [0] * (n+1)
        dp[1] = 1
        for i in range(2,n+1):
            # we just need to iterate from 1 to n//2
            for j in range(1,i//2+1):
                dp[i] = max(max(j, dp[j]) * max(i-j, dp[i-j]), dp[i])
        return dp[-1]
