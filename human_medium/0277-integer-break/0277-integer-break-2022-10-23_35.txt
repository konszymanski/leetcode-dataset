class Solution:
    def integerBreak(self, n: int) -> int:
        if n <= 3 : return n -1 # ugly way to handle edge cases, let me know if you have suggestion
        dp = {} # (i, n_left) : max product
        def helper(n):
            if n < 0:
                return 0
            if n == 0:
                return 1
            if n in dp:
                return dp[n]
            dp[n] = 0
            for idx in range(1,n+1):
                dp[n] = max(helper(n-idx) * idx, dp[n])       
            return dp[n]
        return helper(n)
