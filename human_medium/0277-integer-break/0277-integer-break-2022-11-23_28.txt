class Solution:
    def integerBreak(self, n: int) -> int:
        if n == 2:
            return 1
        if n == 3:
            return 2
        remainder = n%3
        numberOfThree = n//3
        if remainder == 0:
            return 3 ** numberOfThree
        if remainder == 1:
            return (3 ** (numberOfThree-1)) * 4
        else:
            return (3 ** numberOfThree) * 2
