class Solution:
    def integerBreak(self, n: int) -> int:
        dp = {}
        
        def dfs(n):
            if n == 1:
                return 1
            if n in dp:
                return dp[n]
            ans = float(\'-inf\')
            for l in range(1,n):
                r =  n-l
                left,right = dfs(l),dfs(r)
                ans = max(ans ,left * r , l * right ,r*l)
            dp[n] = ans
            return ans
        return dfs(n)
