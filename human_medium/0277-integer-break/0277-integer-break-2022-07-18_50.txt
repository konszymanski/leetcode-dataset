class Solution:
    def integerBreak(self, n: int) -> int:
        dp=[0]*(n+1)
        dp[2]=1
        for i in range(3,n+1):
            for j in range(1,i):
			#sometimes not divide number can generate larger result, for example,use 3*something is larger than using 2*1*something
                dp[i]=max(dp[i],max(j*dp[i-j],j*(i-j)))
        return dp[n]
