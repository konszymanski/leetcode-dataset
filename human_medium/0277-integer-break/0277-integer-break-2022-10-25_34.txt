class Solution:
    def integerBreak(self, n: int) -> int:
        if n == 2:
            return 1
        if n == 3:
            return 2
        
        if n % 3 == 0:
            return pow(3, n//3)
        else:
            r3 = 0
            while n > 4:
                r3 += 1
                n -= 3
            
            return pow(3, r3) * n
