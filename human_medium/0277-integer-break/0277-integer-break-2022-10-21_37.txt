class Solution:
    partial = 0
    def integerBreak(self, n: int) -> int:
        self.partial = 0
        @cache
        def maxBreak(n):
            for i in range(1,n):
                self.partial = max(self.partial,i*(n-i),i*maxBreak(n-i))
            
            return self.partial
        
        return maxBreak(n)
