class Solution:
    def integerBreak(self, n: int) -> int:
		def maxproduct(n,k):
            if n==0:
                if k>=2:
                    return 1
                return 0
            res=0
            for i in range(1,n+1):
                res=max(res,maxproduct(n-i,k+1)*i)
            return res
        return maxproduct(n,1)
