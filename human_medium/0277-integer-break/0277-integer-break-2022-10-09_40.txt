class Solution:
    def integerBreak(self, n: int) -> int:

        def dfs(a):
            if a == 1:
                return 1
            
            max_res = 1
            for right in range(1, a):
                left = a - right
                max_res = max(max_res, left * dfs(right), left * right)
            
            return max_res
        
        return dfs(n)
