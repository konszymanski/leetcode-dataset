class Solution:
    def integerBreak(self, n: int) -> int:
        
        @cache
        def recurse(no):
            if no==0:
                return 1
            
            ans=float(-inf)
            
            for i in range(1,n):
                if(no-i) >=0:
                    ans=max(ans,recurse(no-i)*i)
            return ans
                
        return recurse(n)
