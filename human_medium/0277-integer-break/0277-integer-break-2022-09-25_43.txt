class Solution:
    def integerBreak(self, n: int) -> int:
        # when k=1, product=the number themselves
        dp = [i for i in range(n+1)]
        # but don\'t initialize last cell because k cannot be 1
        dp[n]=0
        for i in range(1,n+1):
            for j in range(1,i):
                dp[i]=max(dp[i], j*dp[i-j])
        return dp[n]
