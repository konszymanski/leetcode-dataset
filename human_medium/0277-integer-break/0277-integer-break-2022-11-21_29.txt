class Solution:
    def integerBreak(self, n: int) -> int:
        if n == 3:
            return 2
        dp = [1 for i in range(n+1+5)]
        
        dp[2] = 1
        dp[3] = 3
        dp[4] = 4
        dp[5] = 6
        
        for i in range(6 , n+1):
            dp[i] = 3 * dp[i-3]
        return dp[n]
