class Solution:
    def integerBreak(self, n: int) -> int:
        dp = {}
        temp = self.recursion(n, dp, 0, 0)
        return dp[n]
    def recursion(self, n, dp, ii, count):
        if n == 0 and count < 2:
            return -1
        if n == 0 and count >= 2:
            return 1
        if n in dp:
            return dp[n]
        dp[n] = 0
        for i in range(1, n + 1):
            dp[n] = max(self.recursion(n - i, dp, i, count + 1) * i, dp[n])
        return dp[n]
