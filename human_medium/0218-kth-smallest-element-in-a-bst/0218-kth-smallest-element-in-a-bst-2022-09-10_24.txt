class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        
        def inOrder(node):
            if not node: return
            yield from inOrder(node.left)
            yield node.val
            yield from inOrder(node.right)
        
        it = inOrder(root)
        for _ in range(k):
            ret = next(it)
        return ret
