class Solution:
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        self.k, self.res = k, None
        self.inOrder(root)
        return self.res
        
        
    def inOrder(self, node: TreeNode) -> None:
        if node == None: return
        
        self.inOrder(node.left)
        
        self.k -= 1
        if self.k == 0: 
            self.res = node.val
            return
        
        self.inOrder(node.right)
