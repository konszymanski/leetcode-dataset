class Solution:
    def dfs(self, root: TreeNode, result:list) -> None:
        if root:
            self.dfs(root.left, result)
            result.append(root.val)
            self.dfs(root.right, result)
            
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        if root:
            result = []
            self.dfs(root, result)
            return result[k-1]
