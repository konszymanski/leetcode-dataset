def preorder(self, root, heap, k):
        if not root: 
            return
        if len(heap) < k:
            heapq.heappush(heap, -root.val)
        elif -root.val > heap[0]: 
            heapq.heapreplace(heap, -root.val)
        self.preorder(root.left, heap, k)
        self.preorder(root.right, heap, k)
        
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        heap = []
        self.preorder(root, heap, k)
        return abs(heap[0])
    
    def inorder(self, root, k, nums):
        if len(nums) == k or not root: 
            return
        self.inorder(root.left, k, nums)
        if len(nums) < k:
            nums.append(root.val)
            self.inorder(root.right, k, nums)
            
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        nums = []
        self.inorder(root, k, nums)
        return nums[-1] if nums else None
