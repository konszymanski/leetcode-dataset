class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        stack = []
        sort = []
        while root or stack:
            while root:
                stack.append(root)
                root = root.left
            root = stack.pop()
            sort.append(root.val)
            root = root.right
        sort.sort()
        return sort[k-1]
