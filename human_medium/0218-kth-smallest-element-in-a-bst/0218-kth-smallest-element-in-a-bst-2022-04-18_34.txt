class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:

        def inorder(n: TreeNode = root):
            if not n:
                return
            yield from inorder(n.left)
            yield n.val
            yield from inorder(n.right)
                
        return next(v for i, v in enumerate(inorder(), start=1) if i == k)
