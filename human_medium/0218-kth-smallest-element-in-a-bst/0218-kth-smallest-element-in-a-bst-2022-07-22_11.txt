class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        
        heap = []
        heapq.heapify(heap)
        
        def dfs(node):
            if node == None:
                return
            heapq.heappush(heap,node.val)
            dfs(node.left)
            dfs(node.right)
            
        dfs(root)
        
        for i in range(k-1):
            heapq.heappop(heap)
            
        return heapq.heapp
