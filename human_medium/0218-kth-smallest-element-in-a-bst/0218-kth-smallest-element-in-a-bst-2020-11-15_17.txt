class Solution:
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        stack = [root]
        node = root
        while stack:
            
            while node.left:
                node = node.left
                stack.append(node)

            while node:
                node, k = stack.pop(), k-1
                if not k: return node.val
                if node.right:
                    node = node.right
                    stack.append(node)
                    break
