class Solution:
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        res=[]
        def dfs(root):
            if not root:
                return 
            dfs(root.left)
            if len(res)==k:
                return
            res.append(root.val)
            dfs(root.right)
        
        dfs(root)
        return res[-1]
