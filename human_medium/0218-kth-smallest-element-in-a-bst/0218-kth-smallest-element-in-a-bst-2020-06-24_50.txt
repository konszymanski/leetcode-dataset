class Solution:
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        i = 0
        def helper(root, k):
            nonlocal i
            if not root:
                return
            a = helper(root.left, k)
            i += 1
            if i == k:
                return root.val
            b = helper(root.right, k)
            return b if b else a
        return helper(root, k)
