def kthSmallest(self, root: TreeNode, k: int) -> int:
       
        count = 0
        while root:
            if root.left:
                temp = root.left
                while temp.right and temp.right != root:
                    temp = temp.right
                if not temp.right:
                    temp.right = root
                    root = root.left
                else:
                    count += 1
                    if count == k:
                        return root.val
                    temp.right = None
                    root = root.right
            else:
                count += 1
                if count == k:
                    return root.val
                root = root.right
        
        return None
