# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution:
    def create(self,root,ans):
        if root is None:
            return ans
        ans = self.create(root.left,ans)
        ans.append(root.val)
        ans =self.create(root.right,ans)
        
        return ans
    
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        if root is None:
            return 
        ans =  self.create(root,[])
        return ans[k-1]
