def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        result = self.inorder(root)
        return result[k-1]

    def inorder(self, root):
        result = []
        if not root:
            return result
        l = self.inorder(root.left)
        result.append(root.val)
        r = self.inorder(root.right)
        return l + result + r
