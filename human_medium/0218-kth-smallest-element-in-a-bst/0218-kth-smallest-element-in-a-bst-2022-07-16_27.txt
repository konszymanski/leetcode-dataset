class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        
        res = []
        
        def bst(node):
            if (not node or (len(res) >= k)):
                return
            
            bst(node.left)
            res.append(node.val)
            bst(node.right)
            
        bst(root)
        
        return res[k - 1]
