class Solution:
    def kthSmallest(self, root: TreeNode, k: int) -> int:
        
        
        self.K = k
        self.out = None
        
        def inorder(node):
            
            if self.K > 0:
                
                if node.left:
                    inorder(node.left)
                
                self.K -= 1
                if self.K == 0:
                    self.out = node.val

                if node.right:
                    inorder(node.right)
        if root:            
            inorder(root)
        
        return self.out
