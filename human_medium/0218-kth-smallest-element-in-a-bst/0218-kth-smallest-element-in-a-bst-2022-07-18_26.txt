#ALl nodes distance K in a binary tree
class TreeNode:
    def __init__(self, x):
        self.val = x
        self.left = None
        self.right = None

class Solution:
    def kthSmallest(self, root, k):
        """
        :type root: TreeNode
        :type k: int
        :rtype: int
        """
        #bst
        res=[] #store all the nodes 
        def bst(root): #inorder traversal
            if root:  
                bst(root.left) #left
                res.append(root.val) #root 
                bst(root.right) #right
        bst(root) #inorder traversal 
        return res[k-1] #return the kth smallest element
