def kthSmallest(self, root, k):
        # inorder traversal   
        def inordTraverse(root, n): # n is the number which have been seen
			# the function will return the number of nodes under (including) root 
            if not root:
                return (0, None)
            
            s, x = inordTraverse(root.left, n)
            if s == k:
                return (k, x)
            if (n+s) == k-1:
                return (k, root.val)
            t, y = inordTraverse(root.right, n+s+1)
            if t == k:
                return (k, y)
            
            return (s+t+1, None)
        
        return inordTraverse(root, 0)[1]
