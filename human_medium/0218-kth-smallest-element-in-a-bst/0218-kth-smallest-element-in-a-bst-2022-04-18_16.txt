class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        self.cnt = self.res = 0
        
        def traverse(node):
            if not node:
                return
            
            traverse(node.left)
            
            self.cnt += 1     
            if self.cnt == k:
                self.res = node.val
                return
            
            traverse(node.right)
        
        traverse(root)
        return self.res
