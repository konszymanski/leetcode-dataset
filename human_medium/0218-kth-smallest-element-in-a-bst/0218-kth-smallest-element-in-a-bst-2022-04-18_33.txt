#space 17.9 MB, beats 99.08%
class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        sort = []  
        def inOrderTrav(node):
            if node.left:
                inOrderTrav(node.left)
            sort.append(node.val)
            if node.right:
                inOrderTrav(node.right)
            return
        inOrderTrav(root)
        return sort[k-1]
