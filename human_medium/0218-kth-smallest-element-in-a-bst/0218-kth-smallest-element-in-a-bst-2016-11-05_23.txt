class Solution(object):
    def kthSmallest(self, root, k):
        stack = []
        self.inorder(root, k, stack)
        return stack.pop()                  # top of the stack is kthSmallest
        
    def inorder(self, root, k, stack):
        if not root:                        # base case
            return
        
        self.inorder(root.left, k, stack)   # recurse left subtree
        
        if len(stack) < k:                  # visit root
            stack.append(root.val)          # push onto stack if we havent reached k
        else:
            return                          # otherwise, we dont care
        
        self.inorder(root.right, k, stack)  # recurse right subtree
