class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        nums.sort()
        N = len(nums)
        mid_values = {}
        mid_values[nums[N//2]] = 0 
        if N % 2 == 0:
            mid_values[nums[N//2 - 1]] = 0
        for n in nums:
            for k , v in mid_values.items():
                mid_values[k] += abs(k - n)
        ans = min(mid_values.values())
        return ans
