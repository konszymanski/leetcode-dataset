class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        \'\'\'
        Failed Solution #1:
        Find the mean of the array and count differences from that
        Why it Fails - The mean of the array is not always the min distance to travel
        
        Failed Solution #2:
        Nested Loop that generates all of the distances for each value
        Why it Fails - Time Limit Exceeded O(n^n) super crappy time complexity
        
        Successful Solution:
        Sort the array: O(n.logn) for python\'s built-in sort
        Identify the median of the array: O(1)
        Return the sum of the distances to median O(n)
        Solution Time Complexity O(n+n.logn)
        \'\'\'
        nums = sorted(nums)
        if len(nums) % 2 == 0:
            mid_point = int(len(nums)/2)
            median = int(sum(nums[mid_point-1:mid_point+1])/2)
        else:
            median = nums[int(len(nums)/2)]
        return sum([abs(i-median) for i in nums])
