class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        N = len(nums)
        nums.sort()
        median = nums[N // 2]
        ans = 0
        for n in nums:
            ans += abs(n - median)
        return ans
    # Time: O(N * log(N))
    # Space: O(N) for Python\'s sorting algorithm
