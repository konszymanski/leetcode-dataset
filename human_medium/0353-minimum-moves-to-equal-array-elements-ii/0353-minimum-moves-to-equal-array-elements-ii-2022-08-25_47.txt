class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        nums.sort()
        length = len(nums)
        if length % 2 != 0:
            median = nums[length // 2]
        else:
            half = length // 2
            median = (nums[half - 1] + nums[half]) // 2
        ans = 0
        for num in nums:
            ans += abs(num - median)
        return ans
