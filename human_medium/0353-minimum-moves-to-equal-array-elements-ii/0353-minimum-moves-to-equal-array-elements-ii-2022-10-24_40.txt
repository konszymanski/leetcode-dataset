class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        def moves(l):
            cnt=0
            for x in nums:
                cnt+=abs(l-x)
            return cnt
        
        num_mvs=float(inf)
        left=min(nums)
        right=max(nums)
        mid=(left+right)//2
        
        while left<right:
            if moves(mid)<moves(mid+1):
                right=mid
            else:
                left=mid+1     
            mid=(left+right)//2
                
        for i in range(left, right+1):  
            num_mvs=min(num_mvs, moves(i))
            
        return num_mvs
