class Solution(object):
    def minMoves2(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
		# find median value
        nums.sort()
        median = nums[len(nums) // 2]
		# calculate total difference from median to each value
        total = 0
        for elem in nums:
            total += abs(median - elem)
        return total
