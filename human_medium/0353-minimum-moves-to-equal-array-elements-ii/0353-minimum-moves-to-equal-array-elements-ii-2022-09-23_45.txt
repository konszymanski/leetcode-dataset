class Solution:
    def minMoves2(self, nums: List[int]) -> int:
        
        #Time -> O(NlogN) Space -> O(1)
        #sort the array
        nums.sort()
                
        answer = float(inf)
        prefix = 0
        total  = sum(nums)
        
        for i in range(len(nums)):
            
            #getting prefix sum
            prefix  += nums[i] 
            #getting right sum from the current index position which exclude current number
            rightLen = len(nums) - i - 1
            rightSum = (total - prefix)
            #number of changes made to right side 
            #(since its sorted we sub, rightsum - target numbers which are all equal to current number)
            rightN   =  (rightSum - (rightLen * nums[i]) )
            
            #number of changes made to left side (inclusive of current number)
            #(since its sorted we sub, target numbers which are all equal to current number - leftsum)
            leftN    = (((i+1) * nums[i]) - prefix)
            
            answer   = min(answer,leftN + rightN)
        
        return answer
