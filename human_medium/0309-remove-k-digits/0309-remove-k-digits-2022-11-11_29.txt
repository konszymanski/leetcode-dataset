class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        stack = []
        for n in num:
            while stack and k != 0 and int(stack[-1]) > int(n):
                k -= 1
                stack.pop()

            if stack or n != \'0\':
                stack.append(n)

        while stack and k:
            k -= 1
            stack.pop()  

        return \'\'.join(stack) if stack else \'0\'
