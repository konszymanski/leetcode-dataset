def removeKdigits(self, num: str, k: int) -> str:
        l = len(num)
        if(l == k): return \'0\'
        ans = []
        for i in num:
            while(len(ans) and k > 0 and ans[-1] > i):
                ans.pop()
                k -= 1
            ans.append(i)
        if(k > 0): ans = ans[:-k]
        return str(int(\'\'.join(ans)))
