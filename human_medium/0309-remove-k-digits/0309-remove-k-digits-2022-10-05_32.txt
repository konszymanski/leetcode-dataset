class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        s = deque()
        last_checked = 0
        for i,e in enumerate(num):
            while(s and s[-1] > e and k > 0):
                k-=1
                s.pop()
            s.append(e)
        
        while k:
            s.pop()
            k -= 1

        while(s and s[0] == \'0\'):
                s.popleft()
                k-=1

        return "".join(s) if s else \'0\'
