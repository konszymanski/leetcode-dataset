class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        stack = []

        for digit in num:
            while k and stack and int(stack[-1]) > int(digit):
                stack.pop()
                k-=1
            if not stack and digit == "0": continue
            stack.append(digit)
        
        while k and stack:
            stack.pop()
            k-=1

        if not stack: return "0"
        return "".join(stack)
