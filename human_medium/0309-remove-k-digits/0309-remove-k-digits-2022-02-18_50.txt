class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        stack = []
        
        for digit in num:
            while stack and k and digit < stack[-1]:
                stack.pop()
                k -= 1
            
            stack.append(digit)
        
        # trip k remaining elements
        if k > 0:
            stack = stack[:-k]
        
        # trip leading
        begin_idx = 0
        for elem in stack:
            if elem != "0":
                break
            
            begin_idx += 1
        
        return \'\'.join(stack[begin_idx:]) or \'0\' # or \'0\' it is case if in result we have empty seq
