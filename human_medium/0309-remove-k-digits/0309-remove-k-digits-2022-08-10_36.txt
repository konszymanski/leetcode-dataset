class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        i=0
        num=list(num)
        stack=[]
        for i in num:  #iter all values in num
            
            while stack and stack[-1]>i and k>0:
            # comparing whether the last element in stack is > than curr element
            #if yes then we pop the last element and continue till condition not true
            
                stack.pop()
                k-=1 # available removals
            
            if i!=\'0\' or stack: # inserting into the stack 
                stack.append(i) 
            
        # if still available removals remain then we pop the last elements as they are 
        #already in increasing order
        
        if k!=0:
            stack=stack[:-k]
            
        #if stack is empty we return 0
        
        if not stack:
            return \'0\'
        
        return \'\'.join(stack)
