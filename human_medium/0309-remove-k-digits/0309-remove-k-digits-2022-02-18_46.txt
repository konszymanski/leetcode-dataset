class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        # Start from first, check whether current num is less than the previous.
        # - Yes, then remove previous, as it will generate the smaller number.
        # - No, save it somewhere, and check for next num.
        
        # We will use stack to store numbers which we haven\'t deleted.
        stack = []
        
        for ch in num:
            # keep on removing elements until cur is less than the top.
            while stack and k and stack[-1] > ch:
                stack.pop()
                k -= 1
            
            # append cur element in stack
            stack.append(ch)
        
        # remove elements from stack if we can
        while k and stack:
            k -= 1
            stack.pop()
        
        # if stack is empty, add 0
        if not stack:
            stack.append(\'0\')
        
        # stack should have and ans
        ans = \'\'.join(stack)
        
        # remove leading zeros
        return str(int(ans))
