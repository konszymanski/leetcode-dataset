class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        sol = \'\'
		
		# remains[i] == 1 only if the digit on position i will remain in our final number
        remains = [1] * len(num)
        i = 0
		
		# we need to reduce the number of decreasing subarrays formed with the number\'s digits
		
        while i < len(num) - 1:
            if k > 0 and num[i] > num[i + 1]:
                k -= 1
                remains[i] = 0
                j = i - 1
                while j >= 0 and num[j] > num[i + 1] and k > 0:
                    if remains[j] == 1:
                        remains[j] = 0
                        k -= 1
                    j -= 1
            i += 1
            
		#ignore the leading zeroes
        i = 0
        while i < len(num) and (num[i] == \'0\' or remains[i] == 0):
            i += 1
         
		 # if we haven\'t removed k numbers yet, it means we are left with an increasing array, so the largest values will be those right at the end
					 # => we will ignore them too
        j = len(num) - 1
        while j >= 0 and k > 0:
            if remains[j] == 1:
                k -= 1
            j -= 1
            
        for q in range(i, j + 1):
            if remains[q] == 1:
                sol += num[q]
        
        if sol == \'\':
            return \'0\'
        return sol
