class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        stack = []
        for i in range(len(num)):
            # keep removing numbers greater than current number from the stack
            while k and stack and stack[-1] > num[i]:
                stack.pop()
                k -= 1
            stack.append(num[i])
        
        while k:
            stack.pop()
            k -= 1
        
        return "".join(stack).lstrip(\'0\') or \'0\'
