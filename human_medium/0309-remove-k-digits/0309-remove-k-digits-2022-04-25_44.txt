"""
Test Case:
	num = "1111298", k = 3
	num = "1198211" k = 3
"""
class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        
        min_stack = []
        n = len(num)
        for i, val in enumerate(num):
            
            # we only remove k digits
            # so only execute pop from stack until k gets zero
            while min_stack \\
            and k > 0 \\
            and int(min_stack[-1]) > (int(val) if i < n else -float(\'Inf\')):
                min_stack.pop()
                k -= 1
            
            # we keep entering digits here
            # and upto k digits get removed before loop ends
            min_stack.append(val)
            #print(min_stack) # sanity check
        
        # print(k, min_stack) # sanity check
        # if the larger digits are at the end
        # and loop ends before k digits get removed
        # this check ensure all k larger digits are removed
        min_stack = min_stack[:-k] if k else min_stack
        
        return str(int(\'\'.join(min_stack))) if min_stack else \'0\'
