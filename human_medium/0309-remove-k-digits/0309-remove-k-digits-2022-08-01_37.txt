def removeKdigits(self, num: str, k: int) -> str:
        s=[10]
        #since we have an element in s to start pushing onto the stack so k=k+1 because it requires an additional delete from stack
        k=k+1
        for i in range(len(num)):
            
            if k>0:
                
                while len(s) and int(s[-1])>int(num[i]) and k>0:
                    
                    s.pop()
                    
                    
                    k-=1
                #here we append current element after delete all greater than it in less than or equal to k operations of deletion
                s.append(num[i])
            else:
                #we have deleted all elements
                remains=num[i:]
                
                res=\'\'.join(s)+remains
                if len(res) and int(res)==0:
                    return \'0\'
                else:
                    #handling zeros 
                    return res.lstrip(\'0\') if len(res) else \'0\'
                
        # in case we have k>0
        while k and s:
            s.pop()
            k-=1
        res=\'\'.join(s)
        if len(res) and int(res)==0:
            return \'0\'
        
        return res.lstrip(\'0\') if len(res)  else \'0\'
