class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        
        if k>=len(num):
            return "0"
        stack=[]
        
        for val in num:
            while stack and k and int(stack[-1])>int(val):
                stack.pop()
                k-=1
            stack.append(val)
        
        while k>0:
            stack.pop()
            k-=1
        
        return str(int("".join(stack)))
