class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        i=0
        if len(num)<=k:
            num="0"
            return num
        while k > 0:
            if i==len(num)-1 and k>0:            
                return str(int(num[:-k]))
            if num[i]>num[i+1]:
                num=num.replace(num[i],"",1)
                i=0
                k=k-1
            else:
                i=i+1
        return str(int(num))
