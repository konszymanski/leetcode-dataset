class Solution:
    def removeKdigits(self, num: str, k: int) -> str:
        stack = []
        for d in num:
            while k and stack and stack[-1] > d:
                stack.pop()
                k -= 1
                
            stack.append(d)
            
        if k:   
            stack = stack[:-k]
        
        idx = 0
        while idx < len(stack) and stack[idx] == \'0\':
            idx += 1
        
        return "".join(stack[idx:]) if idx < len(stack) else \'0\'
