class Solution(object):
    def removeKdigits(self, num, k):
        n = len(num)
        rem_ele = k
        stack = []
        
        for val in num:
            # pop the element if the previous val is > curr_val
            while rem_ele and stack and stack[-1] > val:
                stack.pop()
                rem_ele -= 1
            stack.append(val)
        
        #join the list element in a string and remove the leading zero
        ans = \'\'.join(stack[0:len(num)-k]).lstrip(\'0\')
        
        #add condition for empty string if it is there then return 0
        if len(ans):
            return ans
        else:
            return \'0\'
