"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""

class Solution:
    def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
        curr = head
        while curr:
            if curr.child:
                self.flatten(curr.child)
                tail = curr.child
                while tail:
                    if tail.child:
                        curr=tail
                        tail = tail.child
                    else:
                        if not tail.next:
                            break
                        tail = tail.next
                break
            curr = curr.next
        if not curr:
            return head

        tail.next=curr.next
        if tail.next:
            tail.next.prev=tail
        curr.next=curr.child
        curr.child = None
        curr.next.prev=curr

        return head
