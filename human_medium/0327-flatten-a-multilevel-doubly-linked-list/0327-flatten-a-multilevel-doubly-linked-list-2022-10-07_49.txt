"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""

"""
Algorithm:
step 1: initialize a stack, stack = [node1]
step 2: build a sentinal node: s
step 3: Iteration
        pop from the queue,
        -> s.next = node, node.prev = s
        -> check whehther it has a child:
            if no, just append the next to stack
            if yes, first append the next, then append the child
step 4: return s.next
"""

class Solution:
    def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
        if not head:
            return None
        stack = [head]
        sentinal = Node(12, None, head, None)
        curr = sentinal
        while stack:
            node = stack.pop()
            curr.next = node
            node.prev = curr
            if node.next:
                stack.append(node.next)
            if node.child:
                stack.append(node.child)
                node.child = None
            curr = node
        sentinal.next.prev = None
        return sentinal.next
