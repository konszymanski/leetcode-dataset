def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
    def helper(head):
        runner = head
        while runner:
            if runner.child:
                # get last node from child list, Recursion
                last = helper(runner.child)

                temp = None
                # to handle edge case where there is no next node, only child nodes
                # ex: [1,null,2,null,3,null]
                if runner.next:
                    temp = runner.next                    

                runner.child.prev = runner
                runner.next = runner.child
                runner.child = None

                last.next = temp
                if temp:
                    temp.prev = last

            # Return the last node.
            if not runner.next:
                # last node
                return runner

            runner = runner.next

    helper(head)

    return head
