class Solution:
    def flatten(self, head: \'Node\') -> \'Node\':
        if not head:
            return head
        stack = []
        temp = head
        while True:
            if not temp:
                break
            if not temp.next:
                if stack:
                    node = stack.pop()
                    temp.next = node
                    if node:
                        node.prev = temp
            if temp.child:
                stack.append(temp.next)
                temp.next = temp.child
                temp.child.prev = temp
                temp.child = None
            temp = temp.next
        return head
