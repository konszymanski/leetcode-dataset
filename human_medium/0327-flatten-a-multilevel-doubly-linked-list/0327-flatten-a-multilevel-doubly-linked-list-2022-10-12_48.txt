"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""
class Solution:
    def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
        if not head:
            return head
        
        cur = head
        temp = head

        while cur:
            print(cur.val)
            if cur.child:
                con= None
                if cur.next:
                    con = cur.next
                cur.child.prev = cur
                cur.next = cur.child
                cur.child = None
                cur = cur.next
                while cur:
                    if cur.next == None:
                        if con:
                            cur.next = con
                            con.prev = cur
                        break
                    print("child", cur.val)
                    cur = cur.next
            cur = temp
            temp = temp.next
            cur = cur.next
        return head
