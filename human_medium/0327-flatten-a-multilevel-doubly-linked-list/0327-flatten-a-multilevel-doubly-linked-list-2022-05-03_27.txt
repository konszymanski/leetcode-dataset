def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
	if not head:
		return head
	def helper(prev, cur):
		if not cur:
			return prev
		prev.next = cur
		prev.child = None
		cur.prev = prev
		last = cur
		curNext = cur.next
		if cur.child:
			last = helper(cur, cur.child)
		if curNext:
			last = helper(last, curNext)
		return last

	dummy = Node()
	helper(dummy, head)
	dummy.next.prev = None
	return dummy.next
