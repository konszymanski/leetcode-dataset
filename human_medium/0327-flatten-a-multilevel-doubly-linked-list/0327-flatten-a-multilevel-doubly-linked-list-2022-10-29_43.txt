"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""

class Solution:
    def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
        def addChild(node):
            if node:
                if node.next:
                    stack.append(node.next)
                if node.child:
                    stack.append(node.child)
        stack=[]
        curr=head
        addChild(curr)
        while stack:
            curr.child=None 
            nxt=stack.pop()
            curr.next=nxt 
            nxt.prev=curr
            curr=curr.next 
            addChild(curr)
        return head
