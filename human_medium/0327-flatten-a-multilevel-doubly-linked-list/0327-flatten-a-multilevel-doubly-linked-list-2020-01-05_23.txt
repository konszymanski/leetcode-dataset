"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""
class Solution:
    def flatten(self, head: \'Node\') -> \'Node\':
        def dfs(head):
            yield head.val
            if head.child is not None:
                yield from dfs(head.child)
            if head.next is not None:
                yield from dfs(head.next)
        
        if head is None:
            return None
        
        vals = list(dfs(head))
        nodes = list(map(lambda x: Node(x, None, None, None), vals))
        for i in range(len(vals) - 1):
            nodes[i].next = nodes[i + 1]
            nodes[i + 1].prev = nodes[i]
        return nodes[0]
