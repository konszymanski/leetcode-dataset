"""
# Definition for a Node.
class Node(object):
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""
class Solution(object):
    def flatten(self, head):
        """
        :type head: Node
        :rtype: Node
        """
        
        if head is None:
            return head
        
        ans = []
        stack = []
        
        stack.append(head)
        
        while stack:
            curr = stack.pop()
            
            while curr:
                ans.append(curr.val)
                
                if curr.child is not None:
                    stack.append(curr.next)
                    curr = curr.child
                else:
                    curr = curr.next
            
        for i in range(len(ans)):
            ans[i] = Node(ans[i])
        
        for i in range(1, len(ans)):
            ans[i-1].next = ans[i]
            ans[i].prev = ans[i-1]
        
        return ans[0]
