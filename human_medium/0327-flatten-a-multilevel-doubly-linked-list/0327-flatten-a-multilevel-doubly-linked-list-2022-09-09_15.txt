class Solution:
    def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
        cur = head
        while cur:
            if cur.child == None:
                cur = cur.next
            else:
                subHead = self.flatten(cur.child)
                nextNode = cur.next
                subHead.prev = cur
                cur.next = subHead
                cur.child = None
                while cur.next:
                    cur = cur.next
                if nextNode:
                    cur.next = nextNode
                    nextNode.prev = cur
        return head
