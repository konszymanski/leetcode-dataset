def flatten(self, head: \'Node\') -> \'Node\':
        if not head:
            return 
        dfs = [] # stack
        val = []
        dummy = Node(None,None,None,None) # dummy head
        ans = dummy        
        dfs.append(head)   
            
        while dfs:
            node = dfs.pop()
            val.append(node.val)
            dummy.next = node 
            if node.next:
                dfs.append(node.next)
            if node.child:
                dfs.append(node.child)
            node.child = None
            node.prev = dummy
            dummy = dummy.next
        print(val)
        ans = ans.next
        ans.prev = None
        return ans
