"""
# Definition for a Node.
class Node:
    def __init__(self, val, prev, next, child):
        self.val = val
        self.prev = prev
        self.next = next
        self.child = child
"""

class Solution:
    def flatten(self, head: \'Optional[Node]\') -> \'Optional[Node]\':
        if head is None:
            return None
        if head.next is None and head.child is None:
            return head
        curr = head
        while curr:
            if curr.child:
                node = curr.child
                while node.next:
                    node = node.next
                if curr.next:
                    node.next = curr.next
                    curr.next.prev = node
                    curr.child.prev = curr
                    curr.next = curr.child
                    curr.child = None
                else:
                    curr.child.prev = curr
                    curr.next = curr.child
                    curr.child = None

            curr = curr.next
        return head
