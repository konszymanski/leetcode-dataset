class Solution:
    def minMutation(self, start: str, end: str, bank: List[str]) -> int:
        count = 0
        visited = set()
        visited.add(start)
        queue = deque()
        queue.append(start)

        while queue:
            for _ in range(len(queue)):
                word = queue.popleft()
                if word == end:
                    return count

                for ch in \'ACGT\':
                    for i in range(len(word)):
                        newWord = list(word)
                        newWord[i] = ch
                        newWord = \'\'.join(newWord)
                        if newWord not in visited and newWord in bank:
                            queue.append(newWord)
                            visited.add(newWord)

            count += 1

        return -1
