class Solution:
    def minMutation(self, start: str, end: str, bank: List[str]) -> int:
        if end not in bank:
            return -1
        patternDict = collections.defaultdict(list)
        for string in bank:
            for i in range(8):
                patternDict[string[:i] + \'*\' + string[i+1:]].append(string)
        seen = set()
        queue = collections.deque([start])
        step = 0
        while queue:
            for _ in range(len(queue)):
                curGene = queue.popleft()
                if curGene == end:
                    return step
                for i in range(8):
                    for nextGene in patternDict[curGene[:i] + \'*\' + curGene[i+1:]]:
                        if nextGene not in seen:
                            seen.add(nextGene)
                            queue.append(nextGene)
            step += 1
        return -1
