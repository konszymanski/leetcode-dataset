infi = sys.maxsize
class Solution:
    # Backtracking solution
	# Replacing bank with \'\' when a word from bank is selected and not checking or computing anything in next recursions
    def minMutation(self, s: str, e: str, bank: List[str]) -> int:
        res = infi
        def diff(a, b):
            cnt = 0
            for c1, c2 in zip(a, b):
                if c1 != c2: cnt += 1
            return cnt
        def helper(a, curr, bank):
            nonlocal e, res
            if a == e:
                res = min(res, curr)
                return
            
            for i, word in enumerate(bank):
                if word == \'\': continue
                d = diff(word, a)
                if d == 1:
                    nb = bank.copy()
                    nb[i] = \'\'
                    helper(word, curr + 1, nb)
                    
        helper(s, 0, bank)
        return res if res != infi else -1
