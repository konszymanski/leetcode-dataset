class Solution:
    def getMin(self,count,previous, target, bank):
        if count>len(bank):
            return 10e9
        if previous==target:
            return 0
        mini = 10e9
        for i in range(len(bank)):
            newStr = bank[i]
            diff = 0
            for i in range(len(previous)):
                if newStr[i]!=previous[i]:
                    diff+=1
            if diff!=1:
                continue
            mini = min(mini, 1+self.getMin(count+1,newStr,target,bank))
        return mini



    def minMutation(self, startGene: str, endGene: str, bank: List[str]) -> int:
        x = self.getMin(0,startGene,endGene,bank)
        if x>=10e9:
            return -1
        return x
