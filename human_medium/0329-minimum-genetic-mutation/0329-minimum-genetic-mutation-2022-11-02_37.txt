class Solution(object):
    def minMutation(self, start, end, bank):
        a=[(start,0)]
        bank=list(set(bank))
        while a:
            b,c=a.pop(0)
            if b==end:
                return c 
            for i in range(len(b)):
                for j in "ACGT":
                    mut=b[0:i]+j+b[i+1:]
                    if mut in bank:
                        bank.remove(mut)
                        a.append((mut,c+1))
        return -1
