from collections import deque

class Solution:
    def minMutation(self, start: str, end: str, bank: List[str]) -> int:
        if len(start) != len(end):
            return -1
    
        if start == end:
            return 0
        
        b = {}
        for i in bank:
            b[i] = True
        
        queue = deque([[start, 0]])
        seen = {}
        
        while queue:
            word, level = queue.popleft()
            
            seen[word] = True
            
            if word == end:
                return level
            
            for i in range(len(word)):
                for j in "AGCT":
                    temp = word[:i] + j + word[i+1: ]
                    if temp in b and temp not in seen:
                        queue.append([temp, level + 1])
                        
        return -1
