class Solution:
    def minMutation(self, start: str, end: str, bank: List[str]) -> int:
        queue = [(start, 0)]
        possible_words = ["A", "C", "G", "T"]
        bank = set(bank)
        while queue:
            current_word, level = queue.pop(0)
            current_word = list(current_word)
            for i in range(len(current_word)):
                replacable_word = current_word[i]
                for possible_word in possible_words:
                    current_word[i] = possible_word
                    temp_word = "".join(current_word)
                    if temp_word in bank:
                        if temp_word == end:
                            return level + 1
                        bank.remove(temp_word)
                        queue.append((temp_word, level+1))
                current_word[i] = replacable_word
        return -1
