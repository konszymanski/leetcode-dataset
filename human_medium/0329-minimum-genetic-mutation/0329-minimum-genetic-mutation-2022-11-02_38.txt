class Solution:
    def minMutation(self, start: str, end: str, bank: List[str]) -> int:
        n = len(start)
        q = deque([(start, 0)])
        visited = set([start])
        while q:
            node, dist = q.popleft()
            if node == end:
                return dist
            
            for i in range(n):
                for char in "ACGT":
                    mutation = node[:i] + char + node[i+1:]
                    if mutation in bank and mutation not in visited:
                        q.append((mutation, dist + 1))
                        visited.add(mutation)
                        
        return -1
