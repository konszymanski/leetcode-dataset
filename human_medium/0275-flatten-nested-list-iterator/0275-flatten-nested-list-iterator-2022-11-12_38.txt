class NestedIterator(object):

    items = []

    def __init__(self, nestedList):
        """
        Initialize your data structure here.
        :type nestedList: List[NestedInteger]
        """
        self.items = []
        self._index = 0
        self._build(nestedList)
    
    def _build(self, nestedList):
        if not nestedList:
            return
        
        i = 0
        first = nestedList[i]
        stack = [first]
        while stack:
            current = stack.pop(-1)
            if current.isInteger():
                self.items.append(current.getInteger())
            else:
                self._recurr(stack, current)
            
            i += 1
            if i < len(nestedList):
                stack.append(nestedList[i])
        
    
    def _recurr(self, stack, current):
        if current.isInteger():
            self.items.append(current.getInteger())
            return
        
        for x in current.getList():
            self._recurr(stack, x)

    def next(self):
        """
        :rtype: int
        """
        item = self.items[self._index]
        self._index += 1
        return item

    def hasNext(self):
        """
        :rtype: bool
        """
        return self._index < len(self.items)
