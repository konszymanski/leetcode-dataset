class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        self.queue = []
        
        for i in nestedList:
            self.__nestedListToQueue(i)

        self.index = 0
        
    
    def next(self) -> int:
        if self.index > len(self.queue):
            return None
        else:
            next = self.queue[self.index]
            self.index += 1
            return next
        
    
    def hasNext(self) -> bool:
        return self.index < len(self.queue)
        
            
    def __nestedListToQueue(self, element):
        if element.isInteger():
            self.queue.append(element.getInteger())
        else:
            nestedList = element.getList()
            for i in nestedList:
                self.__nestedListToQueue(i)
