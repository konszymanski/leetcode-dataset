from itertools import chain
class NestedIterator:
    def __init__(self, nestedList):
        self.iter_nestedList = iter(nestedList)

    def next(self) -> int:
        return self.the_next.getInteger()

    def hasNext(self) -> bool:
        self.the_next = next(self.iter_nestedList, None)
        if self.the_next is None    : return False
        if self.the_next.isInteger(): return True
        self.iter_nestedList = chain(self.the_next.getList(), self.iter_nestedList)
        return self.hasNext() #recursion here
