class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        self.data = [] 
        self.flattern(nestedList)
        self.index = 0
        self.size = len(self.data)
    
    def next(self) -> int:
        self.index += 1
        return self.data[self.index - 1]
        
    
    def hasNext(self) -> bool:
        return self.index < self.size
    
    def flattern(self, nested):
        if type(nested) == list:
            for l in nested:
                self.flattern(l)
            return
        if nested.isInteger():
            self.data.append(nested.getInteger())
            return
        
        for l in nested.getList():
            self.flattern(l)
