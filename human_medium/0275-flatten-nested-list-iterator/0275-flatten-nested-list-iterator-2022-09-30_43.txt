class NestedIterator(object):

    def __init__(self, nestedList):
        """
        Initialize your data structure here.
        :type nestedList: List[NestedInteger]
        """
        self.res = []
        self.idx = -1
        def iterate(nestedList):
            for i in nestedList:
                if i.isInteger():
                    self.res.append(i.getInteger())  
                else:
                    iterate(i.getList())
        iterate(nestedList)

    def next(self):
        """
        :rtype: int
        """
        self.idx += 1
        return self.res[self.idx]

    def hasNext(self):
        """
        :rtype: bool
        """
        return self.idx + 1 < len(self.res)
