class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        def flatten(el):
            if el.isInteger():
                yield el.getInteger()
            else:
                for ent in el.getList():
                    yield from flatten(ent)

        self.lis = [nx for el in nestedList for nx in flatten(el)]
        self.curidx = 0
        
    def next(self) -> int:
        self.curidx += 1
        return self.lis[self.curidx-1]
    
    def hasNext(self) -> bool:
        return True if self.curidx < len(self.lis) else False
