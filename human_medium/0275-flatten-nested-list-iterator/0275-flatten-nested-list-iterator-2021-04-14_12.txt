class NestedIterator:

        def initStack(self,nestedList):

            for ele in (nestedList):
                if ele.isInteger():
                    #print(nestedList[i].getInteger())
                    self.stack.append(ele)
                else:
                    #print(nestedList[i].getList())
                    self.stack.extend(ele.getList()[::-1])

        def __init__(self, nestedList: [NestedInteger]):
            self.currentList = nestedList
            self.stack = []
            self.initStack(self.currentList[::-1])


        def next(self) -> int:
            return self.stack.pop()

        def hasNext(self) -> bool:
            while self.stack:
                if self.stack[-1].isInteger():
                    return True

                self.initStack(self.stack.pop().getList()[::-1])

            return False
