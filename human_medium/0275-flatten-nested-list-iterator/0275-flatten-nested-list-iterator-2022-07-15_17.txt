from collections import deque
class NestedIterator:

    #T=O(n), S=O(n)
	#init a queue and copy the nestesList into it
    def __init__(self, nestedList: [NestedInteger]):
        self.q = deque(nestedList)
    
    #T=O(1), S=O(1)
	#return the first item
    def next(self) -> int:
        #either return  an int or None
        #next() is called after hasNext() execution according to the problem statement
		#make sure to pop the item
        return self.q.popleft().getInteger()
    	
	#if first item is an int, return Truthy
    #if queue is empty, return Falsy
    ##and flatten the nested list until we get the first item to an int
	#WORST case: T=O(n^2), S=O(n)
	#[1,[2,[3,[4]]]] is the worst case input
    #AVERAGE case(ammortized) : T=O(n), S=O(n)
	#[1,[2,3,5],4] is the average case input (two pass)
	#best case: T=O(1), S=O(n)
	#[1,2,3,4] is the best case input
    #we fetch only the head of the queue on average
    def hasNext(self) -> bool:
        #flatten the list if the item is not an int
        #if an int, return True
        #iterate until the current item is flattened
		#we need a while loop and not an if statement
		#without while loop, we would always get Falsy and return an empty final result
        while self.q:
            #first item is an int
            if self.q[0].isInteger():
                return True
            #the item is a nested list
            item = self.q.popleft()
            #extendleft() reverses the order on extension, so do [::-1] to re-reverse it
            self.q.extendleft(item.getList()[::-1])
        #queue is empty
        return False
