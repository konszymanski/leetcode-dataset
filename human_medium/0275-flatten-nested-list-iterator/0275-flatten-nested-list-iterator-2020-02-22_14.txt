from collections import deque

class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        
        def helper( nestedList: [NestedInteger] ):
            
            for element in nestedList:
                
                if element.isInteger():
					# Base case:
					# Get integer and append to deque
					self.sequence.append( element.getInteger() )
					
                else:
					# Inductive step:
					# Keep unpacking to next level
					helper( element.getList() )
                    
        # ---------------------------------------                    
                    
        self.sequence = deque()
        helper( nestedList )
    
    
    
    def next(self) -> int:
        
		# Follow First-in First-out property, pop next element on the Head
        return self.sequence.popleft()
            
    
    def hasNext(self) -> bool:
        
		# Has next element when FIFO sequence is not empty
        return self.sequence
