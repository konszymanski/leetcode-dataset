class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        
        def createGenerator(nl):
            for element in nl:
                if element.isInteger():
                    yield element
                else:
                    yield from createGenerator(element.getList())
                    
        self.gen = createGenerator(nestedList)
    
    def next(self) -> int:
        return self.nxt
        
    def hasNext(self) -> bool:
        self.nxt = next(self.gen, None)
        return self.nxt
