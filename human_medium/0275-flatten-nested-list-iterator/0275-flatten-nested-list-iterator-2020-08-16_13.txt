class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        
        def fn(nums):
            """Return flattened list."""
            ans = []
            for x in nums:
                if x.isInteger(): ans.append(x.getInteger())
                else: ans.extend(fn(x.getList()))
            return ans
        
        self.data = fn(nestedList)
        self.i = 0
    
    def next(self) -> int:
        ans = self.data[self.i]
        self.i += 1
        return ans 
    
    def hasNext(self) -> bool:
         return self.i < len(self.data)
