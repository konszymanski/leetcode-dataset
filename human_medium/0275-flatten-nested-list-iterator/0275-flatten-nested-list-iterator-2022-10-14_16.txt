class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):

        def traverseElement(nested_List):
            _list = []
            for element_list in nested_List:
                if element_list.isInteger() == False:
                    _list.extend(traverseElement(element_list.getList()))
                else:
                    _list.append(element_list.getInteger())

            return _list

        self.flattenList = traverseElement(nestedList)
        self.pointer = -1

    def next(self) -> int:
        self.pointer = self.pointer + 1
        return self.flattenList[self.pointer]
    
    def hasNext(self) -> bool:
         return False if self.pointer + 1 == len(self.flattenList) else True
