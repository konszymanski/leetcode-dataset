class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        def flatten(list_):
			"""
			Flatten a NestedList but as a generator instead of list
			"""
            for i in range(len(list_)):
                if list_[i].isInteger():
                    yield list_[i].getInteger()
                else:
                    for x in flatten(list_[i].getList()):
                        yield x
            return None
        self.nestedList = flatten(nestedList)
        # ^ the above is now a generator.
        # Calling list(flatten(nestedList)) would yield a flattened
        # list as one would expect. Ideally we\'d just peel off of this
        # generator, that would be the solution and that\'s that, but
        # unfortunately we need to handle the hasNext. hasNext can still peel
        # off this generator, but stash away that value for use when 
        # next is called. So rest of the logic is just that.
        self.next_ = None
    
    def next(self) -> int:
        # called only if self.hasNext()
        assert self.next_ is not None        
        next_ = self.next_
        self.next_ = None
        return next_
    
    def hasNext(self) -> bool:
        if self.next_ is not None:
            # because next hasn\'t been called yet since
            # the last time hasNext was called.
            return True
        try:
            self.next_ = next(self.nestedList)
        except StopIteration:
            return False
        else:
            return True
