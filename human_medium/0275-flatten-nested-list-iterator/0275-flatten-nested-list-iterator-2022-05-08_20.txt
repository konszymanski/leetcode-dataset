class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        self.nlist = []
        self.getListItems(nestedList)
        self.pos = 0
         
    def next(self) -> int:
        if self.pos < len(self.nlist):
            val = self.nlist[self.pos]
            self.pos += 1
        return val
    
    def hasNext(self) -> bool:
         return self.pos <= len(self.nlist)-1
        
    def getListItems(self,s):
        if not s: return 
        i = 0 
        while i < len(s):
            val = s[i].getList()
            if val:
                self.getListItems(val)
            else:
                val = s[i].getInteger()
                if val != None: self.nlist += [val]
            i+=1
