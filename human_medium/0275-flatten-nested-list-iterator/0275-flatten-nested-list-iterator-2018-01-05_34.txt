from collections import deque

class NestedIterator(object):

    def __init__(self, nestedList):
        self.que = deque(nestedList)

    def next(self):
        return self.que.popleft()

    def hasNext(self):
        if not self.que:
            return False

        cur = self.que.popleft()
        if cur.isInteger():
            self.que.appendleft(cur)
            return True
        while True:
            if cur.isInteger():
                self.que.appendleft(cur) # we simply push it back to stack if it's integer
                return True
            self.que.extendleft(cur.getList()[::-1])
            if not self.que:
                return False
            cur = self.que.popleft()
