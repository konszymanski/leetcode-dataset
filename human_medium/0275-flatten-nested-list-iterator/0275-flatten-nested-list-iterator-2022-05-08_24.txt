class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        self.l = nestedList[::-1]
            
    def next(self) -> int:
        self.hasNext()
        return self.l.pop().getInteger()      
    
    def hasNext(self) -> bool:
        while self.l and not self.l[-1].isInteger():
            self.l.extend(self.l.pop().getList()[::-1])    
        return bool(self.l)
