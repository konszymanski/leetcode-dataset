from collections import deque

class NestedIterator:
    def __init__(self, nestedList: [NestedInteger]):
        self._generator = self._dfs(nestedList)
        self.number = None
        
    def _dfs(self, nestedList):
        for item in nestedList:
            if item.isInteger():
                yield item.getInteger()
            else:
                yield from self._dfs(item.getList())
    
    def next(self) -> int:
        return self.number
    
    def hasNext(self) -> bool:
        try:
            self.number = next(self._generator)
            return True
        except:
            return False
