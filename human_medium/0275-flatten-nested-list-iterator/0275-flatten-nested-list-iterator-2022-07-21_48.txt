class NestedIterator(object):

    def __init__(self, nestedList):
        """
        Initialize your data structure here.
        :type nestedList: List[NestedInteger]
        """
        ans = []
        def recurse(head):
            is_int = head.getInteger()
            if is_int != None:
                ans.append(is_int)
            else:
                is_list = head.getList()
                for l in is_list:
                    recurse(l)
        for element in nestedList:
            recurse(element)
        self.list = ans
        self.iter = 0

    def next(self):
        """
        :rtype: int
        """
        ans = self.list[self.iter]
        self.iter += 1
        return ans
        

    def hasNext(self):
        """
        :rtype: bool
        """
        if(self.iter>=len(self.list)):
            return False
        return True
