# The Idea: Lets iterate the string from the begining.If current char is less than stack[-1]
# and stack[-1] is not the last appearence in the string than we have to pop it.To know if its last appearence or not 
# I created last_ind

class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        last_ind = {s[i]:i for i in range(len(s))}
        exists = {ss: 0 for ss in s}
        stack = []

        for i in range(0, len(s)):
            while stack and s[i] <= stack[-1] and last_ind[stack[-1]] > i and exists[s[i]] == 0:
                exists[stack[-1]] = 0
                stack.pop()
            if exists[s[i]] == 0:
                stack.append(s[i])
                exists[s[i]] = 1

        return \'\'.join(stack)
