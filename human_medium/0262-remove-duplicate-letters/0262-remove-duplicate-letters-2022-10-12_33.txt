class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        charMaxIndex = {}
        for i in range(len(s)):
            charMaxIndex[s[i]] = i
        
        visited = set()
        stack = []
        
        for i in range(len(s)):
            if s[i] in visited:
                continue
            
            while stack and stack[-1] > s[i] and charMaxIndex[stack[-1]] > i:
                visited.remove(stack[-1])
                stack.pop()
            
            visited.add(s[i])
            stack.append(s[i])
        
        return \'\'.join(stack)
