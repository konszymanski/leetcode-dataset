class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        char_Count, char_Seen, char_Stack = Counter(s), set(), []
        for idx, char in enumerate(s):
            if char not in char_Seen:
                
                while char_Stack and char < char_Stack[-1] and char_Count[char_Stack[-1]] > 0:
                    char_Seen.remove(char_Stack.pop())
                
                char_Stack.append(char)
                char_Seen.add(char)
            
            char_Count[char] -= 1
            
        return \'\'.join(char_Stack)
