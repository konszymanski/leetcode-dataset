class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        r=len(s)-1
        while(r>0):
            if s[r] in s[:r]:
                index=s.find(s[r],0,r)
                if s[index]>=s[index+1]:
                    s=s[:index]+s[index+1:]
                else:
                    s=s[:r]+s[r+1:]
                   
            
            r=r-1
        return s
