class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        stk=[s[0]]
        cnt=Counter(s)
        for ch in s[1:]:
            if ch in stk:
                cnt[ch]-=1
                continue
            while stk and ch<stk[-1] and cnt[stk[-1]]>1:#Use Increasing Monotonic Stack, remove the largest char from the stk only if it will appear in the future and then decrease it\'s count
                cnt[stk[-1]]-=1
                stk.pop()
            stk.append(ch)
        return \'\'.join(stk)
