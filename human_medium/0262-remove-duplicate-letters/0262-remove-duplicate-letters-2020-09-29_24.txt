class Solution:
    def removeDuplicateLetters(self, s: str) -> str:
        freq = [0] * 26
        for i in range(len(s)):
            freq[ord(s[i]) - ord(\'a\')] += 1
        stack = []
        is_present = {}
        for i in range(len(s)):
            if s[i] not in is_present:
			
				# here we are checking if we can displace some characters (in the container) if they are lexicographically greater than the current character
                while(stack and (freq[ord(stack[-1]) - ord(\'a\')] and ord(stack[-1]) > ord(s[i]))):
                    del is_present[stack[-1]]
                    stack.pop()
                stack.append(s[i])
                is_present[s[i]] = True
				
			# we are decreasing the count here because we know that we can\'t include the character at this position again
            freq[ord(s[i]) - ord(\'a\')] -= 1
        return "".join(stack)
