class Solution(object):
    def removeDuplicateLetters(self, s):
        stack=[]
        dictionary={}
        exists=set()
        for i in range(len(s)):
            if(s[i] not in dictionary):
                dictionary[s[i]]=1
            else:
                dictionary[s[i]]+=1
        for i in s:
            dictionary[i]-=1
            if(i not in exists):
                while(stack and ord(stack[-1])>ord(i) and dictionary[stack[-1]]):
                    a=stack.pop()
                    exists.remove(a)
                stack.append(i)
                exists.add(i)
        return "".join(stack)
