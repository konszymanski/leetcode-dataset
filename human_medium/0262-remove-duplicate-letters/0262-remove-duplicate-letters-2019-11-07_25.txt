def removeDuplicateLetters(self, s: str) -> str:
        import collections
        cnt = collections.Counter(s)
        stack = []
        for i,c in enumerate(s):
            cnt[c] -= 1
            if c not in stack:
                while stack and stack[-1] > c and cnt[stack[-1]]:
                    stack.pop()
                stack.append(c)
        return \'\'.join(stack)
