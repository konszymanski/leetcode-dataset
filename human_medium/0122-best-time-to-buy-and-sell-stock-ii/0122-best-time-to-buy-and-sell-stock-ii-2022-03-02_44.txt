class Solution:
    def maxProfit(self, prices):
        profit, currProfit
		buy, sell =, 0, 0
        while sell<len(prices):
            if prices[sell]-prices[buy]>currProfit:
                currProfit = prices[sell]-prices[buy]
                sell += 1
            else:
                profit += currProfit
                currProfit = 0
                buy, sell = sell, sell+1
        profit += currProfit
        return profit
