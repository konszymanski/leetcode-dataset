def productExceptSelf(self, nums: List[int]) -> List[int]:

        N = len(nums)
        out = [1]*N
        # Use the \'out\' array to temporarily store the multiplication of the right elements
        # We don\'t really need the scalar \'right\', but I think it will save us on array dereferences
        right = 1
        for p in reversed(range(N)):
            out[p] = right
            right *= nums[p]

        # The scalar \'left\' will hold the multiplication of the left elements
        left = 1
        for p in range(N):
            out[p] *= left
            left *= nums[p]

        return out
