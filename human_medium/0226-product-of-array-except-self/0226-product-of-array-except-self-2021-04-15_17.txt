class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        # Create an array that will return        
        ret = [1] * len(nums)
        # Store the cumulative product
        cumprod = 1
        
        # Iterate over nums
        for i in range(len(nums)):
            ret[i] = cumprod
            cumprod = nums[i] * cumprod
            
        cumprod = 1
            
        for i in range(len(nums) -1, -1, -1):
            ret[i] *= cumprod
            cumprod = nums[i] * cumprod
            
        return ret
