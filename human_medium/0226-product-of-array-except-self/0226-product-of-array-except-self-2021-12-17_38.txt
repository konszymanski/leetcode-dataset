class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
		# solution array
        ans = [1 for i in nums]
        # keep running product of everything to the left of i
		# do not include first element as there is nothing to its left
        left_so_far = 1
        for i in range(1, len(nums)):
            left_so_far *= nums[i - 1]
            ans[i] *= left_so_far
        # keep running product of everything to the right of i (reverse iteration)
		# do not include last element as there is nothing to its right
        right_so_far = 1
        for i in range(len(nums) - 2, -1, -1):
            right_so_far *= nums[i + 1]
            ans[i] *= right_so_far
        return ans
