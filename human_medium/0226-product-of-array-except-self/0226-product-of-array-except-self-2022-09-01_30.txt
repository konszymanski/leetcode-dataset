class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        # We have [1,2,3,4]
        # we initialize both with 1 and we just mulyiply both to obtain the                 result
        #The prefix product will be  [1,1,2,6]
        #the postfix product will be [24,12,4,1]
        
        
        
        result=[1]*len(nums)
        prefix=1
        for i in range(len(nums)):
            result[i]=prefix
            prefix*=nums[i]
        postfix=1
        for i in range(len(nums)-1,-1,-1):
            result[i]*=postfix
            postfix*=nums[i]
        return result
