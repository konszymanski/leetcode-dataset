class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        n = len(nums)
        result = [1 for i in range(n)]
        
#         adding prefix to output
        pre = 1
        for i in range(n):
            result[i] *= pre
            pre *= nums[i]
        
#         adding postfix to output
        post = 1
        for i in range(n-1,-1,-1):
            result[i] *= post
            post *= nums[i]
        
        return result
