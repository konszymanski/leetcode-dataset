class Solution {
public:
    vector<int> productExceptSelf(vector<int>& nums) {
    	vector<int> res = {1};
    	
    	for (int i = 0; i < nums.size()-1; ++i)
    		res.push_back(res.back()*nums[i]);

    	temp = nums.back();
    	for (int i = len(nums)-2; i >= 0; --i) {
        	res[i] *= temp;
        	temp *= nums[i];
        }

        return res;
    }
};
