def productExceptSelf(self, nums: List[int]) -> List[int]:
        n = len(nums)
        output = [0]*n
        left = 1
        right = 1
		
		#Left array
        for i in range(n):
            output[i] = left
            left *= nums[i]
			
		# Right array
        for i in range(n-1, -1, -1):
            output[i] *= right
            right *= nums[i]
		
		#Used only one output array modified inside the array O(1) space.
        return output
