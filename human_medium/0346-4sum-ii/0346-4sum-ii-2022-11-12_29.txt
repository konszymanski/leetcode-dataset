class Solution:
    def fourSumCount(self, nums1: List[int], nums2: List[int], nums3: List[int], nums4: List[int]) -> int:
        # 1. initialize a dictionary to keep records
        records = defaultdict(int)
        # 2. using 2 for loops to find all possible complements A
        for num1 in nums1:
            for num2 in nums2:
                complementA = num1+num2
                records[complementA] += 1
        # 3. initializing a counter to zero
        counter = 0
        # using another 2 for loops to find the possible complementB
        for num3 in nums3:
            for num4 in nums4:
                complementB = num3 + num4
                # 5. if complement A == -complementB, and value to counter
                counter += records[(-1*complementB)]
        # return counter which represent the number of times complementA + complementB = 0
        return counter
