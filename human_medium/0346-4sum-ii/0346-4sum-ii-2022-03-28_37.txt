from collections import defaultdict
class Solution:
    def fourSumCount(self, nums1: List[int], nums2: List[int], nums3: List[int], nums4: List[int]) -> int:
        d1 = defaultdict(lambda:0)
        d2 = defaultdict(lambda:0)
        cnt=0
        
        for i in nums1:
            for j in nums2:
                    d1[i+j]+=1
                    
        for i in nums3:
            for j in nums4:
                    d2[i+j]+=1

        for key,value in d1.items():
            if -key in d2:
                cnt+=value
                
        return cnt
