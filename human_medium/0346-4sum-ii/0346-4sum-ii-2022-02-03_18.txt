def fourSumCount(self, nums1: List[int], nums2: List[int], nums3: List[int], nums4: List[int]) -> int:
	cnt = 0
	n = len(nums1)
	hash_table = defaultdict(int)
	for n1 in nums1:
		for n2 in nums2:
			hash_table[n1+n2] += 1

	for n3 in nums3:
		for n4 in nums4:
			cnt+=hash_table[-(n3+n4)]

	return cnt
