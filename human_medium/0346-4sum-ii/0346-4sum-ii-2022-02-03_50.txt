class Solution:
    def fourSumCount(self, nums1: List[int], nums2: List[int], nums3: List[int], nums4: List[int]) -> int:
        \'\'\'
        Approach:
        
        Step1: Take any two lists from input count number of values after adding one element from each 
               and store them in dic1
        Step2: Follow the same process for remaining two lists and store them in dic2.
        Step3: If we have -2 in dic1 then we need to search for 2 in dic2. As total sum is \'0\'.
        \'\'\'
        dic1, dic2, n, ans = defaultdict(int), defaultdict(int), len(nums1), 0
        for a, b in product(nums1, nums2):
            dic1[a+b] +=1
        for c, d in product(nums3, nums4):
            dic2[c+d] +=1
        for i in dic1:
            if -i in dic2:
                ans += dic1[i]*dic2[-i]
        return ans
