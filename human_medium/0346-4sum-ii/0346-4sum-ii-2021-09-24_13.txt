class Solution {
public:
    int fourSumCount(vector<int>& nums1, vector<int>& nums2, vector<int>& nums3, vector<int>& nums4) {
        unorderedmap<int,int> ref;
        for (auto a:nums1)
            for(auto b:nums2)
                ++ref[a+b];
        int ans=0;
        for (auto a:nums3)
            for(auto b:nums4)
                    ans+=ref[-a-b];
    return ans;
    }
};
