class Solution:
    def fourSumCount(self, A: List[int], B: List[int], C: List[int], D: List[int]) -> int:
        sum_cd = dict()
        res = 0
        for i in C:
            for j in D:
                    sum_cd[i+j] = sum_cd.get(i+j,0) + 1
                    
        for i in A:
            for j in B:
                    res += sum_cd.get(-i-j,0)
        return res
