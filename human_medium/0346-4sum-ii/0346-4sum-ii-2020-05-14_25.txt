from typing import List


class Solution:
    def fourSumCount(self, A: List[int], B: List[int], C: List[int], D: List[int]) -> int:
        sum_dict = {}
        zero_count = 0
        for a in A:
            for b in B:
                sum_dict[a+b] = sum_dict.get(a+b, 0) + 1
        for c in C:
            for d in D:
                zero_count += sum_dict.get(-c-d, 0)
        return zero_count
