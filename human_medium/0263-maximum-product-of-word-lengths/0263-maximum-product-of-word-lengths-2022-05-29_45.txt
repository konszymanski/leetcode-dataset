class Solution:
    def maxProduct(self, words: List[str]) -> int:
        
        answer = 0 
        
        def encodeWord(word):
            n = 0 
            for i in word:
                index = ord(i) - ord(\'a\')
                n = n | (1<<index)
            
            return n 
        
        numberWords = list(map(lambda x:encodeWord(x), words))
        
        for i in range(len(numberWords)):
            for j in range(i+1, len(numberWords)):
                if numberWords[i] & numberWords[j] == 0:
                    answer = max(answer, len(words[i])*len(words[j]))
        
        return answer
