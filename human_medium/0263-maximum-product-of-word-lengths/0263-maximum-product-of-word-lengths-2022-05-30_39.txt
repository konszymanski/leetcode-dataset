#### Main Loop for getting single word out of words List.
#### Then, inner loop j to compare the characters two words at a time.

class Solution:
    def maxProduct(self, words: List[str]) -> int:
        # print(" len ", len(words))
        all_results = []
        for i in range(len(words)):
            # print(" i ", i)
            for j in range(i+1, len(words)):
                # print(" j ", j)
                word1 = words[i]
                word2 = words[j]
                result = True
                for ch in word1:
                    if ch in word2:
                        result = False                        
                # print(" result ", result, \' word1 \', word1, \' word2 \', word2)
                if result == True:
                    all_results.append(len(word1) * len(word2))
                    
        
        # print(" all_results ", all_results)
        if len(all_results) > 0:
            return max(all_results)
        else:
            return 0
