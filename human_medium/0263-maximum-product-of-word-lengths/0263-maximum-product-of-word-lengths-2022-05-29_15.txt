class Solution:
    def maxProduct(self, words: List[str]) -> int:
        m=0
        for i in range(len(words)-1):
            for j in range(i+1, len(words)):
                s=words[i]+words[j]
                if len(set(s))==(len(set(words[i])) + len(set(words[j]))):
                    p=(len(words[i]))*(len(words[j]))
                    if p>m:
                        print(words[i],words[j])
                        m=p
        return m
