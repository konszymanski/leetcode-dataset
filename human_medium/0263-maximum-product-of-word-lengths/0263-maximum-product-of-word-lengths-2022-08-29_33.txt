class Solution:
    def maxProduct(self, words: List[str]) -> int:
        
        max_product = 0
        arr = []
        for index in range(len(words)):
            num = 0
            for char in words[index]:
                num|= 1<< (ord(char) - ord(\'a\'))
            arr.append(num)
        for index in range(len(arr)):
            for begin in range(index+1, len(arr)):
                if not (arr[index] & arr[begin]):
                    max_product = max(max_product, len(words[index]) * len(words[begin]))
        return max_product
