class Solution:
    def maxProduct(self, words: List[str]) -> int:
        l = len(words)
        bitmasks = [0]*l
        for i in range(l):
            for c in words[i]:
                bitmasks[i] |= 1 << (ord(c) - ord(\'a\'))
        
        max_ = 0
        for i in range(l):
            for j in range(i+1, l):
                if bitmasks[i] & bitmasks[j] == 0:
                    max_= max(max_, len(words[i] * len(words[j])))
        return max_
