class Solution:
    def maxProduct(self, words: List[str]) -> int:
        _max=0
        for i in range(len(words)):
            for j in range(i+1,len(words)):
                for k in words[i]:
                    if k in words[j]:break
                else:
                    _max=max(_max,len(words[i])*len(words[j]))
        return _max
