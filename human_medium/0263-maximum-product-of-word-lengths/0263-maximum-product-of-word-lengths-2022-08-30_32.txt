class Solution:
    def maxProduct(self, words: List[str]) -> int:
        
        dic = {}
        
        def hasNoCommon(word1, word2):
            for char in word1:
                if char in word2:
                    return False
            return True
        
        for index in range(len(words)):
            dic[index] = set(words[index])
        ans = 0
        for index1 in range(len(words)):
            for index2 in range(index1+1, len(words)):
                if hasNoCommon(dic[index1], dic[index2]):
                    ans = max(ans, len(words[index1])*len(words[index2]))
        return ans
