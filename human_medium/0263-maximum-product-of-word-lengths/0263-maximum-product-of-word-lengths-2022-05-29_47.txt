class Solution:
    def maxProduct(self, words: List[str]) -> int:
        ans = 0
        words.sort(key = lambda x:-len(x))
        d = {i:set(c) for i,c in enumerate(words)}
        
        for i in range(len(words)-1):
            for j in range(i+1,len(words)):
                if len(d[i] & d[j]) == 0:
                    ans = max(ans,len(words[j])*len(words[i]))
        
        return ans
