class Solution(object):
    def maxProduct(self, words):
        count=0
        for i in range(len(words)):
            for j in range(i+1,len(words)):
                if len(words[i])*len(words[j])<=count:continue   
                if self.check_common(words[i],words[j]):
                    count=len(words[i])*len(words[j])    
        return count
    
    def check_common(self, s1, s2):
        for i in range(len(s1)):
            if s1[i] in s2: return False
        return True
