class Solution:
    def maxProduct(self, words: List[str]) -> int:
        
        char_set = []
        
        for  word in words:
            char_set.append(set(word))
            
        max_product =0 
        
        print(char_set)
            
        for i in range(len(char_set)):
            
            for j in range(i+1,len(char_set)):
                if len(char_set[i].intersection(char_set[j])) ==0 :
                    max_product = max(max_product, len(words[i])*len(words[j]))
                        
        if max_product ==0 :
            return 0 
        return max_product
