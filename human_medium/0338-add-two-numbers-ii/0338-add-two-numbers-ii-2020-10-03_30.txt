stack1 = []
stack2 = []
while l1:
		stack1.append(l1.val)
		l1 = l1.next
while l2:
		stack2.append(l2.val)
		l2 = l2.next
result = ListNode(0)
summ = 0
while stack1 or stack2 or summ:
	if stack1:
		summ += stack1.pop()
	if stack2:
		summ += stack2.pop()
	result.val = summ%10
	head = ListNode(0)
	head.next = result
	result = head
	summ //= 10
return result  if result.val else result.next
