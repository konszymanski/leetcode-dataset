# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def reverse(self, head):
        prev,cur = None,head
        while cur:
            tp = cur.next
            cur.next = prev
            prev = cur
            cur = tp
        return prev

    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        rl1 = self.reverse(l1)
        rl2 = self.reverse(l2)
        res = ListNode(0)
        cur = res
        c1,c2 = rl1,rl2
        car = 0
        while c1 or c2 or car:
            v1 = c1.val if c1 else 0
            v2 = c2.val if c2 else 0
            sm = v1+v2+car
            car = sm//10
            cur.next = ListNode(sm%10)
            cur = cur.next
            if c1: c1 = c1.next
            if c2: c2 = c2.next
        return self.reverse(res.next)
