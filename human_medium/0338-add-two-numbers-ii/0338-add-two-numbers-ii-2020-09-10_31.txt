# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        s1 = []
        s2 = []
        # Traverse through lls and add their nodes to the respective stacks
        while l1 or l2:
            if l1:
                s1.append(l1.val)
                l1 = l1.next
            if l2:
                s2.append(l2.val)
                l2 = l2.next
        carry = 0
        added_numbers_ll = None
        while s1 or s2:
            new_val = carry
            # case specific addition
            if s1 and s2:
                new_val += s1.pop() + s2.pop()
            elif s1:
                new_val += s1.pop()
            elif s2:
                new_val += s2.pop()
            # a digit in a specific position will be from 0-9, so mod 10
            # set the next of the new node to the head of the current linked list
            new_ref = ListNode(new_val % 10, added_numbers_ll)
            # determine if there is a value to carry over
            carry = new_val // 10
            # advance the head of the linked list to recently created node
            added_numbers_ll = new_ref
        # most significant digit + other most significant digit(or carry) led to another carry over
        if carry > 0:
            ref = ListNode(carry, added_numbers_ll)
            added_numbers_ll = ref
        return added_numbers_ll
