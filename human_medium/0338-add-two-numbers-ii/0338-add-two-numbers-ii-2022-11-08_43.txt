# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        lst = ""
        lst1 = ""
        while l1 or l2:
            if l1 :
                lst+=str(l1.val)
                l1 = l1.next
            if l2:
                lst1+=str(l2.val)
                l2 = l2.next
     
        res = int(lst)+int(lst1)
        dummy = None
        temp = dummy
        for i in str(res)[::-1]:
            print(i)
            if dummy is not None:
                curr = ListNode(i,dummy)
                dummy = curr
            else:
                dummy = ListNode(i,None)
           
        return dummy
