# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        def reverseList(head):
            if not head or not head.next:
                return head
            
            prev = None
            curr = head
            while curr:
                nxt = curr.next
                curr.next = prev
                prev = curr
                curr = nxt
            return prev
        
        l1 = reverseList(l1)
        l2 = reverseList(l2)
        
        carry = 0
        ans = l3 = ListNode(carry)
        
        while l1 or l2:
            currSum = 0
            if l1:
                currSum += l1.val
                l1 = l1.next
            if l2:
                currSum += l2.val
                l2 = l2.next
            # nodesum value is current sum + carry from prev
            nodeSum = (currSum + carry)
            # add new node with nodesum 1s place value
            l3.next = ListNode(nodeSum % 10)
            l3 = l3.next
            # update carry
            carry = nodeSum // 10
        
        if carry != 0:
            l3.next = ListNode(carry)
            l3 = l3.next
        
        return reverseList(ans.next)
