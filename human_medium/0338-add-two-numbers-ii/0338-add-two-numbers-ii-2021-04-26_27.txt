class Solution:
    def list2int(self, lst):
        num = ""
        while lst:
            num += str(lst.val)
            lst = lst.next
        return int(num)
		
    def int2list(self, num):
        lst = list(map(int, str(num)))
        begin = ListNode(0)
        end = begin
        for i in lst:
            end.next = ListNode(i)
            end = end.next
        return begin.next
		
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        num1 = self.list2int(l1)
        num2 = self.list2int(l2)
        ans = num1 + num2
        return self.int2list(ans)
