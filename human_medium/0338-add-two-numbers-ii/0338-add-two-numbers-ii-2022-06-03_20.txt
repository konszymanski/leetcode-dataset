class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        def toNumber(head):
            num = 0
            while head:
                num = 10 * num + head.val
                head = head.next
            
            return num
        
        def toLinkedList(num):
            head = None
            while num:
                num, d = divmod(num, 10)
                head = ListNode(d, head)
                
            return head if head else ListNode(0)
        
        return toLinkedList(toNumber(l1) + toNumber(l2))
