class Solution:
    def addTwoNumbers(self, l1: ListNode, l2: ListNode) -> ListNode:
        n1,n2 = 0,0
        while l1:
            n1 = n1*10 + l1.val
            l1 = l1.next
        while l2:
            n2 = n2*10 + l2.val
            l2 = l2.next
        ans = str(n1+n2)
        head = dummy = ListNode(0)
        n = len(ans)
        i = 0
        while i < n:
            dummy.next = ListNode(int(ans[i]))
            dummy = dummy.next
            i += 1
        return head.next
