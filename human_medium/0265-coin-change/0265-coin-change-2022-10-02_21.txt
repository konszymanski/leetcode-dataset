class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        n, to_collect = 0, 1 << amount
        while to_collect & 1 == 0:
            collected = to_collect
            for coin in coins:
                collected |= (to_collect >> coin)
            if collected == to_collect:
                return -1
            n += 1
            to_collect ^= collected
        return n
