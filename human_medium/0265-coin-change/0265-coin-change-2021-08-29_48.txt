class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        memo = dict()
        ways = self.dfs(amount, coins, memo)
        if ways == float(\'inf\'):
            return -1
        return ways


    def dfs(self, amount, coins, memo):
        if amount == 0:
            return 0
        
        if amount in memo:
            return memo[amount]

        mini = float(\'inf\')
        for c in coins:
            new_amount = amount - c
            if new_amount >= 0:
                mini = min(mini, self.dfs(new_amount, coins, memo)+1)
        
        memo[amount] = mini

        return mini
