def fun(cin,amt,memo):
    if(amt in memo):
        return memo[amt]
    mn=inf
    if(amt==0):
        return 0
    if(amt<0):
        return inf
    for x in cin:
        if(x>amt):
            continue
        a=fun(cin,amt-x,memo)
        mn=min(mn,a)
    memo[amt]=mn+1
    return mn+1
        
class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:
        coins.sort(reverse=True)
        ans=fun(coins,amount,{})
        if(ans==inf):
            return -1
        return ans
