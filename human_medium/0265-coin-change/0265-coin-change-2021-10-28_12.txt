class Solution:  # 556ms
    def coinChange(self, coins: List[int], amount: int) -> int:
        step, seen = 0, [amount]
        checked = set()
        while seen:
            cur = []
            for node in seen:
                if node == 0:
                    return step
                for coin in coins:
                    rem = node - coin
                    if rem >= 0 and rem not in checked:
                        cur.append(rem)
                        checked.add(rem)
            step, seen = step + 1, cur
        return -1
