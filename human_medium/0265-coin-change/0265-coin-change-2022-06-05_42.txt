class Solution:
    def coinChange(self, coins: List[int], amount: int) -> int:

        n, coins = len(coins), sorted(coins, reverse=True)

        @lru_cache(None)
        def dp(remaining_amount: int) -> int:
            if remaining_amount == 0: return 0

            best = float(\'inf\')
            for coin in coins:
                if coin > remaining_amount: continue
                best = min(best, 1 + dp(remaining_amount - coin))

            return best

        ans = dp(amount)
        return -1 if ans == float(\'inf\') else ans
