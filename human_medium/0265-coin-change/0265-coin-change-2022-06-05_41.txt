class Solution:
    def coinChange(self, coins: List[int], amt: int) -> int:
        def rec(amt):
            if amt < 0:
                return inf
            if amt == 0:
                return 0
            ans = min(rec(amt-i) for i in coins)+1
            return ans
			
        x = rec(amt) 
        return x if x != inf else -1
