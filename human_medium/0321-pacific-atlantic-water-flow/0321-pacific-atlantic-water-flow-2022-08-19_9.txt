class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        ROWS, COLS = len(heights), len(heights[0])
        pacificSet, atlanticSet = set(), set()
        
        def dfs(row, col, visited, prevHeight):
            if (row not in range(ROWS) or
                col not in range(COLS) or
                (row, col) in visited or
                heights[row][col] < prevHeight
               ):
                return
            
            visited.add((row, col))
            dfs(row + 1, col, visited, heights[row][col])
            dfs(row - 1, col, visited, heights[row][col])
            dfs(row, col + 1, visited, heights[row][col])
            dfs(row, col - 1, visited, heights[row][col])        
        
        for row in range(ROWS):
            dfs(row, 0, pacificSet, heights[row][0])
            dfs(row, COLS - 1, atlanticSet, heights[row][COLS - 1])
            
        for col in range(COLS):
            dfs(0, col, pacificSet, heights[0][col])
            dfs(ROWS - 1, col, atlanticSet, heights[ROWS - 1][col])
        
        return pacificSet & atlanticSet
    # Time: O(m * n) where m and n are the dimensions of the grid
    # Space: O(m * n)
