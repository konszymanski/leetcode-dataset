def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
	numRows = len(heights)
	numCols = len(heights[0])

	def dfs(heights: List[List[int]], ocean: List[List[bool]], row: int, col: int, prevVal: int):
		if row < 0 or col < 0 or row >= numRows or col >= numCols or ocean[row][col] or heights[row][col] < prevVal:
			return

		ocean[row][col] = True
		dfs(heights, ocean, row - 1, col, heights[row][col])
		dfs(heights, ocean, row + 1, col, heights[row][col])
		dfs(heights, ocean, row, col - 1, heights[row][col])
		dfs(heights, ocean, row, col + 1, heights[row][col])

	pacific = [[False for col in range(numCols)] for row in range(numRows)]
	atlantic = [[False for col in range(numCols)] for row in range(numRows)]
	for row in range(numRows):
		dfs(heights, pacific, row, 0, heights[row][0])
		dfs(heights, atlantic, row, numCols - 1, heights[row][numCols - 1])

	for col in range(numCols):
		dfs(heights, pacific, 0, col, heights[0][col])
		dfs(heights, atlantic, numRows - 1, col, heights[numRows - 1][col])

	canFlowToBoth = []
	for i in range(len(heights)):
		for j in range(len(heights[0])):
			if pacific[i][j] and atlantic[i][j]:
				canFlowToBoth.append([i,j])

	return canFlowToBoth
