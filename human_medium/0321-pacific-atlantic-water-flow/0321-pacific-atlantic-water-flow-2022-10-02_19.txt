class OceanType:
    def __init__(self):
        self.pacific = \'Pacific Ocean\'
        self.atlantic = \'Atlantic Ocean\'

class Solution:
    def pacificAtlantic(self, heights: List[List[int]]) -> List[List[int]]:
        if not heights: return 0

        rows, cols = len(heights), len(heights[0])
        po_visited, ao_visited = set(), set()
        ocean_directions = [[1,0], [-1,0], [0,1], [0,-1]]
        ocean_type = OceanType()

        def handleOceanVisited(row, col, ocean):
            if ocean == ocean_type.pacific:
                po_visited.add((row,col))
            else:
                ao_visited.add((row,col))
        
        def getOceanVisitedSet(ocean):
            return po_visited if ocean == ocean_type.pacific else ao_visited

        def bfs(row, col, ocean):
            queue = collections.deque()
            queue.append((row, col))
            handleOceanVisited(row, col, ocean)

            while queue:
                row, col = queue.popleft()
                for dr, dc in ocean_directions:
                    r, c = row + dr, col + dc
                    if(r in range(rows) and 
                    c in range(cols) and 
                    heights[r][c] >= heights[row][col] and
                    (r, c) not in getOceanVisitedSet(ocean)
                    ):
                        queue.append((r, c))
                        handleOceanVisited(r, c, ocean)

        for row in range(rows):
            for col in range(cols):
                # pacific ocean neighbours
                if row == 0 or col == 0:
                    bfs(row, col, ocean_type.pacific)
                # atlantic ocean neighbours
                if row == rows - 1 or col == cols - 1:
                    bfs(row, col, ocean_type.atlantic)

        
        return list(po_visited.intersection(ao_visited))
