class Solution:
    def frequencySort(self, s: str) -> str:
        hashmap = collections.Counter(s)
        keys = list(hashmap.keys())
        length = len(keys)
        values = list(hashmap.values())
        res = \'\'
        i = 0
        while i < length:
            maxy = max(values)
            idx = values.index(maxy)
            res += keys[idx] * values[idx]
            i += 1
            values.pop(idx)
            keys.pop(idx)
        return res
