class Solution:
    def frequencySort(self, nums: str) -> str:
        hashmap={}
        res=\'\'
        for has in nums:
            if has in hashmap:
                hashmap[has]+=1
            else:
                hashmap[has]=1
        heap=[]
        for key in hashmap:
            heapq.heappush(heap,(-hashmap[key],key))
        while heap:
            count,letter=heapq.heappop(heap)
            res+=letter*-count
        return res
