class Solution:
    def frequencySort(self, s: str) -> str:
        res = ""
        dic = {}
        for i in s:
            if i in dic:
                dic[i] +=1
            else:
                dic[i]=1
        sort_dic = sorted(dic.items(),key = lambda x:x[1],reverse=True)
        print(sort_dic)
        for i in range(len(sort_dic)):
            res+=sort_dic[i][0]*sort_dic[i][1]
        return res
