class Solution(object):
    def frequencySort(self, s):
        """
        :type s: str
        :rtype: str
        """
        
        # create a dictionary to store the occurrence of each characters
        hash_dict = {}

        for i in s:
            if i not in hash_dict:
                hash_dict[i] = 0
            hash_dict[i] = hash_dict[i] + 1
            
        # create a function to return the count of a character
        def get_count(character):
            return hash_dict[character]
        
        # intially sort the string alphabetically because some of character can have same number of count
        # take example = "loveleetcode"
        s=\'\'.join(sorted(s))
        
        # call sorted function and pass the function "get_count" 
        res = \'\'.join(sorted(s, key=get_count, reverse=True))       # time complexity: O(nlogn)
        
        # below code will give "Time Limit Exceeded" because of "count" function ---- O(n^2)
		# Hence we are not using this approach
        # res = \'\'.join(sorted(s, key=lambda x: s.count(x), reverse=True))
        
        return res
