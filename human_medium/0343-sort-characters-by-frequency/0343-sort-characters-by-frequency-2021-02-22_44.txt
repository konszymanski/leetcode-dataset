class Solution:
    def frequencySort(self, s: str) -> str:
        dct = {}
        output = \'\'
		# the number of each character in the string
        for i in set(s):
            dct[i] = s.count(i)
		# sort dict by value and reverse it
        for key, value in sorted(dct.items(), key=lambda x: x[1], reverse=True):
			# build new string based on decreasing order based on the frequency of characters
            output += key*value
        return output
