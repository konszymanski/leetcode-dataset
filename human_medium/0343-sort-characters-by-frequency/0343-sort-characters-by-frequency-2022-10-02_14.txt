class Solution:
    def frequencySort(self, s: str) -> str:
        dic = {}
        for ch in s:
            if ch not in dic:
                dic[ch]=0
            dic[ch]+=1
        freq = {}
        for key,val in dic.items():
            if val not in freq:
                freq[val] = []
            freq[val].append(key)
        res = ""
        i=len(s)
        while i>=0:
            if i in freq:
                for val in freq[i]:
                    res += i*val
            i-=1
        return res
