class Solution:
    def frequencySort(self, s: str) -> str:
        # initialize a dictionary and maintain a count of each character
        count_dict = {}
        for i in s:
            if i not in count_dict:
                count_dict[i] = 0
            count_dict[i] += 1
        # convert the dictionary to a list of tuples and sort it on the value of second element and reversing it
        count_dict = sorted(count_dict.items(),
                            key=lambda x: x[1], reverse=True)
        final_string = ""
        for char, count in count_dict:
            final_string += char*count
        return final_string
