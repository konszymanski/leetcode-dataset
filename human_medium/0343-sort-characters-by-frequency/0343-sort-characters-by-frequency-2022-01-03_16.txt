import operator
class Solution:
    def frequencySort(self, s: str) -> str:
        arr_dict={}
        string=""
        for i in s:
            if i not in arr_dict:
                arr_dict[i]=1
            else:
                arr_dict[i]+=1
        arr_dict=dict(sorted(arr_dict.items(),key=operator.itemgetter(1),reverse=True))
        # print(arr_dict)
        for key,values in arr_dict.items():
            string+=key*values
        return string
