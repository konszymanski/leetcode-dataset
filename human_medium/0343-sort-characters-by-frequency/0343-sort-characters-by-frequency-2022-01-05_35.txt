import heapq
class Solution:
    def frequencySort(self, s: str) -> str:
        # 0. base case, if the length of s is less than 3, return the string
        if len(s) < 3:
            return s
        
        # 1. set up a hashmap and add each letter and their count from s into it
        hashmap = {}        
        for letter in s:
            hashmap[letter] = hashmap.get(letter, 0) + 1
        
        # 2. set up a heap array, heapify it, and heappush each -count and letter into it. do -count so it\'s easier to access the most common letters when needed
        heap = []
        heapq.heapify(heap)
        for key, value in hashmap.items():
            heapq.heappush(heap, [-value, key])
        
        # 3. set up the answer array, and as long as there is a heap, heappop values, and append each letter -count times. count is - already so it will even out
        answer = []
        while heap:
            count, letter = heapq.heappop(heap)
            answer.append(letter * -count)
        
        # 4. join the values in the answer array using an empty string and return it
        return "".join(answer)
