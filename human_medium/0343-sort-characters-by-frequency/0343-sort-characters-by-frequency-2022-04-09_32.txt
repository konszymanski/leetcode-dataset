class Solution:
    def frequencySort(self, s: str) -> str:
        fq = {}
        for i in s:
            fq[i] = 1 + fq.get(i, 0)
        tp = [(k, v) for k, v in fq.items()]
        c = list(sorted(tp, key=lambda item: item[1], reverse=True))
        res = \'\'
        for j, f in c:
            res = res + (j * f) 
        return res
