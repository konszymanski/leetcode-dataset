import heapq
class Solution:
    def frequencySort(self, s: str) -> str:
        words,heap,ans = {},[],\'\'
        for char in s:
            if char not in words:
                words[char] = 0
            words[char] += 1
        for key in words:
            heap.append((words[key]*-1, key))
        heapq.heapify(heap)
        while heap:
            count,char = heapq.heappop(heap)
            while count < 0:
                ans += char
                count += 1
        return ans
