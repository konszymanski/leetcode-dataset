class Solution:
    def frequencySort(self, s: str) -> str:
        cnt = {}
        for i, char in enumerate(s):
            cnt[char] = 1+cnt.get(char,0)
        cnt_sorted = {k: v for k, v in sorted(cnt.items(), key=lambda item: -item[1])}
        res = \'\'
        for k, v in cnt_sorted.items():
            res = res + k*v
        return res
