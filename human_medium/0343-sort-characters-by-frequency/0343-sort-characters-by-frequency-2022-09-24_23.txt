class Solution:
    def frequencySort(self, s: str) -> str:
        #use counter
        x = Counter(s)
        #empty list
        _l=[]
        #loop thru keys and vals in Counter dict
        for k,v in x.items():
            #add the key multiplied by value
            _l.append(k*v)
        #sort the list by length of item
        _l.sort(key=len)
        #reverse it
        _l = _l[::-1]
        #join it back to a single string
        return "".join(_l)
