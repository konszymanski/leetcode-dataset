def btrack(inp,out,ans):
    if not inp:
        ans.append(out)
        return
    btrack(inp[1:],out+[inp[0]],ans)
    if inp[0] not in out:
        btrack(inp[1:],out,ans)


class Solution:
    def subsetsWithDup(self, nums: List[int]) -> List[List[int]]:
        ans=[]
        btrack(nums,[],ans)
        return ans
