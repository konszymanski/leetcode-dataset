def maximalSquare(self, matrix: List[List[str]]) -> int:
        queue = deque()
        res = 0
        for i in range(len(matrix)):
            for j in range(len(matrix[i])):
                if matrix[i][j] == "1":
                    matrix[i][j] = 1
                    queue.append((i, j))
                    res = 1
                else:
                    matrix[i][j] = 0
        
        subgrid_n = 2
        while queue:
            length = len(queue)
            for _ in range(len(queue)):
                i, j = queue.popleft()
                
                rows = matrix[i:i+subgrid_n]
                subgrid = [row[j:j+subgrid_n] for row in rows]
                bottom_row = rows[-1][j:j+subgrid_n]
                right_col = [row[-1] for row in subgrid]
                
                # sum up bottom and right
                bot_sum = sum(bottom_row) == subgrid_n
                right_sum = sum(right_col) == subgrid_n
                
                if bot_sum and right_sum:
                    res = subgrid_n * subgrid_n
                    queue.append((i, j))
            subgrid_n += 1
        return res
