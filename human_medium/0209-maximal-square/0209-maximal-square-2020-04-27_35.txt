class Solution:
    def maximalSquare(self, matrix: List[List[str]]) -> int:
		# take care of simple inputs
        if not matrix:
            return 0
        rows = len(matrix)
        if not rows:
            return 0
        columns = len(matrix[0])
        if not columns:
            return 0
        if rows == 1:
            return 1 if "1" in matrix[0] else 0
        elif columns == 1:
            return 1 if ["1"] in matrix else 0

        # initialize with values from first row
        row_values = [int(x) for x in matrix[0]]
		# keep track of the maximum square edge length
        max_edge = 1 if 1 in row_values else 0
		
        for row in matrix[1:]:
            # updates values for first column
			# top_left keeps track of the value in position (row-1, column-1)
            top_left = row_values[0]
            row_values[0] = int(row[0])
            if not max_edge and row_values[0]:
                max_edge = 1
            # loop over column values for given row
            for column in range(1, columns):
                if row[column] == "0":
                    current_value = 0
                else:
                    current_value = min(top_left, row_values[column - 1], row_values[column]) + 1
				# check if we see a longer square edge
                if current_value > max_edge:
                    max_edge = current_value
				# update row values
                top_left = row_values[column]
                row_values[column] = current_value

		# return maximum edge length squared
        return max_edge ** 2
