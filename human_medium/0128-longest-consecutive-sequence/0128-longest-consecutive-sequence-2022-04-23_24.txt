# Solution to start with 
# Time Complexity - O(NlogN) - Because sorting has been done
# Space Complexity - O(1) or O(N) =>
# O(1) because only one value is stored for both res and count.
# O(N) can be possible too if interviewer does not allow inplace sorting and we are forced to use extra space
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        res, count = 0,1
        nums.sort()
        for i in range(len(nums)):
            if i+1<=len(nums)-1 and nums[i]+1 == nums[i+1] : count+=1
            elif i+1<=len(nums)-1 and nums[i] == nums[i+1] : continue
            else:
                res = max(res,count)
                count = 1
        return res
    
# Actual O(N) Solution required by the question
# Time Complexity - O(N)
# Space Complexity - O(N)
class Solution:
    def longestConsecutive(self, nums: List[int]) -> int:
        arr, res = set(nums), 0
        for i in arr:
            if (i-1) not in arr:
                count = 0
                while (i+count) in arr: count+=1
                res = max(res,count)
        return res
