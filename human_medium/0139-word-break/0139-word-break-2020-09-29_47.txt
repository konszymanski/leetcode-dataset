class TrieNode:
    def __init__(self):
        self.isWord = False
        self.children = defaultdict(TrieNode)
        
class Trie:
    def __init__(self):
        self.root = TrieNode()
        
    @lru_cache(None)
    def search(self, s):
        node = self.root
            
        for i, c in enumerate(s):
            node = node.children.get(c)

            if not node:
                return False

            if node.isWord:
                if self.search(s[i + 1:]):
                    return True
        
        return node.isWord
    
    def add(self, word):
        node = self.root
        
        for c in word:
            node = node.children[c]
        
        node.isWord = True

class Solution:
    def wordBreak(self, s: str, wordDict: List[str]) -> bool:
        if not s or not wordDict:
            return False
        
        trie = Trie()
        
        for word in wordDict:
            trie.add(word)
        
        return trie.search(s)
