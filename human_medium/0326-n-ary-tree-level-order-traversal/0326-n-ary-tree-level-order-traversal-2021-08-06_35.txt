class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        if not root:
            return []
        arr = [[root, 0]]
        d = defaultdict(list)
        while len(arr) > 0:
            node, level = arr.pop(0)
            if node:
                d[level].append(node.val)
                for child in node.children:
                    arr.append([child, level+1])
        return d.values()
