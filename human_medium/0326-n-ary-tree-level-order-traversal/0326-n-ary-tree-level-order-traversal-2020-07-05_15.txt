class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        if not root: return []
        queue = [root]
        result = []
        while queue:
            curLevelNodes = []
            nextLevelNodes = []
            while queue:
                node = queue.pop(0)   
                curLevelNodes.append(node.val)
                nextLevelNodes += node.children
            queue = nextLevelNodes
            result.append(curLevelNodes)
        return result
