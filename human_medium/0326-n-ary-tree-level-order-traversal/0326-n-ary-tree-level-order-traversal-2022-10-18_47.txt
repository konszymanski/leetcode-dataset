class Solution(object):
    def levelOrder(self, root):
        if not root:
            return []
        output=[]
        queue=[root]
        while queue:
            b=[]
            for i in range(len(queue)):
                a=queue.pop(0)
                b.append(a.val)
                for j in a.children:
                    queue.append(j)
            output.append(b)
        return output
