class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        groups = []
        queue = deque()
        if root is not None:
            queue.append((root,0))
            while len(queue)>0:
                current,dist = queue.popleft()
                if len(groups)-1>=dist:
                    groups[dist].append(current.val)
                else:
                    groups.append([current.val])
               
                for i in current.children:
                    queue.append((i,dist+1))
        return groups
