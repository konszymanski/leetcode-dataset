class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        hash_val=defaultdict(list)
        ans=[]
        def dfs(root,lvl):
            if root is None:
                return
            hash_val[lvl].append(root.val)
            for i in root.children:
                dfs(i,lvl+1)
                
        dfs(root,1)
        for jk in hash_val.keys():
            ans.append(hash_val[jk])
            
        return ans
