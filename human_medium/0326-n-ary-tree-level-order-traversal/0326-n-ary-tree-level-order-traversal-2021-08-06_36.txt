def levelOrder(self, root):
    # Initalize queue and result:
    q = collections.deque([root] if root else [])
    result = []

    # BFS:
    while q:
        # Set-Up before processing current level:
        level_size = len(q)
        result.append([])

        # Process current level:
        for _ in range(level_size):
            node = q.popleft()
            result[-1].append(node.val)
            q.extend(child for child in node.children)

    return result
