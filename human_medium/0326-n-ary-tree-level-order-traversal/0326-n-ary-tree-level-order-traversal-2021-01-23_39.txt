class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        if not root:
            return []

        res, deque = [], collections.deque()
        deque.append((root))
        while deque:
            ans, size = [], len(deque)
            for i in range(size):
                tmp = deque.popleft()
                ans.append(tmp.val)
                for node in tmp.children:
                    deque.append(node)
            res.append(ans)
        
        return res
