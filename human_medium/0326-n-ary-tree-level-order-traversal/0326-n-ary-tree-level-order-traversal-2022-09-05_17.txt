def levelOrder(self, root: \'Node\') -> List[List[int]]:
        
        # Base case if root does not exist
        if not root:
            return None
        
        # declare a list,queue and dictionary
        ans = list()
        queue = deque()
        store = defaultdict(list)
        
        # Append root with level in the queue
        queue.append((root,0))
        
        # run the loop while queue is not None
        while queue:
            
            # Pop the first element in the queue 
            curr, level = queue.popleft()
            
            # store the element corresponding to it\'s level
            store[level].append(curr.val)
            
            # if children of the curr exist
            if curr.children is not None:
                for child in curr.children:
                    queue.append((child,level + 1))
        
        # Loop through the dictionay and store the value in the ans list
        for level in store:
            ans.append(store[level])
            
        # return ans
        return ans
