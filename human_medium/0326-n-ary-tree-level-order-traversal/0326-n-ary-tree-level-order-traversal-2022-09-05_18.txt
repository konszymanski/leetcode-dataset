class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        result = []
        stack = deque()
        stack.append((0, root))

        while stack:
            level, node = stack.popleft()
            
            if not node:
                continue

            if level < len(result):
                result[level].append(node.val)
            else:
                result.append([node.val])
            
            for child in node.children:
                stack.append((level + 1, child))

        return result
