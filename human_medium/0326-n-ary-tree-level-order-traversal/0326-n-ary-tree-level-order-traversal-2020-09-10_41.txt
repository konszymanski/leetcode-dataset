class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        ans = []
        if root: 
            queue = [root]
            while queue: 
                newq, vals = [], []
                for x in queue: 
                    vals.append(x.val)
                    newq.extend(x.children)
                ans.append(vals)
                queue = newq
        return ans
