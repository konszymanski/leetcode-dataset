# Definition for a Node.
class Node:
    def __init__(self, val=None, children=None):
        self.val = val
        self.children = children
"""

class Solution:
    # BFS level-order-traversal solution
    
    # The idea is to do a simple level order traversal
    # Where we iterate through the entire deque to remove 
    # nodes from the current level while adding nodes in the
    # next level (i.e., the children nodes)
    
    # However, instead of adding the left or right (existing)
    # child of the node like we do for binary tree level-order
    # traversal, we go through the list of children for the 
    # current node and add all the children

        q.append(root)

            level = []
            numOfNodes = len(q)
            for i in range(numOfNodes):

                level.append(node.val)
                
                if node.children:
                    for child in node.children:
                        q.append(child)
            res.append(level)
        
        return res
