class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        if root is None:
            return []
        #print(root.children)
        valList=[[root.val]]
        flst=[]
        xlst=[]
        lst=[root]
        while lst:
            x=lst.pop(0)
            if x.children:
                for i in x.children:
                    flst.append(i)
                    xlst.append(i.val)
            if len(lst)==0:
                lst=flst[:]
                flst=[]
                if xlst:
                    valList.append(xlst)
                xlst=[]
        return valList
