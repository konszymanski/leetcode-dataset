class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        res = []
        
        # edge case
        if not root:
            return res
        
        # level order traversal / list comprehension
        level = [root]
        while level:
            res.append([node.val for node in level])
            level = [child for node in level for child in node.children if child]
        return res
