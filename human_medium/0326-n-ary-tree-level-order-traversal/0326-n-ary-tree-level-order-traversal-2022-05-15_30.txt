class Solution:
    def _visit(self, node, result, level):
        if len(result) <= level: result.append([])
        result[level].append(node.val)
        nextLevel = level + 1
        if node.children:
            for child in node.children: 
                self._visit(child, result, nextLevel)
            
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        if not root: return []
        result = []
        self._visit(root, result, 0)
        return result
