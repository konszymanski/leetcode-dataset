class Solution {
    public List<List<Integer>> levelOrder(Node root) {
        List<List<Integer>> list = new ArrayList<>();
        lvls(root, list, 0);
        return list;
    }
    private void lvls(Node root, List<List<Integer>> list, int level){
        if(root == null)    return;
        if(list.size() <= level)    list.add(new ArrayList());
        list.get(level).add(root.val);
        root.children.forEach(child -> lvls(child, list, level+1));
    }
}
