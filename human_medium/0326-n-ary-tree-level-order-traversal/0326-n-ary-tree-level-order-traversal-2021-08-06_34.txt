class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        
        if not root: return []
        output=[]
        
        level=[root]
        
        while level:
            currLevel=[]
            nextLevel=[]
            
            for node in level:
                currLevel.append(node.val)
                for child in node.children:
                    nextLevel.append(child)
            output.append(currLevel)
            level=nextLevel
        return output
