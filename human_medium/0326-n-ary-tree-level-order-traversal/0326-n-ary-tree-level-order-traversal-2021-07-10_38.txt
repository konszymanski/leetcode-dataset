from collections import deque
class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        
        if not root: return []
        
        level_order = [[]]
        def bfs(node = root, level = 0):
            queue = deque()
            queue.append((node, level))
            while queue:
                current_node, current_level = queue.popleft()
                if len(level_order) <= current_level:
                    level_order.append([])
                level_order[-1].append(current_node.val)
                for child in current_node.children:
                    queue.append((child, current_level+1))
                
        bfs()
        return level_order
