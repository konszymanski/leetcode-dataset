class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        samelevel = defaultdict(list)
        def dfs(node,levels):
            if not node: return
            samelevel[levels].append(node.val)
            for c in node.children:
                dfs(c,levels+1)
        dfs(root,0)
        # dict_items([(0, [1]), (1, [3, 2, 4]), (2, [5, 6])])
        return [lst for k,lst in samelevel.items()]
