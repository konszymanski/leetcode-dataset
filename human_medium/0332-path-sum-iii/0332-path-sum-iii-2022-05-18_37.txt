class Solution:
    
    def __init__(self):
        self.num = 0
        
    def dfs(self,root,s):
        if root is None:
            return 
        s -=root.val
        if s == 0:
            self.num+=1
        self.dfs(root.left,s)
        self.dfs(root.right,s)
    
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> int:
        if root is None:
            return 0
        self.dfs(root,targetSum)
        self.pathSum(root.left,targetSum)
        self.pathSum(root.right,targetSum)
        return self.num
