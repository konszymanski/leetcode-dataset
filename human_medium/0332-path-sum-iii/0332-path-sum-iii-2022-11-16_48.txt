def pathSum(self, root: Optional[TreeNode], targetSum: int) -> int:
	self.ans = 0 # answer
	def pathSums(node): # will return all the possible sums we could have including a certain node
		sums = defaultdict(int) # start with a defaultdict (every value is initialized to 0)
		if node: # if the node exists
			for key,val in pathSums(node.left).items(): # add all the possible sums from the left subnode
				sums[key+node.val] += val # but including this node, so the key increases by node.val
			for key,val in pathSums(node.right).items(): # do the same with the right subnode
				sums[key+node.val] += val
			sums[node.val] += 1 # we could also just have this node alone
			self.ans += sums[targetSum] # add the number of ways to sum to targetSum to our answer
		return sums # return the count of possible sums

	pathSums(root) # go through the tree
	return self.ans # return the answer
