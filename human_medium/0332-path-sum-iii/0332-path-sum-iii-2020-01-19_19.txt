# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def pathSum(self, root: TreeNode, sum: int) -> int:
        sum=[sum]
        count=[0]
        def bfs(root,sums):
            if not root:
                return
            sums=[sum+root.val for sum in sums]+[root.val]
            count[0]+=sums.count(sum[0])
            bfs(root.left,sums)
            bfs(root.right,sums)
        bfs(root,[])
        return count[0]
