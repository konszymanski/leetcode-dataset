def pathSum(self, root: Optional[TreeNode], targetSum: int) -> int:

	if root is None:
		return 0

	total_count = 0

	queue = [(root, targetSum)]

	visited = set()

	while queue:
		node, target_sum = queue.pop()

		visited.add((node, target_sum))

		new_target_sum = target_sum - node.val

		if new_target_sum == 0:
			total_count += 1

		if node.left is not None:
			queue.append((node.left, new_target_sum))
			if (node.left, targetSum) not in visited:
				queue.append((node.left, targetSum))

		if node.right is not None:
			queue.append((node.right, new_target_sum))
			if (node.right, targetSum) not in visited:
				queue.append((node.right, targetSum))

	return total_count
