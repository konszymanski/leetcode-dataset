# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
class Solution:
    def pathSum(self, root: TreeNode, sum: int) -> int:
        def dfs(root,prev_sum,sum):
            if not root:
                return 0
            count = 0
            current_sum = prev_sum + root.val
            if current_sum - sum in cache:
                count += cache[current_sum - sum]
            cache[current_sum] = cache.get(current_sum,0) + 1
            count += dfs(root.left,current_sum,sum)
            count += dfs(root.right,current_sum,sum)
            cache[current_sum] -= 1
            return count
        cache = {0:1}
        return dfs(root,0,sum)
                
class Solution1:
    def pathSum(self, root: TreeNode, sum: int) -> int:
        if root:
            return self.findPaths(root,sum) + self.pathSum(root.left,sum) + self.pathSum(root.right,sum)
        return 0
    
    def findPaths(self,root,target):
        if root:
            return int(root.val == target) + self.findPaths(root.left,target-root.val) + self.findPaths(root.right,target-root.val)
        return 0
