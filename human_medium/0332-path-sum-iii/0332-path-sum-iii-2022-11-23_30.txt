class Solution: # without prefix sum
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> int:
        res = 0
        
        def preorder(node, currPath, currSum):
            if not node: return
            nonlocal res
            currSum += node.val
            temp = currSum
            
            if currSum == targetSum: res += 1
                
            for num in currPath:
                temp -= num
                if temp == targetSum: res += 1
            
            currPath.append(node.val)
            preorder(node.left, currPath, currSum)
            preorder(node.right, currPath, currSum)
            currPath.pop()
        
        preorder(root, [], 0)
        return res
    
class Solution: # with prefix sum
    def pathSum(self, root: Optional[TreeNode], targetSum: int) -> int:
        res = 0
        prefixSums = defaultdict(int)
        
        def preorder(node, currSum):
            if not node: return
            nonlocal res
            currSum += node.val
            
            if currSum == targetSum: res += 1
            
            res += prefixSums[currSum - targetSum] 
            prefixSums[currSum] += 1
            preorder(node.left, currSum)
            preorder(node.right, currSum)
            prefixSums[currSum] -= 1
        
        preorder(root, 0)
        return res
