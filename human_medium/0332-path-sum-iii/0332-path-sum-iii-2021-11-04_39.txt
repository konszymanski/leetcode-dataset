def pathSum(self, root: Optional[TreeNode], targetSum: int) -> int:
        self.cnt = 0
        def dfs(root,curr):
            if not root: return 0
            
            curr.append(root.val) # backtracking added element
            
            sum = 0
            for idx in range(len(curr)-1,-1,-1):
                sum+=curr[idx]
                if sum == targetSum: # can\'t break after finding target or if sum exceeds target because of -ve numbers and 0
                    self.cnt += 1

            dfs(root.left,curr)
            dfs(root.right,curr)
            curr.pop() # backtracking removed element
            
        dfs(root,[])
        return self.cnt
