# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

from collections import defaultdict


class Solution:
    def pathSum(self, root: TreeNode, target: int) -> int:
        if not root:
            return 0
        
        # \'SEEN\' KEEPS TRACK OF A COUNT OF ALL POSSIBLE PATH SUMS FROM THE ROOT TO THE CURRENT NODE IN PRE_ORDER()
        seen = defaultdict(int)
        seen[0] = 1
        
        # START A PRE-ORDER TRAVERSAL
        return self.pre_order(root, 0, target, seen)
        
        
    def pre_order(self, root, path_sum, target, seen):
        if not root:
            return 0
        
        # ADD THE CURRENT NODE\'S VALUE TO THE PATH SUM
        path_sum += root.val
        
        # GET THE NUMBER OF PATHS WITH SUM = PATH_SUM - TARGET
        paths = seen.get(path_sum - target, 0)
        
        # RECORD THE PATH SUM IN SEEN
        seen[path_sum] += 1
        
        # RECURSE ON LEFT AND RIGHT SUBTREES
        paths += self.pre_order(root.left, path_sum, target, seen)
        paths += self.pre_order(root.right, path_sum, target, seen)
        
        # REMOVE THE PATH SUM IN SEEN
        seen[path_sum] -= 1
        return paths
