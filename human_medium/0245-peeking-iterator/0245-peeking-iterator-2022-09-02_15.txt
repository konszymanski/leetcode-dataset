class PeekingIterator:
    def __init__(self, iterator):
        self.iter = iterator
        self.nextVal = self.iter.next()  

    def peek(self):
        return self.nextVal

    def next(self):
        val = self.nextVal
        self.nextVal = self.iter.next() if self.iter.hasNext() else None
        return val

    def hasNext(self):
        return self.nextVal is not None
