class PeekingIterator:
    def __init__(self, iterator):
        self.it = iterator
        self.peekVal = self.it.next() if self.it.hasNext() else None

    def peek(self):
        return self.peekVal

    def next(self):
        val = self.peekVal
        self.peekVal = self.it.next() if self.it.hasNext() else None
        return val

    def hasNext(self):
        return self.peekVal is not None
