class PeekingIterator:
    def __init__(self, iterator):
        """
        Initialize your data structure here.
        :type iterator: Iterator
        """
        self.nums = []
        self.pointer = 0 
        while iterator.hasNext():
            self.nums.append(iterator.next())
 
    def peek(self):
        """
        Returns the next element in the iteration without advancing the iterator.
        :rtype: int
        """
        pointer2 = self.pointer
        return self.nums[pointer2]
        
    def next(self):
        """
        :rtype: int
        """
        res = self.nums[self.pointer]
        self.pointer += 1
        return res
        
    def hasNext(self):
        """
        :rtype: bool
        """
        if self.pointer >= len(self.nums):
            return False
        else:
            return True
