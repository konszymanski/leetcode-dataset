class PeekingIterator:
    def __init__(self, iterator):
        """
        Initialize your data structure here.
        :type iterator: Iterator
        """
        self._iterator = iterator
        self._next_item = None

    def peek(self):
        """
        Returns the next element in the iteration without advancing the iterator.
        :rtype: int
        """
        if not self.hasNext():
            raise StopIteration()
            
        self._next_item = self._next_item if self._next_item else self.next()
        return self._next_item 
        

    def next(self):
        """
        :rtype: int
        """
        if not self.hasNext():
            raise StopIteration()
            
        if self._next_item:
            temp, self._next_item  = self._next_item, None
            return temp
        else:
            return self._iterator.next()

    def hasNext(self):
        """
        :rtype: bool
        """
        return True if self._next_item else self._iterator.hasNext()
