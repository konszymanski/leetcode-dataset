class PeekingIterator:
    
    def __init__(self, iterator):
        self.i=iterator
        self.val=self.i.next() 

    def peek(self):
        return self.val
    
    def next(self):
        temp=self.val
        self.val=self.i.next()
        return temp
    
    def hasNext(self):
        return self.val!=-100000
