class PeekingIterator:
    
    iterator = None
    peeked = None
    
    def __init__(self, iterator):
        self.iterator = iterator
        

    def peek(self):
        if not self.peeked:
            self.peeked = self.iterator.next()
        return self.peeked

    def next(self):
        if self.peeked:
            ret = self.peeked
            self.peeked = None
            return ret
        else:
            return self.iterator.next()

    def hasNext(self):
        return self.peeked is not None or self.iterator.hasNext()
