class PeekingIterator:
    def __init__(self, iterator):
        self.iterator = iterator
        self.peek_val = self.iterator.next()

    def peek(self):
        return self.peek_val
        
    def next(self):
        _next = self.peek_val
        self.peek_val = None
        
        if self.iterator.hasNext():
            self.peek_val = self.iterator.next()
        
        return _next
        
    def hasNext(self):
        return self.peek_val is not None
