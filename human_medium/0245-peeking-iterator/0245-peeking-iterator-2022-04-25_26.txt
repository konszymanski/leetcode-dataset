class PeekingIterator:
    def __init__(self, iterator):
        self.l=[]
        self.it=iterator
        iter=iterator
        while iter.hasNext():
            self.l.append(iter.next())
        self.i=-1

    def peek(self):
        return self.l[self.i+1] 
        
    def next(self):
        self.it.next()
        self.i+=1
        return self.l[self.i]
        
    def hasNext(self):
        return (len(self.l)!=(self.i+1))
