class PeekingIterator:
    def __init__(self, iterator):
        self.itr = iterator
        self.curr = None

    def peek(self):
        if self.curr != None:
            return self.curr
        self.curr = self.itr.next()
        return self.curr

    def next(self):
        if self.curr != None:
            val = self.curr
            self.curr = None
            return val
        return self.itr.next()
        
    def hasNext(self):
        return self.curr != None or self.itr.hasNext()
