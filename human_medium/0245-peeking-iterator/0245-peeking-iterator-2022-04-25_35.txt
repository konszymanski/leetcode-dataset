# Below is the interface for Iterator, which is already defined for you.
#
# class Iterator:
#     def __init__(self, nums):
#         """
#         Initializes an iterator object to the beginning of a list.
#         :type nums: List[int]
#         """
#
#     def hasNext(self):
#         """
#         Returns true if the iteration has more elements.
#         :rtype: bool
#         """
#
#     def next(self):
#         """
#         Returns the next element in the iteration.
#         :rtype: int
#         """

class PeekingIterator:
    def __init__(self, iterator):
        """
        Initialize your data structure here.
        :type iterator: Iterator
        """
        #pre-cache the next/hasNext for recall
        self.iterator = iterator
        self.hasNxt   = iterator.hasNext()
        self.nxt      = iterator.next()
        

    def peek(self):
        """
        Returns the next element in the iteration without advancing the iterator.
        :rtype: int
        """
        #return cached iterator.next()
        return self.nxt
        

    def next(self):
        """
        :rtype: int
        """
        #return cached result, and update cache
        res = self.nxt
        self.hasNxt  = self.iterator.hasNext()
        self.nxt     = self.iterator.next()
        return res
        

    def hasNext(self):
        """
        :rtype: bool
        """
        #return cached iterator.hasNext()
        return self.hasNxt
