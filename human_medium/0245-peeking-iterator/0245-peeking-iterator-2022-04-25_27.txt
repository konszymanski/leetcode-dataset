# STATE THE BUFFER EMPTY AT BEGNING
#IF THE BUFFER IS EMPTY MOVE FORWARD ELSE RETURN THE VALUE IN BUFFER AND MAKE IT EMPTY
class PeekingIterator:
    def __init__(self, iterator):
        """
        Initialize your data structure here.
        :type iterator: Iterator
        """
        self.i = iterator
        self.buffer = None

    def peek(self):
        """
        Returns the next element in the iteration without advancing the iterator.
        :rtype: int
        """
        if self.buffer == None:
            self.buffer = self.i.next()
        return self.buffer

    def next(self):
        """
        :rtype: int
        """
        if self.buffer == None:
            return self.i.next()
        else:
            t = self.buffer
            self.buffer = None
            return t
        
    def hasNext(self):
        """
        :rtype: bool
        """
        if self.buffer:
            return True
        return self.i.hasNext()
