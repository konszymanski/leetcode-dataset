class PeekingIterator:
    def __init__(self, iterator):
        self.iterator = iterator
        if self.iterator.hasNext():
            self.nextValue = self.iterator.next()
        else:
            self.nextValue = None
        
    def peek(self):
        return self.nextValue

    def next(self):
        toReturn = self.nextValue
        if self.iterator.hasNext():
            self.nextValue = self.iterator.next()
        else:
            self.nextValue = None
        return toReturn

    def hasNext(self):
        return self.nextValue != None
