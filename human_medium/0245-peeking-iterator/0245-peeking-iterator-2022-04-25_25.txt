class PeekingIterator:
    def __init__(self, iterator):
        """
        Initialize your data structure here.
        :type iterator: Iterator
        """
        self.hasnext = True  #what I have done here is, I have created a nextElement variable and it is storing the first elemnt of the existing iterator and my current iterator pointing to its second element.
        self.itr = iterator
        self.nextElement = self.itr.next()
        

    def peek(self):
        """
        Returns the next element in the iteration without advancing the iterator.
        :rtype: int
        """
        return self.nextElement   #As we are already keeping record of the next element in advance 
        

    def next(self):
        """
        :rtype: int
        """
        temp = self.nextElement    # we know the next element already so before returning the value make sure we update the nextElement variable and hasnext variable as well
        if(self.itr.hasNext()):
            self.nextElement = self.itr.next()
        else:
            self.hasnext = False
            
        return temp
        

    def hasNext(self):
        """
        :rtype: bool
        """
        return self.hasnext
