class PeekingIterator:
    def __init__(self, iterator):
        self.iterable = iterator
        self.top = None

    def peek(self):
        if self.top == None:
            self.top = self.iterable.next()
        
        return self.top
        
    def next(self):
        if self.top == None:
            return self.iterable.next()
        temp = self.top
        self.top = None
        return temp
        

    def hasNext(self):
        if self.iterable.hasNext() or self.top:
            return True
        else:
            return False
