class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
		#base condition
        if len(s)<len(p):
            return []
        #hash for p, 26 yes from a-z which is 0-25 
        wordhash = [0]*26
        #hash for current window
        currhash = [0]*26
        
        #populating hash for p
        for ch in p:
            wordhash[ord(ch)-ord(\'a\')]+=1
        
        l, r = 0, len(p)-1
        
		#populating currhash(TAKE A LOOK THAT ITS RANGE)
        #Here I am leaving the first right index
        for i in range(l,r):
                currhash[ord(s[i])-ord(\'a\')]+=1
        
        result = []
        while r<len(s):
			#adding to current window\'s hash, ord(\'a\') because ord(character) returns ascii number for that
            currhash[ord(s[r])-ord(\'a\')]+=1
            #Checking if characters match
            if wordhash == currhash:
                result.append(l)
			#removing from current window\'s hash
            currhash[ord(s[l])-ord(\'a\')]-=1
            #Moving Forward
            l+=1
            r+=1
            
        return result
