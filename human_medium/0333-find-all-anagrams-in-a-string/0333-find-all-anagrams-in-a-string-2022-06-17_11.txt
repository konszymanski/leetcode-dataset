class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        # if p bigger than s then automatically disqualify
        if len(p) > len(s):
            return []
        
        def generateFreq(seq):
            \'\'\'
            generate frequencies based on allowable window size
            which is length of p  
            \'\'\'
            freq = [0 for _ in range(26)]
            for i in range(len(p)):
                freq[ord(seq[i]) - ord(\'a\')] += 1
            return freq
        
        # freqencies of each letter in p
        pfreq = generateFreq(p)
        
        # freqencies of each letter in s
        sfreq = generateFreq(s)
                        
        def freqMatches(first, second):                        
            \'\'\'
            Do the frequencies match given two freuquency lists
            \'\'\'
            for i in range(26):
                if first[i] != second[i]:
                    return False            
            return True
        
        res = []
        
        if freqMatches(sfreq, pfreq):
            res.append(0)
            
        for i in range(1, len(s) - len(p) + 1):
            # previous start of the window before sliding
            prevStart = i - 1
            # new end of window slid to the right
            currEnd = i + len(p) - 1            
            # update the new freqncies after sliding window
            # previous start is not valid so remove it\'s count
            sfreq[ord(s[prevStart]) - ord(\'a\')] -= 1
            # new count for the end to be included in the frequencies
            sfreq[ord(s[currEnd]) - ord(\'a\')] += 1
            if freqMatches(sfreq, pfreq):
                res.append(i)
                
        return res
