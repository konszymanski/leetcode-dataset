from collections import Counter
class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        #Maintaing a dictionary of characters in the string p
        ptrcnt=Counter(p)
		\'\'\'
		Initializing the start pointer at 0 and end pointer at length of p since we need to check the substrings for the window size of whatever the length of string p is
		\'\'\'
        st,en=0,len(p)
        ans=[]
		#Traversing till the end pointer doesn\'t reach the last of string s
        while en<=len(s):
		   \'\'\'
		   Creating a frequency map for substring of window of size of p, if it is equivalent to the frequency map of p then add the start index of that substring to my answer
		   \'\'\'
            if Counter(s[st:en])==ptrcnt:
                ans.append(st)
			#Slide the window
            st+=1
            en+=1
        return ans
