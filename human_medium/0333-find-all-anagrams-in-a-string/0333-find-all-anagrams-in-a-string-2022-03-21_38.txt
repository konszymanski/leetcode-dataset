class Solution:
	#sliding window
	#T=O(n), S=O(1)
	#constant space because of lowercase english chars (26 chars)
    def findAnagrams(self, s: str, p: str) -> List[int]:
        #init state counter
        counter = collections.Counter(p)
        #init pointers
        l, r = 0, 0
        res = []
        #count var for inner loop exec
        count = len(p)
        while r < len(s):
            counter[s[r]] -= 1
            #duplicate values in p
            #\'==\' operator for unique values in p
            if counter[s[r]] >= 0:
                count -= 1
            #shift right ptr
            r += 1
            while count == 0:
                #len of anagram
                if (r-l) == len(p):
                    res.append(l)
                counter[s[l]] += 1
                if counter[s[l]] > 0:
                    count +=1
                l += 1
        return res
