class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]: 
        #sorting p string because we will check everytime the sorted string which 
        #will be chosen from the string s
        p=sorted(p)
        #pointer to the left of the string s
        l=0 
        #for storing the indeces of anagrams
        a=[]
        #r is moving as right pointer
        #r start from len(p)-1 because the length of anagram should be equal to p
        #and index will be start from len(p)-1 obviously
        for r in range(len(p)-1,len(s)): 
            #choosing substring
            n=s[l:r+1] 
            #if chosen substring is anagram (check by sorting n , as we sorted p too) then store in a
            if p==sorted(n):
                a.append(l)
            #move the left pointer (this forming sliding window)
            l+=1 
        #return array of indeces
        return a
