class Solution:
    def findAnagrams(self, s: str, p: str) -> List[int]:
        if len(p) > len(s):
            return []
        pCounter = {}
        sCounter = {}
        
        # Count the elemnts in str p and str s
        for i in range(len(p)):
            pCounter[p[i]] = 1 + pCounter.get(p[i], 0)
            sCounter[s[i]] = 1 + sCounter.get(s[i], 0)
        # Check if both dicts are equal and return 0 if true (base case)
        res = [0] if pCounter == sCounter else []
        # Start traversing the string s in the range len(str p) to len(str s) and keep updating the sCounter until it is equal to pCounter
        l = 0
        for i in range(len(p), len(s)):
            sCounter[s[i]] = 1 + sCounter.get(s[i], 0)
            sCounter[s[l]] -= 1

            if sCounter[s[l]] == 0:
                sCounter.pop(s[l])
            l += 1
            if sCounter == pCounter:
                res.append(l)
        return res
