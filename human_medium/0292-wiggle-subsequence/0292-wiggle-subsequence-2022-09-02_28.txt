class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        dp = [nums[i+1]-nums[i] for i in range(len(nums)-1) if nums[i]!=nums[i+1]]
        if not dp:
            return 1
        cur = 2 # base case 
        for i in range(len(dp)-1):
            if dp[i]>0 and dp[i+1]<0 or dp[i]<0 and dp[i+1]>0:
                cur+=1
        return cur
