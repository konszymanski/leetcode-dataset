class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        dp = []
        
        for j in range (1,len(nums)) :
            diff = nums[j] - nums[j-1]
            if diff != 0 :
                dp.append(diff)
                
        if not dp :
            return 1
        
        count = 2       # If an array has atleast two unique numbers, then their minimum subsequence is 2 
            
        for i in range (1,len(dp)) :
            if dp[i] > 0 :
                if dp[i-1] < 0 :
                    count+=1
                    
            if dp[i] < 0 :
                if dp[i-1] > 0 :
                    count+=1
                   
        return count
