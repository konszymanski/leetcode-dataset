class Solution:
    def wiggleMaxLength(self, nums):
        up = 1
        down = 1
        n = len(nums)
        # print(nums)
        
        for i in range(1, n):
            # downhill
            if(nums[i] < nums[i-1]):
                down = up + 1
            #uphill
            elif(nums[i] > nums[i-1]):
                up = down+1
            # print(up,down,i)
        return max(down, up)
