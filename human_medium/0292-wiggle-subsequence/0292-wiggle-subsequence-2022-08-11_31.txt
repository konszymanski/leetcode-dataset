class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        if not nums:
            return 0
        curr_up_state_len = 1
        curr_down_state_len = 1
        for i in range(1, len(nums)):
            if nums[i] > nums[i - 1]:
                curr_up_state_len = curr_down_state_len + 1
            elif nums[i] < nums[i - 1]:
                curr_down_state_len = curr_up_state_len + 1
            else:
                continue  # same value, not wiggling
        return max(curr_up_state_len, curr_down_state_len)
