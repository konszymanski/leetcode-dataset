class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        l=len(nums)
        if l==1:
            return 1
        
        n=[]
		# array N for storing whether the difference is positive or negative
        i=1
        
        while i<l:
            if nums[i]-nums[i-1]>0:
				# if difference between consective numbers is positive append \'+\'
                n.append("+")
            elif nums[i]-nums[i-1]<0:
				# if difference between consective numbers is negative append \'-\'
                n.append("-")
            else:
				# if numbers are same then remove one of them
                nums.pop(i)
                l=l-1
                continue
            i=i+1
        c=0
        l=len(n)
        if l==0:
            return c+1
        
        # now calculating total number of Consecutive + and - and returning by adding 1 to it
        prev=n[0]
        c=c+1
        i=1
        
        while i<l:
            if prev != n[i]:
                c=c+1
                prev=n[i]
                i=i+1
            else:
                i=i+1
        return c+1
