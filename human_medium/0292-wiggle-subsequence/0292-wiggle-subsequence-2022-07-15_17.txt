class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        n = len(nums)
        down, up = 1, 1
        
        for i in range(1, n):
            if nums[i] > nums[i-1]:
                up = max(down + 1, up)
            elif nums[i] < nums[i-1]:
                down = max(up + 1, down)
                
        return max(up, down)
