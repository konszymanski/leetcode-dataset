class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        if len(nums) < 2:
            return len(nums)

        length = 1
        was_positive: bool = None

        for i in range(1, len(nums)):
            prev = nums[i - 1]
            current = nums[i]

            if prev == current:
                continue

            is_positive = prev > current

            if was_positive == is_positive:
                continue

            length += 1
            was_positive = is_positive

        return length
