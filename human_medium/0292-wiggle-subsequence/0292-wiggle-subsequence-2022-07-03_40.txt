class Solution:
    def wiggleMaxLength(self, nums: List[int]) -> int:
        ans = 1
        prev = nums[0]
        wait_big = None
        for num in nums:
            if num > prev:
                if wait_big is None or wait_big:
                    ans += 1
                    wait_big = False
            elif num < prev:
                if not wait_big:
                    ans += 1
                    wait_big = True
            prev = num
        return ans
