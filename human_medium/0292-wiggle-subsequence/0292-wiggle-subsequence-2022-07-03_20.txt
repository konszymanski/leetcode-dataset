class Solution:
    def wiggleMaxLength(self, arr: List[int]) -> int:
        n=len(arr)
        i=1
        WIGGLES=1
        
        #THIS WHILE LOOP IF TO ENSURE THAT IF ALL ELEMENTS OF ARRAY ARE SAME
        #THEN THERE IS NO WIGGLE THEREFORE WIGGLE IS 1
        while i<n and arr[i-1]==arr[i]:
            i+=1
            
        #NOW WE RUN A WHILE LOOP TILL THE LENGTH OF THE ARRAY
        while i<n:
            # WE KEEP GOING ON TILL ARRAY KEEPS INCREASING AND WHEN THERE IS A PEAK VALUE,
            # WE BREAK THE LOOP AND WNCLREASE WIGGLE BY ONE
            if arr[i]>=arr[i-1]:
                while i<n and arr[i]>=arr[i-1]:
                    i+=1
                WIGGLES+=1
            else:
            # WE KEEP GOING ON TILL ARRAY KEEPS DECREASING AND WHEN THERE IS A PEAK VALUE,
            # WE BREAK THE LOOP AND WNCLREASE WIGGLE BY ONE
                while i<n and arr[i]<=arr[i-1]:
                    i+=1
                WIGGLES+=1
        return (WIGGLES)    #RETURN WIGGLES
