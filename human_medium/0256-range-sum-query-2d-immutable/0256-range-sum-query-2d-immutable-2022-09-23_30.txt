class NumMatrix(object):

    def __init__(self, matrix):
        """
        :type matrix: List[List[int]]
        """
        # Create a matrix to store the prefix sums. Add one row of zeros and one column of zeros at the beginning
        # to facilitate the substraction operations involved in the `sumRegion` function.
        pre_sums = [[0] * (len(matrix[0]) + 1) for _ in range(len(matrix) + 1)]
        for index_row in range(len(matrix)):
            row_pre_sum = 0
            for index_col in range(len(matrix[0])):
                row_pre_sum += matrix[index_row][index_col]
                pre_sums[index_row+1][index_col+1] = pre_sums[index_row][index_col+1] + row_pre_sum
        self.pre_sums = pre_sums

    def sumRegion(self, row1, col1, row2, col2):
        """
        :type row1: int
        :type col1: int
        :type row2: int
        :type col2: int
        :rtype: int
        """
        # All the pre_sums start from the origin point. So the sum of one region is basically one big pre_sum minus two small pre_sums
        # plus one overlap area which has been substracted twice.
        return self.pre_sums[row2+1][col2+1] - self.pre_sums[row2+1][col1] - self.pre_sums[row1][col2+1] + self.pre_sums[row1][col1]
