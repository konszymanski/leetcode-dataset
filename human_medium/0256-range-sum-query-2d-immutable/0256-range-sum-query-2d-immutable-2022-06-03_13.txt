class NumMatrix:
    def __init__(self, matrix):
        self.matrix = matrix
        
    def sumRegion(self, row1, col1, row2, col2):
        summation = 0
        for i in range(row1, row2 + 1):
            for j in range(col1, col2 + 1):
                summation += self.matrix[i][j]
        return summation
