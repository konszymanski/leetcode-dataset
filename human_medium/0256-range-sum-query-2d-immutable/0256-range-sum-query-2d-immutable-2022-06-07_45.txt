class NumMatrix:
    def __init__(self, matrix: List[List[int]]):
        m, n = len(matrix), len(matrix[0])
        # init prefixSum matrix of size (m+1)*(n+1)
        self.prefixSum = [[0] * (n+1) for r in range(m+1)]
        for i in range(m):
            prefix = 0
            for j in range(n):
                # sum the whole i-th row from 0-th to j-th column
                prefix += matrix[i][j]
                # sum the prefix with the prefixSum of everything above(above row)
                self.prefixSum[i+1][j+1] = self.prefixSum[i][j+1] + prefix
                

    def sumRegion(self, row1: int, col1: int, row2: int, col2: int) -> int:
        # bottom right corner - left cols - above rows + top left corner
        return self.prefixSum[row2+1][col2+1] - self.prefixSum[row2+1][col1] - self.prefixSum[row1][col2+1] + self.prefixSum[row1][col1]
