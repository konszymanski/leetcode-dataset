def decodeString(self, s: str) -> str:
        stack = []
        
        for ch in s:
		
			# whenever a closing bracket arrives in s.
            if ch == "]":
                value  = "" # set value to default
				
                #store each character from stack in reverse order.
                while stack and stack[-1] != "[":
                    value = stack.pop() + value
                # for example: If stack contains [3, [, c, 12, [, a, b]. Loop will continue till "[" and value will store "ab".
				
                stack.pop() # Removing Opening Bracket "[" from stack.
                value *= int(stack.pop()) # Multiplying string to number stored after "[" that is 12.
                stack.append(value) # Now append the result again in stack for further brackets.
                
            else:
                if ch.isdigit() and stack and stack[-1].isdigit(): 
					# If a number already inserted in stack[-1]. For eg. [3, [, c, 1]
                    stack[-1] += ch 
					# now stack becomes [3, [, c, 12]. if ch is 2 this time.
                else:
					# for character other then a number
                    stack.append(ch)
        
        return "".join(stack)
