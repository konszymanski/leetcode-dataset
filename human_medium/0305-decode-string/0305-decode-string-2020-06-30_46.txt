class Solution:
    def decodeString(self, s: str) -> str:
        resultStr = ""
        countStack , stringStack = [] , []
        idx = 0
        while idx < len(s):
            if s[idx].isdigit():
                stringCount = ""
                while s[idx].isdigit():
                    stringCount += s[idx]
                    idx += 1
                countStack.append(int(stringCount))
            elif s[idx] == "[":
                stringStack.append(resultStr)
                resultStr =  ""
                idx += 1                
            elif s[idx] == "]":
                count = countStack.pop()
                tempString = stringStack.pop()
                resultStr = tempString + count * resultStr
                idx += 1
            else:
                resultStr += s[idx]
                idx += 1
        return resultStr
