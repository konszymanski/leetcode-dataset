def decodeString(self, s: str) -> str:
        k, ans, stack = 0, "", []
        for c in s:
            if c == \'[\':
                stack.extend([k, ans])
                k, ans = 0, ""
            elif c == \']\': ans = stack.pop() + stack.pop()*ans
            elif c.isdigit(): k = 10*k + int(c)
            else: ans += c
        return ans
