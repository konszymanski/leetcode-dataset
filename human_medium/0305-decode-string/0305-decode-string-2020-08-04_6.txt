class Solution:
    def decodeString(self, s: str) -> str:
        stack = []
        a, d = \'\', \'\'
        for letter in s:
            if letter.isdigit():
                d+=letter
            elif letter.isalpha():
                a+=letter
            elif letter==\'[\':
                stack.append((a, int(d)))
                a, d = \'\', \'\'
            elif letter==\']\':
                p, n = stack.pop()
                a = p+a*n
            
        return a
