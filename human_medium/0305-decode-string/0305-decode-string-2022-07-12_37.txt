class Solution:
    def decodeString(self, s: str) -> str:
      stack = [] # empty stack
      temp = \'\' # Temporary string we have to build within [....]
      num = \'\' # create number i\'e 123 or 232 . Given no can be 1 <= n <= 300
      for char in s: # process one char
        if char.isdigit(): num += char # if it is digit, collect digits
        elif char == "[": 
          stack.append(num) # push into stack as one digit
          num = \'\' # for future, make it empty
        elif char.isalpha(): stack.append(char) # if alphabet, just push in stack normally
        if char == "]": # Process stack here for most recent OR Inner Most bracket
          temp = [] # collect chars for innermost bracket
          while stack: # either stack will be empty or we\'ll hit a digit which we pushed
            c = stack.pop()
            if c.isdigit(): # if digit, repeat the already created string these many times and push onto stack and break
              stack.append(\'\'.join(temp[::-1]*int(c)))
              break
            else:
              temp.append(c) # if not digit, create innermost string

      return \'\'.join(stack) # add pieces of stack to create a single string
