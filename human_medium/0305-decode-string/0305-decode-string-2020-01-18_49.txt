import re
class Solution:
    def decodeString(self, s: str) -> str:
        s = re.sub(r\'(\\])(?=[a-zA-Z0-9])\', r\'\\1+\', s)
        s = re.sub(r\'(?<=[a-zA-Z])([\\d]+)\', r\'+\\1\', s)
        s = re.sub(r\'([a-zA-Z]+)\', r\'"\\1"\', s)
        return eval(s.replace("]", ")").replace("[", "*(")) if s else s
