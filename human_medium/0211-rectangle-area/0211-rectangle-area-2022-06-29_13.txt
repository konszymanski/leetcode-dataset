class Solution:
    def computeArea(self, ax1: int, ay1: int, ax2: int, ay2: int, bx1: int, by1: int, bx2: int, by2: int) -> int:
        ans = (ay2-ay1)*(ax2-ax1)+(by2-by1)*(bx2-bx1)
        h = min(ay2,by2)-max(ay1,by1)
        w = min(bx2,ax2)-max(ax1,bx1)
        if h>0 and w>0:
            ans-=h*w
        return ans
