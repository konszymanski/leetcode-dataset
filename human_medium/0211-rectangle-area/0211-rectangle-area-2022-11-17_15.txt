class Solution:
    def computeArea(self, ax1: int, ay1: int, ax2: int, ay2: int, bx1: int, by1: int, bx2: int, by2: int) -> int:
        y_high = min(ay2, by2) #find upper y of common area
        y_low = max(ay1, by1) #find lower y of common area
        x_high = min(ax2, bx2) #find upper x of common area
        x_low = max(ax1, bx1) #find lower x of common area 
        if y_high <= y_low or x_high <= x_low: #if upper y <= lower y of common area or upper x <= lower x of common area 
			#there is no common area between a and b 
            common_area = 0
        else :
            common_area = (y_high - y_low)*(x_high - x_low) #common area
        a_area = (ay2 - ay1)*(ax2 - ax1) #area of a 
        b_area = (by2 - by1)*(bx2 - bx1) #area of b 
        return a_area + b_area - common_area #total area = sum of both areas - common_area
