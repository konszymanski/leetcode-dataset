class Solution:
    def computeArea(self, ax1: int, ay1: int, ax2: int, ay2: int, bx1: int, by1: int, bx2: int, by2: int) -> int:
        r1area=(ax2-ax1)*(ay2-ay1)
        r2area=(bx2-bx1)*(by2-by1)
        cx1=max(ax1,bx1)
        cy1=max(ay1,by1)
        cx2=min(ax2,bx2)
        cy2=min(ay2,by2)
        r3cmnarea=0
        c1=(cx2-cx1)
        c2=(cy2-cy1)
        if c1>0 and c2>0:
            r3cmnarea=c1*c2
        return (r1area+r2area)-r3cmnarea
