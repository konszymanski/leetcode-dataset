class Solution:
    def computeArea(self, ax1: int, ay1: int, ax2: int, ay2: int, bx1: int, by1: int, bx2: int, by2: int) -> int:
        pa = abs(ay2 - ay1) * abs(ax2 - ax1)
        pb = abs(by2 - by1) * abs(bx2 - bx1)

        dx = min(ax2, bx2) - max(ax1, bx1)
        dy = min(ay2, by2) - max(ay1, by1)

        if dx > 0 and dy > 0:
            return (pa + pb) - (dx * dy)
        
        return pa + pb
