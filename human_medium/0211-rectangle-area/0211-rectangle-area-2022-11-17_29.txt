class Solution:
    def computeArea(self, ax1: int, ay1: int, ax2: int, ay2: int, bx1: int, by1: int, bx2: int, by2: int) -> int:
        cx2 = min(ax2, bx2)
        cy2 = min(ay2, by2)
        cx1 = max(ax1, bx1)
        cy1 = max(ay1, by1)
        a = (ax2 - ax1) * (ay2 - ay1)
        b = (bx2 - bx1) * (by2 - by1)
        c = max(0, cx2 - cx1) * max(0, cy2 - cy1)
        return a + b - c
