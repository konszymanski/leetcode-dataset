def computeArea(ax1: int, ay1: int, ax2: int, ay2: int, bx1: int, by1: int, bx2: int, by2: int) -> int:
    overlap = 0
    x = sorted([(ax1, 1), (ax2, 1), (bx1, 2), (bx2, 2)])
    y = sorted([(ay1, 1), (ay2, 1), (by1, 2), (by2, 2)])
    if x[0][1] != x[1][1] and x[2][1] != x[3][1] and y[0][1] != y[1][1] and y[2][1] != y[3][1]:
        x1, y1 = x[1][0], y[1][0]
        x2, y2 = x[2][0], y[2][0]
        overlap = (x2 - x1) * (y2 - y1)

    return (ax2 - ax1) * (ay2 - ay1) + (bx2 - bx1) * (by2 - by1) - overlap
