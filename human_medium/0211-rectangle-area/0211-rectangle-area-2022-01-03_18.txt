class Rectangle:

	def __init__(self, x1, y1, x2, y2):
		self.x1 = x1
		self.y1 = y1
		self.x2 = x2
		self.y2 = y2
	
	def compute_rectangle_area(self):
		width = abs(self.x1 - self.x2)
		height = abs(self.y1 - self.y2)
		return height * width

	@staticmethod
	def compute_shared_rectangle_area(rectangle_a, rectangle_b):
		x1_of_shared_rectangle = max(rectangle_a.x1, rectangle_b.x1)
		x2_of_shared_rectangle = min(rectangle_a.x2, rectangle_b.x2)
		width_of_shared_rectangle = max(abs(x1_of_shared_rectangle - x2_of_shared_rectangle), 0)
		
		y1_of_shared_rectangle = max(rectangle_a.y1, rectangle_b.y1)
		y2_of_shared_rectangle = min(rectangle_a.y2, rectangle_b.y2)
		height_of_shared_rectangle = max(abs(y1_of_shared_rectangle - y2_of_shared_rectangle), 0)
		
		return height_of_shared_rectangle * width_of_shared_rectangle

	@staticmethod
	def compute_total_area_of_shared_rectangles(rectangle_a, rectangle_b):
		total_area_of_a_and_b = sum(
				[rectangle_a.compute_rectangle_area(), rectangle_b.compute_rectangle_area(),
					- Rectangle.compute_shared_rectangle_area(rectangle_a, rectangle_b)]
		)
		return total_area_of_a_and_b

# Rectangle a
# bottom-left-corner
ax1 = -3
ay1 = 0
# top-right-corner
ax2 = 3
ay2 = 4

# Rectangle b
# bottom-left-corner
bx1 = 0
by1 = -1
# top-right-corner
bx2 = 9
by2 = 2

rectangle_a = Rectangle(ax1, ay1, ax2, ay2)
print("Area of rectangle A: ", rectangle_a.compute_rectangle_area())
rectangle_b = Rectangle(bx1, by1, bx2, by2)
print("Area of rectangle B: ", rectangle_b.compute_rectangle_area())

shared_area_of_a_and_b = Rectangle.compute_shared_rectangle_area(rectangle_a, rectangle_b)
print("Shared area of A and B: ", shared_area_of_a_and_b)

total_area_of_a_and_b = Rectangle.compute_total_area_of_shared_rectangles(rectangle_a, rectangle_b)
print("Total area incl. shared area of A and B: ", total_area_of_a_and_b)
