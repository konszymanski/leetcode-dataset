# Create a undirected graph
        # check each queries 
        # Time: O(len(values)*(undirected graph edges)), Space(undirected graph edges)
        
        hmap = defaultdict(dict)
        for eq,v in zip(equations,values):
            hmap[eq[0]][eq[1]] = v
            hmap[eq[1]][eq[0]] = 1/v
        
        
        def checkval(s,e,multi):
            if s in visited:
                return -1.0
            visited.add(s)
            for nei in hmap[s]:
                ans = multi*hmap[s][nei]
                if nei == e:
                    return ans
                else:
                    temp = checkval(nei,e,ans)
                    if temp != -1.0:
                        return temp    
            return -1.0
        
        
        
        res = []
        for q in queries:
            visited = set()
            if q[0] not in hmap:
                res.append(-1.0)
            else:
                res.append(checkval(q[0],q[1],1))
        return res
