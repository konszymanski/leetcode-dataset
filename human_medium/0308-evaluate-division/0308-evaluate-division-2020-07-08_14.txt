class Solution:
    def calcEquation(self, equations: List[List[str]], values: List[float], queries: List[List[str]]) -> List[float]:
        # Floyd-Warshall\'s Algo
        graph = collections.defaultdict(dict)
        res = []
        for (num, den), val in zip(equations, values):
            graph[num][num] = 1.0
            graph[den][den] = 1.0
            graph[num][den] = val
            graph[den][num] = 1/val

        for k in graph:
            for i in graph[k]:
                for j in graph[k]:
                    graph[i][j] = graph[i][k] * graph[k][j]
        
        for n,d in queries:
            if n in graph and d in graph[n]:
                res.append(graph[n][d])
            else:
                res.append(-1.0)

        return res
