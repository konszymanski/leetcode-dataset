class Solution:
    def diffWaysToCompute(self, expression: str) -> List[int]:
                
        def findIndices(expression: str) -> list[int]:
            indicesOps = []

            for i in range(len(expression)):
                if expression[i] in \'+-*\':
                    indicesOps.append(i)
                    
            return indicesOps
        
        def rec(remainingExp: str) -> list[int]:
            curAnswers = []
            indicesOps = findIndices(remainingExp)
            
            if not indicesOps:
                return [int(remainingExp)]
            
            for i in indicesOps:
                op = remainingExp[i]
                left = rec(remainingExp[:i])
                right = rec(remainingExp[i+1:])
                
                for x in left:
                    for y in right:
                        if op == \'+\':
                            curAnswers.append(x + y)
                        elif op == \'-\':
                            curAnswers.append(x-y)
                        else:
                            curAnswers.append(x*y)
                            
            return curAnswers
                            
        return rec(expression)
