class Solution:
    def diffWaysToCompute(self, expression: str) -> List[int]:
        # dfs with memo using divide and conquer
        memo = {}  # str: list of result values
        operators = {\'-\', \'+\', \'*\'}
        
        def dfs(expre):
            # return the list of possible values of expre
            # result should be a list to track all possible ways(duplicate ele allowed)
            result = []
            if expre in memo:
                return memo[expre]
                
            for i, char in enumerate(expre):
                # divide the expre into two sub-expression
                if char in operators:
                    left = dfs(expre[:i])
                    right = dfs(expre[i+1:])
                
                    # conquer - iterate two lists to calculate the result 
                    for num_left in left:
                        for num_right in right:
                            result.append(eval(str(num_left) + char + str(num_right)))
            
            if not result:  # means expre is a number
                result.append(eval(expre))
            
            memo[expre] = result
            return result
        
        return dfs(expression)
        # Time Complexity: O(N * 2^N), N is the length of the expression
