class Solution:
    def diffWaysToCompute(self, expression: str) -> List[int]:
        
        # can memoize this solution, but this will not help reducing asymptotic time complexity. 
        # Don\'t get cought up in leetcode run time
        
        def helper(s, start, end):
            

            if s[start:end+1].isdigit():
                return [int(s[start:end+1])]
            
            result = []
            
            for idx in range(start, end+1):
                
                if not s[idx].isdigit():
                    
                    leftresults = helper(s, start, idx-1)
                    rightresults = helper(s,idx+1, end)

                    for lval in leftresults:
                        for rval in rightresults:
         
                            if s[idx] == "+":

                                result.append(lval + rval)
                            elif s[idx] == "-":

                                result.append(lval - rval)
                            else:

                                result.append(lval * rval)
            return result          
                  
        return helper(expression,0, len(expression)-1)
