from collections import defaultdict

class Solution:
    def getHint(self, secret: str, guess: str) -> str:
        
        b, c = 0, 0
        charCountMap = defaultdict(int)
        avoidIndex = set()
        
        str_len = len(secret)
        
        for i in range(str_len):
            if secret[i] == guess[i]:
                b += 1
                avoidIndex.add(i)
            else:
                charCountMap[secret[i]] += 1


        for j in range(str_len):
            if j not in avoidIndex and guess[j] in charCountMap and charCountMap[guess[j]] > 0:
                c += 1
                charCountMap[guess[j]] -= 1

        
        return "{}A{}B".format(b, c)
