class Solution:
    def getHint(self, secret: str, guess: str) -> str:
        myCount=collections.Counter()
        seen=set()
        bulls=0
        cows=0
        
        for idx,n in enumerate(secret):
            if guess[idx]==n:
                seen.add(idx)
                bulls+=1
            else:
                myCount[n]+=1
                
        for idx,n in enumerate(guess):
            if idx not in seen and myCount[n]>0:
                    cows+=1
                    myCount[n]-=1
                    
        hint=str(bulls)+"A"+str(cows)+"B"    
        return  hint
