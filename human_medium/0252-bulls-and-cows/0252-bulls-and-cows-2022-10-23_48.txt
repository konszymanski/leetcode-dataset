from collections import Counter
class Solution:
    def getHint(self, secret: str, guess: str) -> str:
      # so, we need to return the number "right place right number" guesses
      # and the number of "right number wrong place" guesses, after removing the above
      # that really sounds like removing items and then using a counter
      correct = set()
      for i, (s, g) in enumerate(zip(secret, guess)):
        if s == g:
          correct.add(i)
      # rewrite the secret and guess strings by removing the numbers we got right
      secret_number_count = Counter([s for i,s in enumerate(secret) if i not in correct])
      guess_number_count = Counter([g for i,g in enumerate(guess) if i not in correct])
      close = 0
      for num, occurrences in secret_number_count.items():
        guess_occurrences = guess_number_count[num]
        close += min(occurrences, guess_occurrences)
      return f"{len(correct)}A{close}B"
