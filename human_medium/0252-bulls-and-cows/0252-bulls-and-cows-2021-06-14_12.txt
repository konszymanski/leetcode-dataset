class Solution(object):
    def getHint(self, secret, guess):
        ct1, ct2 = [0] * 10, [0] * 10
        x, y = 0,0
        for i in range(len(secret)): 
            if guess[i] == secret[i]: x += 1
            else:
                ct1[ord(secret[i])-ord(\'0\')] += 1
                ct2[ord(guess[i])-ord(\'0\')] += 1
        y = sum(min(x,y) for x,y in list(zip(ct1,ct2)))
        return  str(x) + "A" + str(y) + "B"
