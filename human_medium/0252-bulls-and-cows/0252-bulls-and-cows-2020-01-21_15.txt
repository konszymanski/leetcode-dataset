class Solution:
    def getHint(self, secret: str, guess: str) -> str:       
        na, stk= 0, {str(i):[0, 0] for i in range(10)}         
        for i in range(len(secret)):
            vs, vg = secret[i], guess[i]
            if vs==vg:
                na += 1
            else:
                stk[vs][0] += 1
                stk[vg][1] += 1
        nb = sum(min(stk[i][0], stk[i][1]) for i in stk)
        return "{0:d}A{1:d}B".format(na, nb)
