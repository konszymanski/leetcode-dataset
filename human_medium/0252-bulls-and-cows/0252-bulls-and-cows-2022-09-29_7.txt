class Solution:
    def getHint(self, se: str, gu: str) -> str:
        dcse=defaultdict(lambda:0)
        dcgu=defaultdict(lambda:0)
        a=0
        b=0
        for i in range(len(se)):
            if(se[i]==gu[i]):
                a+=1
            else:
                dcse[se[i]]+=1
                dcgu[gu[i]]+=1
        for x in dcse:
            if(dcgu[x]>=dcse[x]):
                b+=dcse[x]
            else:
                b+=dcgu[x]
        return(str(a)+"A"+str(b)+"B")
