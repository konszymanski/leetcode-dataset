from collections import Counter

class Solution:
    def getHint(self, secret: str, guess: str) -> str:
        b = 0
        c = 0
        

        counter_s = Counter(secret)        
        counter_g = Counter(guess)        
        
		# Count exact matches
		for i in range(len(secret)):
            if secret[i] == guess[i]:
                b += 1
				
		# Count matches without respect to location
        for k,v in counter_s.items():
            c += min(v, counter_g[k])
			
		# Remove exact matches from c.
        c -= b
		
        return f\'{b}A{c}B\'
