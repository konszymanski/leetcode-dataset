def getHint(self, secret: str, guess: str) -> str:
        #counting Bulls here
        s=sum([1 for i in range(len(secret)) if secret[i]==guess[i]])
        a={}
        #Creating a dictionary with values and their counts in secret
        for i in range(len(secret)):
            if secret[i] in a:
                a[secret[i]]=a[secret[i]]+1
            else:
                a.update({secret[i]:1})
        counter=0
        #Counting how many characters from guess are in secret. Counter here will be cows + Bulls
        for i in range(len(guess)):
            if guess[i] in a and a[guess[i]]>0:
                counter=counter+1
                a[guess[i]]=a[guess[i]]-1
        return str(s)+"A"+str(counter-s)+"B"
