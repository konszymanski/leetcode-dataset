def getHint(self, secret: str, guess: str) -> str:
        cb,cc = 0,0                                #Count bulls and Count cows 
        seen = set()
        
        ds,dg = Counter(secret), Counter(guess)                              #dictionary secret and dictionary guess
        
        for x,y in zip(secret,guess):              # For bulls
            if x==y:
                cb+=1
        for i in guess:                                 # For cows and bulls 
            if i in ds and i not in seen:
                cc+=min(ds[i],dg[i])
                seen.add(i)
        
        cc = cc-cb                            # total Cows 
        res = ""
        res = str(cb)+"A"+str(cc)+"B"
        return res
