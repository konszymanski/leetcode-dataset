class Solution:
    def getHint(self, secret: str, guess: str) -> str:

        bulls = sum([1 if x==y else 0 for x,y in zip(secret,guess)])
        
        # remove bull characters
        secret_new = \'\'.join([x for x,y in zip(secret,guess) if x!=y])
        guess_new  = \'\'.join([y for x,y in zip(secret,guess) if x!=y])

        chars_secret = Counter(guess_new)
        chars_guess = Counter(secret_new)
        
        cows = sum( [min(chars_secret[c],chars_guess[c]) for c in chars_secret] )
                
        return \'{0}A{1}B\'.format(bulls,cows)
