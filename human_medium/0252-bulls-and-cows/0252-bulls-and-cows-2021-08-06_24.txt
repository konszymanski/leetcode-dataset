def getHint(self, secret: str, guess: str) -> str:
        secret_dict, guess_dict = defaultdict(set), defaultdict(set)
        
        for i,e in enumerate(secret):
            secret_dict[e].add(i)
        for i,e in enumerate(guess):
            guess_dict[e].add(i)
            
        bulls = cows = 0
        for e in secret_dict.keys():
            if e not in guess_dict.keys():
                continue
            intersection_len = len(secret_dict[e] & guess_dict[e])
            bulls += intersection_len
            cows += min(len(secret_dict[e]), len(guess_dict[e]))-intersection_len
            
        return "{}A{}B".format(bulls, cows)
