class Solution:
    def getHint(self, secret: str, guess: str) -> str:
        return f"{sum([1 for g,s in zip(guess, secret) if g==s])}A{sum([min(gs[x], ss[x]) for x in set(guess) for (gs, ss) in [((Counter([g for g,s in zip(guess, secret) if g!=s])), Counter([s for g,s in zip(guess, secret) if g!=s]))]])}B"
