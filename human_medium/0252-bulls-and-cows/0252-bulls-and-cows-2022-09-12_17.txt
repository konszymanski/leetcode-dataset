class Solution:
    def getHint(self, secret: str, guess: str) -> str: 
        s1="" #making for \'secret\' after removing the same position elements from it
        s2=""  #making for \'guess\' after removing the same position elements from it

        x=0 #for counting the same place elements of \'secret\' and \'guess\'
        for i in range(len(secret)):  
            if secret[i]==guess[i]:  #searching the same place elements
                x+=1  #increment the count if found
            else:  #otherwise remaining elements of \'secret\' and \'guess\' will be stored in s1 & s2 respectively
                s1=s1+secret[i] 
                s2=s2+guess[i] 
		#now y is initialised for counting how many elements of \'secret\' are present in \'guess\' but they are in different positions
        y=0 
        for i in range(len(s1)): 
            if s1[i] in s2: #searching every element of s1 in s2
                y+=1  #if found increment the count
                s2=s2.replace(s1[i],\'\',1) #if found the element in s2 then remove that element from s2 so that it can not be counted again
        return str(x)+"A"+str(y)+"B" #returning the string as asked in question
