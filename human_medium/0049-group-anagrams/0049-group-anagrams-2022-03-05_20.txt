# Runtime: 118 ms, faster than 74.17% of Python3 online submissions for Group Anagrams.
# Memory Usage: 17.4 MB, less than 74.19% of Python3 online submissions for Group Anagrams.
class Solution:
    def groupAnagrams(self, strs):
        result = {}
        for s in strs:
            key = str(sorted(s))
            if key in result:
                result[key] += [s]
            else:
                result[key] = [s]
        return list(result.values())
