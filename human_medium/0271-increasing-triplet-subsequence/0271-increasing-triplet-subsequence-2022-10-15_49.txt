# TC= O(N),SC=O(1)
class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:
        m1 = m2=float(\'inf\')# INT_MAX(Values)
        for i,n in enumerate(nums):
            if m1<m2<n:
                return True
            elif n<m1: #1st MIN Value
                m1=n
            elif m1<n<m2: #2nd MIN Value
                m2=n
            
        return False
