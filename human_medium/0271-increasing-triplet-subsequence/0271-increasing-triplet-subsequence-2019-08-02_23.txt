class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:
        dp=[1]*len(nums)
        
        for j in range(len(nums)):
            for i in range(j):
                if nums[j]>nums[i]:
                    dp[j]=max(dp[i]+1,dp[j])
                    if dp[j]==3:
                        return True
        return False
