class Solution:
def increasingTriplet(self, nums: List[int]) -> bool:
        f_min = float(\'inf\')
        s_min = float(\'inf\')
        for i in range(len(nums)):
            if nums[i] <= f_min:
                f_min = nums[i]
            elif nums[i] <= s_min:
                s_min = nums[i]
            else:
                return True
return False
