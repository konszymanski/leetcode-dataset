class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:
        a=[0]
        mn=nums[0]
        for x in nums:
            mn=min(mn,x)
            a.append(mn)
        a.pop()
        b=[0]
        mx=nums[-1]
        for i in range(len(nums)-1,0,-1):
            mx=max(mx,nums[i])
            b.append(mx)
        b.reverse()
        for i in range(1,len(nums)-1):
            if(a[i]<nums[i]<b[i]):
                return True
        return False
