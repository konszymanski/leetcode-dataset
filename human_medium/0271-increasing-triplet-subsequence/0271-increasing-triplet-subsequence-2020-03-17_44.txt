class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:
        mins = list(itertools.accumulate(nums, min))
        maxs = list(itertools.accumulate(nums[::-1], max))
        c = 0
        for i, num in enumerate(nums):
            if i == 0 or i == len(nums) - 1:
                continue
            else:
                if mins[i] < num < maxs[i]:
                    return True
                
        return False
