class Solution:
    def increasingTriplet(self, s: List[int]) -> bool:
        
        # if length is less than 3, alway return False
        if len(s) < 3:
            return False
        
        # length greater than 3
        else:
            # current length of the maintaining increasing subsequence
            cur = 1
            
            # first item in the maintaining increasing subsequence
            smallest = s[0]
            
            middle = s[0]
            
            # every time the possible second item in the increasing subsequence is found,
            # keep the smallest among all possible
            # s.t once any value larger than ``found`` is reached, a 3-length increasing subsequence is found
            found = max(s)
            for i in range(1, len(s)):
                if s[i] > found:
                    return True
                if cur==1:
                    if s[i] <= smallest:
                        smallest = s[i]
                    else:
                        cur = 2
                        middle = s[i]
                        found = min(found, middle)
                elif cur == 2:
                    # a new smallest item 
                    # the variable ``found`` is handling the previous part
                    # from this item one, need to find whether there will be a middle
                    # then possibly updata ``found``
                    if s[i] <= smallest:
                        smallest = s[i]
                        cur = 1
                    elif s[i] <= middle:
                        found = min(found, s[i])
                    else:
                        return True        
            return False
