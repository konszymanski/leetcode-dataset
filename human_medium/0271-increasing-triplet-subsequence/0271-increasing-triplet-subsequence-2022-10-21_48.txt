import bisect
class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:
        size = len(nums)
        
        minP = nums[0]
        maxP = nums[0]
        count = 1
        
        lastMax = float("infinity")
        lastMaxValue = float("infinity")
        
        for i in range(1, len(nums)):
            
            if lastMax != float("infinity"):
                if lastMaxValue < nums[i] and lastMax == 2:
                    return True
            
            if nums[i] > maxP:
                count += 1
                maxP = nums[i]
                if count >= 3:
                    return True
                continue
            
            if nums[i] > minP and nums[i] < maxP:
                maxP = nums[i]
                continue
            
            if nums[i] < minP:
                lastMax = count
                lastMaxValue = maxP
                count = 1
                minP = nums[i]
                maxP = nums[i]
                
        return False
