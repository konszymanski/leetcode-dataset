class Solution:
    def increasingTriplet(self, nums: List[int]) -> bool:
        k=3
        res=[]
        for x in nums:
            i= bisect.bisect_left(res, x)
            if i==len(res):
                res.append(x)
            else:
                res[i]=x
                
            if len(res)==k:
                return True
            
        return False
