class Solution:
    def numSquares(self, n: int) -> int:
        # minComp is the list of summing component count for each integer from 0 to n.
        # We can start by assuming all int can be made of that exact numbers of 1, as I do below
        # or we can use float("inf"), whatever.
        minComp = [i for i in range(n+1)]
        # minComp = [float("inf") for i in range(n+1)]
        # minComp[0] = 0
        
        # Now set up a list of valid squares. Make sure all valid squares are not greater than n.
        sqroot = int(n**0.5)
        sqs = [i*i for i in range(1, sqroot+1)]
        
        # Start looping at idx-1 (minComp[0] can only be 0 by definition.)
        for thisInt in range(2, n+1):    # Leave 0 and 1, they are correct by default. (4rund3v)
            # Update minComp value with appropriate value by comparing...
            # -> What exists now (minComp[thisInt])
            # -> The mininum component of what makes up the difference between thisInt and sq
            # In all, let\'s consider the loop where thisInt is 12:
            # we are comparing for minimum count of...
            # -> (whatever-makes-11) + 1:       minComp[12-1]+1
            # -> (whatever-makes-8)  + 4:       minComp[12-4]+1
            # -> (whatever-makes-3)  + 9:       minComp[12-9]+1
            minComp[thisInt] = min([minComp[thisInt-sq]+1 for sq in sqs if thisInt>=sq])    # Consider only sq not greater than thisInt (4rund3v)
        return minComp[-1]
