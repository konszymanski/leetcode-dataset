let numSquares = n => {
    let go = (i = n) => {
        if (!i)
            return 0;
        let best = 123456789;
        for (let k = 1; k <= Math.floor(Math.sqrt(i)); ++k)
            if (0 <= i - k * k)
                best = Math.min(best, 1 + go(i - k * k));
        return best;
    };
    return go();
};
