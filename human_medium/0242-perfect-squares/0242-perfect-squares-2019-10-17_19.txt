def numSquares(self, n: int) -> int:
	squares = [i**2 for i in range(1, int(n**0.5)+1)]
	res, q, sub_q = 1, {n}, set()
	while q:
		for node in q:
			for square in squares:
				if node == square: return res
				if node < square: break
				sub_q.add(node-square)
		q, sub_q, res = sub_q, set(), res+1
	return res
