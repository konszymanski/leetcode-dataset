class Solution(object):
    def numSquares(self, n):
        """
        :type n: int
        :rtype: int
        """
        
        def bfs(n):
            
        
            q = collections.deque([n])
            
            lvl = 1
            while q:
                
                
                lq = len(q)
                for i in range(lq):
                    
                    k = q.popleft()
                    
					
					# loop though all possible perfect squares that we could have for this number
                    i = int(sqrt(k))+1
                    while i>=1:
                        
                        res = k - i**2
						# if we have 0 that means weve added perfect squares up to get n and because its BFS 
						# we know the first time we see zero itll be the sortest path to n
                        if res == 0:
                            return lvl
                        if res > 0:
                            q.append(res)
                        i-=1
                    
                lvl+=1
        return bfs(n)
