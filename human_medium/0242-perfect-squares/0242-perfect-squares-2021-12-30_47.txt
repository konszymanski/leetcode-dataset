class Solution:
    def numSquares(self, n: int) -> int:
        sq = []
        i = 1
        while i*i <= n:
            sq.append(i*i)
            i += 1
        
        dp = [sys.maxsize for i in range(n+1)]
        dp[0] = 0
        
        for s in sq:
            for i in range(1, n+1):
                if i-s >= 0:
                    dp[i] = min(dp[i], dp[i-s] + 1)
        
        return dp[n]
