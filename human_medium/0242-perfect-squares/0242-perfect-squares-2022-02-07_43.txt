class Solution:
    def dp(self, n, dp):
        if n in dp:
            return dp[n]

        if n < 4:
            return n

        ans = float(\'inf\')
        for i in range(1, int(sqrt(n)) + 1):
            ans = min(ans, 1 + self.dp(n - (i * i), dp))

        dp[n] = ans
        return ans

    def numSquares(self, n):
        dp = {}
        return self.dp(n, dp)
