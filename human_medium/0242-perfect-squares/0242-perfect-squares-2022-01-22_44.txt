class Solution:
    def numSquares(self, n: int) -> int:
        
        # try combination of numbers (get greedy first)
        cache = {}
        
        def backtracking(currSum):
            
            if currSum in cache: return cache[currSum]
            
            # print(currSum, currCount)
            if currSum == 0:
                return 0
            
            elif currSum < 0: return math.inf
            
            tempCount = math.inf
            currNum = 1
            while currNum * currNum <= currSum:
                tempCount = min(tempCount, 1 + backtracking(currSum - (currNum*currNum)))
                currNum += 1
            
            cache[currSum] = tempCount
            
            return tempCount
        
        return backtracking(n)
