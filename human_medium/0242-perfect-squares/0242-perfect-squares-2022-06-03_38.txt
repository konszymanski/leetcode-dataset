class Solution:
    def numSquares(self, n: int) -> int:
        def solve(n,i,k,a):
            if n==0:
                return min(k,a)
            if i<=0 or k>=a:
                return a
            if (i**2)<=n:
                a=solve(n-(i**2),i,k+1,a)
            a=solve(n,i-1,k,a)
            return a
        
        return solve(n,int(n**0.5),0,1001)
