# IDEA: define DP element as maximum subarray that ends on this number, then the next DP element is largest out of sum of
#       previous DP and next number or next number itself:   dp[i] = max(dp[i-1] + n[i], n[i])
#       loop array and store only last DP value, keep maxSum value as result
#
class Solution:
    def maxSubArray(self, nums: List[int]) -> int:
        dp = 0
        maxSum = None
        for n in nums:
            dp = max(dp+n, n)
            if maxSum is None or dp > maxSum:
                maxSum = dp
        return maxSum
