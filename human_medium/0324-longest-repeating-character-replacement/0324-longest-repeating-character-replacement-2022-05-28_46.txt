def characterReplacement(self, s: str, k: int) -> int:
        count = defaultdict(int)
        l = 0
        maxcount = 0
        ans = 0
        for r in range(len(s)):
            count[s[r]] += 1
            maxcount = max(count[s[r]], maxcount)
            while (r-l+1) - maxcount > k:
                count[s[l]] -= 1
                l += 1
            ans = max(r-l+1, ans)
        return ans
