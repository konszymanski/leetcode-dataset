class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        res,l = 0, 0
        
        hashmap = {}
        for r in range(len(s)):
            hashmap[s[r]] = 1 + hashmap.get(s[r], 0)
			maxOccur = max(hashmap.values())
    
            while (r-l+1) - maxOccur > k:
                hashmap[s[l]] -= 1
                l += 1
                
            res = max(res, r-l+1)
            
        return res
