class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        result = 0
        counts = defaultdict(int)
        start = 0
        for end in range(len(s)):
            counts[s[end]] += 1
            while end - start + 1 - max(counts.values()) > k:
                counts[s[start]] -= 1
                start += 1
            result = max(result, end - start + 1)
            
        return result
