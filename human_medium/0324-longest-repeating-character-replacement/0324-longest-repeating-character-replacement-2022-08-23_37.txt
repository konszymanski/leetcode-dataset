def characterReplacement(self, s: str, k: int) -> int:
	count = defaultdict(int)         
	p1 = max_size = 0
	
	for p2 in range(len(s)):
		# add char count at each iterration 
		count[s[p2]] += 1 
		
		while p2 > p1 and p2 - p1 - max(count.values()) > k:
			# remove char count if scenario is incorrect, i.e. char in dict are more than what k can replace
			count[s[p1]] -= 1  
			p1 += 1
			
		# if size of window is greater than max_size
		max_size = max(p2 - p1, max_size)
	return max_size
