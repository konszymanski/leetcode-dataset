class Solution:
    def characterReplacement(self, string, k):
        left, right = 0, 0
        frequencyOfChars, mostCommonElementCount = {}, 0
        for right in range(len(string)):
            frequencyOfChars[string[right]] = frequencyOfChars.get(string[right], 0) + 1
            mostCommonElementCount = max(mostCommonElementCount, frequencyOfChars[string[right]])
            windowSize = right - left + 1
            if windowSize - mostCommonElementCount > k:
                frequencyOfChars[string[left]] -= 1
                left += 1
        return right - left + 1
