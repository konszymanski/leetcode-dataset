class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        count = {}
        res = 0
        
        lt = 0
        for rt in range(len(s)):
            # Update the frequency count for the character
            count[s[rt]] = count.get(s[rt], 0) + 1
            
            window_size = rt - lt + 1
            if window_size - max(count.values()) <= k:
                # Update the res using the current window size
                # but only if the window is valid based on the 
                # number of available replacements, k
                res = max(res, window_size)
            else:
                # Decrement the frequency count for the character
                # we\'ll be removing from the window
                count[s[lt]] -= 1
                lt += 1
                
        return res
