class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        """
        Sliding window approach
        Time complexity: O(N)
        Space complexity: O(1)
        """
        res = 0
        l = 0 # left pointer
        count = dict()
        # instead of using max(count.values()) we can only store the maximum frequency of the character
        max_freq = 0
        n = len(s)
        for r, char in enumerate(s): # right pointer
            if r == n:
                # if the right pointer has reached the end we can stop moving the left pointer
                return res
            count[char] = 1 + count.get(char, 0)
            max_freq = max(max_freq, count[char])
            if (r - l + 1) - max_freq <= k: # compute the replacement cost
                res = max(res, r - l + 1)
            else:
                count[s[l]] -= 1 
                l += 1
        return res
