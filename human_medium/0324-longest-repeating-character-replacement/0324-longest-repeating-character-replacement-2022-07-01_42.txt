class Solution:
    def characterReplacement(self, s: str, k: int) -> int:
        freq={}
        for i in s:
            if i in freq:
                continue
            else:
                freq[i]=0
        i=0
        j=0
        res=0
        while j<len(s):
            windowsize=(j-i)+1  
            freq[s[j]]+=1
            maxi=max(list(freq.values()))  # O(26) for calculating maximum frequency
            if windowsize-maxi<=k:    #Checking if the window is valid or not.
                j+=1
                res=max(res,windowsize)
            else:
                freq[s[i]]-=1   #if the window is not valid increase the left and right pointer.
                i+=1
                j+=1
        return res
