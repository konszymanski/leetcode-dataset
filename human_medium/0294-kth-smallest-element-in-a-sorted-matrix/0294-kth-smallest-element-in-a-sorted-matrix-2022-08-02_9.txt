class Solution:
    def kthSmallest(self, matrix, k: int) -> int:
        min_heap = []
        for row in range(len(matrix)):
            for col in range(len(matrix[0])):
                heapq.heappush(min_heap, matrix[row][col])
        for _ in range(k-1):
            heapq.heappop(min_heap)
        
        return min_heap[0]
