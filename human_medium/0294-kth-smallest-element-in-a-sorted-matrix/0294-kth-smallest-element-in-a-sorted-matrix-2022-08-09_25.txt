# space: O(n) for heap
        # time: O(k log n)
        n = len(matrix)
        h = [(v, 0, col) for col, v in enumerate(matrix[0])]
        for i in range(k - 1):
            val, row, col = heappop(h)
            row += 1
            if row < n:
                heappush(h, (matrix[row][col], row, col))
        return heappop(h)[0]
