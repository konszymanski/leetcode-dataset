import heapq
class Solution:
    def kthSmallest(self, matrix: List[List[int]], k: int) -> int:
        
        res = sum(matrix,[]) # this will convert 2d arr to 1d arr with O(N) time \uD83D\uDE32\uD83D\uDE32
        
        heapq.heapify(res)
        
        for i in range(k-1):
            heapq.heappop(res)
        
        return heapq.heappop(res)
