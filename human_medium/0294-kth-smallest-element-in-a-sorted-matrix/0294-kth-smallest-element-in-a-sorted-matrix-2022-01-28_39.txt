# this approach has a pointer in every row that keeps incrementing until it reaches the last index or until k is reached.
# if matrix has n rows our space complexity is O(n)
# since our loop runs kxn times time complexity is O(kn)
class Solution:
    def kthSmallest(self, matrix: List[List[int]], k: int) -> int:
        ref = [1]*len(matrix)
        ctr=1
        while ctr<=k:
            s,s_i = math.inf,0
            for i in range(len(ref)):
                if ref[i]<=len(matrix[0]) and matrix[i][ref[i]-1]<s:
                    s,s_i = matrix[i][ref[i]-1],i
            if ctr<k:
                ref[s_i]+=1 
                ctr+=1
            else:
                return matrix[s_i][ref[s_i]-1]
