# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        mi=min(p.val,q.val) #selecting min value
        ma=max(p.val,q.val) #selecting max value 
        #now we will compare val of nodes in bst to search which is ancestor
        while root!=None: #till root is null
            if root.val<mi: #means the mi value is present in right subtree
                root=root.right 
            elif root.val>ma: #means the ma value is present in left subtree
                root=root.left 
            else: #means the mi value is present in between p and q nodes
                return root 
        return None #if no ancestor then return none
