class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        # if one of the values is the root itself, 
        # or if p and q are at the different subtrees (left and right) with respect to the root, 
        # we are done: we have found the LCA - which is the root itself:
        if (p.val == root.val) or (q.val == root.val) or (p.val < root.val and root.val < q.val) or (q.val < root.val and root.val < p.val):
            return root
