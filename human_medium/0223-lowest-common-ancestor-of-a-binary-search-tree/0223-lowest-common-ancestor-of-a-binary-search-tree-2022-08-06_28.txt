# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        def f(root,p,q):
            if root:
                if p<=root.val<=q:
                    return root
                if q<root.val:
                    return f(root.left,p,q)
                return f(root.right,p,q)
            
        return f(root,min(p.val,q.val),max(p.val,q.val))
