# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        node=root
        while node: #to run infinite loop since node will always exist
            if p.val>node.val and q.val>node.val:
                node=node.right # going towards right subtree
            elif p.val<node.val and q.val<node.val:
                node=node.left # going towards left subtree
            else:
                return node  # when one node is in right subtree and another in left subtree, this guarantees us that root is the lowest common ancestor
