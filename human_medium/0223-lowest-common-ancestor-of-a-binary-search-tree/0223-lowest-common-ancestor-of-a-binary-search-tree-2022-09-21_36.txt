# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None


class Solution:
    def lowestCommonAncestor(
        self, root: "TreeNode", p: "TreeNode", q: "TreeNode"
    ) -> "TreeNode":
        if root is p or root is q or root is None:
            return root

        lsubtree = self.lowestCommonAncestor(root.left, p, q)
        rsubtree = self.lowestCommonAncestor(root.right, p, q)

        if lsubtree and rsubtree:
            return root  # This is the LCA.

        return lsubtree or rsubtree
