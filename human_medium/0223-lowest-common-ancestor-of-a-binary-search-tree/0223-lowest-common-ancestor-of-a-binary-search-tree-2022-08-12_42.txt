class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        p_val, q_val = p.val, q.val
        while root:
            val = root.val
            if val > max(p_val, q_val):
                root = root.left
            elif val < min(p_val, q_val):
                root = root.right
            else:
                return root
