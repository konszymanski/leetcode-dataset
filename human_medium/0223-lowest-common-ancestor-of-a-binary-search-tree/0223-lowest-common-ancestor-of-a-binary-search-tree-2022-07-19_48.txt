class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        node = root
        if p.val>q.val:
            p,q=q,p
        while True:
            if p.val<=node.val<=q.val:
                return node
            if p.val>node.val:
                node=node.right
            else:
                node=node.left
