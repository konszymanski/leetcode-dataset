# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        if not root: return
        # if root is greater than both p and q move left
        if root.val>p.val and root.val>q.val :return self.lowestCommonAncestor(root.left,p,q)
        # else if the root is smaller than both p and q, move right 
        elif root.val<p.val and root.val<q.val :return self.lowestCommonAncestor(root.right,p,q)
        # else root in the LCA
        else: return root
