class Solution(object):
    def lowestCommonAncestor(self, root, p, q):
        """
        :type root: TreeNode
        :type p: TreeNode
        :type q: TreeNode
        :rtype: TreeNode
        """
        node = root
        while node:
            
            if node.val<p.val and node.val<q.val:
                node = node.right
            elif node.val>p.val and node.val>q.val:
                node = node.left
            else:
                return node
