# Memory Usage: 18.8 MB, less than 29.53% of Python3 online submissions for Lowest Common Ancestor of a Binary Search Tree.

class Solution:
    def lowestCommonAncestor(self, root, p, q):
        minNode, maxNode = p, q
        if p.val>q.val:
            minNode, maxNode = q, p
        while root!=None:
            if minNode.val<root.val<maxNode.val or minNode.val==root.val or maxNode.val==root.val:
                return root
            elif maxNode.val<root.val:
                root = root.left
            elif root.val<minNode.val:
                root = root.right
