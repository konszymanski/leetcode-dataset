# 1 Recursive
    def lowestCommonAncestor(self, root: TreeNode, p: TreeNode, q: TreeNode) -> TreeNode:

        def lca(node):
            value = node.val
            if left <= value <= right:
                return node
            elif right < value:
                lca(node.left)
            elif left > value:
                lca(node.right)

        left = min(p.val, q.val)
        right = max(p.val, q.val)
        return lca(root)

    # 2 Iterative
    def lowestCommonAncestor(self, root: TreeNode, p: TreeNode, q: TreeNode) -> TreeNode:

        left = min(p.val, q.val)
        right = max(p.val, q.val)

        while root:
            value = root.val
            if left <= value <= right:
                return root
            elif right < value:
                root = root.left
            elif left > value:
                root = root.right
