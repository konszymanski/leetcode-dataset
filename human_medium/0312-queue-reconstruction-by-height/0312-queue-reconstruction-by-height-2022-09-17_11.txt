class Solution:
    def reconstructQueue(self, people: List[List[int]]) -> List[List[int]]:
        ans = [None] * len(people)
        people.sort()
        
        for h, k in people:
            i = 0
            count = 0
            
            while count <= k:
                if ans[i] is None and count == k:
                    ans[i] = (h, k)
                    break
                    
                if ans[i] is None or ans[i][0] >= h:
                    count += 1
                i += 1
        
        return ans
