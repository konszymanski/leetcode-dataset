class Solution:
    def reconstructQueue(self, people: List[List[int]]) -> List[List[int]]:
        
        people.sort(key=lambda x: (x[0], -x[1]), reverse=True)
        n = len(people)
        for i in range(1, n):
            cur, k = i, people[i][1]
            while cur > k:
                people[cur], people[cur-1] = people[cur-1], people[cur]
                cur -= 1
        return people
