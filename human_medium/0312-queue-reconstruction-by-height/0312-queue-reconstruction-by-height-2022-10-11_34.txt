class Solution:
    def reconstructQueue(self, people: List[List[int]]) -> List[List[int]]:
        inds = list(range(len(people)))
        temp, prev = [], None
        for p in sorted(people):
            if p[0] != prev:
                prev = p[0]
                for i in temp:
                    inds.remove(i)
                temp.clear()
            people[inds[p[1]]] = p
            temp.append(inds[p[1]])
        return people
