class Solution(object):
    def reconstructQueue(self, people):
        """
        :type people: List[List[int]]
        :rtype: List[List[int]]
        """
        people.sort(key = lambda x: (x[0], -x[1]), reverse = True)
        output = []
        for i in range(len(people)):
            output.insert(people[i][1], people[i])
        return output
