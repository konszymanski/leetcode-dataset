class Solution:
    def reconstructQueue(self, people: List[List[int]]) -> List[List[int]]:
        n = len(people)
		#sorting the list on the basis of height in decending order and if there is same height scenario
		#we will sort by minimum value of k, i.e.[h, k]
        people = sorted(people, key=lambda x: (x[0], -x[1]), reverse = True)
        
        ans = []
        for l in people:
            ans.insert(l[1], l) 
        return ans
