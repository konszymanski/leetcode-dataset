class Solution:
    def reconstructQueue(self, people: List[List[int]]) -> List[List[int]]:
        people.sort(key=lambda x: (-x[0], x[1]))
        ans = []
        for person, num in people:
            if num >= len(ans):
                ans.append([person, num])
            else:
                ans.insert(num, [person, num])
        return ans
