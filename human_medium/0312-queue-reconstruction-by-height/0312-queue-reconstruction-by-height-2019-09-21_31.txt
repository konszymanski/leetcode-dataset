class Solution(object):
    def reconstructQueue(self, people):
        """
        :type people: List[List[int]]
        :rtype: List[List[int]]
        """
        people.sort(key=lambda x:(-x[0],x[1]))
        res=[]
        print(people)
        for p in people:
            print(p)
            res.insert(p[1],p)
            print(res)
        return res
