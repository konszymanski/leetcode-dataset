class Solution:
    def reconstructQueue(self, people: List[List[int]]) -> List[List[int]]:
        people.sort()
        N=len(people)
        ans=[None]*N
        for height,q in people:
            i,j=0,-1
            while i<N:
                if not ans[i] or ans[i][0]==height:
                    j+=1
                if j==q:
                    break
                i+=1
            ans[i]=[height,q]
        return ans
