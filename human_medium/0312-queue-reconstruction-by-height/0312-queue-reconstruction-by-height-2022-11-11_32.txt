class Solution(object):
    def reconstructQueue(self, people):
        """
        :type people: List[List[int]]
        :rtype: List[List[int]]
        
        """
        output=[];
        people.sort(key = lambda  num : (-num[0] , num[1]));
        
        for a in people:
            output.insert(a[1], a)
        return output
