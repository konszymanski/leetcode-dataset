class Solution(object):
    def reconstructQueue(self, people):
        
		#AA#> Sort heights in descending order and then
		#AA#> sort number of people for each height (i.e. index) in ascending order
        people.sort(key = lambda x: (-x[0],x[1])) 
        output = []
        
        for ph in people:
                output.insert(ph[1],ph)

        return output
