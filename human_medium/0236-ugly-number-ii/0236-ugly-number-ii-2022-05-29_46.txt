class Solution:
    def nthUglyNumber(self, n: int) -> int:
        ans = [1]
        u2, u3, u5 = 0,0,0
        
        while len(ans) < n:
            temp = min(2*ans[u2], 3*ans[u3], 5*ans[u5])
            ans.append(temp)
            
            if temp == 2*ans[u2]:
                u2+=1
            
            if temp == 3*ans[u3]:
                u3 +=1
                
            if temp == 5*ans[u5]:
                u5+=1
            
        return ans[-1]
