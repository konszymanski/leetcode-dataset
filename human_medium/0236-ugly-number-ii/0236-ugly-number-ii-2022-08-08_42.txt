class Solution:
    def nthUglyNumber(self, n: int) -> int:
        Ugly=[1]*n
        a2,a3,a5=0,0,0
        for i in range(1,n):
            Ugly[i]=min(2*Ugly[a2],3*Ugly[a3],5*Ugly[a5])
            while a2<i and 2*Ugly[a2]<=Ugly[i]:
                a2+=1
            while a3<i and 3*Ugly[a3]<=Ugly[i]:
                a3+=1
            while a5<i and 5*Ugly[a5]<=Ugly[i]:
                a5+=1
        return Ugly[-1]
