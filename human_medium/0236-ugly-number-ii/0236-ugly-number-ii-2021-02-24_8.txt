heap=[1]
ans=[]
seen=set()
while len(ans)<n:
	min_element = heapq.heappop(heap)
	ans.append(min_element)
	a,b,c = 2*min_element, 3*min_element, 5*min_element
	if a not in seen:
		seen.add(a)
		heapq.heappush(heap,a)
	if b not in seen:
		seen.add(b)
		heapq.heappush(heap,b)
	if c not in seen:
		seen.add(c)
		heapq.heappush(heap,c)

return ans[n-1]
