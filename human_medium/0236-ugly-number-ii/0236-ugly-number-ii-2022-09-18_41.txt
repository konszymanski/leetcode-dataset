class Solution:
    def nthUglyNumber(self, n: int) -> int:
        primeFactor = [2,3,5]
        heap = [1]
        uglySet = set()
        j = 1
        while j < n:
            for i in primeFactor:
                curMin = heap[0]
                if curMin*i not in uglySet:
                    uglySet.add(curMin*i)
                    heapq.heappush(heap, curMin*i)
            heapq.heappop(heap)
            j += 1
        return heap[0]
