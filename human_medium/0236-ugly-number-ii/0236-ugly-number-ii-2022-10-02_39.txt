class Solution:
    def nthUglyNumber(self, n: int) -> int:
        if n == 0:
            return 1
        res = [1] + [0]*(n-1)
        i2 = i3 = i5 = 0

        for i in range(1, n):
            nxt = min(res[i2]*2, res[i3]*3, res[i5]*5)
            if res[i2] * 2 == nxt:
                i2 += 1
            if res[i3]*3 == nxt:
                i3 += 1
            if res[i5]*5 == nxt:
                i5 += 1
            res[i] = nxt
        
        return res[-1]
