class Solution:
    def nthUglyNumber(self, n: int) -> int:
        res=[1]
        two_p=0
        three_p=0
        five_p=0
        while(len(res)<n):
            by2=res[two_p]*2
            by3=res[three_p]*3
            by5=res[five_p]*5
            mn=min(by2,by3,by5)
            res.append(mn)
            if(by2==mn):
                two_p+=1
            if(by3==mn):
                three_p+=1
            if(by5==mn):
                five_p+=1
        return res[-1]
