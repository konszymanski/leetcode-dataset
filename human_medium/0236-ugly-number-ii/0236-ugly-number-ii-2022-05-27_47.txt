import heapq
class Solution:
    def nthUglyNumber(self, n: int) -> int:
        heap = [1]
        visited = set()
        visited.add(1)
        
        for _ in range(1, n):
            x = heapq.heappop(heap)
            for dx in [2, 3, 5]:
                nx = x * dx
                if nx in visited:
                    continue
                heapq.heappush(heap, nx)
                visited.add(nx)

        return heap[0]
