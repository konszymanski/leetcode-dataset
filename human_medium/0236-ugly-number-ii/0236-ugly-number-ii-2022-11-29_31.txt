class Solution:
    def nthUglyNumber(self, n: int) -> int:
        # populate the minHeap with multiples of prime factors
        # And heappop n-1 times which
        # leaves the nth ugly number at the top of the minHeap
        
		# Min Heap will store numbers that are factors of our prime numbers [2,3,5]
        minHeap = [1] 
        seen = set([1]) # so we dont recalculate numbers already in our minHeap
        prime = (2,3,5)
        
        for i in range(n-1):  # for n-1 keep popping until we are at our nth number
            ugly = heapq.heappop(minHeap)
            for factor in prime:
                product = ugly * factor
                if product not in seen:
                    seen.add(product)
                    heapq.heappush(minHeap, product)
        return minHeap[0]
