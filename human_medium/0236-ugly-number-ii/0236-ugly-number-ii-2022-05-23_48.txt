class Solution:
    def nthUglyNumber(self, n: int) -> int:
        d = [1]
        i2,i3,i5 = 0,0,0
        c = 1
        if n==1:
            return 1
        while c < n:
            d.append(min(d[i2]*2,d[i3]*3,d[i5]*5))
            c += 1
            if d[-1] == d[i2]*2:
                i2 += 1
            if d[-1] == d[i3]*3:
                i3 += 1
            if d[-1] == d[i5]*5:
                i5 += 1
        return d[n-1]
