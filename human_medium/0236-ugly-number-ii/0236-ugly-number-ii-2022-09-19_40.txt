class Solution:
    def nthUglyNumber(self, n: int) -> int:
        if n == 1:
            return 1
        i=1
        j = 1
        k=1
        l=[1]
        t,e,f=1,1,1
        for _ in range(n-1):
            a = min(i*2,j*3,k*5)
            l.append(a)
            if a == i*2:
                i=l[t]
                t+=1
            if a == j*3:
                j=l[e]
                e+=1
            if a == k*5:
                k=l[f]
                f+=1

        return a
