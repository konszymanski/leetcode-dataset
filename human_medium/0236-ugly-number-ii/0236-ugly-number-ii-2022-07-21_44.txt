class Solution:
    def nthUglyNumber(self, n: int) -> int:
        c2 = 1
        c3 = 1
        c5 = 1
        dp = [0] * (n + 1)
        dp[1] = 1
        for i in range(2, n + 1):
            dp[i] = min(2 * dp[c2], 3 * dp[c3], 5 *dp[c5])
            
            if dp[i] == 2 * dp[c2]:
                c2+=1
            if dp[i] == 3 * dp[c3]:
                c3+=1
            if dp[i] == 5 * dp[c5]:
                c5+=1
        
        
        return dp[n]
