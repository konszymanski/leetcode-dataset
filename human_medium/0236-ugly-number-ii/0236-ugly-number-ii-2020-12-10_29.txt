from heapq import *
class Solution:
    def nthUglyNumber(self,n):
        seen = {1}
        self.nums = []
        heap = []
        heappush(heap, 1)

        for _ in range(n):
            curr_ugly = heappop(heap)
            self.nums.append(curr_ugly)
            for i in [2, 3, 5]:
                new_ugly = curr_ugly * i
                if new_ugly not in seen:
                    seen.add(new_ugly)
                    heappush(heap, new_ugly)
        
        return self.nums[-1]
