# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def rotateRight(self, head: Optional[ListNode], k: int) -> Optional[ListNode]:
        if head is None or head.next is None:
            return head
        lst=[]
        itr=head
        while itr:
            lst.append(itr.val)
            itr=itr.next
        k=k%len(lst)
        for i in range(k):
            itr=head
            while itr.next:
                prev=itr
                itr=itr.next
            prev.next=None
            itr.next=head
            head=itr
        return head
