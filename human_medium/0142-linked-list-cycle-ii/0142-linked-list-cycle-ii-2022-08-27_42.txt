# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def detectCycle(self, head: Optional[ListNode]) -> Optional[ListNode]:
        s=set()        ## using set for constant time access
        start=head
        while start:
            if start not in s:     ######if node already exists in set this means there is a cycle(\'pointer back to visited node\')
                s.add(start)
            else:
                return start
            start=start.next
