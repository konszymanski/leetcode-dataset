def topKFrequent(self, nums: List[int], k: int) -> List[int]: #I\'ll try O(n+klogk)
        count = collections.Counter(nums)
        heap = []
        for num in count:
            if len(heap) < k:
                heapq.heappush(heap,(count[num],num))
            elif count[num] > heap[0][0]:
                heapq.heapreplace(heap,(count[num],num))
        
        result = []
        heap.sort()
        heap.reverse()
        for item in heap:
            result.append(item[1])
        return result
