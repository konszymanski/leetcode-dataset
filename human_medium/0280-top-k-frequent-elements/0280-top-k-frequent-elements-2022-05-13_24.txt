class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        
        mapping = collections.defaultdict(int)
        for num in nums:
            mapping[num] += 1
            
        heap = []
        for num in mapping:
            heapq.heappush(heap, (mapping[num], num))
            if len(heap) > k:
                heapq.heappop(heap)
    
        res = []
        for frequence, num in heap:
            res.append(num)
        return res
