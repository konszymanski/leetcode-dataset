class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        stats = {}
        for i in nums:
            if i not in stats:
                stats[i] = 1
            else:
                stats[i] += 1
        
        buckets = [[] for i in range(len(nums)+1)]
        for i, j in stats.items():
            buckets[j].append(i)
        
        buckets = [j for i in buckets for j in i]
        
        return buckets[::-1][:k]
