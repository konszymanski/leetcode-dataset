def topKFrequent(self, words: List[str], k: int) -> List[str]:
        
        FreqMap = {}
        final_ans = []
        
        # sorting strings lexico.... ( alphabetical sorting condition would be satisfied by this )
        words.sort()

        # making word : frequency map
        for i in words:
            if i in FreqMap:
                FreqMap[i] += 1
            else:
                FreqMap[i] = 1
        
        #  Sorting keys on the basis of increasing frequency
        ans = sorted(FreqMap.items(), key=lambda x: (-x[1], x[0]))

        
        # getting top k words by iterating through the list up to k
        for i in range(k):
            final_ans.append(ans[i][0])
        return(final_ans)
