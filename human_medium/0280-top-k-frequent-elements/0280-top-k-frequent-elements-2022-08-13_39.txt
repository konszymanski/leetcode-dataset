class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        results = []
        lookup = {}
        for num in nums:
            lookup[num] = lookup[num] + 1 if num in lookup else 1

        sortedNums = sorted(lookup.items(), key=lambda x: x[1], reverse=True)
        for idx in range(0, k):
            results.append(sortedNums[idx][0])
        return results
