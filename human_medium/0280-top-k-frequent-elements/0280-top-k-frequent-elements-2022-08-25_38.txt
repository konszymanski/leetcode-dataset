class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        
        hashmap = {}
        output = []
        
        for num in nums:
            if num in hashmap:
                hashmap[num] += 1
            else:
                hashmap[num] = 1

        target = sorted(hashmap.values(), reverse = True)[k - 1]
        
        for num in hashmap:
            if hashmap[num] >= target:
                output.append(num)
        
        return output
