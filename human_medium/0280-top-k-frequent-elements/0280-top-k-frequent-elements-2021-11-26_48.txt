class Solution(object):
    def topKFrequent(self, nums, k):
        count = {}
        res = []
        reps = []
        
        for i in range(len(nums) + 1):
            reps.append([])
        
        # use hashmap to find the frequency of all numbers
        for n in nums:
            count[n] = 1 + count.get(n, 0)
        
        # use indices of array to identify the max frequency of a number
        # and append the number itself to its index served as its frequency
        # key is number, value is the sum of repetitions
        for key, value in count.items():
            reps[value].append(key)
        
        # start iteration from the end of the reps list
        # because the last element has largest reps count
		# and stop iteration when result\'s length is k
        for i in range(len(reps) - 1, 0, -1):
            for n in reps[i]:
                res.append(n)
                if len(res) == k:
                    return res
