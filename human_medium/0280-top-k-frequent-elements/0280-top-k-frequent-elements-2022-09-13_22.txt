import heapq
class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        a = []
        d = Counter(nums)
        res = []
        s =set()
        for i in nums:
            if i not in s:
                heapq.heappush(a,(d[i],i))          # pushing frequnecy of element and element. heap will sort based on frequency 
                s.add(i)                                     #since traversing over the array element,  we dont want element to come again in heap
            if len(a)>k:
                heapq.heappop(a)
                

        for i in range(len(a)):
            res.append(a[i][1])
        return res
