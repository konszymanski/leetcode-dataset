class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        # T: O(n log k), S: O(k)
        count = Counter(nums)
        min_heap = []
        for key, val in count.items():
            heappush(min_heap, (val, key))
            if len(min_heap) > k:
                heappop(min_heap)
        return [tup[1] for tup in min_heap]
