def topKFrequent(self, nums: List[int], k: int) -> List[int]:
	minHeap = []
	count = {}
	# make a hash map
	for num in nums:
		count[num] = count.get(num,0) + 1

	# convert dict to list((key,value))
	count = zip(count.keys(),count.values())
	count = list(count)

	# add first k elements to min heap
	for num, cnt in count[:k]:
		heappush(minHeap,[cnt,num])

	# add new element to heap at the same time remove smallest element in heap thus always maintaining only k elements in heap => O(N*log(K))
	for num, cnt in count[k:]:
		if len(minHeap) > k:
			heappop(minHeap)
		minCount = minHeap[0][0]
		# no point in adding to min heap if cnt is already less than minimum count in min heap
		if cnt > minCount: 
			heappush(minHeap,[cnt,num])

	# extract only elements, neglect count
	result = [val[1] for val in minHeap[:k]]
	return result
