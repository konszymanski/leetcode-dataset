class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        
        # To make sure that complexity is less than O(n*logn)
        if k == len(nums):
            return nums
        
        # Create a hashmap and add the freq of num
        hashMap = {}
        for num in nums:
            hashMap[num] = hashMap.get(num, 0)+1
        
        # Create a heap with value and key. As python support min heap we will include an \'-\' infront of the value (freq) to techincally make it work like a max heap.
        
        hp = []
        for key, value in hashMap.items():
            hp.append((-value, key))
            
        heapq.heapify(hp)
        
        # Pop the k element out of heap.
        res = []
        for _ in range(k):
            res.append(heapq.heappop(hp)[1])
            
        return res
