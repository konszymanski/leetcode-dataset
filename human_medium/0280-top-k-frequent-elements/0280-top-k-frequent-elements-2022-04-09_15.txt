class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]: 
        minHeap = [] 
        res = [] 
        for key, val in Counter(nums).items(): 
            heapq.heappush(minHeap,(val, key)) 
            if len(minHeap)>k: 
                heapq.heappop(minHeap) 
        while minHeap: 
            res.append(heapq.heappop(minHeap)[1]) 
        return res
