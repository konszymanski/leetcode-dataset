import heapq
class Solution:
    def topKFrequent(self, nums: List[int], k: int) -> List[int]:
        c = collections.Counter(nums)
        output = []
        #print(c)
        for v,f in c.items():
            #print(output)
            if len(output) < k:
                heapq.heappush(output,[f,v])
            else:
                if f > output[0][0]:
                    heapq.heappop(output)
                    heapq.heappush(output,[f,v])
        #print(output)
        res = []
        for i in range(k):
            res.append(heapq.heappop(output)[1])  
        return res[::-1]
