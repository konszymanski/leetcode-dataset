class Solution(object):
    def lowestCommonAncestor(self, root, p, q):
        """
        :type root: TreeNode
        :type p: TreeNode
        :type q: TreeNode
        :rtype: TreeNode
        """
        if root == p or root == q or root is None:
            return root
        n1 = self.lowestCommonAncestor(root.left, p, q)
        n2 = self.lowestCommonAncestor(root.right, p, q)
        if n1 and n2:
            return root
        else:
            return n1 or n2
