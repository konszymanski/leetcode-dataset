class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        if not root:
            return None
        \'\'\'
		If root itself is p or q, then do not need to find if the subtrees contain the other node. 
		If the other node is not in the subtree, then it is reasonable to return root; 
		if it is in the subtree, then the root becomes the LCA, still returns root. 
		\'\'\'
        if root.val == p.val or root.val == q.val:
            return root 
        leftnode = self.lowestCommonAncestor(root.left, p, q)
        rightnode = self.lowestCommonAncestor(root.right, p, q)
        if leftnode and rightnode: # p and q are from decendants
            return root
        return leftnode if leftnode else rightnode
    \'\'\'
    If both of them are None, then returning one of them is enough,
    does not need extra if statement to confirm.
    \'\'\'
