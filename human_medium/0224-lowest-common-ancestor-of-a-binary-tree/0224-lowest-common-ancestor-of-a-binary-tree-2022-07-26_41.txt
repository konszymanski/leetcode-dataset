class Solution:
    def lowestCommonAncestor(self, root: \'TreeNode\', p: \'TreeNode\', q: \'TreeNode\') -> \'TreeNode\':
        def backtrackMe(node,p,q):
            if node==p or node==q or node==None:
                return node
            left=backtrackMe(node.left,p,q)
            right=backtrackMe(node.right,p,q)
            if left==None:
                return right
            elif right==None:
                return left
            else:
                return node
        return backtrackMe(root,p,q)
