class Solution:
    def canPlaceFlowers(self, flowerbed: List[int], n: int) -> bool:
        # edge cases:
        if n == 0: return True
        if len(flowerbed) == 0: return False
        if len(flowerbed) == 1: return flowerbed[0] == 0
            
        # add empty plots in the beginning and end:
        flowerbed.insert(0, 0)
        flowerbed.append(0)
        
        # check if there are three empty plots in a row:
        i=0
        while i < len(flowerbed)-2 and n > 0:
            if flowerbed[i] == 0 and flowerbed[i+1] == 0 and flowerbed[i+2] == 0:
                flowerbed[i+1] = 1
                n-=1
            i+=1
            
        return n == 0
