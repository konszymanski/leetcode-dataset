def sumRootToLeaf(self, root: Optional[TreeNode]) -> int:
        
        stack = []
        def dfs(root, s):
            
            if not root:
                return
            each_val = str(root.val)
            if not root.left and not root.right:
                stack.append(int(s+each_val, 2))
                return
            
            dfs(root.left, s+each_val) 
            
            dfs(root.right, s+each_val) 

        dfs(root, "")
        return(sum(stack))
