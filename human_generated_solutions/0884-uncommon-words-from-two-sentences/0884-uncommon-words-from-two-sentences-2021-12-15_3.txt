class Solution {
public:
    vector<string> uncommonFromSentences(string s1, string s2) {
        stringstream ss(s1 + " " + s2);        
        unordered_map<string, int> m;
        for(string word;ss>>word;)  ++m[word];
        vector<string> ans;
        for(auto& [k, v]: m)
            if(v == 1) ans.emplace_back(k);
        return ans;
    }
};
