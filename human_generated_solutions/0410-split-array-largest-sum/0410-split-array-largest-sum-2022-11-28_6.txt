class Solution:
    def splitArray(self, nums: List[int], k: int) -> int:
        
        
        @cache
        def dp(ci, remains): # returns, from ci onwards, min possible sum 
            
            if remains == 1:
                return sum(nums[ci:])
            
            m = inf
            curr_sum = 0
            for i in range(ci, len(nums)):
                curr_sum += nums[i]
                temp = max(curr_sum, dp(i + 1, remains - 1))
                m = min(m, temp)
                
                if curr_sum >= m:
                    break # minor optimization
                
            return m
                
        return dp(0, k)
