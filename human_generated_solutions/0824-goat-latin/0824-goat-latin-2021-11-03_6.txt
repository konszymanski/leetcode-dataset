def toGoatLatin(self, sentence: str) -> str:
        words = sentence.split()
        vowels = "AEIOUaeiou"
        m="maa"
        l=[]
        for i in words:
            if i[0] in vowels:
                l.append(i + m)
            else:
                l.append(i[1:] + i[0] + m)
            m+="a"   
        return " ".join(l)
