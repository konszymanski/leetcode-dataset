class Solution:
    def matrixScore(self, grid: List[List[int]]) -> int:
        return sum(map(lambda pair : (pair[0] if pair[0] > (len(grid) / 2) else  (len(grid) - pair[0])) << (len(grid[0]) - pair[1] - 1),zip([sum([1 if i == 0 else v if grid[r][0] else -1*v+1 for r in range(len(grid)) for v in list([grid[r][i]])]) for i in range(len(grid[0]))],range(len(grid[0])))))
