def decodeAtIndex(self, s: str, k: int) -> str:
	size = 0
	l = len(s)
	for i in range(l):
		if s[i]>=\'a\' and s[i]<=\'z\':
			size+=1
		else:
			size*=(ord(s[i])-48)

	for i in range(l-1, -1, -1):
		k%=size
		if not k and s[i]>=\'a\' and s[i]<=\'z\':
			return s[i]
		if s[i]>=\'0\' and s[i]<=\'9\':
			size//=(ord(s[i])-48)
		else:
			size-=1
