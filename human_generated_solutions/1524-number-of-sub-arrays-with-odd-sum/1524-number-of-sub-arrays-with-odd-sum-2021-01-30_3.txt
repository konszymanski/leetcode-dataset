class Solution:
    def numOfSubarrays(self, arr: List[int]) -> int:
        ans = 0 
        even = 1 #considering 0 as even number 
        odd = 0 
        s = 0 
        mod = 10**9 +7
        for i in arr :
            s += i 
            #print(s,ans,even,odd)
            if s%2 == 1 :
                """
                let the sum = 3 
                we can think like 0 + 3
                here 0 is odd even number 
                so whenever we come across a odd sum that means that some odd value is added to some +ve value
                """
                odd += 1
                ans = (ans+even)%mod
            else:
                """
                if the sum is even there is two possibilities 
                1. this can be sum of two even values 
                2. this can be sum of two odd values 
                so if the odd count is >0 we add that to our answer 
                """
                even += 1
                ans = (ans+odd)%mod
        return ans%mod
