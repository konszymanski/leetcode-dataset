class Solution:
    def findKDistantIndices(self, nums: List[int], key: int, k: int) -> List[int]:
        """
        find all that == key into a list, idx
        
        traverse in nums, using i: (using binary search)
            if abs(i - idx) <= k: res.append(i)
        """
        keys = [i for (i, e) in enumerate(nums) if e == key]    
        res = set(keys)
        
        for i in range(len(nums)):
            idx = bisect.bisect_left(keys, i)
            
            # to keys[idx] and keys[idx - 1] are all <= k
            if 0 <= idx < len(keys) and abs(i - keys[idx]) <= k:
                res.add(i)
            if 0 < idx <= len(keys) and abs(i - keys[idx - 1]) <= k:
                res.add(i)
                
        
        return sorted(list(res))
