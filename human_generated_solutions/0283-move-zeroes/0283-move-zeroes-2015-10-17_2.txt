class Solution:
    def moveZeroes(self, nums: List[int]) -> None:
        slow, fast = 0, 0
        N = len(nums)
        for fast in range(N):
            while nums[slow] != 0 and slow < fast:
                slow += 1

            if nums[fast] != 0:
                nums[fast], nums[slow] = nums[slow], nums[fast]
