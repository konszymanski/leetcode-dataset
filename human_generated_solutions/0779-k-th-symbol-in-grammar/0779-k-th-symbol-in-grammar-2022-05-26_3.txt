class Solution:
    def kthGrammar(self, n: int, k: int) -> int:
        def solve(n,k):
            x=0
            if n==1 and k==1:
                return 0
            mid= (2**(n-1))/2
            if k<=mid:
                return solve(n-1,k)
            else :
                x!= (solve(n-1,k-mid)^1)
                return x
        return solve(n,k)
