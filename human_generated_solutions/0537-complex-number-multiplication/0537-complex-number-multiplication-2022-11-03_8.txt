class Solution:
    def complexNumberMultiply(self, num1: str, num2: str) -> str:
        n1, n2 = num1.split(\'+\'), num2.split(\'+\')
        n1[1], n2[1] = n1[1][:-1], n2[1][:-1]
        real = int(n1[0])*int(n2[0])
        real += int(n1[1])*int(n2[1])*(-1)
        unreal = int(n1[0])*int(n2[1]) + int(n1[1])*int(n2[0])
        return str(real) + \'+\' + str(unreal) + \'i\'
