class Solution:
    def complexNumberMultiply(self, a: str, b: str) -> str:
        a = ComplexNum(a)
        b = ComplexNum(b)
        re = a.re * b.re - (a.im * b.im) 
        im = a.re * b.im + a.im * b.re
        return str(re)+\'+\'+str(im) + \'i\'
        
class ComplexNum:
    def __init__(self, strr):
        parsed = strr.split(\'+\')
        self.re = int(parsed[0])
        self.im = int(parsed[1][:-1])
