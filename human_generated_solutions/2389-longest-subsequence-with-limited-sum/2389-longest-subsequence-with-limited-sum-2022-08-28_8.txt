class Solution:
    def answerQueries(self, nums: List[int], queries: List[int]) -> List[int]:
        nums.sort()
        n, m = len(nums), len(queries)
        ans = [0] * m
        
        for i, query in enumerate(queries):
            s = 0
            k = -1
            for j in range(n):
                if s + nums[j] > query:
                    break
                
                s += nums[j]
                k = j
            ans[i] = k + 1
        
        return ans
