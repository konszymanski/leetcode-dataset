class Solution:
    def findSolution(self, customfunction: \'CustomFunction\', z: int) -> List[List[int]]:
        res=[]
        x,y=1,1000
        while x<1000 and y>0:
            if customfunction.f(x,y)==z:
                res.append([x,y])
                x+=1
                y-=1
            elif customfunction.f(x,y)>z:
                y-=1
            else:
                x+=1
        return res
