def longestDiverseString(self, a: int, b: int, c: int) -> str:
    t = sorted([[a, \'a\'], [b, \'b\'], [c, \'c\']], reverse=True)
    s = \'\'
    k = max(0, t[0][0] - 2 * t[1][0])
    for i in range(t[1][0]):
        s += (1 + (i < t[0][0] - t[1][0])) * t[0][1] + t[1][1]
        if i < t[2][0]:
            s += min(2, k) * t[0][1] + t[2][1]
            k -= min(2, k)
    return s + min(2, k) * t[0][1]
