class Solution(object):
    def arrayNesting(self, nums):    
        count , sum_ , n = 1, [] , len(nums)
        for i in range(n):
            temp = i
            if nums[i] != n:
                while 1:
                    if nums[temp] == i:
                        sum_.append(count)
                        nums[temp] = n
                        count = 1
                        break
                    else:
                        temp1 = nums[temp]
                        nums[temp] = n
                        temp = temp1
                        count += 1
        return max(sum_)
