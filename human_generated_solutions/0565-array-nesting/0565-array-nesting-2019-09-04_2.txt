def arrayNesting(self, nums: List[int]) -> int:
        mx = 0
        for i in range(len(nums)):
            if nums[i] == -1: continue # continue if we already met it
            length = 0
            while nums[i] != -1: # stop when we meet the same number twice
                nums[i], i = -1, nums[i]
                length += 1
            if length > mx: mx = length
        return mx
