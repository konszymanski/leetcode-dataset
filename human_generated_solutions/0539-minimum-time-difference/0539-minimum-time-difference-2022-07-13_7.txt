def findMinDifference(self, timePoints: List[str]) -> int:
        #create a lookup table for every second of the day
        lt = [False] * 1440
        minTime, maxTime = 1440, 0
        
        #populate the lookup table
        for t in timePoints:
            h, m = t.split(":")
            s = (int(h) * 60) + int(m)
            if lt[s]:
                return 0
            lt[s] = True
            minTime = min(minTime, s)
            maxTime = max(maxTime, s)
        
        #set a reference time to compare
        op = maxTime - minTime
        if op > 720:
            op = 1440 - op
        prev = minTime
        
        #update op
        for i in range(minTime + 1, maxTime + 1):
            if lt[i]:
                diff = i - prev
				if diff > 720:
					diff = 1440 - diff
                op = min(op, diff)
                prev = i
        return op
