class Solution:
    def isOneBitCharacter(self, bits: List[int]) -> bool:
        n = len(bits)
        
        # if bits[i] == 0 means it\'s a 1-bit char
        # if bits[i] == 1 means it\'s the start of a 2-bit char
        i = 0
        while i < n - 1:
            if bits[i] == 0:
                i += 1
            else:
                i += 2
                
        return i == n - 1 and bits[i] == 0
