# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def removeNodes(self, head: Optional[ListNode]) -> Optional[ListNode]:
        cur = head
        stack = []
        while cur:
            while stack and stack[-1].val < cur.val:
                if len(stack) == 1: 
                    stack[-1] = stack[-1].next
                    stack.pop()
                else:
                    stack[-2].next = stack[-1].next
                    stack[-1].next = None
                    stack.pop()
            stack.append(cur)
            cur = cur.next

        return stack[0]
