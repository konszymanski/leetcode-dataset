class Solution:
    def maxSatisfied(self, customers: List[int], grumpy: List[int], X: int) -> int:
        ans = val = ii = mx = 0
        for i in range(len(customers)): 
            if not grumpy[i]: ans += customers[i]
            else: 
                val += customers[i]
                while ii <= i-X: 
                    if grumpy[ii]: val -= customers[ii]
                    ii += 1
                mx = max(mx, val)
        return ans + mx
