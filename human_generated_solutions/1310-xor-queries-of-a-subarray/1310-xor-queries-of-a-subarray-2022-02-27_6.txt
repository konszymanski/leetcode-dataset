class Solution:
    def xorQueries(self, arr: List[int], queries: List[List[int]]) -> List[int]:
        l=[]
        
        l.append(arr[0])
        
        for i in range(1,len(arr)):
            l.append(arr[i]^l[-1])
        res=[]
        for q in queries:
            
            if q[0]==0:
                res.append(l[q[1]])
            else:
                res.append(l[q[1]]^l[q[0]-1])
        return res
