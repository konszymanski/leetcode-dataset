class Solution:
    def countSubarrays(self, nums: List[int], k: int) -> int:
        left = right = 0
        windowSum = 0
        valid = 0
        while right < len(nums):
            windowSum += nums[right]
            right += 1
            
            while left < right and windowSum * (right - left) >= k:
                windowSum -= nums[left] 
                left += 1
            
            valid += right - left
        
        return valid
