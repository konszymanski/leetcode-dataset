def solve(n):
    if (n % 3 == 0):
        return n // 3
    if (n == 2):
        return 1
    if (n == 1):
        return -1
    three_count = n // 3 # y
    res = n // 2 # result
    while(n - 3*three_count < n):
        if ((n - 3*three_count) % 2 == 0):
            res = three_count + (n - 3*three_count) // 2
            break
        three_count -= 1
    return res
