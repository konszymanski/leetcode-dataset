class Solution:
    def minimumRounds(self, tasks: List[int]) -> int:
        
        freq=dict()
        
        for i in tasks:
            if i not in freq:
                freq[i]=1
            else:
                freq[i]+=1
        minRounds=0
        for i in freq:
            if freq[i]==1:
                minRounds=-1
                break
            rem=freq[i]%3
            r=int(freq[i]/3)
            if rem==1 or rem==2:
                r+=1
            
            minRounds+=r
        return minRounds
