from threading import Lock

class MyCircularQueue:

    def __init__(self, k: int):
        self.k = k
        self.lock = Lock()
        self.curSize = 0
        self.queue = []

    def enQueue(self, value: int) -> bool:
        self.lock.acquire()
        if self.curSize==self.k:
            self.lock.release()
            return False
        self.queue.append(value)
        self.curSize+=1
        self.lock.release()
        return True

    def deQueue(self) -> bool:
        self.lock.acquire()
        if self.curSize==0:
            self.lock.release()
            return False
        self.queue.pop(0)
        self.curSize-=1
        self.lock.release()
        return True

    def Front(self) -> int:
        self.lock.acquire()
        if self.curSize==0:
            self.lock.release()
            return -1
        ans = self.queue[0]
        self.lock.release()
        return ans

    def Rear(self) -> int:
        self.lock.acquire()
        if self.curSize==0:
            self.lock.release()
            return -1
        ans = self.queue[self.curSize-1]
        self.lock.release()
        return ans

    def isEmpty(self) -> bool:
        self.lock.acquire()
        if self.curSize==0:
            self.lock.release()
            return True
        self.lock.release()
        return False

    def isFull(self) -> bool:
        self.lock.acquire()
        if self.curSize==self.k:
            self.lock.release()
            return True
        self.lock.release()
        return False
