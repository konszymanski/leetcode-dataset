class Solution:
    def findSubsequences(self, nums):
        if not nums:
            return []
        res = []
        def dfs(start, path):
            if len(path) >= 2:
                res.append(path)
            used = set()
            for i in range(start, len(nums)):
                if not path or path[-1] <= nums[i]:
                    if nums[i] not in used:
                        used.add(nums[i])
                        dfs(i + 1, path+[nums[i]])
        dfs(0, [])
        return res
