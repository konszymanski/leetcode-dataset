# Unoptimized approach - Kadanes algorithm is better

import collections
class Solution:
    def largestVariance(self, s: str) -> int:
        @cache
        def dp(i):
            if i == len(s):
                return 0
            
            res = 0
            for j in range(i,len(s)):
                count = collections.Counter(s[i:j+1])
                variance = abs(count.most_common(1)[-1][1] - count.most_common()[-1][1]) #This gets us the most and least frequently occuring elements in the substring, to help us calculate variance
                res = max(res, variance, dp(j+1))
    
            return res
        
        return dp(0)
