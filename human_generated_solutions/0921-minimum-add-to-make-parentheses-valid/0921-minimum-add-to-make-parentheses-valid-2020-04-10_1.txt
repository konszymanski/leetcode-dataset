def without_stack(self, S):
        opening = count = 0
        for i in S:
            if i == \'(\': 
                opening += 1
            else:
                if opening: opening -= 1
                else: count += 1
        return count + opening
        
    def using_stack(self, S):
        stack = []
        for ch in S:
            if ch == \'(\': stack.append(ch)
            else:
                if len(stack) and stack[-1] == \'(\': stack.pop()
                else: stack.append(ch)
        return len(stack)
