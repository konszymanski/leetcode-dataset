class Solution:
    def minAddToMakeValid(self, s: str) -> int:
        n , idx , needed , stack = len(s) , 0 , 0 , []
        while idx < n:
            if s[idx] == "(":
                stack.append(s[idx])
            elif s[idx] == ")":
                if stack and stack[-1] == "(":
                    stack.pop()
                else:
                    needed += 1
            idx += 1
        return needed + len(stack)
