class Solution:
    def countPalindromicSubsequence(self, s: str) -> int:
        
        ctr = 0
        
        for i in range(97,123):
        
            first = s.find(chr(i))
            
            if first != -1:
                last = s.rfind(chr(i))
                ctr += len(set(s[first+1:last])) # count of unique characters between first and last character
    
        return ctr
