class Solution:
    def getNoZeroIntegers(self, n: int) -> List[int]:
        def check(num):
            while num>0:
                if num%10==0:
                    return False
                num//=10
            return True
        for i in range(1,n):
            t=n-i
            if check(t) and check(i):
                return [i,t]
