class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        
        
        n=len(boxes)
        
        ans=[]
        
        for i in range(n):
            
            p=0
            
            for j in range(n):
                if boxes[j]==\'1\':
                    p+= abs(i-j)
            ans.append(p)
        
        return ans
