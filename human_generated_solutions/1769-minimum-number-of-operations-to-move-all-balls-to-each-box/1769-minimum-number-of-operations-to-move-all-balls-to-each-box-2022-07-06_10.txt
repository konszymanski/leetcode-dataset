class Solution:
    def minOperations(self, boxes: str) -> List[int]:
        indices = set()
        for i, box in enumerate(boxes):
            if box == \'1\':
                indices.add(i)
                
        num_ones = len(indices)
        total = sum(indices)
        
        out = [total]
        before = 0
        after = num_ones
        for idx in range(1, len(boxes)):
            if boxes[idx-1] == \'1\':
                before += 1
                after -= 1 
            total += before
            total -= after
            out.append(total)
        return out
