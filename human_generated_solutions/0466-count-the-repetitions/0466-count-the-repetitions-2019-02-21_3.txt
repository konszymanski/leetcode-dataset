def Contain(L1,L2):
    for i in range(len(L2)):
        if L2[i] in L1:
            L1 = L1[L1.index(L2[i]):]
        else:
            return False
    return True

def getMaxRepetitions(s1,n1,s2,n2):
    S1 = s1*n1
    S2 = s2*n2
    M = int(len(S1)//len(S2))
    for m in range(M,1,-1):
        if Contain(S1,S2*m):
            return m
