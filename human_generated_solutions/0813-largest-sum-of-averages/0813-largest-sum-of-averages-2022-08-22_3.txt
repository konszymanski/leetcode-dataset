def largestSumOfAverages(self, nums: List[int], k: int) -> float:
	n = len(nums)
	pre = [0]
	for i in nums:
		pre.append(pre[-1] + i)
	def solve(i, k):
		if(k == 1): return sum(nums[i:])/(n-i)

		if(n-i < k): return -1

		temp = 0
		for j in range(i+1, n):
			temp = max(temp, solve(j, k-1)+(pre[j]-pre[i])/(j-i))
		d[(i, k)] = temp
		return temp
	d = dict()            
	return solve(0, k)
