class LinkedListNode:
    
    def __init__(self, val):
        self.val = val
        self.prev = None
        self.next = None

    def attachTo(self, node):
        node.next = self
        self.prev = node
        
    def detach(self):
        prev, next = self.prev, self.next
        if prev:
            prev.next = next
        if next:
            next.prev = prev
            
    def __str__(self):
        return self.val + "\	" + str(self.next)

class Solution:
    def predictPartyVictory(self, senate: str) -> str:
        delR, delD = 0,0
        head = LinkedListNode(senate[0])
        totalD, totalR = 0, 0
        if senate[0] == "D":
            totalD += 1
        else:
            totalR += 1
        curr = head
        for ch in senate[1:]:
            node = LinkedListNode(ch)
            node.attachTo(curr)
            curr = node
            if ch == "D":
                totalD += 1
            else:
                totalR += 1
        
        while head:
            curr = head
            while curr:
                nextNode = curr.next
                if curr.val == "D":
                    if delD < 0:
                        delD += 1
                        if curr is head:
                            head = nextNode
                        curr.detach()
                        totalD -= 1
                    else:
                        delR -= 1
                else:
                    if delR < 0:
                        delR += 1
                        if curr is head:
                            head = nextNode
                        curr.detach()
                        totalR -= 1
                    else:
                        delD -= 1
                curr = nextNode
            
            if not totalR:
                return "Dire"
            if not totalD:
                return "Radiant"
