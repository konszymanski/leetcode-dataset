class Solution:
    def findLHS(self, nums: List[int]) -> int:
        letter = {}
        ans = 0
        for i in nums:
            if i not in letter:
                letter[i] = 1
            else:
                letter[i]+=1
        for i in letter:
            if i+1 in letter.keys():
                ans = max(ans, letter[i]+letter[i+1])
        return ans
