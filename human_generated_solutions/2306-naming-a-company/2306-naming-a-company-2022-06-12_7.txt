class Solution:
    def distinctNames(self, ideas: List[str]) -> int:
        initial = defaultdict(lambda: set()) 
        chars = [chr(ord(\'a\') + c) for c in range(26)]
        for s in ideas:
            initial[s[0]].add(s[1:]) 
        res = 0
        for i in range(26):
            for j in range(i + 1, 26):
                a, b = chars[i], chars[j] 
                res += len(initial[a] - initial[b]) * len(initial[b] - initial[a])
        return res * 2
