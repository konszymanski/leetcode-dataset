class Solution:
    def maxAbsoluteSum(self, nums: List[int]) -> int:
        def maxSum(nums: List[int]) -> int:
            result, total = nums[0], 0
            for num in nums:
                total += num
                if total < 0: total = 0
                result = max(result, total)
            return result
        invNums = [-num for num in nums]
        return max(maxSum(nums), maxSum(invNums))
