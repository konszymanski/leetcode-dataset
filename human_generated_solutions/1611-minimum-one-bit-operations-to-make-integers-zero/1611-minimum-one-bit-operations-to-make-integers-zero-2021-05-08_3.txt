class Solution:
    def minimumOneBitOperations(self, n: int) -> int:
        x = \'{0:b}\'.format(n)
        y = list(x)
        r = 0
        s = True
        for k in range(len(y)):
            if x[k]==\'1\' and s:
                r+=(2**(len(y)-k))-1
                s=False
            elif x[k]==\'1\':
                r-=(2**(len(y)-k))-1
                s=True
        return r
