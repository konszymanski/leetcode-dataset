class Solution {
public:
    string convert(int n) {
        string s = to_string(n);
        sort(s.begin(), s.end());
        return s;
    }
    
    bool reorderedPowerOf2(int n) {
        string s = convert(n);
        for (int i = 0; i < 30; i++) {
            if (s == convert(1 << i)) return true;
        }
        return false;
    }
};
