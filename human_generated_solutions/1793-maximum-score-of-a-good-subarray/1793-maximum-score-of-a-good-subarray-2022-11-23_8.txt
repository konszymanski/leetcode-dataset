def maximumScore(self, nums: List[int], k: int) -> int:
    n = len(nums)
    i = j = k
    score =  c_min = nums[k]

    while i >= 0 and j < n:
        if j < n-1 and (i==0 or nums[i-1] < nums[j+1]):
            j += 1
        else:
            i -= 1
        c_min = min(c_min, nums[i], nums[j])
        score = max(score, c_min*(j - i + 1))

        if i == 0 and j == n-1: break

    return score
