class BrowserHistory:

    def __init__(self, homepage: str):
        self.back_stack = []
        self.fw_stack = []
        self.back_stack.append(homepage)
        
    def visit(self, url: str) -> None:
        self.back_stack.append(url)
        self.fw_stack = []
        
    def back(self, steps: int) -> str:
        while steps != 0:
            len_back = len(self.back_stack)
            if len_back > 0 and len_back != 1:
                temp = self.back_stack.pop(-1)
                self.fw_stack.append(temp)
            steps = steps - 1
        return self.back_stack[-1]
        
    def forward(self, steps: int) -> str:
        while steps != 0:
            if len(self.fw_stack) > 0:
                temp = self.fw_stack.pop(-1)
                self.back_stack.append(temp)
            steps = steps - 1
        return self.back_stack[-1]
