class Solution:
    def binaryGap(self, n: int) -> int:
        if n < 3: return 0
        n = bin(n)[2:]

        r = len(n)
        i = 0
        
        ct = 0
        max = 0
        while i < r:
            if n[i] == \'1\':
                if ct > max:
                    max = ct
                ct=1
            else:
                ct+=1
            i+=1
            
        return max
