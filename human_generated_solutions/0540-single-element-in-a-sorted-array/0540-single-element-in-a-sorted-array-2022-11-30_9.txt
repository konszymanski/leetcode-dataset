class Solution:
    def singleNonDuplicate(self, nums: List[int]) -> int:
        if len(nums)==1:
            return nums[0]
        i=0
        j=len(nums)-1
        while i<=j:
            if nums[i]!=nums[i+1]:
                return nums[i]
            elif nums[j]!=nums[j-1]:
                return nums[j]
            if nums[i]==nums[i+1]:
                i+=2
            elif nums[j]==nums[j-1]:
                j-=2
