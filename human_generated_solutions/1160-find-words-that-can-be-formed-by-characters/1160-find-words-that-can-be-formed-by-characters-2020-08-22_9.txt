class Solution:
    def countCharacters(self, words: List[str], chars: str) -> int:
        from collections import Counter
        d = Counter(chars)
        ans=0
        for word in words:
            flag=0
            newD = Counter(word)
            for key in newD.keys():
                try:
                    if d[key]>=newD[key]:
                        flag=1
                    else:
                        flag=0
                        break
                except:
                    pass
            if flag:
                ans+=len(word)
            
            
        return(ans)
