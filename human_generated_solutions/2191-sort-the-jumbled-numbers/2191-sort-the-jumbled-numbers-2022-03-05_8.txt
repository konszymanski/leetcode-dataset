class Solution:
    def sortJumbled(self, mapping: List[int], nums: List[int]) -> List[int]:
        h = {}
        ans = []
        for num in nums:
            res = 0
            for i in str(num):
                val = mapping[int(i)]
                res = res*10 + val
            ans.append((num,res))
        ans.sort(key=lambda x:x[1])
        return [each[0] for each in ans]
