class Solution {                                //  O(log(n)) // Most Optimal
public:
    bool isPerfectSquare(int num) {
        int low = 1, high = num; 
        while(low <= high){
            long long int mid = low + (high - low) / 2;
            if(mid * mid == num)
                return 1;
            else if(mid * mid < num)
                low = mid + 1;
            else
                high = mid - 1;
        }
        return 0;
    }
};
