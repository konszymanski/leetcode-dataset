class Solution {
    public int findLucky(int[] arr) {
        HashMap<Integer, Integer> map = new HashMap<>();
        Arrays.sort(arr);
        for (int i : arr) map.put(i, map.getOrDefault(i, 0) + 1);
        for (int j = arr.length - 1; j >= 0; j--) {
            if (map.containsKey(arr[j]) && map.get(arr[j]) == arr[j]) return arr[j];
        }
        return -1;
    }
}
