def winnerOfGame(self, colors: str) -> bool:
        
        count_AAA = count_BBB = 0
        
        for i in range(1, len(colors) - 1):
            if colors[i-1] == colors[i] == colors[i+1] == \'A\':
                count_AAA += 1
            
            if colors[i-1] == colors[i] == colors[i+1] == \'B\':
                count_BBB += 1
                
        
        return count_AAA > count_BBB
