class Solution:

    def __init__(self, radius: float, x_center: float, y_center: float):
        self.x = x_center
        self.y = y_center
        self.r = radius

    def randPoint(self) -> List[float]:
        r = sqrt(random.random()) * self.r
        x = random.uniform(0, 2*pi)
        
        return [self.x + r*math.cos(x), self.y + r*math.sin(x)]
