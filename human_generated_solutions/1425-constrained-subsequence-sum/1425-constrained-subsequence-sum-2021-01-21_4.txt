import heapq
class Solution:
    def constrainedSubsetSum(self, nums: List[int], k: int) -> int:
        dp = [0 for _ in range(len(nums))] 
        dp[0]=nums[0]
        heap = []
        heapq.heapify(heap)
        for i in range(1,len(nums)):
            while len(heap)>0 and heap[0][1]<i-k:
                heapq.heappop(heap)
            heapq.heappush(heap, (-dp[i-1],i-1))
            max_=-heap[0][0]
            dp[i] = max(0,max_)+nums[i]
        return max(dp)
