def maxLevelSum(self, root: Optional[TreeNode]) -> int:
        
        q = deque([(root,1)])
        
        max_sum = -float(\'inf\')
        
        while q:
            
            level_sum = 0
            
            for _ in range(len(q)):
                cur, level = q.popleft()
                
                level_sum += cur.val
                
                if cur.left:
                    q.append((cur.left,level+1))
                if cur.right:
                    q.append((cur.right,level+1))
            
            if level_sum > max_sum:
                    max_sum, min_level = level_sum, level
                
        return min_level
