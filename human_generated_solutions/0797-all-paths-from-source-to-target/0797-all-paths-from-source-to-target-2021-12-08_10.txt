class Solution:
    def allPathsSourceTarget(self, graph: List[List[int]]) -> List[List[int]]:
        queue = []
        paths = []
		max_item = len(graph) - 1

        queue.append([0])
        
        while len(queue):
            item = queue.pop(0)

            for next in graph[item[-1]]:
                if next == max_item:
                    paths.append(item + [next])
                else:
                    queue.append(item + [next])

        return paths
