class Solution:
    def largestTriangleArea(self, points: List[List[int]]) -> float:
        max_area = 0
        for [x1, y1], [x2, y2], [x3, y3] in combinations(points, 3):
            area = 0.5 * abs(y1 * (x2 - x3) + y2 * (x3 - x1) + y3 * (x1 - x2))
            if area > max_area: max_area = area
        
        return max_area
