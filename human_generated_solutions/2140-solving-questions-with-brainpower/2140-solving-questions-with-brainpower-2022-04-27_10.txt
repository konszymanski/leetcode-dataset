class Solution:
    def mostPoints(self, questions: List[List[int]]) -> int:
		dp = [0] * len(questions)
		dp[-1] = questions[-1][0]
		for i in range(len(questions)-2, -1, -1):
			ind = i + questions[i][1] + 1
			if ind >= len(questions):
				dp[i] = max(questions[i][0], dp[i+1])
			else:
				dp[i] = max(questions[i][0] + dp[ind], dp[i+1])
		return dp[0]
