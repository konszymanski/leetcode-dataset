def getMaximumGenerated(self, n: int) -> int:
        l = n+1
        nums = [0]* l
        nums[0] = 0
        if n > 1:
            nums[1] = 1
            for i in range(1, n+1//2):
				# checking the values as per the directions in the question
                if 2 <= 2 * i + 1 <= n:
					# creating the elements as per the directions in the question
                    nums[2*i] = nums[i]
                    nums[(2*i)+1] = nums[i] + nums[i + 1]
                elif 2 <= 2 * i <= n:
					# creating the element as per the directions in the question
                    nums[2*i] = nums[i]

            return max(nums)
        else:
			# If n is not greater than 1 return n
            return n
