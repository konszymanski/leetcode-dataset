class Solution:
    def getMaximumGenerated(self, n: int) -> int:
        if n==0:
            return 0
        t=[0]*(n+1)
        t[1] = 1    
        for i in range(2,n+1):
            
            if i%2==0:
                t[i] = t[i//2]
            else:
                t[i] = t[i//2] +t[i//2 + 1]
        return max(t)
