class Solution:
    def mincostToHireWorkers(self, quality: List[int], wage: List[int], K: int) -> float:
        
        
        workers=[]
        for w,q in zip(wage, quality):
            workers.append([float(w) / q, q] )
        workers.sort()
        print(workers)                   
        res = float(\'inf\')
        qsum = 0
        heap = []
        for r, q in workers:
            
            heapq.heappush(heap, -q)
            #print(heap)
            qsum += q
            if len(heap) > K:
                qsum += heapq.heappop(heap)
            if len(heap) == K:
                res = min(res, qsum * r)
        print(heap)
        return res
