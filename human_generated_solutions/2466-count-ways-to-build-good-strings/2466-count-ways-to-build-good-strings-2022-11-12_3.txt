class Solution:
    def countGoodStrings(self, low: int, high: int, zero: int, one: int) -> int:
        """
            . find total number of good strings possible to make using within the string length range [low-high]
            
            dp(count):
                
            state:
                count: total elements used till now
            transition
                use o + use 1
            exit
                count == high
            
        """
        
        MOD = 1_000_000_007
        
        @lru_cache(100*100)
        def dp(c):
            if c == high:
                return 1
            if c > high:
                return 0
            
            if c >= low:
                return 1 + dp(c + zero) + dp(c + one)
            else:
                return dp(c + zero) + dp(c + one)
        
        return dp(0) % MOD
