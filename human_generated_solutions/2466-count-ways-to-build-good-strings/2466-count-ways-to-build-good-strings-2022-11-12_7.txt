class Solution:
    def countGoodStrings(self, low: int, high: int, zero: int, one: int) -> int:
        dp = [0]*(high+1)
        dp[zero] += 1
        dp[one] += 1
        mod = (10**9)+7
        for i in range(min(zero,one)+1,high+1):
            if i-zero>=0:
                dp[i]+=dp[i-zero]
            if i-one>=0:
                dp[i]+=dp[i-one]
        s = 0
        for i in range(low,high+1):
            s+=dp[i]
        return s%mod
