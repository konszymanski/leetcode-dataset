class Solution:
    def findLongestChain(self, pairs: List[List[int]]) -> int:
        pairs.sort()
        lis = [1] * len(pairs)
        for i in range(len(pairs) - 1, -1, -1):
            for j in range(i + 1, len(pairs)):
                if pairs[i][1] < pairs[j][0]:
                    lis[i] = max(lis[i], 1 + lis[j])
        return max(lis)
