class Solution:
    def maximumImportance(self, n: int, roads: List[List[int]]) -> int:
        table, result, ans = {}, [],0
        for road in roads:
            pointA, pointB = road
            if pointA not in table:
                table[pointA] = 0
            if pointB not in table:
                table[pointB] = 0
            table[pointA] += 1
            table[pointB] += 1
        for key in table:
            result.append([key,table[key]])
        result.sort(key=lambda x: x[1])
        for idx in range(len(result)-1,-1,-1):
            table[result[idx][0]] = n
            n -= 1
        for a,b in roads:
            ans += (table[a]+table[b])
        return ans
