class Solution:
    def maximumDifference(self, nums: List[int]) -> int:
        
        max_diff, temp =-1, nums[0] #set max_diff default value -1
        
        for value in nums:
            diff = value -temp
            if temp<value and diff > max_diff:
                max_diff = diff
            if temp>value:
                temp = value
        return max_diff
