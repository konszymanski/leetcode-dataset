class Solution:
    def maxSubarraySumCircular(self, nums: List[int]) -> int:
        if max(nums)<=0:
            return max(nums)
        maxt=[i for i in nums]
        mint=[i for i in nums]
        for i in range (1,len(nums)):
            if maxt[i-1]>0:
                maxt[i]+=maxt[i-1]
            if mint[i-1]<0:
                mint[i]+=mint[i-1]
        return max (max(maxt),sum(nums)-min(mint))
