class Solution:
    def maxVowels(self, s: str, k: int) -> int:
        vowel = {\'a\', \'e\', \'i\', \'o\', \'u\'}
        c=0
        for i in range(k):
            if(s[i] in vowel):
                c+=1
        n=len(s)
        m=c
        j=0
        for i in range(k,n):
            if(s[j] in vowel):
                c-=1
            if(s[i] in vowel):
                c+=1
            m=max(m,c)
            j+=1
        return m
