def judgePoint24(self, cards: List[int]) -> bool:
    def dfs(pos):
        if len(pos) == 1 and round(pos[0], 1) == 24:
            return True
        for (i,a),(j,b) in itertools.combinations(enumerate(pos), 2):
            for x in [a+b, a-b, b-a, a*b, b and a/b, a and b/a]:
                if dfs([x]+[v for idx,v in enumerate(pos) if idx not in [i,j]]):
                    return True
        return False
    return dfs(cards)
