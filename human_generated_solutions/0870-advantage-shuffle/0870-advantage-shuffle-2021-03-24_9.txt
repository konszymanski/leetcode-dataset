def advantageCount(self, A: List[int], B: List[int]) -> List[int]:
	N = len(A)
	A.sort()
	c = sorted([(v, i) for i, v in enumerate(B)])  # sorting B, while keeping the positions
	result = [-1] * N

	li, ri = 0, N-1  # pointers for A
	rj = N-1  # pointer for c
	while rj >= 0:
		cv, ci = c[rj]
		if cv < A[ri]:
			result[ci] = A[ri]
			ri -= 1
		else:
			result[ci] = A[li]
			li += 1
		rj -= 1

	return result
