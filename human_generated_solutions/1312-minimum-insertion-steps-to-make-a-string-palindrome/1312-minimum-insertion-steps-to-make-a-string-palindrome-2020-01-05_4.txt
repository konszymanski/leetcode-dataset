from functools import lru_cache


class Solution:
    def minInsertions(self, s: str) -> int:
        @lru_cache(None)
        def dp(s):
            if s == s[::-1]:
                return 0
            elif s[0] == s[-1]:
                return dp(s[1:-1])
			else:
				return 1 + min(dp(s[1:]), dp(s[:-1]))
        
        return dp(s)
