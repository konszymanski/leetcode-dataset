class Solution:
    def minInsertions(self, s: str) -> int:
        rs= s[::-1]
        n,m = len(s),len(rs)
        t = [[0 for j in range(m+1)]for i in range(n+1)]
        for i in range(1,n+1):
            for j in range(1,m+1):
                if s[i-1] == rs[j-1]:
                    t[i][j] = 1+t[i-1][j-1]
                else:
                    t[i][j] = max(t[i-1][j],t[i][j-1])      
        return n-t[i][j]
