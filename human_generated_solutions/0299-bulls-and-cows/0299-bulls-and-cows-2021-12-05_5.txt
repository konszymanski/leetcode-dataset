from collections import Counter

class Solution:
    def getHint(self, secret: str, guess: str) -> str:
        secret_non_bull, guess_non_bull = [], []
        num_bulls = 0
        for char1, char2 in zip(secret, guess):
            if char1 == char2:
                num_bulls += 1
            else:
                secret_non_bull.append(char1)
                guess_non_bull.append(char2)
        
        secret_counter, guess_counter = Counter(secret_non_bull), Counter(guess_non_bull)
        
        num_cows = 0
        for word, cnt in guess_counter.items():
            num_cows += min(secret_counter[word], cnt)
    
        
        return str(num_bulls) + \'A\' + str(num_cows) + \'B\'
