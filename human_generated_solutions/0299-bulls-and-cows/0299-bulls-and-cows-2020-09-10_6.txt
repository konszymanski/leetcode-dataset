class Solution:
    def getHint(self, secret: str, guess: str) -> str:
        bull, cow = 0, 0
        # keep in dict indexes of each numer:
        my_dict = defaultdict(list)
        for i, num in enumerate(secret):
            my_dict[num].append(i)
        temp = ""
        # find exact matches:
        for i, num in enumerate(guess):
            if my_dict.get(num):
                if i in my_dict.get(num):
                    bull+=1
                    my_dict[num].remove(i)
                    temp+=\'a\'
                else:
                    temp+=num
        # find numbers in wrong positions:
        for i, num in enumerate(temp):
            if my_dict.get(num):
                cow+=1
                my_dict[num].pop(-1)
                
        return str(bull)+\'A\'+str(cow)+\'B\'
