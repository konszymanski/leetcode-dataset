class Solution:
    def kthSmallest(self, mat: List[List[int]], k: int) -> int:
        N = len(mat)
        min_heap = []
        for i in range(N):
            heappush(min_heap, (mat[i][0], 0, mat[i][:]))
            
        while k > 0:
            k -= 1
            number, i, row = heappop(min_heap)
            if i + 1 < N:
                heappush(min_heap, (row[i+1], i+1, row))
            
        return number
