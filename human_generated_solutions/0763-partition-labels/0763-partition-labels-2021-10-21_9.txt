# count how often letter appears in string
        d = Counter(s)
        
        # create empty dictionary for tracking letter count (see below)
        my_set = {}
        
        # create empty array (could be a string) to track current partition
        partition = []
        
        # your results
        result = []
        
        # iterate thgouth each letter
        for letter in s:
            
            # track your current letter >> check if not tracking >> add it to the tracker
            if letter not in my_set:
                my_set[letter] = d.get(letter)

            # add your letter to current partition
            partition.append(letter)
            
            # decrease the letter occurance count in your tracker
            my_set[letter] -= 1
            
            # as you iterate though letters in s, the number of tracked letters will grow. 
            # Once there are no more letter occurances in your tracker - i.e. sub-set of letters only appear in your current partition, 
            # you\'ve reached end of your partition. 
            # none of your tracking letters will appear anywhere else in the string going forward.
            # each of the tracked letters appears only in current partition >> all other partitions will have their own set of letters
            
            if sum(my_set.values()) == 0:
                
                # reset for next partition
                result.append(len(partition))
                partition.clear()
        
        return result
