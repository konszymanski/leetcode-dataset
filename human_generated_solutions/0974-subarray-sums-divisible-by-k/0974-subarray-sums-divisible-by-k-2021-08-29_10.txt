class Solution:
    def subarraysDivByK(self, nums: List[int], k: int) -> int:
        pre=[0]
        hs={0:1}
        c=0
        for i in range(len(nums)):
            z=pre[-1]+nums[i]
            c=c+hs.get(z%k,0)
            hs[z%k]=hs.get(z%k,0)+1

            
            pre.append(z)
        
        return c
