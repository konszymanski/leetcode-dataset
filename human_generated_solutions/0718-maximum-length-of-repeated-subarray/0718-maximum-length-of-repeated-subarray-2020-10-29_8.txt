class Solution:
    def findLength(self, A: List[int], B: List[int]) -> int:
        m, n = len(A), len(B)
        dp = [[0]*(n+1) for _ in range(m+1)] # (m+1) x (n+1)
        ans = 0
        for i in reversed(range(m)):
            for j in reversed(range(n)):
                if A[i] == B[j]: dp[i][j] = 1 + dp[i+1][j+1]
                ans = max(ans, dp[i][j])
        return ans
