class Solution:
    def constructDistancedSequence(self, n: int) -> List[int]:
        m = 1+ 2*(n-1)
        ans=[\'*\']*m
        left_nodes = set([i for i in range(1, n + 1)])
        def func(index):
            if(index>=m):
                return True
            if(ans[index]!=\'*\'):
                return func(index+1)
            for _ in range(n,0,-1):
                if(_ in left_nodes):
                    if(_ == 1 ):
                        ans[index] = _
                        left_nodes.remove(_)
                        if(func(index+1)):
                            return True
                        else:
                            left_nodes.add(_)
                            ans[index]=\'*\'
                    else:
                        next_index = index + _
                        if(next_index<m and ans[next_index] == \'*\'):
                            ans[index] = _
                            ans[next_index]= _
                            left_nodes.remove(_)
                            if(func(index+1)):
                                return True
                            else:
                                left_nodes.add(_)
                                ans[index] = \'*\'
                                ans[next_index]= \'*\'
            return False
                        
        func(0)
        return ans
