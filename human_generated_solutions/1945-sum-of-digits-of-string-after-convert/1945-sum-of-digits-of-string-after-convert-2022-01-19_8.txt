class Solution:
    def getLucky(self, s: str, k: int) -> int:
        output = ""
        for i in s:
            output += str((ord(i)-97)+1)
        while k !=0:
            sum = 0
            res = output
            for i in res:
                sum += int(i)
            output = str(sum)
            k -= 1
        return int(output)
