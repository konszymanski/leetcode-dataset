def rotatedDigits(self, n: int) -> int:
        invalid_nums = set([3,4,7])
        change_nums = set([2,5,6,9])
        
        def isGood(num):
            changed_num = False
            while num:
                num, rem = divmod(num, 10)
                if rem in invalid_nums:
                    return False
                if rem in change_nums:
                    changed_num = True
            return changed_num
    
        return sum(isGood(i) for i in range(1,n+1))
