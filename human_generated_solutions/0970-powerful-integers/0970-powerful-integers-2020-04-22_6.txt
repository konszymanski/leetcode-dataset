class Solution:
    def powerfulIntegers(self, x: int, y: int, bound: int) -> List[int]:
          
		# Catch edge cases where the bound is 1 or 0
        if bound <= 1:
            return []
        
		# make sure x is the larger value
        if x < y:
            x,y = y, x
            
        if x > 1: # store all powers of x that are below the bound
            store = []
            i = 0
            while x**i <= bound:
                store.append(bound-x**i)
                i += 1
        else:
            store = [bound - 1] # store just one value if x == 1

        out = set()
        if y > 1:
            i = 0    
            while i < len(store): # for each power of x below the bound try to add powers of y
                j = 0
                while y**j <=store[i]:
                    out.add(y**j+bound-store[i])
                    j += 1
                i += 1
        else: # add 1 to all sums in the x store
            for item in store:
                out.add(bound - item+1)
				
	return out
