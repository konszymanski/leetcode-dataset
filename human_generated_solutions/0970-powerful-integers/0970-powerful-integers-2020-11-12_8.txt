class Solution:
    def powerfulIntegers(self, x: int, y: int, bound: int) -> List[int]:
        result = set()        
        for i in range(int(log(bound, x))+1 if x != 1 else 1):
            for j in range(int(log(bound, y))+1 if y != 1 else 1):
                num = x**i + y**j 
                if num <= bound:
                    result.add(num)
                else:
                    break
                    
        return list(result)
