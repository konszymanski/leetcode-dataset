class Solution:
    def zeroFilledSubarray(self, nums: List[int]) -> int:
        n=len(nums)
        count=s=0
        for i in range(n):
            if nums[i]==0:
                s+=1
            else:
                count+=(s*(s+1))//2
                s=0
        if nums[n-1]==0:
            count+=(s*(s+1))//2
        return count
