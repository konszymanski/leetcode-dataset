def slowestKey(self, releaseTimes: List[int], keysPressed: str) -> str:
        ld,lex = releaseTimes[0],keysPressed[0]
        for i in range(1,len(releaseTimes)):
            t = releaseTimes[i]- releaseTimes[i-1] 
            if ld <t:
                ld = t
                lex = keysPressed[i]
            elif ld == t:
                lex = max(lex,keysPressed[i])
        return lex
