class Solution(object):
    def slowestKey(self, releaseTimes, keyPressed):
        maxDiff = -1
        
        maxDuration = releaseTimes[0]
        ans = keyPressed[0]
        
        for i in range(1, len(releaseTimes)):
            diff = releaseTimes[i] - releaseTimes[i-1]
            
            if diff >= maxDuration:
                if diff > maxDuration:
                    ans = keyPressed[i]
                    maxDuration = diff
                else:
                    if keyPressed[i] > ans:
                        ans = keyPressed[i]
                    maxDuration = diff
                        
        return ans
