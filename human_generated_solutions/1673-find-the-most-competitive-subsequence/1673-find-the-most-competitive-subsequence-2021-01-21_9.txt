class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        
        #monotone increasing stack
        stack = []
        for index,num in enumerate(nums):
            
            #append to stack if decreasing and still enough for k-length
            remaining = len(nums) - index
            while (stack and num < stack[-1]) and (len(stack) + remaining > k):
                stack.pop()
            
            #only append if we\'re below k
            if len(stack) < k:
                stack.append(num)
        
        return stack
