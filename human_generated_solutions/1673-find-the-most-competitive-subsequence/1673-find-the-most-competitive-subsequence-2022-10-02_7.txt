class Solution:
    def mostCompetitive(self, nums: List[int], k: int) -> List[int]:
        n=len(nums)
        a=[nums[0]]
        for i in range(1,n):
            while a and a[-1]>nums[i] and n-i>k-len(a):
                a.pop()
            if len(a)<k:
                a.append(nums[i])
        return a
