class Solution {
public:
    vector<vector<int>> restoreMatrix(vector<int>& rs, vector<int>& cs) {
        int rows = rs.size();
        int cols = cs.size();
        int req;
        vector<vector<int>> ans(rows, vector<int>(cols, 0));
        for(int i=0; i<rows; i++) {
            for(int j=0; j<cols; j++) {
                req = min(rs[i], cs[j]);
                ans[i][j] = req;
                rs[i] -= req;
                cs[j] -= req;
            }
        }
        return ans;
    }
};
