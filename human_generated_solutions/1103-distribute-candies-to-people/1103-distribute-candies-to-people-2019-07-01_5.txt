            rounds = n // num_people 
        else:
            rounds = n // num_people + 1
		
		# The last index to fill
        last = int(n % num_people - 1)
        edge = last
        if edge < 0:
            edge += num_people
		
        for i in range(edge):
            res[i] = int((2*i + 2 + num_people * rounds - num_people) * rounds / 2
            candies -= res[i]
        for i in range(edge + 1, len(res)):
            res[i] = int((2*i + 2 + num_people * (rounds - 1) - num_people) * (rounds - 1) / 2)
            candies -= res[i]
        res[last] = candies
        return res
