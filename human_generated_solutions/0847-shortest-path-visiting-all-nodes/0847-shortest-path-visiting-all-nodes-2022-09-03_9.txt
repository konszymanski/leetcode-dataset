class Solution:
    def shortestPathLength(self, graph: List[List[int]]) -> int:
        n = len(graph)
        distance = [[float(\'inf\') for _ in range(n)] for _ in range(n)]
        
        for i, edges in enumerate(graph):
            distance[i][i] = 0
            for edge in edges:
                distance[i][edge] = 1
          
        # Floyd-Warshall
        for k in range(n):
            for i in range(n):
                for j in range(n):
                    distance[i][j] = min(distance[i][j], distance[i][k] + distance[k][j])

        @cache         
        def visit(node, visited):
            # Check if all nodes have already been visited
            if visited == (1 << n) - 1:
                return 0
            
            shortest = float(\'inf\')
            for i in range(n):
                # We check if the i-th bit has been visited or not
                if ((1 << i) & visited) == 0:
                    shortest = min(shortest, visit(i, (1 << i) | visited) + distance[node][i])
            return shortest
        
        shortest = float(\'inf\')
        for i in range(n):
            shortest = min(shortest, visit(i, (1 << i)))
        return shortest
    # Time: O(n^2 * 2^n)
    # Space: O(n * 2^n)
