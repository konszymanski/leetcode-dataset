def minCost(self, grid: List[List[int]]) -> int:
	m, n = len(grid), len(grid[0])
	dir = [(0, 1), (0, -1), (1, 0), (-1, 0)] 
	# in order of grid value for direction
	dis = [[float(\'inf\')] * n for _ in range(m)]
	dis[0][0] = 0
	heap = [(dis[0][0], [0, 0])]
	while(heap):
		d, [i, j] = heappop(heap)
		for x, y in dir:
			if(0<=i+x<m and 0<=j+y<n):
				if((x, y) == dir[grid[i][j]-1] and dis[i+x][j+y] > d):
					dis[i+x][j+y] = d
					heappush(heap, (d, [i+x, j+y]))
				elif(dis[i+x][j+y] > 1+d):
					dis[i+x][j+y] = 1+d
					heappush(heap, (d+1, [i+x, j+y]))
	return dis[-1][-1]
