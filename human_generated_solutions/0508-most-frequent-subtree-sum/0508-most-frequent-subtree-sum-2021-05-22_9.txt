class Solution:
    def findFrequentTreeSum(self, root: TreeNode):
        sums = defaultdict(int)

        def tree_sum(node):
            if node is None:
                return 0
            res = node.val
            if node.left:
                res += tree_sum(node.left)
            if node.right:
                res += tree_sum(node.right)
            sums[res] += 1
            return res

        tree_sum(root)
        max_frequency = max(sums.values())
        return [key for key, val in sums.items() if val == max_frequency]
