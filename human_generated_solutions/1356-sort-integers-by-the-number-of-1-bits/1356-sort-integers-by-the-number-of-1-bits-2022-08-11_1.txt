def sortByBits(self, arr: List[int]) -> List[int]:
	# Function to calculate number of ones
	def countOnes(n):
		cnt = 0
		while n:
			n = n & n-1
			cnt += 1

		return cnt
	
	# Calculate number of ones and add them to tmp list as tuple of (number of ones, num)
