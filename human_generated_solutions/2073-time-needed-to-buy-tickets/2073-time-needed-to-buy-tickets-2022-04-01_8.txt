class Solution:
    def timeRequiredToBuy(self, tickets: List[int], k: int) -> int:
        c=0
        while(tickets[k]!=0):
            c+=1
			#check if kth element is now the first element of our queue.
            if(tickets[0]==1):
                tickets.pop(0)
                if(k==0):
                    return(c)
                else:
                    k-=1
            else:
			#if no of tickets of person at first position in queue are more than 1 we move to the back
                tickets.append(tickets[0]-1)
                tickets.pop(0)
                if(k==0):
                    k=len(tickets)-1
                else:
                    k-=1
