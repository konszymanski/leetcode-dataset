class Solution:
    def largestTimeFromDigits(self, arr: List[int]) -> str:

        arr.sort(reverse = True)
        if arr[2] == 2 and arr[1] >=6 and arr[1] <=9:
            iterate = [1, 9, 5, 9] #handle special cases

    def smallerThanKLargest(self, arr, k):
        for index, value in enumerate(arr):
            if value <= k:
                arr.pop(index)
                return value
        return -float(\'inf\')
