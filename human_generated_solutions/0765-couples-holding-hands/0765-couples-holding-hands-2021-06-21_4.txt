class Solution:
    def minSwapsCouples(self, row: List[int]) -> int:
        s=0
        dic={min(i,j):max(i,j)for i,j in zip(row[0::2], row[1::2])}
        for i in range(0,len(row),2):
            a=dic[i]
            if abs(i-a)>1:
                s+=1
                d=dic.pop(i+1)
                dic[min(a,d)],dic[i]=max(a,d),i+1
        return s
