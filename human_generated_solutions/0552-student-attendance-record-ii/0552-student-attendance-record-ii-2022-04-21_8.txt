@lru_cache(None)
def dp(i, j):
    if i == 0:
        return 1

    count = 0

    if j < 4:
        count += dp(i - 1, j + 2)
    if j % 2 == 0:
        count += dp(i - 1, 1)
        count += dp(i - 1, 0)
    else:
        count += dp(i - 1, 1)
    return count % 1_000_000_007

class Solution:
    def checkRecord(self, n: int) -> int:
        return dp(n, 0)
