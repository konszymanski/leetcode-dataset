class Solution:
    def countOperations(self, num1: int, num2: int) -> int:
        
        count=0
        
        if (num1==0) | (num2==0):
            return 0
        
        while (num1!=0) | (num2!=0):
            
            if num1>num2:
                num1=num1-num2
                count+=1
                
            elif num1<num2:
                num2=num2-num1
                count+=1
                
            else:
                return count+1
