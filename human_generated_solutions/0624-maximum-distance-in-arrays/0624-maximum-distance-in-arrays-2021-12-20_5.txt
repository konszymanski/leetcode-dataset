class Solution:
    def maxDistance(self, arrays: List[List[int]]) -> int:
        cur_min = arrays[0][0]
        cur_max = arrays[0][-1]
        cur_dist = 0
        for arr in arrays[1:]:
            cur_dist = max(cur_dist, abs(arr[-1] - cur_min), abs(cur_max - arr[0]))
            cur_min = min(cur_min, arr[0])
            cur_max = max(cur_max, arr[-1])
        return cur_dist
