class Solution:
    def maxDistance(self, arrays: List[List[int]]) -> int:
        res = 0
        _min = arrays[0][0]
        _max = arrays[0][-1]
        
        for idx in range(1, len(arrays)):
            res = max(res, abs(_max - arrays[idx][0]), abs(_min - arrays[idx][-1]))
            
            _min = min(_min, arrays[idx][0])
            _max = max(_max, arrays[idx][-1])
        
        return res
