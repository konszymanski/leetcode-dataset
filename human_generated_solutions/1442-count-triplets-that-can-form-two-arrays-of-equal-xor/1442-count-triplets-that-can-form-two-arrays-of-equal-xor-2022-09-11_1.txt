class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        count = 0

        for i in range(len(arr) - 1):
            a = arr[i]

            for k in range(i + 1, len(arr)):
                if a ^ arr[k] == 0:
                    count += k - i

                a ^= arr[k]

        return count
