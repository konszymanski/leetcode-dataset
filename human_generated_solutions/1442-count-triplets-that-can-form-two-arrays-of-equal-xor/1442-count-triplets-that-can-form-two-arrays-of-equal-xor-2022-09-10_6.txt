class Solution:
    def countTriplets(self, arr: List[int]) -> int:
        ans = 0
        n = len(arr)
        for i in range(n):
            xor = arr[i]
            for j in range(i+1,n):
                xor ^= arr[j]
                if xor == 0:
                    ans += (j-i)
        return ans
