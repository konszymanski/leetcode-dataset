def other_way(self, root):
        #as per the hints
        def rec(node, parent, grand):
            if node:
                if grand%2==0: self.s+= node.val
                rec(node.left, node.val, parent)
                rec(node.right, node.val, parent)
        rec(root, 1, 1)
        return self.s
        
        
    def one_way(self, root):
        def addSum(node):
            if node.left:
                if node.left.left: self.s+= node.left.left.val
                if node.left.right: self.s+= node.left.right.val
            if node.right:
                if node.right.left: self.s+= node.right.left.val
                if node.right.right: self.s+= node.right.right.val
  
            
        def rec(node):
            if node:
                if node.val%2==0:
                    addSum(node)
                rec(node.left)
                rec(node.right)
                
        rec(root)
        return self.s
