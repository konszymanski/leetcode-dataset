class Solution:
    def findMin(self, nums: List[int]) -> int:
        lo, hi = 0, len(nums) - 1
        while hi > 0 and nums[0] == nums[hi]:
            hi -= 1
        while lo < hi:
            mid = lo + hi >> 1
            if nums[mid] > nums[hi]:
                lo = mid + 1
            else:
                hi = mid
        return nums[lo]
