        def path(s,e):
            if (s,e) in self.cache:  #if in cache
                return (self.cache[(s,e)])
            if s == e:     # edge case
                self.cache[(s,e)] = True
                return True
            else:

                    for j in g[s]: # recursilvly call to all neighbours
                        if path(j,e):

        # 1. Graph making
