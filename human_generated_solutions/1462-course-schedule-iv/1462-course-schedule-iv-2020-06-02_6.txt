def checkIfPrerequisite(self, n: int, prerequisites: List[List[int]], queries: List[List[int]]) -> List[bool]:
    g = [[] for _ in range(n)]
    q = [set() for _ in range(n)]         
    for u, v in prerequisites:
        g[u].append(v)

    def dfs(i):
        if q[i]:
            return q[i]
        q[i].update(g[i])
        for j in g[i]:
            q[i].update(dfs(j))
        return q[i]

    for i in range(n):
        dfs(i)
    return [b in q[a] for a, b in queries]
