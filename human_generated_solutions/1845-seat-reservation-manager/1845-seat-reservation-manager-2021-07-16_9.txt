class SeatManager:

    def __init__(self, n: int):
        self._toAlot = 1
        self._freed = collections.deque([])
        

    def reserve(self) -> int:
        if self._freed:
            val = self._freed[0]
            self._freed.popleft()
            return val
        else:
            self._toAlot +=1
            return self._toAlot-1

    def unreserve(self, seatNumber: int) -> None:
        bisect.insort(self._freed, seatNumber)
