class Solution:
    def findJudge(self, n: int, trust: List[List[int]]) -> int:
        degree = [0]*(n+1)
        
        # calculate the absolute degree
        for x,y in trust:
            degree[x]-=1
            degree[y]+=1
            
        # find the node with degree n-1
        for i in range(1,n+1):
            if degree[i] == n-1: return i
            
        # if no node is a valid judge
        return -1
