def findJudge(self, N: int, trust: List[List[int]]) -> int:
        graph = { x: [] for x in range(1, N + 1) }    
        indegrees = { x: 0 for x in range(1, N + 1) }
        outdegrees = { x: 0 for x in range(1, N + 1) }
        
        # build graph and count indegress/outdegrees
        for u, v in trust:
            # directed edge u -> v
            graph[u].append(v)
            # v as incoming connection
            indegrees[v] += 1 
            # u has outgoing connection 
            outdegrees[u] += 1
        
        # unconnected verticies have outdegree 0
        for i in range(1, N + 1):
            if i not in outdegrees:
                outdegrees[i] = 0 
        
        # check for vertice that has indegree zero 
        for v in range(1, N + 1):
            if v in indegrees:
                if indegrees[v] == N - 1 and outdegrees[v] == 0:
                    return v
        return -1
