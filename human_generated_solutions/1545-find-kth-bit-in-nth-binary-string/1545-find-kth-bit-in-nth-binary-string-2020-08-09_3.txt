class Solution(object):
    
    def findKthBit(self, n, k):
        """
        :type n: int
        :type k: int
        :rtype: str
        """
        if n == 1:
            return "0"
        s = \'011\'
        
        for i in range(2,n):
            mid = len(s)//2
            news = s + \'1\'+ s[:mid] + \'0\' + s[mid+1:]
            s = news
            #print(i,s)
        return s[k-1]
