class Solution(object):
    def totalHammingDistance(self, nums):
        """
        :type nums: List[int]
        :rtype: int
        """
        result=0
        if nums == []:
            return 0
        nums_len = len(nums)
        for i in range(nums_len):
            for n in range (i+1,nums_len):
                result += bin(nums[i]^nums[n]).count(\'1\')
        return result
