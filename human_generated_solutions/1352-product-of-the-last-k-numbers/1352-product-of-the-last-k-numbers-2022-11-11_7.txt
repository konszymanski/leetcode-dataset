class ProductOfNumbers:

    def __init__(self):
        self.s = [1]

    def add(self, num: int) -> None:
        if num == 0:
            self.s = [1]
        else:
            self.s.insert(0,self.s[0] * num)

    def getProduct(self, k: int) -> int:
        if k >= len(self.s):
            return 0
        else:
            return self.s[0] // self.s[k]
