class ProductOfNumbers:

    def __init__(self):
        self.lst = []

    def add(self, num: int) -> None:
        if num:
            if num != 1:
                self.lst = [n * num for n in self.lst]
            self.lst.append(num)
        else:
            self.lst = []

    def getProduct(self, k: int) -> int:
        return self.lst[-k] if k <= len(self.lst) else 0
