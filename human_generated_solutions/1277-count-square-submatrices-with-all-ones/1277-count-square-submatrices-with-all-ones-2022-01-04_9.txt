class Solution:
    def countSquares(self, matrix: List[List[int]]) -> int:
        m,n=len(matrix),len(matrix[0])
        dp=[[0]*(n+1) for i in range(m+1)]
        res=0
        for r in range(m):
            for c in range(n):
                if matrix[r][c]==1:
                    dp[r+1][c+1]=min(dp[r][c+1],dp[r+1][c],dp[r][c])+1
                    res=res+dp[r+1][c+1]
        return res
