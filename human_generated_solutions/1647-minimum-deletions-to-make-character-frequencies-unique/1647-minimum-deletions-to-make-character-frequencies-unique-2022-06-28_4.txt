from collections import Counter
from string import ascii_lowercase
class Solution:
    def minDeletions(self, s: str) -> int: 
        freq = Counter(s)
        result = 0
        seen = set()
        
        for i in ascii_lowercase:
            while freq[i] and freq[i] in seen:
                freq[i]-=1
                result+=1
            seen.add(freq[i])
        
        return result
