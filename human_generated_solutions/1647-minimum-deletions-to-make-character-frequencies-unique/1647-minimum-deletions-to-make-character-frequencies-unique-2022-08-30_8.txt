class Solution:
    def minDeletions(self, s: str) -> int:
        sd={}
        for i in s:
            if(i in sd):
                sd[i]+=1
            else:
                sd[i]=1
        s=[]
        for i in sd:
            s.append(sd[i])
        n=len(s)
        tot=0
        for i in range(n):
            while(s[i] in s[:i]+s[i+1:] and s[i]>0):
                s[i]-=1
                tot+=1
        return(tot)
