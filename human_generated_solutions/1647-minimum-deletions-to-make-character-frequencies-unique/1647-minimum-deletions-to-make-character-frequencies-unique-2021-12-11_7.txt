from collections import Counter
import heapq

class Solution:
    def minDeletions(self, s: str) -> int:
        counter = Counter(s) # O(n)
        heap = [-x for x in counter.values()] # max heap
        heapq.heapify(heap) # O(1) : only 26 alphabet exists
        
        prev = -1
        cnt = 0
        while heap:
            now = -heapq.heappop(heap)  # O(1) : only 26 alphabet exists
            if prev == now:
                now -= 1
                cnt += 1
                if now != 0:
                    heapq.heappush(heap, -now)  # O(1) : only 26 alphabet exists
            else:
                prev = now
        return cnt
