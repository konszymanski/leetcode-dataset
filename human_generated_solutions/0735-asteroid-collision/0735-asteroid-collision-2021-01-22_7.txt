def asteroidCollision(self, asteroids: List[int]) -> List[int]:
        
        stack = []
        
        stack.append(asteroids[0])
        
        for i in range(1,len(asteroids)):
            
            stack.append(asteroids[i])
            
            while len(stack)>=2:
                
                a = stack.pop()
                b = stack.pop()
                
                if a<0 and b>0:
                    if abs(a)>b:
                        stack.append(a)
                    elif abs(a)<b:
                        stack.append(b)
                else:
                    stack.append(b)
                    stack.append(a)
                    break
        
        return stack
