# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeNodes(self, head: Optional[ListNode]) -> Optional[ListNode]:
        l=ListNode()
        ans=l
        s=0
        while(head!=None):
            if(head.val==0):
                if(s==0):
                    pass
                else:
                    f=ListNode(s)
                    l.next=f
                    l=l.next
                    s=0
            else:
                s=s+head.val
                
            head=head.next
        
        return(ans.next)
