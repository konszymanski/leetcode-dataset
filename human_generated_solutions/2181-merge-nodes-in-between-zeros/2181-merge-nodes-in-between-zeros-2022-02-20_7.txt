def mergeNodes(self, head: Optional[ListNode]) -> Optional[ListNode]:
        
        head = head.next
        node = head
        while node:
            while node.next.val != 0:
                node.val += node.next.val
                node.next = node.next.next
            
            node.next = node.next.next
            node = node.next

        return head
    
# Time: O(N)
# Space: O(1)
