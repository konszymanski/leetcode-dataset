/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
public:
    int maxVal = 0;
    int leftVal = 0;
    int rightVal = 0;
    int longestZigZag(TreeNode* root);
    int helper(TreeNode*, bool);

};

int Solution::longestZigZag(TreeNode* root) {
    if (!root) return 0;
    if (root->left != NULL){
        this->leftVal = helper(root->left, true) + 1;
    }
    if (root->right != NULL){
        this->rightVal = helper(root->right, false) + 1;
    }
    return (std::max(this->maxVal, std::max(this->leftVal, this->rightVal)));
}

int Solution::helper(TreeNode* root, bool left){
    int leftVal = 0;
    int rightVal = 0;
    if (root->left != NULL)
        leftVal = helper(root->left, true) + 1;
    if (root->right != NULL)
        rightVal = helper(root->right, false) + 1;
    this->maxVal = std::max(this->maxVal, std::max(leftVal, rightVal));
    if (left) return rightVal;
    else return leftVal;
}
