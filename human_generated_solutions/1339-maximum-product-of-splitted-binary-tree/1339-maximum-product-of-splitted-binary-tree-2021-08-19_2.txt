class Solution:
    def maxProduct(self, root: Optional[TreeNode]) -> int:
        
        sums = []
        def dfs(node):
            if node is None:
                return 0
            subtree_sum = dfs(node.left) + dfs(node.right) + node.val
            sums.append(subtree_sum)
            return subtree_sum
        
        
        m = -inf
        total = dfs(root)
        for i in sums:
            prod = i * (total-i)
            if prod > m: m = prod
        
        return m % (10**9 + 7)
