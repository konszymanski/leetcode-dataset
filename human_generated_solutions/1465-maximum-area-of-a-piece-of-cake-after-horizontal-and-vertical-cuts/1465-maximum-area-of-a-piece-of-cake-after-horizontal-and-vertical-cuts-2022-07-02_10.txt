class Solution {
    public int maxArea(int h, int w, int[] horizontalCuts, int[] verticalCuts) {
        Arrays.sort(horizontalCuts);
        Arrays.sort(verticalCuts);
        int max_hor_dis = Math.max(horizontalCuts[0], h - horizontalCuts[horizontalCuts.length-1]);
        int max_ver_dis = Math.max(verticalCuts[0], w - verticalCuts[verticalCuts.length-1]);
        int mod = (int)Math.pow(10,9)+7;
        for(int i=1; i<horizontalCuts.length; i++){
            max_hor_dis = Math.max(max_hor_dis, horizontalCuts[i] - horizontalCuts[i-1]);
        }
        for(int i=1; i<verticalCuts.length; i++){
            max_ver_dis = Math.max(max_ver_dis, verticalCuts[i] - verticalCuts[i-1]);
        }
        return (int)((long) max_hor_dis * max_ver_dis % mod);
    }
}
