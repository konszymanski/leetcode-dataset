def triangularSum(self, nums: List[int]) -> int:
        if not nums:
            return 0
        if len(nums) == 1:
            return nums[0]
        
        n = len(nums)
        
        while n != 1:
            nums = [(nums[i] + nums[i+1]) % 10 for i in range(n-1)]
            n = len(nums)
        return nums[0]
