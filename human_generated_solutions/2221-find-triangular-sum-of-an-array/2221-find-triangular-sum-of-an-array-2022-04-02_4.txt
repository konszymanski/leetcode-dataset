class Solution:
    def helper(self, nums: List[int]) -> List[int]:
        dp = []
        i = 0
        while i < len(nums) - 1:
            dp.append((nums[i] + nums[i + 1]) % 10)
            i += 1
        return dp
    
    def triangularSum(self, nums: List[int]) -> int:
        while len(nums) > 1:
            nums = self.helper(nums)
        return nums[0]
