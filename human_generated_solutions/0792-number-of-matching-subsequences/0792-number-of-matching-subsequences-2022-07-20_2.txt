class Solution:
    def numMatchingSubseq(self, s: str, words: List[str]) -> int:
        if not words or not s: return 0
        
        ans = 0
        for word in words:
            if self.is_subsequence(s, word):
                ans += 1
        return ans
        
    
    def is_subsequence(self, s: str, ss: str) -> bool:
        if len(ss) > len(s): return False
        
		# sub string pointer
        ss_p = 0
		
		# s_p = string pointer
        for s_p in range(len(s)):
            if ss[ss_p] == s[s_p]:
                ss_p += 1
				# if we matched the whole sub string
                if ss_p == len(ss):
                    return True
        return False
