# Buckets: Most Optimal 
# TC: O( len(S)  + sum of all words[i] length)

def numMatchingSubseq(self, s: str, words: List[str]) -> int:
	store_words = defaultdict(list)

	for word in words:
		store_words[word[0]].append(word)

	cnt = 0
	for ch in s:

		arr = list(store_words[ch])
		store_words[ch] = []
		while arr:
			word = arr.pop()[1:]

			if len(word) == 0:
				cnt += 1
			else:
				store_words[word[0]].append(word)

	return cnt
