class Solution:
    def numMatchingSubseq(self, s: str, words: list[str]) -> int:
        
        @lru_cache(1000)
        def checkWord(word):

            start = 0

            for ch in word:
                start = s.find(ch, start) + 1
                if not start: return False
                
            return True
        
        return sum(map(checkWord,words))
