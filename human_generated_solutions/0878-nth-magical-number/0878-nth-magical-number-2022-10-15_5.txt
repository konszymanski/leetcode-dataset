class Solution:
    def nthMagicalNumber(self, n: int, a: int, b: int) -> int:
        MOD, left, right, LCM = 1000000007, min(a, b), n * min(a, b), lcm(a, b)
        while left < right:
            mid = (left + right) // 2
            if (mid // a) + (mid // b) - (mid // LCM) < n:
                left = mid + 1
            else:
                right = mid
        return left % MOD
