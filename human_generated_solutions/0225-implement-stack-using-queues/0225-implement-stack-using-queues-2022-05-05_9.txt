from collections import deque

class MyStack:

    def __init__(self):
        self._stack = deque()

    def push(self, x: int) -> None:
        deq = self._stack
        deq.append(x)
        for i in range(len(deq) - 1):
            deq.append(self.pop())

    def pop(self) -> int:
        return self._stack.popleft()

    def top(self) -> int:
        return self._stack[0]        

    def empty(self) -> bool:
        return len(self._stack) == 0
