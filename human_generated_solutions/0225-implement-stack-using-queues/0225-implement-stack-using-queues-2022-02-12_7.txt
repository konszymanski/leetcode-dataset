class Queue:
    def __init__(self):
        self.queue = []
    def push(self,val):
        self.queue.append(val)
    def pop(self):
        return self.queue.pop(0)
    def top(self):
        return self.queue[0]
    
    def isEmpty(self):
        if len(self.queue)== 0:
            return True
        else:
            return False
