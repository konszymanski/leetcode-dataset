var isBalanced = function(root) {
    
    if(!root) return true
    
    let leftHeight  = height(root.left)            // is waaale node ka, left aur right subtree ka height kya hai
    let rightHeight = height(root.right)          
    
    
    
    if(Math.abs(rightHeight - leftHeight) <= 1     //  (height <= 1) hua toh koi dikkat nai 
      && isBalanced(root.left)                     //  LIKEWISE,  LEFT WALO KE LIYE DEKHO
      && isBalanced(root.right)                    //  LIKEWISE, RIGHT WALO KE LIYE DEKHO
    ) return true
    
    else return false

};

function height(root){
    if(!root) return 0
    return Math.max(height(root.left), height(root.right)) + 1
}
