class Solution {
public:
    int surfaceArea(vector<vector<int>>& grid) {
        int res = 0;

        for (int i = 0; i < grid.size(); ++i)
            for (int j = 0; j < grid[0].size(); ++j) {
                if (grid[i][j])
                    res += 4*grid[i][j] + 2;

                if (j) {
                    res -= min(grid[i][j],grid[i][j-1])*2;
                    res -= min(grid[j][i],grid[j-1][i])*2;
                }
            }

        return res;
    }
};
