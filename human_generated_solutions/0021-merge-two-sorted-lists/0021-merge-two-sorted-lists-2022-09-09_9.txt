# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def mergeTwoLists(self, list1: Optional[ListNode], list2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(0) # making a dummy node
        curr = dummy # creating a pointer pointing to the dummy node
        
        while list1 and list2: # we must be having both list to perform merge operation.
            if list1.val < list2.val: # in case list value is less then list2,
                curr.next = list1  # then we move our pointer ahead in list 1.
                list1 = list1.next # having value of next element of list2
            else:
                curr.next = list2 # in case list2 value is greaer then list 1 value.
                list2 = list2.next #having value of next element of list2
            curr = curr.next # moving our curr pointer
        
        # In case all the elements of any one of the list is travered. then we`ll move our pointer to the left over. 
        # As lists are soted already, technically we could do that
        # Method : 1
#         if list1:
#             curr.next = list1
#         elif list2:
#             curr.next = list2
        
        # Method : 2
        curr.next = list1 or list2
        
        return dummy.next # return next bcz first node is dummy.
