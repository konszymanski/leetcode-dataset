func validPartition(nums []int) bool {
    dp := make([]bool, len(nums)+1)
    dp[0] = true
    if nums[0] == nums[1] {
        dp[2] = true
    }
    for i := 3; i < len(nums)+1; i++ {
        // Condition 1
        if nums[i-1] == nums[i-2] {
            dp[i] = dp[i] || dp[i-2] 
        }
        // Condition 2
        if nums[i-1] == nums[i-2] && nums[i-2] == nums[i-3] {
            dp[i] = dp[i] || dp[i-3] 
        }
        // Condition 3
        if nums[i-1]-2 == nums[i-2]-1 && nums[i-1]-2 == nums[i-3] {
            dp[i] = dp[i] || dp[i-3]
        }
    }
    return dp[len(dp)-1]
}
