row = [0]*len(grid)
        col = [0]*len(grid[0])
        count = 0
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if grid[i][j] == 1:
                    row[i] += 1
                    col[j] += 1
                    count += 1
        for i in range(len(grid)):
            for j in range(len(grid[i])):
                if grid[i][j] == 1 and row[i] == 1 and col[j] == 1:
                    count -= 1
        return count
