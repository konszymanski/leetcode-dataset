class Solution:
    def nthUglyNumber(self, n: int) -> int:
        h = [1]
        d = {}
        while n-1:
            x = heapq.heappop(h)
            if x*2 not in d:
                heapq.heappush(h, x*2)
                d[x*2] = 1
            if x*3 not in d:
                heapq.heappush(h, x*3)
                d[x*3] = 1
            if x*5 not in d:
                heapq.heappush(h, x*5)
                d[x*5] = 1
            n -= 1
        return heapq.heappop(h)
