class Solution:
    def frequencySort(self, nums: List[int]) -> List[int]:
        d={}
        nums.sort()
        nums=nums[::-1]
        for i in nums:
            if i not in d:
                d[i]=0
            d[i]+=1
        ans=[]
        d=dict(sorted(d.items(),key=lambda x:x[1]))
        for i,j in d.items():
            while d[i]!=0:
                ans.append(i)
                d[i]-=1
        return ans
