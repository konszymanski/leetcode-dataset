class Solution:
    def canFormArray(self, arr: List[int], pieces: List[List[int]]) -> bool:
        d = {p[0]: p for p in pieces}
        i = 0
        while i < len(arr):
            if arr[i] not in d or len(arr) - i < len(d[arr[i]]) or arr[i: i+len(d[arr[i]])] != d[arr[i]]:
                return False
            i += len(d[arr[i]])
        return True
