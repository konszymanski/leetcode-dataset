class Solution:
    def duplicateZeros(self, arr: List[int]) -> None:
        w = len(arr) - 1 + arr.count(0)  # \'write to\' index
        
        for r in range(len(arr) - 1, -1, -1): # \'read from\' index
            if arr[r] == 0:  # insert extra 0
                if w < len(arr):
                    arr[w] = 0
                w -= 1
                
            if w < len(arr):
                arr[w] = arr[r]
            w -= 1
