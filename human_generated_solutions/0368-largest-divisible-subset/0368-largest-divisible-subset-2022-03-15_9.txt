def largestDivisibleSubset(self, nums: List[int]) -> List[int]:
        nums.sort()
        length = len(nums)
        
        dp = [[nums[i]] for i in range(length)]
        
        for i in range(length):
            for j in range(i):
                if nums[i]%nums[j]==0 and len(dp[j])+1>len(dp[i]):
                    dp[i] = dp[j] + [nums[i]]
                    
        return max(dp, key=len)
