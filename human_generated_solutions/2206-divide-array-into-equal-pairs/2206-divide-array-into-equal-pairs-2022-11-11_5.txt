#1- by hash table

class Solution:
    def divideArray(self, nums: List[int]) -> bool:
        hash=set()
        for i in nums:
            if i in hash:
                hash.remove(i)
            else:
                hash.add(i)
        if len(hash)==0:
            return True
        else:
            return False
