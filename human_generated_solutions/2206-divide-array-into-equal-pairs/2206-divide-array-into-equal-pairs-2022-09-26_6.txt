class Solution:
    def divideArray(self, nums: List[int]) -> bool:
        if len(nums)%2!=0:
            return False
        n=len(nums)
        nums.sort()
        p1,p2=[],[]
        for i in range(n):
            if i%2==0:
                p1.append(nums[i])
            else:
                p2.append(nums[i])
        for i in range(len(p1)):
            if p1[i]!=p2[i]:
                return False
        return True
