# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, val=0, next=None):
#         self.val = val
#         self.next = next
class Solution:
    def numComponents(self, head: Optional[ListNode], nums: List[int]) -> int:
        itr=head
        c=0
        s=set(nums)
        while itr:
            if itr.val in s and (itr.next==None or itr.next.val not in s):
                c+=1
            itr=itr.next
        return c
