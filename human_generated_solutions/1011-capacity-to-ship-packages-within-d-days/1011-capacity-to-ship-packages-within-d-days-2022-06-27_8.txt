class Solution:
    def shipWithinDays(self, weights: List[int], days: int) -> int:
		def feasable(capacity):
            required_days = 1
            total = 0
            
            for weight in weights:
                total += weight
                
                if total > capacity: # too heavy, wait for the next day
                    required_days += 1
                    total = weight
                    
                    if required_days > days: # cannot ship within D days
                        return False 
                    
                    
            return True
            
        # left = max(weights) as otherwise the conveyor belt couldn\'t ship the heaviest package
		# right = sum(weights) then we can ship all packages in just one day.
        left, right = max(weights), sum(weights)
        
        while left < right: 
            mid = (left + right) // 2
            
            if feasable(mid):
                right = mid
            else:
                left = mid + 1
                
        return left
