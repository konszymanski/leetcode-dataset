class Solution:
    def minSwap(self, nums1: List[int], nums2: List[int]) -> int:
        # swap means the minimum swaps if we swap A[i] and B[i] for ith element
        # fix means for the minimum swaps if we DONOT swap A[i] and B[i] for ith element
        swap, fix, sz = 1, 0, len(nums1)
        for i in range(1, sz):
            # In this case, the ith manipulation should be same as the i-1th manipulation
            if nums1[i - 1] >= nums2[i] or nums2[i - 1] >= nums1[i]:
                swap += 1
            # In this case, the ith manipulation should be the opposite of the i-1th manipulation
            elif nums1[i - 1] >= nums1[i] or nums2[i - 1] >= nums2[i]:
                swap, fix = fix, swap; swap += 1
            # Either swap or fix is OK. Let\'s keep the minimum one
            else:
                minimum = min(swap, fix)
                swap, fix = minimum + 1, minimum
        return min(swap, fix)
