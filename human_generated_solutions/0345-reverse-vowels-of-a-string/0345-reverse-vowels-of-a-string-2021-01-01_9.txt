class Solution:
    def reverseVowels(self, s: str) -> str:
        stack = [i for i in s if i in "aeiouAEIOU"]
        res = []
        for i in s:
            if(i in "aeiouAEIOU"):
                res.append(stack.pop())
            else:
                res.append(i)
        return("".join(res))
