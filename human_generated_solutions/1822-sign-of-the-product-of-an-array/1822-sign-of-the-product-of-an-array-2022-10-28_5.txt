1. If nums has 0 it means the result of multiplication is guranteed to be 0
2. Otherwise count the number of negative signs
3. If the number of negative signs is even it means they are all cancelled during multiplication and we can return 1 else -1


class Solution:
    def arraySign(self, nums: List[int]) -> int:
        if 0 in nums:
            return 0
        
        count = 0
        
        for n in nums:
            if n < 0:
                count += 1
        
        return 1 if count % 2 == 0 else -1
