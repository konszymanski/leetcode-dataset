class Solution:
    def searchMatrix(self, matrix, target):
        m = len(matrix)
        if m < 1:
            return False
        n = len(matrix[0])
        if n < 1:
            return False
        i = 0
        while i < m and target >= matrix[i][0]:
            low, high = 0, n
            while low < high:
                mid = (low+high)//2
                if target > matrix[i][mid]:
                    low = mid + 1
                elif target < matrix[i][mid]:
                    high = mid
                else:
                    return True
            i += 1
        return False
