class Solution:
    def countCollisions(self, directions: str) -> int:
        #collsion are only possible in following cases
        #SL,RL,RS
        
        n=len(directions)
        total=0
        right=0
        s=False
        for j in range(n):
            
            #When S is found all previous R changes to S.
            
            if directions[j]==\'S\':
                total+=right
                right=0
                s=True
                
            #when L is found then:
              #check if there is any previous R,if yes then change them to S.
              #check if there is any previous S,if yes then total+=1
            
            elif directions[j]==\'L\':
                if right:
                    total+=right+1
                    s=True
                    right=0
                elif s:
                    total+=1
                    
            else:
                right+=1
                
        return total
