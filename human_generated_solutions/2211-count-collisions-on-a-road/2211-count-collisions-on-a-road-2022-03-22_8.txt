class Solution(object):
    def countCollisions(self, directions):
        """
        :type directions: str
        :rtype: int
        """
        ans = 0
        carsFromRight = 0
        carsFromLeft = 0
        for car in directions:
            if car == \'R\':
                carsFromRight += 1
            if car == \'S\':
                ans += carsFromRight
                carsFromRight = 0
            if car == "L":
                ans += carsFromRight
                carsFromRight = 0

        for car in reversed(directions):
            if car == "L":
                carsFromLeft += 1
            if car == \'S\':
                ans += carsFromLeft
                carsFromLeft = 0
            if car == \'R\':
                ans += carsFromLeft
                carsFromLeft = 0

        return ans
