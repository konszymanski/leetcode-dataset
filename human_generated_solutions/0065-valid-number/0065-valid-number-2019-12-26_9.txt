class Solution:
    def isNumber(self, s: str) -> bool:
        s = s.lower().replace(\'+\',\'-\').strip()
        if not s: return False
        if not (s[-1].isnumeric() or s[-1] == \'.\'): return False
        if not (s[0].isnumeric() or s[0] in \'-.\'): return False
        if s[0] == \'-\': s = s[1:]
        s, p, e = s.replace(\'e-\',\'\'), s.find(\'.\'), s.find(\'e\')
        if \'e\' in [s[0],s[-1]] or s.count(\'.\') > 1 or s.count(\'e\') > 1: return False
        if s.isnumeric(): return True
        if len(s) <= 1: return False
        if any(c in s for c in \'abcdfghijklmnopqrstuvwxyz- \'): return False
        if s[:2] == \'.e\' or not (-1 in [e,p] or e > p): return False
        return True
