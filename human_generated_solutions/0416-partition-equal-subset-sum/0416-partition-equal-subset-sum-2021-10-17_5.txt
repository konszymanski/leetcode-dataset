class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        s = sum(nums)
        if s%2 != 0: return False
        target = s//2
        memo = {}
        n = len(nums)
        def canPartRec(nums,idx,target):
            if target ==0: 
                return True
            if idx >= n or target < 0: return False
            if (idx,target) not in memo:
                memo[(idx,target)] = canPartRec(nums,idx+1,target-nums[idx]) or canPartRec(nums,idx+1,target) 
            return memo[(idx,target)]
        return canPartRec(nums,0,target)
