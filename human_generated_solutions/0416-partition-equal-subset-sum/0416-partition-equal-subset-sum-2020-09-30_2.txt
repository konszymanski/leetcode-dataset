class Solution:
    def canPartition(self, nums: List[int]) -> bool:
        if not nums:
            return True
        n = len(nums)
        if sum(nums) % 2 != 0:
            return False
        target = sum(nums)//2
        answer = 0
        
        def helper(total, i):
            nonlocal nums, answer
            if i == len(nums):
                return
            if total == 0:
                answer += 1
            helper(total-nums[i], i+1)
            helper(total, i+1)
        
        helper(target, 0)
        return answer
