def canPartition(self, nums: List[int]) -> bool:  
		if sum(nums) % 2 != 0:
				return False
			trgt = sum(nums)//2
			# res = []
			stack = [(0, [])]
			while stack:
				indx, subset = stack.pop()
				if sum(subset) == trgt:
					return True

				for i in range(indx, len(nums)):
					stack.append((i+1, subset+[nums[i]]))
			return False
