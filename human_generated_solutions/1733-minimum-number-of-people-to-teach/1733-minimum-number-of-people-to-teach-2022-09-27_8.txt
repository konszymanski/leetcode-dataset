class Solution:
    def minimumTeachings(self, n: int, languages: List[List[int]], friendships: List[List[int]]) -> int:
        hmap = {}
        for i, l_list in enumerate(languages):
            hmap[i+1] = set()
            for item in l_list:
                hmap[i+1].add(item)
        
        mis_friend = set()
        
        for friend in friendships:
            a, b = friend[:]
            if not len(hmap[a].intersection(hmap[b])):
                mis_friend.add(a)
                mis_friend.add(b)
                
        res = float(\'inf\')
        for lang in range(1, n+1):
            count = 0
            for fri in mis_friend:
                if lang not in hmap[fri]:
                    count+=1
            res = min(res, count)
        return res
