class Solution:
    def idealArrays(self, n, m):
        mod, ans, comb = 10**9 + 7, m, 1
        dp = [0] + [1] * m
        
        for i in range(1, min(n, 14)):
            for j in range(m, 0, -1):
                if dp[j] == 0: continue
                for k in range(j << 1, m + 1, j):
                    dp[k] += dp[j]
                dp[j] = 0
            comb = (comb * (n-i)) // i
            ans = (ans + sum(dp) * comb) % mod

        return ans
