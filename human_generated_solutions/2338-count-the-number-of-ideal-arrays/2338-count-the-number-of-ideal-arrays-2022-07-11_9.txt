def idealArrays(self, n: int, maxValue: int) -> int:
        M = 10 ** 9 + 7
        A = [[0] * (maxValue+1) for _ in range(15)]
        res = 0
        comb = A[0][1] = A[1][1] = 1
        for j in range(1, min(15, n+1)):
            for i in range(1, maxValue+1):
                for k in range(i*2, maxValue+1, i):
                    A[j][k] = (A[j][k]+A[j-1][i]) % M
            res = (res + sum(A[j]) * comb % M) % M
            comb = comb * (n-j) * pow(j,M-2,M) % M 
        return res
