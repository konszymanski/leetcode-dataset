def optimalDivision(self, nums):
        """
        :type nums: List[int]
        :rtype: str
        """
        if len(nums)<3:
            result=''.join([str(num)+'/' for num in nums])
            return result[:-1]
        result=[]
        result.append(str(nums[0])+'/(')
        for i in range(1,len(nums)):
            result.append(str(nums[i]))
            result.append('/')
        result=result[:-1]
        result.append(')')
        return ''.join(result)
