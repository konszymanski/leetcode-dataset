# Definition for a binary tree node.
# class TreeNode(object):
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right
class Solution(object):
    def removeLeafNodes(self, root, target):
        """
        :type root: TreeNode
        :type target: int
        :rtype: TreeNode
        """
        stack = [(root, None, "root", False)]
        while(stack != []):
            node, parent, state, backtrack = stack.pop()
            
            if(node.left == None and node.right == None):
                if(node.val == target):
                    if(state == "left"):
                        parent.left = None
                    elif(state == "right"):
                        parent.right = None
                    elif(state == "root"):
                        root = None
                        return

            if(node.val == target):
                if(not backtrack):
                    stack.append((node, parent, state, True))
            
            if(not backtrack):
                if(node.left):
                    stack.append((node.left, node, "left", False))
                if(node.right):
                    stack.append((node.right, node, "right", False))
        return root
