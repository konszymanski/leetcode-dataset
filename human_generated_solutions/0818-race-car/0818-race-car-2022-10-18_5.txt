class Solution:
    def racecar(self, target: int) -> int:

        q = deque([(\'\', 0,  0, 1)])
        visited = set()
        
        while q:
            path, moves, pos, speed = q.popleft()
            
            if pos == target:
                # return path if path is needed in the question
                print(path)
                return moves
            
            if (pos, speed) in visited:
                continue
            visited.add((pos, speed))

            q.append((path + \'A\', moves + 1,  pos + speed, speed*2))
        
            if (pos + speed > target and speed > 0) or (pos + speed < target and speed < 0):
                new_speed = -1 if speed > 0 else 1
                q.append((path + \'R\', moves + 1, pos, new_speed))
