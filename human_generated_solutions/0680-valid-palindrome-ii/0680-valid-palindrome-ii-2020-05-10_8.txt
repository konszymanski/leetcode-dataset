class Solution:
    def validPalindrome(self, s: str) -> bool:
        def is_valid(i, j, s, d=1):
            while i < j:
                if s[i] != s[j]:
                    return is_valid(i+1, j, s, d-1) or is_valid(i, j-1, s, d-1) if d else False
                i, j = i+1, j-1
            return True
        return is_valid(0, len(s)-1, s)
