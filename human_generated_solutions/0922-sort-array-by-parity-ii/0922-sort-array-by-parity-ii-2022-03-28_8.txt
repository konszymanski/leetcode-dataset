class Solution:
    def sortArrayByParityII(self, nums: List[int]) -> List[int]:
        j = 1 # j is used to check all the odd indexes
        
        # check all the even indexes with i
        for i in range(0, len(nums), 2):
            # nums[i] is odd while it should be even
            if nums[i] & 1:
                # look for the next even number in an odd index
                # it\'s guaranteed to exist because we already found
                # ad odd number with even index
                while (j < len(nums)) and (nums[j] & 1):
                    j += 2
                # now swap them
                # by putting even numbers in positions with even indexes
                # we are also putting odd numbers in odd indexes as a
                # consequence of the swap, all in one pass
                nums[i], nums[j] = nums[j], nums[i]
        return nums
