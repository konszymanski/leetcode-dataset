class Solution:
    def sortArrayByParityII(self, nums: List[int]) -> List[int]:
        odd = [x for x in nums if x%2!=0]
        even = [x for x in nums if x%2==0]
        
        nums = []
        for x,y in zip(odd,even):
            nums.append(y)
            nums.append(x)
        return nums
