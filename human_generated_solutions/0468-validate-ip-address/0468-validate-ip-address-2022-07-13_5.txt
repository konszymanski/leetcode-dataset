class Solution:
    def validIPAddress(self, queryIP: str) -> str:
        new_string = queryIP.replace(":" , ".")
        formatted_string = new_string.split(".")
        res = "Neither"
        if len(formatted_string) == 4 or len(formatted_string) == 8:
            if len(formatted_string) == 4:
                for item in formatted_string:
                    for character in item:
                        if not character.isdigit():
                            return "Neither"
                    if len(item) == 0 or len(item) > 3 or int(item) >255 or (len(item)!=1 and item[0] == "0"):
                        return "Neither"
                res = "IPv4"

            if len(formatted_string) == 8:
                for item in formatted_string:
                    if len(item) > 4 or not item:
                        return "Neither"
                    for character in item:
                        if character.isalpha() and character.lower() not in "abcdef":
                            return "Neither"
                res = "IPv6"
        return res
