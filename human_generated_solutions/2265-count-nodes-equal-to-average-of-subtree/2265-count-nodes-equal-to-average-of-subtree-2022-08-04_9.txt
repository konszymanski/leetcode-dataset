class Solution:
    def averageOfSubtree(self, root: Optional[TreeNode]) -> int:
        self.count = 0
        def dfs(root):
			# if node is None, sum and count will be 0, 0
            if not root:
                return 0, 0
            
            if not root.left and not root.right:
                self.count += 1
                return root.val, 1
            
			# Sum of nodes in subtree, and count of nodes
            left, nl = dfs(root.left)
            right, nr = dfs(root.right)
            
            if (left + right + root.val) // (nl + nr + 1)   == root.val:
                self.count += 1
            return left + right + root.val, (nl + nr + 1)
        dfs(root)
        return self.count
