def averageOfSubtree(self, root: Optional[TreeNode]) -> int:
        self.output=0 #count number of nodes with values same as average
        
        def recur(root):
            if root:
                left_count,left_s=recur(root.left) #getting number of nodes and sum of left sub tree
                right_count,right_s=recur(root.right) #getting number of nodes and sum of right sub tree

                avg=(left_s+right_s+root.val)//(left_count+right_count+1) #calculating the average
                if root.val==avg:
                    self.output+=1
                
                return left_count+right_count+1,left_s+right_s+root.val #returning the sum and count of the given subtree. 
            return 0,0 # if root is none then return sum=0 and count=0
        
        recur(root) #calling the function
        return self.output
