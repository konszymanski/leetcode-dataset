class Solution:
    def checkArithmeticSubarrays(self, nums: List[int], l: List[int], r: List[int]) -> List[bool]:
        res = []
        for i in range(len(l)):
            temp = nums[l[i]:r[i]+1]
            temp.sort()
            for i in range(2, len(temp)):
                if temp[i]-temp[i-1] != temp[i-1] - temp[i-2]:
                    res.append(False)
                    break
            else:
                res.append(True)
        return res
