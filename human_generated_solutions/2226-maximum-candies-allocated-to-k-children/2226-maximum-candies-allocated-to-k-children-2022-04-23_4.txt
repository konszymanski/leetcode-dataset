class Solution:
    """
    approach: 
    the problem can be tackled using binary search 
    max_candies that can be allocated to children = max(candies)
    min_candies that can be allocated to children = 1
    
    we will apply binary search on finding the optimal values of candies that can be
    distributed to children. 
    """
    def check_fulfilment(self, candies, elem, k):
        count = 0
        for candy in candies:
            count+= candy // elem
        if count >= k:
            return True
        return False
    def maximumCandies(self, candies: List[int], k: int) -> int:
        low = 1
        high = max(candies)
        if sum(candies) < k:
            return 0
        while low <= high:
            mid = (low+high) // 2
            # check if I can fulfil that order
            status = self.check_fulfilment(candies, mid, k)
            if status:
                low = mid+1
            else:         
                high = mid-1
        # low is the maximum number of candies that can be  distributed
        return high
