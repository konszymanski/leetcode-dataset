def maximumCandies(self, candies: List[int], k: int) -> int:
        
        def can_take(target):
            count = 0
            for candy in candies:
                if candy >= target:
                    count += (candy // target) # Find out how many children this pile can feed :) 
            return count >= k   # Make sure that we have enough to feed all the children
        
        l, r = 1, max(candies)
        res = 0
        while l <= r:
            m = (l + r) // 2
            if can_take(m):
                res = m
                l = m + 1
            else:
                r = m - 1
        return res
