class Solution:
    def constructArray(self, n: int, k: int) -> List[int]:
        digits, res = {*range(1, n)}, [n]
        while True:
            if res[-1] - k in digits:
                digits -= {res[-1] - k}
                res.append(res[-1] - k)
                k = -(k - 1) if k > 0 else -(k + 1)
            else:    
                break
        return res + sorted(list(digits), reverse=True)
