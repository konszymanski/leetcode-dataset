class Solution:
    def countPrimeSetBits(self, L, R):
        top = []
        for i in range(L,R+1):
            bit_rep = bin(i)[2:]
            ones = len([item for item in bit_rep if item == "1"])
            top.append(self.dictionary(ones))
        return (len([item for item in top if item == True ]))
        
    def dictionary(self,value):
        if value == 1:
            return False
        for i in range(2,value):
            if value%i ==0:
                return False 
        return True
