class Solution(object):
    def countPrimeSetBits(self, L, R):
        """
        :type L: int
        :type R: int
        :rtype: int
        """
        res = 0
        for i in range(L,R+1):
            bits= bin(i).count(\'1\')
            if bits!=1 and math.factorial(bits - 1)  % bits == bits - 1:
                res+=1
        return res
