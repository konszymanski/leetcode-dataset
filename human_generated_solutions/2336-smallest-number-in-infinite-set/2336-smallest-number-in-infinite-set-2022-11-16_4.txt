import heapq

class SmallestInfiniteSet:

    def __init__(self):
        self.hq = []
        self.numSet = set()
        
        for num in range(1, 1001):
            heapq.heappush(self.hq, num)
            self.numSet.add(num)

    def popSmallest(self) -> int:
        num = heapq.heappop(self.hq)
        self.numSet.discard(num)
        
        return num
        

    def addBack(self, num: int) -> None:
        if num in self.numSet:
            return
        
        heapq.heappush(self.hq, num)
        self.numSet.add(num)
