class Solution:
    def smallestDistancePair(self, nums, k: int) -> int:

        Given an array of integers (nums) and a parameter k,
        this program uses binary search to determine the kth
        smallest pair distance within nums.

        :param nums: array of non-negative integers
        :type nums: list[int]
        :param k: defines order of smallest pair distance to
                  find in nums
        :type k: int
        :return: value of kth smallest pair distance in nums
        :rtype: int
        """

            distances less than or equal to distance.

            :type distance: int

        right = nums[-1] - nums[0]
