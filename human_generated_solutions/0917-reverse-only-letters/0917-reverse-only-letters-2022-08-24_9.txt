const reverseOnlyLetters = (s) => {
		let i = 0;
		let result = \'\';
		const arr = [];

		while (i < s.length) {
			const isFirst = (s[i] >= \'A\' && s[i] <= \'Z\') || (s[i] >= \'a\' && s[i] <= \'z\');

			if (isFirst) arr.push(s[i])
			i++;
		}
    
		for (const val of s) {
			const alphaCheck = (val >= \'A\' && val <= \'Z\') || (val >= \'a\' && val <= \'z\');
			result += !alphaCheck ? val : arr.pop();
		}
    
    
		return result;
};
