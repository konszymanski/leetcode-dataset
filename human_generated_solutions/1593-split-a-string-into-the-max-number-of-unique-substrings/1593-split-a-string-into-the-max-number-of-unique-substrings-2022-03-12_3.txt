class Solution {
public:
    int maxUniqueSplit(string s) {
        unordered_map<string,int>mp;
        for(int i=0; i<s.length(); i++){
            string temp="";
            temp+=s[i];
            while(mp[temp]!=0 && i<s.length()){
                temp+=s[++i];
            }
            if(mp[temp] != 0){
                mp[temp]++;    
            }
            
        }
        return mp.size();
    }
};
