class Solution {
    fun countStudents(students: IntArray, sandwiches: IntArray): Int {
        var m = mutableMapOf<Int, Int>()
        for (x in students)
            m[x] = 1 + (m[x] ?: 0)
        for (x in sandwiches) {
            if (m[x] ?: 0 == 0)
                break
            m[x] = -1 + m[x]!!
        }
        return m.values.sum()
    }
}
