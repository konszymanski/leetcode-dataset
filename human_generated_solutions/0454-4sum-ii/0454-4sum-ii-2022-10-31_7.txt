class Solution:
    def fourSumCount(self, nums1: List[int], nums2: List[int], nums3: List[int], nums4: List[int]) -> int:
        dc1=defaultdict(lambda:0)
        for a in nums1:
            for b in nums2:
                dc1[a+b]+=1
        ans=0
        for c in nums3:
            for d in nums4:
                ans+=dc1[-c-d]
        return ans
