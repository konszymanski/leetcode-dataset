def smallestDivisor(self, nums: List[int], threshold: int) -> int:
	def isValid(num):
		temp = 0
		for i in nums:
			temp += ceil(i/num)
		return temp <= threshold
	i, j = 1, max(nums) + 1   
	while(i <= j):
		mid = i + (j-i)//2
		if(isValid(mid)):
			j = mid-1
			ans = mid
		else:
			i = mid+1
	return ans
