class Bank:

    def __init__(self, balance: List[int]):
        self.balance = balance
        self.size = len(balance)
    
    def checkValidAccount(self, account):
        if account-1<0 or account-1>=self.size:
            return False
        return True

    def transfer(self, account1: int, account2: int, money: int) -> bool:
        if not self.checkValidAccount(account1) or not self.checkValidAccount(account2) or self.balance[account1-1] < money:
            return False
        
        self.balance[account1-1] -= money
        self.balance[account2-1] += money
        return True
        

    def deposit(self, account: int, money: int) -> bool:
        if not self.checkValidAccount(account):
            return False
        self.balance[account-1]+=money
        return True
        

    def withdraw(self, account: int, money: int) -> bool:
        if not self.checkValidAccount(account) or self.balance[account-1] < money:
            return False

        self.balance[account-1]-=money
        return True
