class Solution:
    def canReorderDoubled(self, arr: List[int]) -> bool:
        n = len(arr)
        if n % 2 == 1 or n == 1:
            return []

        numbers = Counter(arr)
        
        for i in sorted(arr):
            
            double = i*2
            if numbers.get(i, 0) > 0 and numbers.get(double, 0) > 0:
                numbers[i] -= 1
                numbers[double] -= 1
        for i in numbers.values():
            if i > 0:
                return False
        return True
