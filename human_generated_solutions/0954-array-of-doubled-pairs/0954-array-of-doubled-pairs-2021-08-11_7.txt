class Solution:
    def canReorderDoubled(self, arr: List[int]) -> bool:
        
        # key: number
        # value: occurrence
        num_occ_dict = Counter(arr)
        
        # scan each number in arr, and sorted in absolute value
        for number in sorted(arr, key = abs ):
            
            # find all double pairs as many as possible
            while num_occ_dict[number] > 0 and num_occ_dict[2*number] > 0:
                
                # update occurrence of current double pair
                num_occ_dict[number] -= 1
                num_occ_dict[2*number] -= 1 
        
        # Accept if all number are paired
        # Reject, otherwise
        return all( occurrence == 0 for occurrence in num_occ_dict.values() )
