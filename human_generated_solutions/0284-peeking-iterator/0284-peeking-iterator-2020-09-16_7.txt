class PeekingIterator:
    def __init__(self, iterator):
        self._iterator = iterator
        self.cache = None
        
    def peek(self):
        if self.cache is None: self.cache = self._iterator.next()
        return self.cache

    def next(self):
        if self.cache is not None: 
            val, self.cache = self.cache, None
            return val
        else: return self._iterator.next()
        
    def hasNext(self):
        return self._iterator.hasNext() or self.cache is not None
