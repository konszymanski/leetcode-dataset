def waysToSplitArray(self, nums: List[int]) -> int:

    left_sum = nums[0]
    right_sum = sum(nums[1: ])
    count = 0
    queue = deque(nums[1:])

    while queue:
        if left_sum >= right_sum:
            count += 1

        ele = queue.popleft()

        # add to left_sum
        left_sum += ele
        # remove from right
        right_sum -= ele

    return count
