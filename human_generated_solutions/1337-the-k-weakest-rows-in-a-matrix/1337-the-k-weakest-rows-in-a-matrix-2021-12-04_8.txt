from heapq import heappush, heappop

class Solution:
  def kWeakestRows(self, mat: List[List[int]], k: int) -> List[int]:
    def binarySearch(row, isLeftBias):
      res = -1
      left, right = 0, len(row)-1
      while left <= right:
        mid = left + (right-left)//2
        if row[mid] == 1:
          res = mid
          if isLeftBias:
            right = mid - 1
          else:
            left = mid + 1
        else:
          right = mid - 1
      return res
    
    maxHeap = []
    for i in range(len(mat)):
      left = binarySearch(mat[i], True)
      if left == -1:
        heappush(maxHeap, (0, -i, i))
      else:
        right = binarySearch(mat[i], False)
        heappush(maxHeap, (-(right-left+1), -i, i))
      
      while len(maxHeap) > k:
        heappop(maxHeap)

    res = [0]*k
    for i in range(k-1, -1, -1):
      res[i] = heappop(maxHeap)[2]    
    return res
