class Solution:
    # just check if k lies between [mink, maxk].
    # max k is the length of s, cuz every character of s can be a 
    # Palindrome. min k is the count of odd occurence of characters.
    # say we have character \'t\' appear odd amount of times, then 
    # \'t\' must have a Palindrome on its own with \'t\' at the center
    # position of the palindrome. It must have a palindrome that looks
    # like \'...t...\'. for every odd occurence of character, we need to
    # make a separate palindrome for it. so the min k value is the 
    # count of odd occurence of characters.
    def canConstruct(self, s: str, k: int) -> bool:
        return k >= sum(ct & 1 for ct in Counter(s).values()) and k <= len(s)
