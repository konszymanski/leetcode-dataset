class Solution:
    def getMaximumXor(self, nums: List[int], maximumBit: int) -> List[int]:
        max_possible=2**maximumBit-1        #max possible ans
        array_xor=0                         #current array xor
        n=len(nums)
        
        ans=[]                             #stores ans of queries
        
        #finding xor of nums
        for i in nums:
            array_xor^=i
            
        ans.append(max_possible^array_xor)
        
        for i in range(n-1,0,-1):
            array_xor^=nums[i]              #removing largest values from ans
            ans.append(array_xor^max_possible)
        
        return ans
