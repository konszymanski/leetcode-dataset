class Solution:
    def getMaximumXor(self, nums: List[int], maximumBit: int) -> List[int]:
        li=[]
        l=2**maximumBit-1
        for i in range(1,len(nums)):
            nums[i]=nums[i]^nums[i-1]
        nums=nums[::-1]
        for i in range(len(nums)):
            li.append(nums[i]^l)
        return li
