class Solution:
    def count_zeros(self,num):
        d=5
        zeros=0
        while d<=num:
            zeros+=num//d
            d*=5
        return zeros
		
    def preimageSizeFZF(self, K: int) -> int:
        start=0
        end=int(5e9)
        while start<=end:
            mid=start+(end-start)//2
            temp=self.count_zeros(mid)
            if temp==K:
                return 5
            if temp>K:
                end=mid-1
            else:
                start=mid+1
        return 0
