class Solution:
    def findzeroes(self,num):
        # This part takes log(n) time
        tmp=0
        val=5
        while val<=num:
            tmp+=num//val
            val*=5
        return tmp
    def preimageSizeFZF(self, k: int) -> int:

        if k==0:return 5
        high=5
        while True:
            tmp=self.findzeroes(high)
            if tmp==k:return 5
            if tmp>k:break
            high*=5
        low=high//5
        while low<=high:
            mid=(low+high)//2
            tmp=self.findzeroes(mid)
            if tmp==k:return 5
            if tmp<k:low=mid+1
            else:high=mid-1
        return 0
