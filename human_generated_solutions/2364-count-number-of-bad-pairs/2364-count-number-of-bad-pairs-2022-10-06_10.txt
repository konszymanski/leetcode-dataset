class Solution:
    def countBadPairs(self, nums: List[int]) -> int:
        count = Counter()
        res = 0
        nums = [n - i for i, n in enumerate(nums)]
        for i, n in enumerate(nums):
            res += i - count[n]
            count[n] += 1
        return res
    # Time: O(n)
    # Space: O(n)
