import functools

class Solution:
    def numDistinct(self, s: str, t: str) -> int:
        @functools.lru_cache(None)
        def dp(i, j):
            if j == len(t):
                return 1
            if i == len(s):
                return 0
            count = 0
            if s[i] == t[j]:
                count += dp(i+1, j+1)
            count += dp(i+1, j)
            return count
        return dp(0, 0)
