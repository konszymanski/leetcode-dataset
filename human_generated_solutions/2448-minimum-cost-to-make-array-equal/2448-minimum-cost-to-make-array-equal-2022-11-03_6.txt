/**
 * @param {number[]} nums
 * @param {number[]} cost
 * @return {number}
 */
var minCost = function(nums, cost) {
    const n = nums.length;
    
    let res = Infinity;
    for(let i = 0; i < n; i++){
        const target = nums[i];
        let total = 0;
        for(let j = 0; j < n; j++){
            total += Math.abs(target - nums[j]) * cost[j];
        }
        res = Math.min(res, total);
    }
    return res;
};
