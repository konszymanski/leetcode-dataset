class Solution:
    def numFactoredBinaryTrees(self, arr: List[int]) -> int:
        arr.sort()
        
        mem = {}
        for parent in arr:
            mem[parent] = 1
            
            for child in arr:
                other_child = parent/child
                if other_child in mem:
                    mem[parent] += (mem[other_child] * mem[child]) % 1000000007

        
        
        return (sum(mem.values()) % 1000000007)
