matrix = []
        rows, cols = len(box), len(box[0])
        
		# Creation of the new matrix skeleton
        for col in range(cols):
            temp = []
            for row in range(rows):
                temp.append(" ")
            matrix.append(temp)
        
		# Rotating the box
        for row in range(cols):
            for col in range(rows):
                matrix[row][col] = box[rows - col - 1][row]
        
        rows, cols = cols, rows
        
		# Main logic
        for col in range(cols):
            blank, stone = rows - 1, rows - 1
			# Iterating till all the rows are iterated over
            while stone >= 0:
				# Iterating the column bottom-up till a stone is found
                while stone >= 0 and matrix[stone][col] != \'#\':
					# If an obstacle is found make its previous index the starting point for the search for blanks
                    if matrix[stone][col] == \'*\':
                        blank = stone - 1
                    stone -= 1
				# Iterating the column bottom-up till a blank is found
                while blank >= 0 and matrix[blank][col] != \'.\':
                    blank -= 1
				# Check for the index values for out of bounds and mainly if the index for stone is lesser than the index for blank
                if blank >= 0 and stone >= 0 and stone < blank:
                    matrix[blank][col], matrix[stone][col]  = matrix[stone][col], matrix[blank][col]
                stone -= 1
        return matrix
