class Solution(object):
    def PredictTheWinner(self, nums):
        summ = sum(nums)
        score = self.my_score(nums, summ)
        
        # If our score is greater than equal to opponents score, return True
        return True if score >= summ - score else False
    
    def my_score(self, nums, summ):
        if len(nums) <= 2:
			# If there are only 2 numbers, obviously we\'ll choose the greatest number
            return max(nums)
        
        # my_score function will return the score of the opponent
        choose_first = self.my_score(nums[1:], summ-nums[0])
        choose_last = self.my_score(nums[:-1], summ-nums[-1])
        
        return max(summ - choose_first, summ - choose_last)
