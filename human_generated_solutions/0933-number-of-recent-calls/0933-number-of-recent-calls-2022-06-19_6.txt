class RecentCounter:
#     O(n) || O(n)
# Runtime: 310ms 78.53% || memory: 19.6mb 46.67%

    def __init__(self):
        self.queue = []

    def ping(self, t: int) -> int:
        self.queue.append(t)
        
        while (t - self.queue[0]) > 3000:
            self.queue.pop(0)
            
        return self.queue.__len__()
