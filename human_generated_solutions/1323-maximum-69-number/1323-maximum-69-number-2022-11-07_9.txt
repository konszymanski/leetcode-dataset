class Solution {
    public int maximum69Number (int num) {
        //Counting the digits in num
        int n = num;
        int digits = 0;
        while(n!=0){
            digits++;
            n=n/10;
        }
        
        //storing num\'s digits in an Array
        int digi[] = new int[digits];
        for(int i=0;i<digits;i++){
            digi[digits-1-i]=num%10;
            num/=10;
        }
        
        //changing first 6 in num to 9 as it would give the greatest number
        // and if no 6 present no need of changes as its greatest number
        for(int i=0;i<digits;i++){
            if(digi[i]==6)   {digi[i]=9;break;}
        }
        
        //just converting array of num back to an interger res
        int res = 0;
        for(int i=0,exp=digi.length-1;i<digi.length;i++,exp--)
        res+=digi[i]*Math.pow(10, exp);
        
        //returning the answer
        return res;
        
}
}
