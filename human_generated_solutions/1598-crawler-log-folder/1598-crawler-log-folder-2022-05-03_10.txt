# var for counting directories or depth 
dirs = 0
for i in logs:
	if i == "../":
	    # if there is any folder present then remove it and 
		# if no folder present i.e. dirs = 0 then no need to do anything, 
		# you are already in root directory
		if dirs != 0:
			dirs -= 1
	# you have to remain in same directory hence no increment in dirs ..i.e. depth remains same
	elif i == "./":
		continue
	else:
	# increase the dirs count with 1. because a new directory is encountered and 
	# depth of dirs is increased
	#   root / a / b / c / d     --- > depth = 4
	# so if you see everytime we encounter a new directory our depth increases.
		dirs += 1
		
# if directories left then return count of them 
if dirs > 0:
	return dirs
# else if there are 0 directories then return 0
else:
	return 0
