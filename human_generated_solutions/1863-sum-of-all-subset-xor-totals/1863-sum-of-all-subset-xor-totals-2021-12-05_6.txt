class Solution:
    def subsetXORSum(self, nums: List[int]) -> int:
        self.res = 0
        
        def dfs(path, start):
            curSum = 0
            for n in path:
                curSum = curSum ^ n
            self.res += curSum
            
            for i in range(start, len(nums)):
                num = nums[i]
                path.append(num)
                dfs(path, i + 1)
                path.pop()
            
        dfs([], 0)    
        return self.res
