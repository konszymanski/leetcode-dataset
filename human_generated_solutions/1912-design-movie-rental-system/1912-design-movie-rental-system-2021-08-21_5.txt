import sortedcontainers as sc
class MovieRentingSystem:
    
    SearchTuple = namedtuple(\'SearchTuple\', (\'price\', \'shop\'))
    ReportTuple = namedtuple(\'ReportTuple\', (\'price\', \'shop\', \'movie\'))
    SEARCH_COUNT = 5
    REPORT_COUNT = 5

    def __init__(self, n: int, entries: List[List[int]]):
        self.shop_movie_price = defaultdict(lambda: defaultdict(int))
        self.search_dict = defaultdict(sc.SortedList)
        self.report_list = sc.SortedList()
        for shop, movie, price in entries:
            self.search_dict[movie].add(self.SearchTuple(price, shop))
            self.shop_movie_price[shop][movie] = price

    def search(self, movie: int) -> List[int]:
        return [t.shop for t in self.search_dict[movie][:self.SEARCH_COUNT]]
        
    def rent(self, shop: int, movie: int) -> None:
        price = self.shop_movie_price[shop][movie]
        self.search_dict[movie].remove(self.SearchTuple(price, shop))
        self.report_list.add(self.ReportTuple(price, shop, movie))
    
    def drop(self, shop: int, movie: int) -> None:
        price = self.shop_movie_price[shop][movie]
        self.search_dict[movie].add(self.SearchTuple(price, shop))
        self.report_list.remove(self.ReportTuple(price, shop, movie))
        
    def report(self) -> List[List[int]]:
        return [[t.shop, t.movie] for t in self.report_list[:self.REPORT_COUNT]]
