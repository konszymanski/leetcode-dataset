from collections import Counter


class Solution:
    def heightChecker(self, heights):
        counter = Counter(heights)
        
        i = 0
        
        removals = 0
        for height in heights:
            while counter[i] == 0:
                i += 1
            
            if i != height:
                removals += 1
            
            counter[i] -= 1
            
        return removals
