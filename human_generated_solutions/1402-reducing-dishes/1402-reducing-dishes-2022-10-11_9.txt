class Solution:
    def maxSatisfaction(self, satisfaction: List[int]) -> int:
		n = len(satisfaction)
		satisfaction.sort()

		@cache
		def dp(dish, time):
			if dish == n:
				return 0

			cook = satisfaction[dish] * (time + 1) + dp(dish + 1, time + 1)
			skip = dp(dish + 1, time)

			return max(cook, skip)

		return dp(0, 0)
