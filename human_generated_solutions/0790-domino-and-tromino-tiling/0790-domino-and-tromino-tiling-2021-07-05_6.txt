class Solution:
    def numTilings(self, n: int) -> int:
        
        # x_n = x_(n-1) + x_(n-2) + 2 * (x_(n-3) + ... + x_1) + 2
        # ==> x_n = 2 * x_(n-1) + x_(n-3)
        
        res = [1, 1, 2]
        
        if n < 3:
            return(res[n])
        for i in range(3, n+1):
            res.append(2 * res[-1] + res[-3])
        
        return(res[n] % (10**9+7))
