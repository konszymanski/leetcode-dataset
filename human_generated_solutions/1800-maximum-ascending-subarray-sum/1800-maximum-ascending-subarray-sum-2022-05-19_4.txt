class Solution:
    def maxAscendingSum(self, nums: List[int]) -> int:
        temp=nums[0]
        max_sum=0
        for i in range(1,len(nums)):
            if nums[i-1]>=nums[i]:
                if temp>max_sum:
                    max_sum=temp
                temp=0
            temp=temp+nums[i]
        if temp>max_sum:
            return temp
        else:
            return max_sum
