class Solution(object):
    def maxAscendingSum(self, nums):
        res = nums[0]
        result = []
        for i in range(1, len(nums)):
            if nums[i-1] < nums[i]:
                res += nums[i]
            else:
                result.append(res)
                res = nums[i]
        result.append(res)
        return max(result)
