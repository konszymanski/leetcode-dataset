def minKBitFlips(self, nums: List[int], k: int) -> int:
	n = len(nums)
    diff = [0] * (n + 1) 
    flip = curr = 0
    for i, num in enumerate(nums): 
		curr += diff[i]
        if not (num ^ +(curr % 2)):
			if n - i < k:
				return -1
			curr += 1
            diff[i + k] -= 1
            flip += 1
	return flip
