import heapq
class Solution:
    def minGroups(self, intervals: List[List[int]]) -> int:

        intervals.sort()

        n = len(intervals)
        clusters = [intervals[0][1]]
        heapq.heapify(clusters)

        for i in range(1, n):
            if(intervals[i][0] > clusters[0]):
                heapq.heappop(clusters)
                heapq.heappush(clusters, intervals[i][1])
            else:
                heapq.heappush(clusters, intervals[i][1])
        
        return len(clusters)
