class Solution:
    def pseudoPalindromicPaths (self, root: Optional[TreeNode]) -> int:
        ans = 0
        rxor = 0
        
        def traverse(node,rxor):
            nonlocal ans
            rxor = rxor ^ 1<<node.val
			
            # If it is a Leaf Node (Both Child would be None )
			if node.left == None and node.right == None:
                
				if  rxor == 0: ans += 1
				elif rxor & (rxor-1) == 0 : ans += 1
				return

            if node.left : traverse(node.left,rxor)
            if node.right: traverse(node.right,rxor)
            
        traverse(root,rxor)
        return ans
