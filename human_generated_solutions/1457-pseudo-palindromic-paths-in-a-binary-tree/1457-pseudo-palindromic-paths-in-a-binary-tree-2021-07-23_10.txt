class Solution:
	def pseudoPalindromicPaths (self, root: TreeNode, bitck: int = 0) -> int:
		if not root: return 0
		bitck ^= (1 << root.val)
        if not root.left and not root.right:
			cnt = 0
            while bitck > 0:
				bitck &= (bitck-1)
				cnt += 1
            if cnt <= 1: return 1
            return 0
        return self.pseudoPalindromicPaths(root.left, bitck) + self.pseudoPalindromicPaths(root.right, bitck)
