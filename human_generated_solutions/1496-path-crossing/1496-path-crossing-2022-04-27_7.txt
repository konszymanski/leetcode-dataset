class Solution:
    def isPathCrossing(self, path: str) -> bool:
        moves = {"N": 1, "S": -1, "E": 1, "W": -1}
        seen = [(0, 0)]
        
        for move in path:
		
            if move in "EW":
                location = (seen[-1][0] + moves[move], seen[-1][1])
            else:
                location = (seen[-1][0], seen[-1][1] + moves[move])
            
            if location in seen: return True
            else: seen.append(location)
        
        return False
