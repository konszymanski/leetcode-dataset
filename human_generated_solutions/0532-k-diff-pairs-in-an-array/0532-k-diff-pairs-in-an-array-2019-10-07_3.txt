class Solution(object):
    def findPairs(self, nums, k):
        if k < 0: return 0
        numsSet, pairsSet = set(), set()
        for num1 in nums:
            for num2 in [num1 + k, num1 - k]:
                if num2 in numsSet:
                    pairsSet.add(tuple(sorted([num1, num2])))
            numsSet.add(num1)
        return len(pairsSet)
