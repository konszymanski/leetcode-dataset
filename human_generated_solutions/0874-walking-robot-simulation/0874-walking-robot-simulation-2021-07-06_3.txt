crx=0     #current x axis position of the robot
        cry=0     #current y axis position of the robot
        maxdis=0
        x=0   #currently x=0 and y=1 because we are facing towards north
        y=1
        ss=set()
		
        for _ in obstacles:
            ss.add(tuple(_))   #set of tuples
            
        for _ in commands:
            if _==-2:
                x,y=-y,x   #for anticlockwise turn - eg - if robot turns left, 1,0 becomes 0,1
				
            elif _==-1:
                x,y=y,-x   #for clockwise turn
				
            else:
                for ctr in range(_):
                    if (crx+x,cry+y) in ss:   #find if next step is obstacle
                        break
                    else:
                        crx=crx+x;
                        cry=cry+y;
						
            maxdis=max(maxdis,(crx*crx+cry*cry))
        
        return maxdis
