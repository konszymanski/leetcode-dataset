class Solution:
    def maxResult(self, nums: List[int], k: int) -> int:
        n = len(nums)
        dp = [0] * (n)
        dp[0] = nums[0]
        
        for i in range(1, n):
            best = float(\'-inf\')
            for j in range(max(i-k, 0), i):
                best = max(best, dp[j] + nums[i])
            dp[i] = best
        
        return dp[n-1]
