bool isPalindrome(ListNode* head) {
    // first we iterate over the linked list to get the middle of linked list by two pointers
	ListNode* fast = head;
	ListNode* slow = head;

	while(fast && fast -> next)
	{
		slow = slow -> next;
		fast = fast -> next -> next;
	}

	// reverse the second half of the linked list by two pointers
	ListNode* mid = slow;
	ListNode* prev = NULL;  
	ListNode* next = NULL;

	while(mid)
	{
		next = mid -> next;
		mid -> next = prev;
		prev = mid;
		mid = next;

	}

	ListNode* tail = prev;

	// go over the two linked lists - the first half and the reversed second half
	// and compare the values to check if it\'s palindrome
	while(head && tail && tail != head)
	{
		if(tail -> val != head -> val)
			return false;
		head = head -> next;
		tail = tail -> next;
	}

	return true;
}
