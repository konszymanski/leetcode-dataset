class Solution:
    def orderlyQueue(self, s: str, k: int) -> str:
        if k>1:
            return \'\'.join(sorted(s))
        n=len(s)
        t=s*2 # t=cbacba
        ans=s # ans=cba
        for i in range(1,n):
            s1=t[i:i+n]          # 1st move : s1=t[1:1+3] = bac
            ans=min(ans,s1)      # 2nd move : s1=t[2:2+3] = acb
        return ans
