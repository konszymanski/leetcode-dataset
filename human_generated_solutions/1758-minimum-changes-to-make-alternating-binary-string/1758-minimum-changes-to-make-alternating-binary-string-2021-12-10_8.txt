class Solution:
    def minOperations(self, s: str) -> int:
        return min(self.alt("0",s),self.alt("1",s))
        
    def alt(self,num,string):
        count = 0
        for s in string:
            if s != num:
                count +=1
            num = "0" if num == "1" else "1"
        return count
