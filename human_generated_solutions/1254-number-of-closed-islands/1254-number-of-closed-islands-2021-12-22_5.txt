def closedIsland(self, grid: List[List[int]]) -> int:
        m, n = len(grid), len(grid[0])
        closed_island = 0
        
        def dfs(row: int, col: int) -> bool:
            if row < 0 or row >= m or col < 0 or col >= n:
                return False
            
            if grid[row][col] == 1:
                return True
            
            grid[row][col] = 1
            
            top = dfs(row - 1, col)
            down = dfs(row + 1, col)
            left = dfs(row, col - 1)
            right = dfs(row, col + 1)
            
            return top and down and left and right
        
        for i in range(m):
            for j in range(n):
                if grid[i][j] == 0:
                    closed_island += dfs(i, j)
                    
        return closed_island
