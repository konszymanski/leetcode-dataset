def findMinHeightTrees(self, n, edges):
    """
    :type n: int
    :type edges: List[List[int]]
    :rtype: List[int]
    """
    if n == 1: return [0]
	#create graph
    adj = [set() for _ in xrange(n)]
    for i, j in edges:
        adj[i].add(j)
        adj[j].add(i)
    #find leaves - these are only connected to one other node (this will help us find min height)
    leaves = [i for i in xrange(n) if len(adj[i]) == 1]
    #we then want to remove the leaves from the graph iteratively until we reach the node
    while n > 2:
        #minus the leaves from the total nodes
        n -= len(leaves)
        #newLeaves are the leaves for the next iteration
        newLeaves = []
        #go through each current leaf
        for i in leaves:
            #remove leaf from the graph
            j = adj[i].pop()
            #remove the connection of the leaf to the node its connected to
            adj[j].remove(i)
            #check whether this node is now a leaf node and if it is, append it to newLeaves
            if len(adj[j]) == 1: newLeaves.append(j)
        #update the new leafs in the graph
        leaves = newLeaves
    #return the remaining leaf node which is the node to make the min height of the tree
    return leaves
