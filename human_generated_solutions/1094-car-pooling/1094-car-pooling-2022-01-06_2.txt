class Solution {
public:
    bool carPooling(vector<vector<int>>& trips, int capacity) {
        vector<int> car(1001);
        for (int i = 0; i < trips.size(); i++) {
            car[trips[i][1]] += trips[i][0];
            car[trips[i][2]] -= trips[i][0];
        }
        
        for (auto stop : car) {
            capacity -= stop;
            if (capacity < 0) return false;
        }
        
        return true;
    }
};
