from heapq import heapify, heappush, heappop

class Solution:
#     Runtime: 204ms 95.45%; 15.7mb 49.06%
#      O(logn*n) || O(n)
    def isNStraightHand(self, hand: List[int], groupSize: int) -> bool:
        if not hand:
            return False
        
        if len(hand) % groupSize != 0:
            return False
        
        hashMap = dict()
        
        for i in hand:
            hashMap[i] = 1 + hashMap.get(i, 0)
            
        keys = list(hashMap.keys())
        
        heapify(keys)
        
        while keys:
            pop = keys[0]
            
            for i in range(pop, pop + groupSize):
                if i not in hashMap:
                    return False
                
                hashMap[i] -= 1
                
                if hashMap[i] == 0:
                    if i != keys[0]:
                        return False
                    heappop(keys)
                
                
        return True
