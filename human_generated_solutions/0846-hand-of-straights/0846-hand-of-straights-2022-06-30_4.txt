class Solution:
    def isNStraightHand(self, hand: List[int], groupSize: int) -> bool:
        counter = Counter(hand)
        keys = sorted(list(counter.keys()))
        while keys:
            num = keys[-1]
            counter[num] -= 1
            if counter[num] == 0:
                del counter[num]
                keys.pop()
            for _ in range(groupSize - 1):
                num -= 1
                if num not in counter:
                    return False
                counter[num] -= 1
                if counter[num] == 0:
                    del counter[num]
                    keys.pop()
        return True
