class Solution:
    def numWays(self, words: List[str], target: str) -> int:
        MOD = 10**9+7
        n,m = len(target),len(words[0])
        
        def dfs(i,j):
            if j == n: return 1
            if i == m: return 0
            if n-j > m-i: return 0

            res = 0
            for word in words:
                for k in range(i,m):
                    if word[k] == target[j]:
                        res += dfs(k+1,j+1)

            return res

        return dfs(0,0)%MOD
