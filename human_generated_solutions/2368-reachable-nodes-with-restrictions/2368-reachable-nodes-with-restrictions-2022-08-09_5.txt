def reachableNodes(self, n: int, edges: List[List[int]], restricted: List[int]) -> int:
		graph = defaultdict(list)
        visited = [False]*n
        queue = deque([0])
        res = 0
        
        for u in restricted:
            visited[u] = True
        
        for u,v in edges:
            graph[u].append(v)
            graph[v].append(u)
            
        while queue:
            res += len(queue)
            for _ in range(len(queue)):
                u = queue.popleft()
                visited[u] = True
                queue.extend([v for v in graph[u] if not visited[v]])
        
        return res
