class Solution:
    def rangeSum(self, nums: List[int], n: int, left: int, right: int) -> int:
        sums = []
        for i in range(len(nums)):
             n = 0
             for j in range(i, len(nums)):
                n += nums[j]
                sums.append(n)
        sums.sort()
        
        res = 0
        for i in range(left - 1, right):
            res = (res + sums[i]) % (10 ** 9 + 7)
        
        return res
