class Solution:
    def longestOnes(self, nums: List[int], k: int) -> int:
        
        if not nums:
            return 0
        
        # find the longest subseq that has at most k zeroes
        res = 0
        left = 0
        
        count = 0
        
        for i in range(len(nums)):
            
            cur = nums[i]
            
            # 1. in
            if cur == 0:
                count += 1
            
            # 2. out
            while count > k:
                
                if nums[left] == 0:
                    count -= 1
                
                left += 1
            
            # 3. compute
            res = max(res, i - left + 1)
        
        return res
