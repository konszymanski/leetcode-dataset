class Solution(object):
    def longestOnes(self, A, K):
        i, j, max_length, num_zeros = 0, 0, 0, int(not A[0])
        while j < len(A) - 1:
            if num_zeros > K:
                if A[i] == 0: num_zeros -= 1
                i += 1
            j += 1
            if A[j] == 0: num_zeros += 1
            if num_zeros <= K: max_length = max(max_length, j - i + 1)
        return max_length
