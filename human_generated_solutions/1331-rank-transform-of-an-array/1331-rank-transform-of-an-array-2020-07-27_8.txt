class Solution(object):
    def arrayRankTransform(self, arr):
        """
        :type arr: List[int]
        :rtype: List[int]
        """
        if not arr:
            return []
			
        rank = 1
        hashMap = {}
        result = sorted(arr)
        for num in result:
            if num not in hashMap:
                hashMap[num] = rank
                rank += 1
            
        for idx, val in enumerate(arr):
            arr[idx] = hashMap[val]
			
        return arr
