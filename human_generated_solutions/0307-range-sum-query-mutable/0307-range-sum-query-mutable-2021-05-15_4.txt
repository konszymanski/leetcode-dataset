#[lx,rx] denote the range of the interval of the node.
# left and right denote the left and right children of the node
#sum denotes the sum for the interval represented by the node. Initialized to zero.
class TreeNode:
    def __init__(self,lx,rx):
        self.lx = lx  
        self.rx = rx
        self.left = None 
        self.right = None
        self.sum = 0
