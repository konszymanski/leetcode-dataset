# sc:O(N), addNum\'s tc: O(N) -> due to insert(), getIntervals\'s tc: O(1)
class SummaryRanges:

    def __init__(self):
        self.intervals = []

    def addNum(self, val: int) -> None:
        
        left, right = 0, len(self.intervals) - 1
        # to use binary search to find the right bound in self.intervals where val will be inserted
        while left <= right:
            mid = (left + right) // 2
            e = self.intervals[mid]
            if e[0] <= val <= e[1]: return # if val is in the interval, return
            elif val < e[0]:
                right = mid - 1
            else:
                #  val >= e[0]
                left = mid + 1
        # prev <= val <= next
        pos = left
        self.intervals.insert(pos, [val, val])
		# to check right (next) interval whether to be merged
        if pos + 1 < len(self.intervals) and val + 1 == self.intervals[pos + 1][0]:
            self.intervals[pos][1] = self.intervals[pos + 1][1]
            del self.intervals[pos + 1]
		# to check left (previous) interval whether to be merged
        if pos - 1 >= 0 and val - 1 == self.intervals[pos - 1][1]:
            self.intervals[pos - 1][1] = self.intervals[pos][1]
            del self.intervals[pos]

    def getIntervals(self) -> List[List[int]]:
        return self.intervals
