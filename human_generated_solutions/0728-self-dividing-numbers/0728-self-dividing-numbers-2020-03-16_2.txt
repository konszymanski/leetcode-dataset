def selfDividingNumbers(self, left: int, right: int) -> List[int]:
        return [
            i for i in range(left, right + 1)
            if self.isDiv(i)
        ]
    
    def isDiv(self, i):
        num = i
        while i:
            rem = i%10
            if not rem: return False
            if num%rem != 0: return False
            i = i // 10
        return True
