class Solution:
    def averageWaitingTime(self, customers: List[List[int]]) -> float:
        a=[]
        c=[]
		
        for i in customers:
            a.append(i[0])
            c.append(i[1])
			
        x=min(a)
		
        for i in range(len(c)):
            a[i]=a[i]-x
        t=0
        wt=[]
		
        for i in range(len(a)):
            while(t<a[i]):
                t+=1
        
            t+=c[i]
            wt.append(t-a[i])
       
        return (round(   sum(wt)/len(a)  ,5))
