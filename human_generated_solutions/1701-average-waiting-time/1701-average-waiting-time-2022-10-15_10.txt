class Solution {
public:
    double averageWaitingTime(vector<vector<int>>& customers) {
        int ft = 0;
        double wt = 0;

        for(auto c : customers) {
            ft <= c[0] ? ft = c[0] + c[1] : ft += c[1];
            wt += ft - c[0];
        }
        return wt / customers.size();
    }
};
