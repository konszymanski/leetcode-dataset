class Solution:
    def minDeletionSize(self, strs: List[str]) -> int:
        m, n = len(strs), len(strs[0])
        dp = [1] * n
        for i in range(1, n):
            for j in range(i):
                if all(strs[k][i] >= strs[k][j] for k in range(m)):
                    dp[i] = max(dp[i], dp[j] + 1)
        return n - max(dp)
