MOD = 10**9+7
class Solution:
    def countHomogenous(self, s: str) -> int:
        ans = 0
        prev=None
        count=1
        for i in range(1,len(s)):
                if s[i-1]==s[i]:
                    count+=1
                else:
                    ans+=count*(count+1)//2
                    count=1
        ans+=count*(count+1)//2
        return ans%MOD
