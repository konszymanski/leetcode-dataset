class Solution:
    def countHomogenous(self, s: str) -> int:
        
        answer = 0
        count = 1
        MOD = 10**9 + 7
        
        for i in range(len(s) - 1):
            if s[i] == s[i + 1]:
                count += 1
            else:
                answer += (count * (count + 1) // 2) % MOD
                count = 1
                
        answer += (count * (count + 1) // 2) % MOD
        
        return answer % MOD
