class Solution:
    def uniqueOccurrences(self, arr: List[int]) -> bool:
        # Count of all values in arr
        count = Counter(arr)
        
        # if length of unique values != length of unique elements return False
        if len(set(count.values())) != len(count):
            return False
        
        # If both are equal it means we have unique count return True
        return True
