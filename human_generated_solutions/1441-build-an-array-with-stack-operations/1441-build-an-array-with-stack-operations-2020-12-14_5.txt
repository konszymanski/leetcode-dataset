def buildArray(self, target: List[int], n: int) -> List[str]:
        res = []
        cur = 0
        for val in range(1, n+1):
            if cur == len(target) or val > target[cur]:
                break
            elif val == target[cur]:
                res.append(\'Push\')
                cur += 1
            else:
                res.extend([\'Push\', \'Pop\'])                
        return res
