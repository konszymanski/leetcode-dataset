class Solution:
    def buildArray(self, target: List[int], n: int) -> List[str]:
        a = []
        c = 0
        for i in range(1,n+1):
            if target[c] == i:
                a.append("Push")
                c += 1
            else:
                a.extend(["Push","Pop"])
            if c == len(target):
                break
        return a
