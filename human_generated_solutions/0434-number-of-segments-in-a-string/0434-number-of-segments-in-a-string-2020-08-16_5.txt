class Solution:
    def countSegments(self, s: str) -> int:
        result = 0
        
        # use stack to track segment
        stack = []
        
        for c in s:
            if c != \' \':
                # push non-space character into current segment
                stack.append(c)
            elif stack:
                # pop current segement when encounter space character
                result += 1
                stack = []
        
        # don\'t forget the last segment
        if stack:
            result += 1
        
        return result
