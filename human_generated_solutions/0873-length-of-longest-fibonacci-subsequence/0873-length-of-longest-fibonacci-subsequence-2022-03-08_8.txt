class Solution:
    def lenLongestFibSubseq(self, arr: List[int]) -> int:
        n = len(arr)
        d = {arr[i]:i for i in range(n)}
        best=0
        for i in range(n):         
            for j in range(i+1,n):
                tmp = 2
                a = arr[i]
                b = arr[j]
                while True:
                    if a+b in d:
                        c= a+b
                        a = b
                        b = c 
                        tmp+=1
                    else:break
                if tmp>best:best=tmp
                        
        return best  if best>2 else 0
