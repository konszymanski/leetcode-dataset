def minSteps(self, number: int) -> int:
        # sum of prime factors of n
        
        factor, num_steps = 2, 0 
        M = int(math.sqrt(number))+1
        while factor <= M:
            while number%factor == 0: 
                num_steps += factor
                number = number//factor
            else: 
                # only check odd numbers after 2
                factor += 1 if factor == 2 else 2
                
        if number > 2: 
            # the input number was a prime number
            num_steps += number
            
        return num_steps
