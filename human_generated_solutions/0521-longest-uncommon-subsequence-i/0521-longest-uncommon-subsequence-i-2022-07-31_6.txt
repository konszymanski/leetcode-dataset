def findLUSlength(self, a: str, b: str) -> int:
	def isSub(x, y):
		i = j = 0
		while(i<len(x) and j<len(y)):
			if(x[i] == y[j]):
				i += 1
				j += 1
			else:
				j += 1
		return i != len(x)
	t1, t2 = isSub(a, b), isSub(b, a)
	if(t1 and t2):
		return max(len(a), len(b))
	if(t1):
		return len(a)
	if(t2):
		return len(b)
	return -1
