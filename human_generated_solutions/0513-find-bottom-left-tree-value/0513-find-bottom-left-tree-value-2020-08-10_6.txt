class Solution:
    def findBottomLeftValue(self, root: TreeNode) -> int:
        q=[root]
        ans=None
        while q:
            ans=q[0].val
            for _ in range(len(q)): 
                node=q.pop(0)
                if node.left:q.append(node.left)
                if node.right:q.append(node.right)
        return ans
