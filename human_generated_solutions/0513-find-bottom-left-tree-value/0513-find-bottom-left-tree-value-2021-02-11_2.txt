# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, val=0, left=None, right=None):
#         self.val = val
#         self.left = left
#         self.right = right

import collections

class Solution:
    def findBottomLeftValue(self, root: TreeNode) -> int:
        left = None
        queue = collections.deque([root])
        while queue:
            size = len(queue)
            left = queue[0].val
            for _ in range(size):
                node = queue.popleft()
                if node.left:
                    queue.append(node.left)
                if node.right:
                    queue.append(node.right)
            
            
        return left
