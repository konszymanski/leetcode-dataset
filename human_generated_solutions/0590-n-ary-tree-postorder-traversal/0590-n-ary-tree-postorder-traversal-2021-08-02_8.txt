"""
# Definition for a Node.
class Node:
    def __init__(self, val=None, children=None):
        self.val = val
        self.children = children
"""

class Solution:
    def postorder(self, root: \'Node\') -> List[int]:
        
        ans=[]
        
        def abhi(node):
            
            if not node:
                return 
            
            for i in node.children:
                abhi(i)
                
            ans.append(node.val)
        
        abhi(root)
        return ans
