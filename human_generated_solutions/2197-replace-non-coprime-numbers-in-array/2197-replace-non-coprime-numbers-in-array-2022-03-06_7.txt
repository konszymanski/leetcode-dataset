import math
class Solution:
    def replaceNonCoprimes(self, nums: List[int]) -> List[int]:
        i=1
        while i < len(nums):
            x = math.gcd(nums[i],nums[i-1])
            if x > 1:
                nums[i-1] = nums[i]*nums[i-1]//x
                nums.pop(i)
                i -= 1
            else:
                i += 1
            if i == 0:
                i+= 1
        return nums
