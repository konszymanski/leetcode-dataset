class Solution:

    def __init__(self, m: int, n: int):
        self.m = m
        self.n = n
        self.flipped = Counter()
        

    def flip(self) -> List[int]:
        i, j = random.randint(0, self.m-1), random.randint(0, self.n-1)
        while self.flipped[(i, j)] == 1:
            i, j = random.randint(0, self.m-1), random.randint(0, self.n-1)
        self.flipped[(i, j)] = 1
        return (i, j)

    def reset(self) -> None:
        self.flipped = Counter()
