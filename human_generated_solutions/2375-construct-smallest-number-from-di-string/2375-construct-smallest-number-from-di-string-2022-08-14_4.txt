class Solution:
    def smallestNumber(self, pattern: str) -> str:
        n = len(pattern)
		
		#smallest possible string
        smallest = [f\'{i+1}\' for i in range(n+1)]
        
		#inplace reversal
        def revserse(arr,start,end):
            if start>=end:
                return
            
            arr[start],arr[end] = arr[end],arr[start]
            revserse(arr,start+1,end-1)
        
        i = 0
        while i<n:
		#starting of the sequence of Ds
            if pattern[i]==\'D\':
                j = i #ending index of sequence of Ds
                while j+1<n and pattern[j+1]==\'D\':
                    j+=1
					
				#reverse the sequence
                revserse(smallest,i,j+1)
				
				#move front
                i = j+1
            else:
                i+=1
                
        return "".join(smallest)
