def strWithout3a3b(self, a: int, b: int) -> str:
	res = ""
	while a or b:
		if a >= b:
			if not res or res[-1] != \'a\':
				if a >= 2:
					res += "aa"
					a -= 2
				else:
					res += "a"
					a -= 1
			else:
				res += "b"
				b -= 1
		else:
			if not res or res[-1] != \'b\':
				if b >= 2:
					res += "bb"
					b -= 2
				else:
					res += "b"
					b -= 1
			else:
				res += "a"
				a -= 1
	return res
