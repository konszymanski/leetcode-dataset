import math
class Solution:
    def minimumReplacement(self, nums: List[int]) -> int:
        prev=nums[-1]
        ans=0
        n=len(nums)
        for i in range(n-2,-1,-1):
            noOfTerms=math.ceil(nums[i]/prev)
            if nums[i]%prev!=0:
                prev=nums[i]//noOfTerms
            ans+=noOfTerms-1
        return ans
