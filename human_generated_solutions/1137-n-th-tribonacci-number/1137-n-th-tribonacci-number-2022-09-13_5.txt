class Solution:
		"""
		Time:  O(n)
		Space: O(n)
		"""

		@cache
		def tribonacci(self, n: int) -> int:
			if n < 1:
				return 0
			if n < 3:
				return 1
			return self.tribonacci(n - 1) + self.tribonacci(n - 2) + self.tribonacci(n - 3)

		"""

		"""
