"""
# Definition for a Node.
class Node:
    def __init__(self, val, children):
        self.val = val
        self.children = children
"""
from collections import deque
class Solution:
    def levelOrder(self, root: \'Node\') -> List[List[int]]:
        if root == None:
            return []
        res = []
        mapping = {}
        q = deque([(root,0)])
        while q:
            n,l = q.popleft()
            if l not in mapping:
                mapping[l] = [n.val]
            else:
                mapping[l].append(n.val)
            if n.children:
                for i in n.children:
                    q.append((i,l+1))
        for i in sorted(mapping.keys()):
            res.append(mapping[i])
        return res
