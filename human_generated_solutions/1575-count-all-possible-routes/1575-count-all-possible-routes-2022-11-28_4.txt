class Solution:
    def countRoutes(self, locations: List[int], start: int, finish: int, fuel: int) -> int:
        @lru_cache(None)
        def dp(i, j, f):
            if f < abs(locations[i] - locations[j]): return 0
            return sum(dp(i, k, f - abs(locations[k] - locations[j])) % MOD for k in indices - {j}) + (1 if i != j else 0)
                
        n, indices, MOD = len(locations), set(range(len(locations))), 1000000007
        return (dp(start, finish, fuel) + (1 if start == finish else 0)) % MOD
