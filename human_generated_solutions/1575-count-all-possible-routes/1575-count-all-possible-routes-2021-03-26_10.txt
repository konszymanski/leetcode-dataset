class Solution:
    def countRoutes(self, locations: List[int], start: int, finish: int, fuel: int) -> int:
        n, M = len(locations), 10**9 + 7
        @lru_cache(None)
        def dfs(i, f):
            return (sum([dfs(j,f - abs(locations[i]-locations[j])) for j in range(n) if i != j and f - abs(locations[i] - locations[j]) >= 0] or [0]) + (i == finish)) % M
        return dfs(start, fuel)
