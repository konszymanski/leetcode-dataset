from threading import Event


class Foo:
    def __init__(self):
		# Initialize events for threads
        self.event1 = Event()
        self.event2 = Event()
        

    def first(self, printFirst: \'Callable[[], None]\') -> None:
        printFirst()
		# set flag
        self.event1.set()


    def second(self, printSecond: \'Callable[[], None]\') -> None:
        # wait for flag
		self.event1.wait()
        printSecond()
		# set flag
        self.event2.set()


    def third(self, printThird: \'Callable[[], None]\') -> None:
        # wait for flag
		self.event2.wait()
        printThird()
