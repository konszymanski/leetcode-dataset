class Solution:
    def bestTeamScore(self, scores: List[int], ages: List[int]) -> int:
        zipped = list(zip(scores,ages))
        zipped.sort(key = lambda x:(x[0],x[1]))
        dp = [0]*(len(zipped))
        dp[0] = zipped[0][0]
        for i in range(1,len(zipped)):
            dp[i] = zipped[i][0]
            for j in range(i):
                if zipped[i][1]>=zipped[j][1]:
                    dp[i] = max(dp[i] , dp[j] + zipped[i][0])
        return max(dp)
