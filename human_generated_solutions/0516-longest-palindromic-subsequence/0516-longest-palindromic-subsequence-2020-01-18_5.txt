class Solution:
    def longestPalindromeSubseq(self, s: str) -> int:
        dp = [[0 for i in range(len(s))] for j in range(len(s))]
        
        for i in range(len(dp)-1, -1, -1):
            dp[i][i] = 1
            for j in range(i+1, len(dp)):
                if s[i] == s[j]:
                    dp[i][j] = dp[i+1][j-1]+2
                else:
                    dp[i][j] = max(dp[i][j-1], dp[i+1][j])
                
        return dp[0][-1]
