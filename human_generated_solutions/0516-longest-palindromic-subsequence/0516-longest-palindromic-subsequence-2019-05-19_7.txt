import functools


class Solution:
    def longestPalindromeSubseq(self, s: str) -> int:
        @functools.lru_cache(None)
        def dp(i, j):
            if i == j:
                return 1
            if j < i:
                return 0
            if s[i] == s[j]:
                return 2 + dp(i+1, j-1)
            else:
                return max(dp(i, j-1), dp(i+1, j))
        return dp(0, len(s) - 1)
