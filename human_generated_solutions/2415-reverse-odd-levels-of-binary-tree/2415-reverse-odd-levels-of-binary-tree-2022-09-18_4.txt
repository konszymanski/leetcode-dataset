class Solution:
    def reverseOddLevels(self, root: Optional[TreeNode]) -> Optional[TreeNode]:
        if not root: return None
        
        tree = []
        queue = deque([root])
        level = 0
        
        while queue:
            cur_level = []
            for _ in range(len(queue)):
                node = queue.popleft()
                cur_level.append(node.val)
                if node.left: queue.append(node.left)
                if node.right: queue.append(node.right)
                
            if level % 2 != 0:
                tree += reversed(cur_level)
            else:
                tree += cur_level
            level += 1
                
                
        def build_perfect_tree(array_index):
            if array_index >= len(tree):
                return None
            
            root = TreeNode(tree[array_index])
            root.left = build_perfect_tree(array_index * 2 + 1)
            root.right = build_perfect_tree(array_index * 2 + 2)
            return root
        
        return build_perfect_tree(0)
