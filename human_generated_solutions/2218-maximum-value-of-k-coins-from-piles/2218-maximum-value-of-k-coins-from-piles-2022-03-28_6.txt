class Solution:
    def maxValueOfCoins(self, piles: List[List[int]], k: int) -> int:
        
        @cache
        def fn(i, k): 
            """Return """
            if i == len(piles) or k == 0: return 0 
            ans = fn(i+1, k)
            prefix = 0 
            for j in range(min(k, len(piles[i]))): 
                prefix += piles[i][j]
                ans = max(ans, prefix + fn(i+1, k-j-1))
            return ans 
        
        return fn(0, k)
