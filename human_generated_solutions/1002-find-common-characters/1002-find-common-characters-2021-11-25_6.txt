from collections import defaultdict
class Solution(object):
    def commonChars(self, words):
        """
        :type words: List[str]
        :rtype: List[str]
        """

        
        res = [] 
        d = defaultdict(lambda: 0) 
        # create default dictionary with key as character, and value as the occurence
        # since the constraint is that there is always more than 1 item in the list, we can do this
        for x in words[0]:
            d[x] += 1
        
        
        for key in d: # check each key in the dictinary to see if the character exist in the item in the list
            for word in words[1:]: # since we already looked up the first element, start from the second
                if not key in word: # if key does not exsit in the item, set the occurence as 0
                    d[key] = 0
                else:
                    # check the value of the key in the dictinonary is more than 1
                    # in that case, we need to check how many of the same key(character) exists in the item
                    # and keep the minimun value of the occurence
                    if d[key] > 1:
                        d[key] = min(d[key], word.count(key)) 
                        
        
		# now the dictionary holds the common characters as key and its occurences as value
        for key, value in d.items():
            if value > 0:
                for i in range(value):
                    res.append(key)
        return res
