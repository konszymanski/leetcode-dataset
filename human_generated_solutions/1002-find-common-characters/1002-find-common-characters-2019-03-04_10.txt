def commonChars(self, A: List[str]) -> List[str]:
        dict1 = collections.defaultdict(int)
        dict2 = collections.defaultdict(int)
        k = 0
        
        for letter in A[0]:
            dict1[letter] += 1
        
        for word in A[1:]:
            if k%2 == 0:
                for letter in word:
                    if dict1[letter]:
                        dict2[letter] += 1
                        dict1[letter] -= 1
                dict1.clear()
            else:
                for letter in word:
                    if dict2[letter]:
                        dict1[letter] += 1
                        dict2[letter] -= 1
                dict2.clear()
            k += 1
        
        if k%2 == 0:
            result = [letter for l,cnt in dict1.items() for letter in l*cnt]
        else:
            result = [letter for l,cnt in dict2.items() for letter in l*cnt]

        return result
