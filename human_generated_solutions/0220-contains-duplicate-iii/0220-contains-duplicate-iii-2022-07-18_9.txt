import sortedcontainers
import bisect
class Solution:
    def containsNearbyAlmostDuplicate(self, nums: List[int], k: int, t: int) -> bool:
        
        k +=1
        res = sortedcontainers.SortedList(nums[:k])
        for i in range(1,len(res)):
            if res[i]-res[i-1] <= t :
                return True
        
        
        for i in range(1,len(nums)-k+1):
            
            num = nums[i+k-1]
            res.remove(nums[i-1])
            res.add(nums[i+k-1])
            idx = bisect.bisect_left(res, num)
            
            if (idx-1 >= 0 and num-res[idx-1]<=t) or (idx+1<k and res[idx+1]-num<=t):
                return True
            
        return False
