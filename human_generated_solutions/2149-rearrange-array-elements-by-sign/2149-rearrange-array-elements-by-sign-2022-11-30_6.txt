class Solution:
    def rearrangeArray(self, nums: List[int]) -> List[int]:
        p_numbers = [num for num in nums if num >= 0]
        n_numbers = [num for num in nums if num < 0]
        result = []
        for p_num, n_num in zip(p_numbers, n_numbers):
            result += [p_num, n_num]
        return result
