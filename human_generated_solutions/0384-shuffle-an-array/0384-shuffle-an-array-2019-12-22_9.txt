import random
class Solution:

    def __init__(self, nums: List[int]):
        self.orig = nums[:]
        self.curr = nums[:]

    def reset(self) -> List[int]:
        """
        Resets the array to its original configuration and return it.
        """
        self.curr = self.orig[:]
        return self.curr

    def shuffle(self) -> List[int]:
        """
        Returns a random shuffling of the array.
        """
        temp = []
        while self.curr:
            ind = random.randint(0,len(self.curr)-1)
            self.curr[-1], self.curr[ind] = self.curr[ind], self.curr[-1]
            temp.append(self.curr.pop())
        self.curr = temp[:]                
        return self.curr
