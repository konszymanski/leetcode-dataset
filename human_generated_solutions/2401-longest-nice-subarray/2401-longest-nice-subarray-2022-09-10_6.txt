class Solution {
    public int longestNiceSubarray(int[] nums) {
        int count = 0, startidx = 0, val = 0;
        for(int i=0; i<nums.length; i++){
            while((val & nums[i]) != 0){
                val ^= nums[startidx++];
            }
            val |= nums[i];
            count = Math.max(count, i - startidx + 1);
        }
        return count;
    }
}
