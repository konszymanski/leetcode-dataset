def totalMoney(self, n: int) -> int:
        """
            thought:
                There is a pattern that every seven day appears.
                This pattern we can use Trapezoidal formula to solve.
                And the rest of day we just sum up.
        """
        
        remains = n % 7
        head = n // 7 + 1
        ans = 0
        
        for i in range(1, head):
            ans += sum([i for i in range(i, i+7)])
        for i in range(head, head + remains):
            ans += i
        
        return ans
