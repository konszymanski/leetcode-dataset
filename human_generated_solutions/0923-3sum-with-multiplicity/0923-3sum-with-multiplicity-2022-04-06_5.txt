n = len(arr)
mode = 1e9 + 7
output = 0

for i in range(n - 2):
    if arr[i] > target:
        continue
    for j in range(i + 1, n - 1):
        if arr[j] + arr[i] > target:
            continue
        for k in range(j + 1, n):
            if arr[i] + arr[j] + arr[k] == target:
                output += 1
                
    output %= mode
    
return int(output)
