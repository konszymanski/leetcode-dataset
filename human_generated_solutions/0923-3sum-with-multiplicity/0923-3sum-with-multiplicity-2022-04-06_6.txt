given target;
given arr[n]; # sorted array

i = 0;
j = n-1;
ans = 0;
while i<j:
	value1 = arr[i];
	value2 = arr[j];
	sum = value1 + value2
	if sum < target:
		# we need to shift a pointer to next greater element
		i++;
	else if sum > target:
		# we need to shift a pointer to a lower element
		j--;
	else:
		# for the case that sum==target
		ans++
		i++
		j--

return ans
