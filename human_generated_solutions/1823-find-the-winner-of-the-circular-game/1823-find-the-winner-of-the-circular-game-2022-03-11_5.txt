# Runtime: 45 ms, faster than 77.07% of Python3 online submissions for Find the Winner of the Circular Game.
# Memory Usage: 13.9 MB, less than 61.73% of Python3 online submissions for Find the Winner of the Circular Game.
class Solution:
    def findTheWinner(self, n, k):
        result, start = [i for i in range(1,n+1)], 0
        while len(result)>1:
            loser = (start+k-1)%len(result)
            del result[loser]
            start = loser if loser<=len(result) else 0
        return result[0]
