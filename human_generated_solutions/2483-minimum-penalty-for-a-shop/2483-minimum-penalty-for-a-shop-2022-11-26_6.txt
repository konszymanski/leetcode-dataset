class Solution {
public:
    int bestClosingTime(string customers) {
        int ans = 0, prefix = count(customers.begin(), customers.end(), \'Y\'), least = prefix; 
        for (int i = 0; i < customers.size(); ++i) {
            if (customers[i] == \'N\') ++prefix; 
            else --prefix; 
            if (prefix < least) ans = i+1, least = prefix; 
        }
        return ans; 
    }
};
