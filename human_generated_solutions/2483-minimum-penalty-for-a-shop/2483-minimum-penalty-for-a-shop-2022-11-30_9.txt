class Solution:
    def bestClosingTime(self, customers: str) -> int:
        lossCustomersAfterClosing = customers.count("Y")
        if lossCustomersAfterClosing == 0:
            return 0
        minCost = lossCustomersAfterClosing 
        minIndex = 0
        lossCustomerBeforeClosing = 0
        for index, char in enumerate(customers):
            if char == "N":
                # customer didn\'t visit
                lossCustomerBeforeClosing += 1
            else:
                lossCustomersAfterClosing -= 1
            currentCost = lossCustomerBeforeClosing + lossCustomersAfterClosing
            if currentCost < minCost:
                minCost = currentCost
                minIndex = index + 1
        return minIndex
