#sliding window problem
        
        windowStart = 0
        windowSum = 0.0
        maxWindowSum = float(-inf)
        
        for windowEnd in range(len(nums)):
            windowSum += nums[windowEnd]
            
            if windowEnd >= k-1:
                avg = windowSum/k
                if avg > maxWindowSum:
                    maxWindowSum = avg
                    
                windowSum -= nums[windowStart]
                windowStart += 1
        
        return maxWindowSum
