class Solution(object):
    def getTargetCopy(self, original, cloned, target):
        """
        :type original: TreeNode
        :type cloned: TreeNode
        :type target: TreeNode
        :rtype: TreeNode
        """
        p, children = [cloned], []
        while p:
            for x in p:
                if x.val == target.val:
                    return x
                if x.left:
                    children.append(x.left)
                if x.right:
                    children.append(x.right)
            p, children = children, []

        return None
