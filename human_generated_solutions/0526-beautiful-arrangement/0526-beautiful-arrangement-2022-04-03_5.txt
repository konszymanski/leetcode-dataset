class Solution:
    def countArrangement(self, n: int) -> int:
        ans=[]
        def permute(l,p,index):
            if l==[]:
                ans.append(p)
            for i,v in enumerate(l):
                if v%(index+1)==0 or (index+1)%v==0:
                    permute(l[:i]+l[i+1:],p+[v],index+1)
        
        permute(list(range(1,n+1)),[],0)
        return len(ans)
