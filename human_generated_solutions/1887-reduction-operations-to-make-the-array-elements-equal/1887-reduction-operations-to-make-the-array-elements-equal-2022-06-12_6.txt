class Solution:
    def reductionOperations(self, nums: List[int]) -> int:
        
        d_ = {}
        
        for num in nums:
            if num in d_:
                d_[num] += 1    
            else:
                d_[num] = 1
                
        d = sorted(d_.keys(), reverse=1)
        len_d = len(d)
        operations, i = 0, 0
        
        for k in d:
            i += 1
            operations += (d_[k])*(len_d-i)
            
        return operations
