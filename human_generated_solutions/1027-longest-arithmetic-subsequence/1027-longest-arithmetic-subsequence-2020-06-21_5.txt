class Solution:
    def longestArithSeqLength(self, nums: List[int]) -> int:
        ans = 0
        dp = defaultdict(lambda: 1)
        for i in range(len(nums)-1, -1, -1):
            for j in range(i+1, len(nums)):
                d = nums[j] - nums[i]
                dp[i,d] = max(dp[i,d], 1 + dp[j,d])
                ans = max(dp[i,d], ans)
        return ans
