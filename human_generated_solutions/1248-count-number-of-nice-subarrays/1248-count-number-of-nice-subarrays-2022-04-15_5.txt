class Solution:
    """
    approach: 
    Firstly I can think of a brute force approach that will have time complexity of O(n^3)
    The idea is to exhaust all the possible subarrays and then find the ones having k odd numbers
    if subarray has k odd numbers, add 1 to count
    The outer loop will signify the subarray ending at i
    The inner loop will signify the subarray starting at j
    The innermost loop will be used to calculate the number of odd terms in subarray nums[j..i]
    return the count
    
    Once I thought of above approach, we can try to remove the innermost loop by storing the 
    number of odd numbers till nums[i] position in another array and use that array for lookup
    dp[i] will signify the number of odd numbers till nums[i]
    Then we can have the running time complexity of O(n^2)
    we need to find indexes where dp[i] - dp[j] == k where i ranges from 0 to len(nums) 
    and j ranges from 0 to i
    
    Third approach:
    we can identify the number of subarrays in a single pass
    keep track of the indexes where odd no has occured.
    basically, we need to keep track of the first odd occurence (of the k odd numbers)
    and one previous to the first odd occurence (of the k odd numbers)
    example: 1 2 2 1 1 2 1
    k = 2
    we will start iterating, 
    for i = 0, odd_indices = [0]
    for i = 1, odd_indices = [0]
    for i = 2, odd_indices = [0, 3]
        since len(odd_indices) == k:
        count+=1
    for i = 3, odd_indices = [0, 3, 4]
        since len(odd_indices) > k:
            update j = min(odd_indices) #j = 0
            odd_indices.pop(j)
            count += odd_indices[0] - j
    for i = 4, odd_indices = [3, 4]
        count += odd_indices[0] - j # 3 - 0
        
    for i = 5, odd_indices = [3, 4, 6]
        since len(odd_indices) > k:
            update j = min(odd_indices) # j= 3
            odd_indices.pop(j)
            count += odd_indices[0] - j # 4 - 3
    """
    # O(n^3) approach
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        dp = [0 for _ in range(len(nums)+1)]
        for i in range(1, len(nums)+1):
            if nums[i-1] % 2 != 0:
                dp[i] = dp[i-1] + 1
            else:
                dp[i] = dp[i-1]
        count = 0
        for i in range(1, len(nums)+1):
            for j in range(1, i+1):
                if dp[i] - dp[j-1] == k:
                    count+=1
                elif dp[i] - dp[j-1] <k:
                    break
        return count
    
    #--------------------------
    # O(n^2) approach
    #---------------------------
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        odd_count_so_far = 0
        seen_k_odd = False
        for i in range(len(nums)):
            if nums[i] % 2 != 0:
                # nums[i] is odd then check the odd_count seen so far
                odd_count_so_far += 1
                if odd_count_so_far == k:
                    count = count + 1 
                elif odd_count_so_far > k:
                    
                if seen_k_odd:
                    count = count + odd_count_so_far
                if not seen_k_odd:
                    odd_count_so_far+=1
                    if odd_count_so_far == k:
                        count+=1
                        seen_k_odd = True
    
    #----------------------
    # O(n) approach
    #----------------------
    def numberOfSubarrays(self, nums: List[int], k: int) -> int:
        j = -1
        odd_indices = []
        count = 0
        for i in range(len(nums)):
            if nums[i] % 2 != 0:
                odd_indices.append(i)
            if len(odd_indices) == k:
                count += odd_indices[0] - j
            elif len(odd_indices) > k:
                j = odd_indices[0]
                odd_indices.pop(0)
                count += odd_indices[0] - j
        return count
