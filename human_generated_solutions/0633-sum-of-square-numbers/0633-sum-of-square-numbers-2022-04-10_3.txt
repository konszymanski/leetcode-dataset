def judgeSquareSum(self, c: int) -> bool:
    for a in range(int(sqrt(c)) + 1):
        target = c - a**2
        l, r = 0, sqrt(c)
        while l <= r:
            b = l + (r - l) // 2
            if b*b == target:
                return True
            elif b*b > target:
                r = b - 1
            else:
                l = b + 1
    return False
