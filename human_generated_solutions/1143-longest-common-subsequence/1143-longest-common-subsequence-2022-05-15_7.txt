def f(ind1,ind2,s,t):
    if(ind1<0 or ind2<0):
        return 0
    if(s[ind1] == t[ind2]):
        return 1 + f(ind1-1,ind2-1,s,t)
    return max(f(ind1-1,ind2,s,t),f(ind1,ind2-1,s,t))
	

class Solution:
    def longestCommonSubsequence(self, text1: str, text2: str) -> int:
        n = len(text1)
        m = len(text2)
        return f(n-1,m-1,text1,text2)
