        hmap = defaultdict(list)

            q.append(res_temp.pop())
        # End of Post order traversal Recie

       # Actual Logic begins
        
        while q:
            node = q.popleft()
            if not node.left and not node.right:
                hmap[node].append(1)
            else:
                left = hmap[node.left] if node.left else []
                right = hmap[node.right] if node.right else []

                for l in left:
                    for r in right:
                        if l + r <= distance:
                            count += 1
                
                heights = left + right
                heights = [c+1 for c in heights]
                hmap[node] = heights
                    
            
        return count
