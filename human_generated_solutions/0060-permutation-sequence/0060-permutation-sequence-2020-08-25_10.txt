class Solution:
    def getPermutation(self, n: int, k: int) -> str:
        res = ""
		# 0 index:
        k = k-1
        take = [str(i) for i in range(1,n+1)]
		# store the factorial of n:
        fact = [1 for _ in range(n+1)]
        for i in range(1,n):
            fact[i] = fact[i-1] * i
        for i in range(1,n+1):
            target = k // fact[n-i]
            k = k % fact[n-i]
			# target is an index of the new "take" 
            res += take.pop(target)
        return res
