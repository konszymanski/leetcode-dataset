class Solution:
    def getPermutation(self, n: int, k: int) -> str:
        perm, fact = "", factorial(n)
        nums = [ i for i in range(1, n+1) ]  #to keep track of the options left
        for i in range(n, 0, -1):
            fact //= i
            cur_pos = (k-1) // fact  #index of next number based on possibilities left
            perm += str( nums[cur_pos] )
            del nums[cur_pos]  #get rid of the number we used
            k %= fact
        return perm
