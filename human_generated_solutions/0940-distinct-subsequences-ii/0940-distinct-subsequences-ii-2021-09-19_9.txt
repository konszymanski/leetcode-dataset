def distinctSubseqII(self, s: str) -> int:
	l = len(s)

	dp = [0]*(l+1)
	dp[0] = 1
	dic = {}
	for i in range(1, l+1):
		dp[i] = 2*dp[i-1]
		if s[i-1] in dic:
			dp[i]-=(dp[dic[s[i-1]]-1])
		dp[i]%=1000000007
		dic[s[i-1]] = i
	#you have to return the number of non empty subsequences
	#but the result will have 1 empty string, so you have to remove that
	return dp[l]-1
