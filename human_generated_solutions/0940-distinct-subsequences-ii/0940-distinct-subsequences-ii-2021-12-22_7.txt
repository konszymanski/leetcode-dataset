def distinctSubseqII(self, s: str) -> int:
        letters = []
        for i in range(ord(\'a\'), ord(\'a\') + 26):
            letters.append(chr(i))
        
        dp = defaultdict(int)
        for i in range(len(s)):
            key = s[i]
            tmp = 0
            for letter in letters:
                if letter == key:
				    # if previous sequence ends with the same letter, we can make one more unique sequence ending with the letter
                    tmp += dp[key] + 1
                else:
				    # if previous sequence ends with different letter, append current letter, still unique
                    tmp += dp[letter]       
            dp[key] = tmp     

        res = 0
        for key in dp:
            res += dp[key]
        
        modulo = pow(10,9) + 7
        
        return res % modulo
