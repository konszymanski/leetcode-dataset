class Solution:
    def findRestaurant(self, list1: List[str], list2: List[str]) -> List[str]:
        temp = defaultdict(list)
        ind = 0
        for i in list1:
            if i in list2:
                ind = list1.index(i)+list2.index(i)
                temp[ind].append(i)
        return(temp[min(list(temp))])
