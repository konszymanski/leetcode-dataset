class Solution:
    def xorAllNums(self, nums1: List[int], nums2: List[int]) -> int:
        n1=len(nums1)%2
        n2=len(nums2)%2
        res1=0
        res2=0
        for i in range(len(nums1)):
            res1^=nums1[i]
        for i in range(len(nums2)):
            res2^=nums2[i]
        if(n1==0 and n2==0):
            return 0
        elif(n1==1 and n2==0):
            return res2
        elif(n1==0 and n2==1):
            return res1
        else:
            return res1^res2
