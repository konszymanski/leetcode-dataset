class Solution {
    public List<Boolean> prefixesDivBy5(int[] nums) {
        int val = 0;
        List<Boolean> list = new ArrayList<>();
        for(int num : nums){
            // % 5 to prevent integer overflow, otherwsie will need to use BigInteger
            val = ((val * 2) + num) % 5;
            list.add(val == 0);
        }
        return list;
    }
}
