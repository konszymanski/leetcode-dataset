class Solution:
    table = dict()
    def numberOfSteps (self, num: int) -> int:
        
        step = 0
        while num != 0 :
            
            step += 1
            
            if num & 1 == 1:
                # odd number, subtract by 1
                num -= 1

            else:
                # even number, divide by 2 <=> right shift one bit
                num >>= 1
        
        return step
