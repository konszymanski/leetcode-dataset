class Solution:
    def findKthNumber(self, m: int, n: int, k: int) -> int:
        low, high = 1, m * n
        while low < high:
            mid = (low + high) // 2
            # how many element is smaller than or equal to current mid
            count = sum(min(mid // i, n) for i in range(1, m + 1)) 
            if count >= k:
                high = mid
            else:
                low = mid + 1
        
        return low
