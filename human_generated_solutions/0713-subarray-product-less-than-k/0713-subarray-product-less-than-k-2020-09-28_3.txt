class Solution {
public:
    int numSubarrayProductLessThanK(vector<int>& nums, int k) {
        int count = 0;
        int n = nums.size();
        
        for(int i = 0; i<n; i++){
            int product = nums[i];
            if(nums[i] < k ) count++;
            for(int j = i+1; j<n; j++){
                product *= nums[j];
                if(product >= k) break;
                else count++;
            }
        }
        return count;
    }
};
