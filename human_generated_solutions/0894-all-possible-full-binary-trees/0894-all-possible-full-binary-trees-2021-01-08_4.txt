class Solution:
    def allPossibleFBT(self, N: int) -> List[TreeNode]:
        dp = [[] for _ in range(N + 1)]
        dp[1] = [TreeNode()]
        for i in range(1, N + 1, 2):
            for j in range(1, i, 2):
                for tree1 in dp[j]:
                    for tree2 in dp[i - j - 1]:
                        dp[i].append(TreeNode(0, tree1, tree2))
        return dp[-1]
