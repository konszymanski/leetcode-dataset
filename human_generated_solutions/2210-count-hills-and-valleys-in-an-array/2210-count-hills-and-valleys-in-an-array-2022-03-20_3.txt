int countHillValley(vector<int>& ns) {
    //Step 1
    std::vector<int> nums;
    int l = ns[0];
    nums.push_back(l);
    for (int n : ns) {
        if (n != l) {
            nums.push_back(n);
            l = n;
        }
    }
        
    //Step 2.
    int ret = 0;
    for (int i = 1; i < nums.size() - 1; i++) {
        if (nums[i-1] < nums[i] == nums[i+1] < nums[i]) ret++;
    }
    return ret;
}
