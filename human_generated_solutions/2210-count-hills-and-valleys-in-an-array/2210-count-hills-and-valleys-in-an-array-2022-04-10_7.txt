def countHillValley(self, nums: List[int]) -> int:
        nums1 = [nums[0]]
        for i in range(1, len(nums)):
            if nums[i] != nums1[-1]:
                nums1.append(nums[i])
        cnt = 0
        for i in range(1, len(nums1) - 1):
            if nums1[i - 1] < nums1[i] > nums1[i + 1] or nums1[i - 1] > nums1[i] < nums1[i + 1]:
                cnt += 1
        return cnt
