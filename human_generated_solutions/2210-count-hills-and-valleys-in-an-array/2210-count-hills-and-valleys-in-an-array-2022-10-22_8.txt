class Solution:
    def countHillValley(self, nums: List[int]) -> int:
        left, right = 1, 2
        total = 0
        while right < len(nums):
            if nums[left] != nums[right]:
                if (nums[left-1] < nums[left] and nums[right] < nums[left]) or (nums[left-1] > nums[left] and nums[right] > nums[left]):
                    total += 1
                left = right
            right += 1
        return total
