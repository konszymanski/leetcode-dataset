class Solution:
    def countHillValley(self, nums: List[int]) -> int:
        increasing, changed = -1, 0
        for i in range(1, len(nums)):
            if nums[i] > nums[i - 1]:
                changed += int(increasing == 0)
                increasing = 1
            elif nums[i] < nums[i - 1]:
                changed += int(increasing == 1)
                increasing = 0
        return changed
