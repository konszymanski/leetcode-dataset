class Solution(object):
    def largestValues(self, root):
        """
        :type root: TreeNode
        :rtype: List[int]
        """
        
        res = []
        
        def helper(node, level):
            if not node:
                return
            if len(res) <= level:
                res.append(node.val)
            res[level] = max(res[level], node.val)
            helper(node.left, level + 1)
            helper(node.right, level + 1)
            
        helper(root, 0)
        return res
