class Solution:
    def isRectangleOverlap(self, rec1: List[int], rec2: List[int]) -> bool:
        """
        ---------(x2(2),y2(3))
        |                 |
        |                 |
        (x1(0),y1(1))-----
        """
        if rec2[0] >= rec1[2] or rec2[1]>= rec1[3] or rec2[2] <= rec1[0] or rec2[3] <= rec1[1]:
            return False
        else:
            return True
