class Solution:
    def longestUnivaluePath(self, root: Optional[TreeNode]) -> int:
        max_num = 0
        
        def dfs(node):
            nonlocal max_num
            
            if not node:
                return 0
            
            left = dfs(node.left)
            if node.left and node.left.val == node.val:
                left += 1
            else:
                left = 0
            right = dfs(node.right)
            if node.right and node.right.val == node.val:
                right += 1
            else:
                right = 0
                
            max_num = max(max_num, left + right)
            
            # recall that we can only split ONCE! Hence the max of left and right
            return max(left, right)
        
        dfs(root)
        
        return max_num
