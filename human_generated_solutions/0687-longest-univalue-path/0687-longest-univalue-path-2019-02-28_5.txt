def longestUnivaluePath(self, root):
        """
        """
        def dfs(node, prev_val):
            if node is None:
                return 0
            left=dfs(node.left, node.val)
            right=dfs(node.right, node.val)
            self.max_val=max(self.max_val, left+right)
            if node.val==prev_val:
                return max(left, right)+1
            return 0
        self.max_val=0
        if root is None:
            return 0
        dfs(root,root.val)
        return self.max_val
