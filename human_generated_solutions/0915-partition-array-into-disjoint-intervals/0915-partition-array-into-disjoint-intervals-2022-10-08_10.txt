def partitionDisjoint(self, nums: List[int]) -> int:
        l, r = 0,0 
        LENGTH = len(nums)
        largest = nums[0]
        while r < LENGTH -1:
            while r + 1 < LENGTH and nums[r+1] >= largest:
                r += 1
            l = l + 1 
            r = l if l > r else r 
            largest = max(nums[l],largest)
        return l
