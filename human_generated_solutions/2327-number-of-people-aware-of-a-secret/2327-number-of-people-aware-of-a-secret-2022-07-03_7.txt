//here  x denotes the day a particular person got to know truth
//and pd denotes the present day

  int mod=1e9+7;
    int dp[1004][1004];
    
    int solve(int x,int pd,int n,int d,int f)
    {
        if(pd>n) return 1;
        //if he doesn\'t forget  till n
		
        if(pd-x==f) return 0;
        //if he forgets then return
		
		
       if(dp[x][pd]!=-1) return dp[x][pd];
        
        long long int ans=0;
        
         
       //if time lapse between  the day he got to know truth and present day is less tahn delay time 
	   //he can\'t spread it
        if(pd-x<d)
        {
            ans=solve(x,pd+1,n,d,f);
        }
        else{
            
        ans=(solve(pd,pd+1,n,d,f)%mod+solve(x,pd+1,n,d,f)%mod)%mod;
            
            }
			//else he is spreading here and to a differnt person and also keep spreading until he forgets 
        
        return dp[x][pd]=ans;
        
    }
    
    int peopleAwareOfSecret(int n, int delay, int forget) {
        
        memset(dp,-1,sizeof(dp));
        
        return solve(1,1,n,delay,forget) ;
    }
