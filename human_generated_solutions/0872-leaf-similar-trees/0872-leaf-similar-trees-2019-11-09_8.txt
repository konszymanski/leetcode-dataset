class Solution:
    def leafSimilar(self, root1: TreeNode, root2: TreeNode) -> bool:
        def helper(root):
            if root.left or root.right:
                if root.left: 
					yield from helper(root.left)
                if root.right: 
					yield from helper(root.right)
            else:
                yield root.val
                
        return all(i == j for i, j in itertools.zip_longest(helper(root1), helper2(root2), fillvalue=None)
