# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None
from collections import deque
class Solution:
    def findTarget(self, root: TreeNode, k: int) -> bool:
        if not root:
            return False
        queue, seen = deque([root]),set()
        while queue:
            cur = queue.popleft()
            if cur.val in seen:
                return True
            else:
                seen.add(k-cur.val)
                if cur.left: queue.append(cur.left)
                if cur.right: queue.append(cur.right)
        return False
