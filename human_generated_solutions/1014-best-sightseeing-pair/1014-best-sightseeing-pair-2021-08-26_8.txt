class Solution:
    def maxScoreSightseeingPair(self, values: [int]) -> int:
        i = 0 
        maxScore = float(\'-inf\')
        for j in range (1,len(values)): 
            distance = j - i
            localScore = values[j] + values[i] - distance
            maxScore = max(maxScore, localScore) 
            delta = values[j] - values[i] + distance 
            if delta > 0: 
                i = j 
        return maxScore
