class Solution:
    def maxScoreSightseeingPair(self, nums: List[int]) -> int:
        a = nums[0]-1
        b = nums[1]
        ans = a+b
        for i in range(2, len(nums)):
            a -= 1
            b -= 1
            a,b,c = sorted([a,b, nums[i]], reverse=True)
            ans = max(ans, a+b)
        return ans
