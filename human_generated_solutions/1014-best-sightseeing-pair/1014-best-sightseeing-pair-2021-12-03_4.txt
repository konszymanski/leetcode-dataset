class Solution:
    def maxScoreSightseeingPair(self, values: List[int]) -> int:
        max_value = 0
        i = 0
        for j in range(1, len(values)):
            value = values[i] + values[j] + i - j
            max_value = max(max_value, value)
            if values[i] - (j - i) < values[j]:
                i = j
                
        return max_value
