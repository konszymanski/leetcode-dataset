class Solution:
    def fullBloomFlowers(self, flowers: List[List[int]], persons: List[int]) -> List[int]:
        ans = [0] * len(persons)
        events = [evt for start, end in flowers for evt in [(start, 0, -1), (end + 1, 1, -1)]] + [(day, 2, person) for person, day in enumerate(persons)]
        events.sort()
        blooming = 0
        for day, event_type, person in events:
            if event_type == 0:
                blooming += 1
            elif event_type == 1:
                blooming -= 1
            else:
                ans[person] = blooming
        return ans
