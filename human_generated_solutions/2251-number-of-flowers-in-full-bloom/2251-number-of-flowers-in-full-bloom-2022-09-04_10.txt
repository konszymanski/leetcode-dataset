class Solution:
    def fullBloomFlowers(self, flowers: List[List[int]], persons: List[int]) -> List[int]:
        startBloom, endBloom = [], []
        for flower in flowers:
            startBloom.append(flower[0])
            endBloom.append(flower[1] + 1)
        startBloom.sort()
        endBloom.sort()
        res = []
        for person in persons:
            numFlowersStartedBlooming = bisect.bisect_right(startBloom, person)
            numFlowersStoppedBlooming = bisect.bisect_right(endBloom, person)
            res.append(numFlowersStartedBlooming - numFlowersStoppedBlooming)
        return res
