class Solution:
    def minimumSize(self, nums: List[int], maxOperations: int) -> int:
        
        def isPossible(k,mxop=maxOperations):
            for i in range(len(nums)):
                if nums[i]>k:
                    if mxop <= 0:
                        return False
                    mxop-= math.ceil(nums[i]/k)-1
            if mxop<0:
                return False
            return True
        left = 1
        right = max(nums)
        ans = right
        while left<=right:
            mid = left + (right-left)//2
            if isPossible(mid):
                ans = min(ans, mid)
                right = mid-1
            else:
                left = mid+1
        return ans
