class Solution { // DFS: in-order traverse
public: // Time/Space Complexity: O(N); O(N)
    int rangeSumBST(TreeNode* root, int low, int high) {
        vector<int> nums;
        dfs(root, nums);
        int ans = 0;
        for(auto e: nums) 
            if(e >= low && e <=high) ans += e;
        return ans;
    }
    
private:
    void dfs(TreeNode* root, vector<int>& nums){
        if(!root) return;
        dfs(root->left, nums);
        nums.push_back(root->val);
        dfs(root->right, nums);
    }
};
