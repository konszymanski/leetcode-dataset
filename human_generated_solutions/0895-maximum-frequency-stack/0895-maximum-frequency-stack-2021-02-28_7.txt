class FreqStack:

    def __init__(self):
        self.dic={}
        self.stack=[]
    def push(self, x: int) -> None:
        self.stack.append(x)
        self.dic[x]=self.dic.get(x,0)+1
    
    def pop(self) -> int:
        maxi=max(self.dic.values())
        i= len(self.stack)-1

        while i >= 0:
            if self.dic[self.stack[i]] == maxi:
                self.dic[self.stack[i]]-=1
                return self.stack.pop(i)
            i-=1
