class Solution {
public:
    vector<int> buildArray(vector<int>& nums) {
        int n=nums.size();
        vector<int>arr(n);
        int i=0;
        int a=0;        
        while (i<n){
            a=nums[i];
            arr[i]=nums[a];
            i++;
            a==0;
        }
        return arr;
        
    }
};
