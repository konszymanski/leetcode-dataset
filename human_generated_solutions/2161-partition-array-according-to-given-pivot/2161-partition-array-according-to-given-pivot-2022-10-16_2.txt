class Solution:
    def pivotArray(self, nums: List[int], pivot: int) -> List[int]:
        n=len(nums)
        a=[0]*n
        l,r=0,n-1
        for i in range(n):
            if nums[i]<pivot:
                a[l]=nums[i]
                l+=1
            if nums[n-i-1]>pivot:
                a[r]=nums[n-i-1]
                r-=1
        while l<=r:
            a[l]=pivot
            l+=1
        return a
