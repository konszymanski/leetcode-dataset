"""https://leetcode.com/problems/minimum-number-of-swaps-to-make-the-string-balanced/"""
class Solution:
    def minSwaps(self, s: str) -> int:
        close,maxClose=0,0
        for i in s:
            if i=="[":
                close-=1
            else:
                close+=1
            maxClose=max(close,maxClose)
        return (maxClose+1)//2 # we will do this because swapping 1 close brackets neutralizes one more. Example:- ]][[, if we swap ONE ] bracket, it will become [][], so it neutralizes TWO , so we will divide our maxCLose by TWO by ADDING one because we need ONE operation of swapping
