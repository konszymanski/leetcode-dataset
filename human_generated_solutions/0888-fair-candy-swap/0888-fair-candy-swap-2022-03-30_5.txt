def fairCandySwap(self, aliceSizes: List[int], bobSizes: List[int]) -> List[int]:
        sum1 = sum(aliceSizes)
        sum2 = sum(bobSizes)
        
        aliceSizes = sorted(aliceSizes)
        bobSizes = sorted(bobSizes)
        
        for a in aliceSizes:
            maybe_b = (2*a - (sum1 - sum2))/2
            b = self.binarySearch(bobSizes, maybe_b)
            if not b:
                continue
            else:
                return [a, b]
                
    def binarySearch(self, nums, n):
        l, r = 0, len(nums)-1
        while l <= r:                     
            mid = (l+r)//2                
            if nums[mid] == n:
                return nums[mid]
            elif nums[mid] > n:
                r = mid-1
            elif nums[mid] < n:
                l = mid+1  
        return False
