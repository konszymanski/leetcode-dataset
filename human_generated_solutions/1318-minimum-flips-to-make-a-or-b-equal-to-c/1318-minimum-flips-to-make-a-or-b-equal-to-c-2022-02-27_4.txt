class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        ans=0
        while(c | a | b):
            if(c&1):
                if((a&1)==0 and (b&1)==0):
                    ans+=1
            
            else:
                if(a&1 and b&1):
                    ans+=2;
                elif(a&1 or b&1):
                    ans+=1
            
            c>>=1
            b>>=1
            a>>=1
        
        return ans;
