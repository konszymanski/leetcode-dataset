class Solution:
    def minFlips(self, a: int, b: int, c: int) -> int:
        count = 0
        while a != 0 or b!= 0 or c!=0:
            x , y, z = b&1,a&1,c&1
            if (x|y)!=z:
                if x == 1 and y == 1:
                    count += 2
                else:
                    count+=1
            b >>= 1
            c >>= 1
            a >>= 1
        return count
