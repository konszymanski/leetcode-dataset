class Trie:
    def __init__(self):
        self.root = {}
        
    def insert(self, word):
        p = self.root
        for char in word:
            if char not in p:
                p[char] = {}
            p = p[char]
        p[\'#\'] = True
    
    def search(self, word):
        p = self.root
        res = \'\'
        for char in word:
            if char in p:
                res += char
                p = p[char]
                if \'#\' in p:
                    break
            else:
                break
        return res if \'#\' in p else word
        
class Solution:
    def replaceWords(self, roots: List[str], sentence: str) -> str:
        t = Trie()
        for root in roots:
            t.insert(root)
        return \' \'.join(map(t.search, sentence.split()))
