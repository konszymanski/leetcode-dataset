class Solution {
    // nums array is sorted
    public int minPatches(int[] nums, int n) {
        long sum = 0;
        int i=0;
        int patch = 0;
        
        while(sum < n) {
            if(i >= (int) nums.length || sum + 1 < nums[i]) {
                patch += 1;
                sum = sum + (sum + 1);
            } else {
                sum = sum + nums[i];
                i++;
            }
        }
        
        return patch;
    }
}
